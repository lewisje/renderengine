component.base.js


Summary

The base class from which all components extend. A component is a piece of functionality used by a HostObject.

 38 Engine.initObject("BaseComponent", "BaseObject", function() {

 40 /**
 41  * @class All components extend from this object class.  A component is one
 42  *        part of an operating whole object (a {@link HostObject}) which is
 43  *        responsible for only a portion of the overall operation of an in-
 44  *        game object.  Components are broken down into five major categories:
 45  *        <ul>
 46  *          <li><b>TYPE_INPUT</b> - Input from controllers (keyboard, mouse, etc.)</li>
 47  *          <li><b>TYPE_TRANSFORM</b> - Performs transformations on the host object</li>
 48  *          <li><b>TYPE_LOGIC</b> - Handles logical operations that are not related to
 49  *              input and collision</li>
 50  *          <li><b>TYPE_COLLIDER</b> - Determines what this object is possibly colliding
 51  *              with, and reports those collisions via callbacks to the host.</li>
 52  *          <li><b>TYPE_RENDER</b> - Performs some sort of rendering operation to the context</li>
 53  *        </ul>
 54  *        Components are executed in the order listed.  First, all inputs are
 55  *        checked, then logic is performed.  Logic may be internal to a host object
 56  *        itself, but some components perform an object-centric type of logic that
 57  *        can be reused.  Next, collisions are checked.  And finally, rendering can
 58  *        occur.
 59  *        <p/>
 60  *        Within each component type set, components can be prioritized so that
 61  *        one component will execute before others.  Such an ordering allows for
 62  *        multiple components of each type to perform their tasks in an order
 63  *        that the host defines.
 64  *
 65  *
 66  * @extends BaseObject
 67  */
 68 var BaseComponent = BaseObject.extend(/** @scope BaseComponent.prototype */{
 69 
 70    priority: 0,
 71 
 72    type: -1,
 73 
 74    host: null,
 75 
 76    /**
 77     * Create a new instance of a component, setting the name, type, and
 78     * update priority of this component compared to all other components
 79     * within the host.
 80     *
 81     * @param name {String} The name of the component
 82     * @param type {Number} The type of the component
 83     * @param priority {Number} A value between 0.0 and 1.0.  Default: 0.5
 84     */
 85    constructor: function(name, type, priority) {
 86       Assert((name != null), "You must assign a name to every Component.");
 87       name = name.toUpperCase();
 88 
 89       Assert((type != null && (type >= BaseComponent.TYPE_INPUT && type <= BaseComponent.TYPE_RENDERING)),
 90              "You must specify a type for BaseComponent");
 91 
 92       this.type = type;
 93 
 94       Assert((priority != null && (priority >= 0.0 && priority <= 1.0)),
 95              "Priority must be between 0.0 and 1.0 for BaseComponent");
 96 
 97       this.priority = priority || 0.5;
 98       this.base(name);
 99    },
100 
101    /**
102     * Release the object back into the object pool.
103     */
104    release: function() {
105       this.base();
106       this.priority = 0;
107       this.type = -1;
108       this.host = null;
109    },
110 
111    /**
112     * Set the host object this component exists within.
113     *
114     * @param hostObject {HostObject} The object which hosts this component
115     */
116    setHostObject: function(hostObject) {
117       this.host = hostObject;
118    },
119 
120    /**
121     * Get the host object this component exists within.
122     *
123     * @type HostObject
124     */
125    getHostObject: function() {
126       return this.host;
127    },
128 
129    /**
130     * Get the type of this component.  Will be one of:
131     * {@link #TYPE_INPUT}, {@link #TYPE_TRANSFORM}, {@link #TYPE_LOGIC},
132     * {@link #TYPE_COLLIDER}, or {@link #TYPE_RENDERING}
133     *
134     * @type Number
135     */
136    getType: function() {
137       return this.type;
138    },
139 
140    /**
141     * Set the execution priority of this component with
142     * 1.0 being the highest priority and 0.0 being the lowest.  Components
143     * within a host object are sorted by type, and then priority.
144     *
145     * @param priority {Number} A value between 0.0 and 1.0
146     */
147    setPriority: function(priority) {
148       this.priority = priority;
149       this.getHost().sort();
150    },
151 
152    /**
153     * Returns the priority of this component.
154     *
155     * @type Number
156     */
157    getPriority: function() {
158       return this.priority;
159    },
160 
161    /**
162     * Run the component, updating its state.  Not all components will need an execute
163     * method.  However, it is important to include one if you need to run calculations
164     * or update the state of the component each engine cycle.
165     *
166     * @param renderContext {RenderContext} The context the component will render within.
167     * @param time {Number} The global engine time
168     * @param rendering {Boolean} <tt>true</tt> during the rendering phase
169     */
170    execute: function(renderContext, time, rendering) {
171       // Does nothing...
172    },
173 
174    /**
175     * Converts the type name to a string.
176     *
177     * @type String
178     */
179    getTypeString: function() {
180       var ts = "";
181       switch (this.getType()) {
182          case BaseComponent.TYPE_INPUT: ts = "TYPE_INPUT"; break;
183          case BaseComponent.TYPE_TRANSFORM: ts = "TYPE_TRANSFORM"; break;
184          case BaseComponent.TYPE_LOGIC: ts = "TYPE_LOGIC"; break;
185          case BaseComponent.TYPE_COLLIDER: ts = "TYPE_COLLIDER"; break;
186          case BaseComponent.TYPE_RENDERING: ts = "TYPE_RENDERING"; break;
187          default: ts = "TYPE_UNKNOWN";
188       }
189 
190       return ts;
191    }
192 
193 
194 }, /** @scope BaseComponent.prototype */{
195 
196    /**
197     * Get the class name of this object
198     *
199     * @return {String} The string "BaseComponent"
200     */
201    getClassName: function() {
202       return "BaseComponent";
203    },
204 
205    /**
206     * An input component
207     * @type Number
208     */
209    TYPE_INPUT:          1,
210 
211    /**
212     * A transformation (move,rotate,scale) component
213     * @type Number
214     */
215    TYPE_TRANSFORM:      2,
216 
217    /**
218     * A logic component
219     * @type Number
220     */
221    TYPE_LOGIC:          3,
222 
223    /**
224     * A collider component
225     * @type Number
226     */
227    TYPE_COLLIDER:       4,
228 
229    /**
230     * A rendering component
231     * @type Number
232     */
233    TYPE_RENDERING:      5
234 });
235 
236 return BaseComponent;
237 
238 });
