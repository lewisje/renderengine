 34 // Includes
 35 Engine.include("/engine/engine.math2d.js");
 36 Engine.include("/components/component.render.js");

 38 Engine.initObject("SpriteComponent", "RenderComponent", function() {
 39 
 40 /**
 41  * @class A render component that renders its contents from a {@link Sprite}.
 42  * @extends RenderComponent
 43  */
 44 var SpriteComponent = RenderComponent.extend(/** @scope SpriteComponent.prototype */{
 45 
 46    currentSprite: null,
 47 
 48    /**
 49     * @constructor
 50     */
 51    constructor: function(name, priority, sprite) {
 52       this.base(name, priority || 0.1);
 53       this.currentSprite = sprite;
 54    },
 55 
 56    release: function() {
 57       this.base();
 58       this.currentSprite = null;
 59    },
 60 
 61    /**
 62     * Calculate the bounding box from the set of
 63     * points which comprise the shape to be rendered.
 64     * @private
 65     */
 66    calculateBoundingBox: function() {
 67       return;
 68       var x1 = 0;
 69       var x2 = 0;
 70       var y1 = 0;
 71       var y2 = 0;
 72       for (var p = 0; p < this.points.length; p++)
 73       {
 74          var pt = this.points[p];
 75 
 76          if (pt.x < x1)
 77          {
 78             x1 = pt.x;
 79          }
 80          if (pt.x > x2)
 81          {
 82             x2 = pt.x;
 83          }
 84          if (pt.y < y1)
 85          {
 86             y1 = pt.y;
 87          }
 88          if (pt.y > y2)
 89          {
 90             y2 = pt.y;
 91          }
 92       }
 93 
 94       var bbox = new Rectangle2D(x1, y1, Math.abs(x1) + x2, Math.abs(y1) + y2);
 95       this.getHostObject().setBoundingBox(bbox);
 96 
 97       // Figure out longest axis
 98       if (bbox.len_x() > bbox.len_y)
 99       {
100          this.fullBox = new Rectangle2D(x1,x1,Math.abs(x1) + x2,Math.abs(x1) + x2);
101       }
102       else
103       {
104          this.fullBox = new Rectangle2D(y1,y1,Math.abs(y1) + y2,Math.abs(y1) + y2);
105       }
106    },
107 
108    /**
109     * Set the sprite the component will render.
110     *
111     * @param sprite {Sprite} The sprite to render
112     */
113    setSprite: function(sprite) {
114       this.currentSprite = sprite;
115    },
116 
117    /**
118     * Get the sprite the component is rendering.
119     *
120     * @return A {@link Sprite} object
121     */
122    getSprite: function() {
123       return this.currentSprite;
124    },
125 
126    /**
127     * Draw the sprite to the render context.  The frame, for animated
128     * sprites, will be automatically determined based on the current
129     * time passed to the method.
130     *
131     * @param renderContext {RenderContext} The context to render to
132     * @param time {Number} The engine time in milliseconds
133     */
134    execute: function(renderContext, time) {
135 
136       if (!this.base(renderContext, time))
137       {
138          return;
139       }
140 
141       if (this.currentSprite) {
142          renderContext.drawSprite(this.currentSprite, time);
143       }
144    }
145 }, /** @scope SpriteComponent */{
146    /**
147     * Get the class name of this object
148     *
149     * @type String
150     */
151    getClassName: function() {
152       return "SpriteComponent";
153    }
154 });
155 
156 return SpriteComponent;
157 
158 });
