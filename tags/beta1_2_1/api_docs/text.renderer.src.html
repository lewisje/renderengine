text.renderer.js


Summary

A text renderer object that uses a specific render object to produce text when a render context cannot.
Class Summary

 35 // Includes
 36 Engine.include("/engine/engine.object2d.js");
 37 Engine.include("/components/component.transform2d.js");

 39 Engine.initObject("TextRenderer", "Object2D", function() {
 40 
 41 /**
 42  * @class A 2d text rendering object.  The object hosts the given text
 43  *        renderer, and a way to position and size the text.  It is up
 44  *        to the rendered provided to present the text within the render
 45  *        context.
 46  *
 47  * @constructor
 48  * @param renderer {AbstractTextRenderer} The text renderer to use
 49  * @param text {String} The text to render
 50  * @param size {Number} The size of the text to render
 51  * @see VectorText
 52  * @see BitmapText
 53  */
 54 var TextRenderer = Object2D.extend(/** @scope TextRenderer.prototype */{
 55 
 56    drawMode: 0,
 57 
 58    /**
 59     * @private
 60     */
 61    constructor: function(renderer, text, size) {
 62 
 63       Assert((renderer instanceof AbstractTextRenderer), "Text renderer must extend AbstractTextRenderer");
 64 
 65       this.base("TextRenderer");
 66 
 67       // Add components to move and draw the asteroid
 68       this.renderer = renderer;
 69       this.add(this.renderer);
 70       this.add(Transform2DComponent.create("transform"));
 71       this.getComponent("TextRenderObject").setText(text || "");
 72       this.getComponent("transform").setScale(size || 1);
 73    },
 74 
 75    /**
 76     * @private
 77     */
 78    release: function() {
 79       this.base();
 80       this.drawMode = 0;
 81    },
 82 
 83    /**
 84     * Called to render the text to the context.
 85     *
 86     * @param renderContext {RenderContext} The context to render the text upon
 87     * @param time {Number} The engine time in milliseconds
 88     */
 89    update: function(renderContext, time) {
 90 
 91       if (this.drawMode == TextRenderer.DRAW_TEXT)
 92       {
 93          renderContext.pushTransform();
 94          this.base(renderContext, time);
 95          renderContext.popTransform();
 96       }
 97 
 98    },
 99 
100    /**
101     * Set the text for this object to render.
102     *
103     * @param text {String} The text to render.
104     */
105    setText: function(text) {
106       this.getComponent("TextRenderObject").setText(text);
107    },
108 
109    /**
110     * Get the text for this object to render.
111     * @return {String} The text to draw
112     */
113    getText: function() {
114       return this.getComponent("TextRenderObject").getText();
115    },
116 
117    /**
118     * Set the size of the text to render.
119     *
120     * @param size {Number} Defaults to 1
121     */
122    setTextSize: function(size) {
123       this.getComponent("transform").setScale(size || 1);
124    },
125 
126    /**
127     * Get the size of the text to render.
128     * @return {Number} The size/scale of the text
129     */
130    getTextSize: function() {
131       this.getComponent("transform").getScale();
132    },
133 
134    /**
135     * Set the weight (boldness) of the text.
136     *
137     * @param weight {Number} The boldness of the text
138     */
139    setTextWeight: function(weight) {
140       this.getComponent("TextRenderObject").setTextWeight(weight);
141    },
142 
143    /**
144     * Get the weight of the text.
145     * @return {Number} The text weight
146     */
147    getTextWeight: function() {
148       this.getComponent("TextRenderObject").getTextWeight();
149    },
150 
151    /**
152     * Get the position where the text will render.
153     * @return {Point2D} The position of the text
154     */
155    getPosition: function() {
156       return this.getComponent("transform").getPosition();
157    },
158 
159    /**
160     * Set the position where the text will render.
161     *
162     * @param point {Point2D} The position to render the text
163     */
164    setPosition: function(point) {
165       this.getComponent("transform").setPosition(point);
166    },
167 
168    /**
169     * Set the horizontal alignment of the text.
170     *
171     * @param alignment {Number} One of {@link AbstractTextRenderer#ALIGN_LEFT}, {@link AbstractTextRenderer#ALIGN_RIGHT}
172     *                  or {@link AbstractTextRenderer#ALIGN_CENTER}
173     */
174    setTextAlignment: function(alignment) {
175       this.getComponent("TextRenderObject").setAlignment(alignment);
176    },
177 
178    /**
179     * Get the horizontal alignment of the text.
180     * @return {Number} The alignment constant for the text
181     */
182    getTextAlignment: function() {
183       this.getComponent("TextRenderObject").getAlignment();
184    },
185 
186    /**
187     * Set the color of the text to render.
188     *
189     * @param color {String} The color of the text
190     */
191    setColor: function(color) {
192       this.getComponent("TextRenderObject").setColor(color);
193    },
194 
195    /**
196     * Get the color of the text to render
197     * @return {String} The text color
198     */
199    getColor: function() {
200       this.getComponent("TextRenderObject").getColor();
201    },
202 
203    /**
204     * Set the color of the text.
205     *
206     * @param textColor {String} Color of the text.
207     */
208    setTextColor: function(textColor) {
209       this.getComponent("TextRenderObject").setColor(textColor);
210    },
211 
212    /**
213     * Get the text color
214     * @return {String} The color or style of the line
215     */
216    getTextColor: function() {
217       return this.getComponent("TextRenderObject").getColor();
218    },
219 
220    /**
221     * Set the text drawing mode to either {@link #DRAW_TEXT} or {@link #NO_DRAW}.
222     *
223     * @param drawMode {Number} The drawing mode for the text.
224     */
225    setDrawMode: function(drawMode) {
226       this.drawMode = drawMode;
227    },
228 
229    /**
230     * Get the current drawing mode for the text.
231     * @return {Number} The text drawing mode
232     */
233    getDrawMode: function() {
234       return this.drawMode;
235    }
236 }, /** @scope TextRenderer.prototype */{
237 
238    /**
239     * Get the class name of this object
240     * @return {String} The string "TextRenderer"
241     */
242    getClassName: function() {
243       return "TextRenderer";
244    },
245 
246    /**
247     * Draw the text to the context.
248     * @type Number
249     */
250    DRAW_TEXT: 0,
251 
252    /**
253     * Don't draw the text to the context.
254     * @type Number
255     */
256    NO_DRAW: 1
257 });
258 
259 return TextRenderer;
260 
261 });
