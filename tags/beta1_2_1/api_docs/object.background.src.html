Summary

An object which loads an image and a collision map for usage as a single layered scrolling background.

 35 /**
 36  * @class A scrolling background render context with an associated
 37  *        collision map.  The render context loads a level which defines
 38  *        the image that will be displayed within a rendering context.
 39  *
 40  * @constructor
 41  * @param name {String} The name of the object
 42  * @param level {Level} A level object to use for this scrolling background
 43  * @param windowWidth {Number} The width of the viewable window, in pixels
 44  * @param windowHeight (Number} The height of the viewable window, in pixels
 45  * @extends HTMLElementContext
 46  */
 47 var ScrollingBackground = HTMLElementContext.extend(/** @scope ScrollingBackground.prototype */{
 48 
 49    canvasContext: null,
 50 
 51    jQ: null,
 52 
 53    constructor: function(name, level, windowWidth, windowHeight) {
 54 
 55       // Create an element for us to use as our window
 56       var e = $("<div style='overflow: hidden; width:" + windowWidth + "px; height:" + windowHeight + "px'></div>");
 57       this.base(name || "ScrollingBackground", e[0]);
 58 
 59       this.jQ = $(this.getSurface());
 60 
 61       // Draw the image
 62       this.jQ.append($(level.getSourceImage()).clone());
 63       //this.canvasContext.drawImage(level.getFrame(), level.getSourceImage());
 64    },
 65 
 66    release: function() {
 67       this.base();
 68       this.canvasContext = null;
 69       this.jQ = null;
 70    },
 71 
 72    setScrollPosition: function(point) {
 73       this.jQ[0].scrollLeft = point.x;
 74       this.jQ[0].scrollTop = point.y;
 75    },
 76 
 77    setHorizontalScroll: function(x) {
 78       this.jQ[0].scrollLeft = x;
 79    },
 80 
 81    setVerticalScroll: function(y) {
 82       this.jQ[0].scrollTop = y;
 83    }
 84 
 85 }, /** @scope ScrollingBackground */{
 86 
 87    /**
 88     * Get the class name of this object
 89     *
 90     * @type String
 91     */
 92    getClassName: function() {
 93       return "ScrollingBackground";
 94    }
 95 });
