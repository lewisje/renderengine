 32 // Includes
 33 Engine.include("/engine/engine.pooledobject.js");
 34 Engine.include("/engine/engine.events.js");
 35 
 36 Engine.initObject("BaseObject", "PooledObject", function() {
 37 
 38 /**
 39  * @class The base object class which represents an object within
 40  * the engine.  All objects should extend from this class mainly due to
 41  * the fact that the object intelligently will inherit a <tt>destroy</tt>
 42  * method.  This method gives the object an opportunity to clean up
 43  * object references, thus keeping the memory requirements low.
 44  * @extends PooledObject
 45  */
 46 var BaseObject = PooledObject.extend(/** @scope BaseObject.prototype */{
 47 
 48    element: null,
 49 
 50    events: null,
 51 
 52 	constructor: function(name) {
 53 		this.base(name);
 54 		this.events = {};
 55 	},
 56 
 57 	destroy: function() {
 58 		// We need to make sure to remove any event's attached to us
 59 		// that weren't already cleaned up
 60 		for (var ref in this.events) {
 61 			var fn = this.events[ref];
 62 			var type = ref.split(",")[1];
 63 			if (fn) {
 64 				EventEngine.clearHandler(this.getElement(), type, fn);
 65 			}
 66 		}
 67 
 68 		this.base();
 69 	},
 70 
 71    release: function() {
 72       this.base();
 73       this.element = null;
 74       this.events = null;
 75    },
 76 
 77    /**
 78     * Set the element which will represent this object within
 79     * its rendering context.
 80     *
 81     * @param element {HTMLElement} The HTML element this object is associated with.
 82     */
 83    setElement: function(element) {
 84       this.element = element;
 85    },
 86 
 87    /**
 88     * Get the element which represents this object within its rendering context.
 89     *
 90     * @return The HTML element
 91     */
 92    getElement: function() {
 93       return this.element;
 94    },
 95 
 96    /**
 97     * Abstract update method to set the state of the object.
 98     *
 99     * @param renderContext {RenderContext} The context the object exists within
100     * @param time {Number} The current engine time, in milliseconds
101     */
102    update: function(renderContext, time) {
103    },
104 
105    /**
106     * Add an event handler to this object if it has an associated HTML element.
107     *
108     * @param ref {Object} The object reference which is assigning the event
109     * @param type {String} The event type to respond to
110     * @param [data] {Array} Optional data to pass to the handler when it is invoked.
111     * @param fn {Function} The function to trigger when the event fires
112     */
113    addEvent: function(ref, type, data, fn) {
114       if (this.getElement()) {
115 			Console.info(ref.getName() + " attach event '" + type + "' to " + this.getName());
116          EventEngine.setHandler(this.getElement(), type, data || fn, fn);
117 
118          // Remember the handler by the reference object's name and event type
119          var func = $.isFunction(data) ? data : fn;
120          this.events[ref.getName() + "," + type] = func;
121       }
122    },
123 
124    /**
125     * Remove the event handler assigned to the object's associated HTML element
126     * for the given type.
127     *
128     * @param ref {Object} The object reference which assigned the event
129     * @param type {String} The event type to remove
130     */
131    removeEvent: function(ref, type) {
132       if (this.getElement()) {
133 			// Find the handler to remove
134 			var fn = this.events[ref.getName() + "," + type];
135 			if (fn) {
136 				Console.info(ref.getName() + " remove event '" + type + "' from " + this.getName());
137 	         EventEngine.clearHandler(this.getElement(), type, fn);
138 			}
139          // Remove the reference
140          delete this.events[ref.getName() + "," + type];
141       }
142    }
143 
144  }, /** @scope BaseObject */{
145 
146    /**
147     * Get the class name of this object
148     *
149     * @type String
150     */
151    getClassName: function() {
152       return "BaseObject";
153    }
154 
155 });
156 
157 return BaseObject;
158 
159 });
