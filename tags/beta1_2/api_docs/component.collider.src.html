 34 // Includes
 35 Engine.include("/components/component.base.js");
 36 
 37 Engine.initObject("ColliderComponent", "BaseComponent", function() {
 38 
 39 /**
 40  * @class A collider component handles collisions by updating the
 41  *        collision model and checking for possible collisions.
 42  *
 43  * @param name {String} Name of the component
 44  * @param collisionModel {SpatialCollection} The collision model
 45  * @param priority {Number} Between 0.0 and 1.0, with 1.0 being highest
 46  *
 47  * @extends BaseComponent
 48  */
 49 var ColliderComponent = BaseComponent.extend(/** @scope ColliderComponent.prototype */{
 50 
 51    collisionModel: null,
 52 
 53    constructor: function(name, collisionModel, priority) {
 54       this.base(name, BaseComponent.TYPE_COLLIDER, priority || 1.0);
 55       this.collisionModel = collisionModel;
 56    },
 57 
 58    release: function() {
 59       this.base();
 60       this.collisionModel = null;
 61    },
 62 
 63    /**
 64     * Update the collision model that this component was initialized with.
 65     * Creates an object called <tt>ModelData</tt> on the host object which contains
 66     * information relative to the collision model.  This data contains:
 67     * <ul>
 68     * <li>lastNode - The last node the object was reported within.
 69     * </ul>
 70     */
 71    updateModel: function() {
 72 
 73       // Get the model data for the object
 74       var obj = this.getHostObject();
 75       if (!obj.ModelData)
 76       {
 77          obj.ModelData = { lastNode: null };
 78       }
 79 
 80       if ( obj.ModelData.lastNode && obj.ModelData.lastNode.getRect().containsPoint(obj.getPosition()) )
 81       {
 82          // The object is within the same node
 83          return;
 84       }
 85 
 86       // Find the node that contains the object
 87       var aNode = this.collisionModel.findNodePoint(obj.getPosition());
 88       if (aNode != null)
 89       {
 90          if (obj.ModelData.lastNode && (obj.ModelData.lastNode.getIndex() != aNode.getIndex()))
 91          {
 92             obj.ModelData.lastNode.removeObject(obj);
 93             aNode.addObject(obj);
 94          }
 95          obj.ModelData.lastNode = aNode;
 96       }
 97    },
 98 
 99    /**
100     * Updates the object within the collision model and determines if
101     * the host object wants to be alerted whenever a potential collision
102     * has occurred.  If a potential collision occurs, an array (referred to
103     * as a Potentail Collision List, or PCL) will be created which
104     * contains objects that might be colliding with the host object.  It
105     * is up to the host object to make the final determination that a
106     * collision has occurred.
107     * <p/>
108     * The list of objects within the PCL will be passed to the <tt>onCollide</tt>
109     * method (if declared) on the host object.  If a collision occurred and was
110     * handled, the <tt>onCollide</tt> method should return <tt>CollisionComponent.STOP</tt>,
111     * otherwise, it should return <tt>CollisionComponent.CONTINUE</tt> to continue
112     * checking objects from the PCL against the host object.
113     *
114     * @param renderContext {RenderContext} The render context for the component
115     * @param time {Number} The current engine time in milliseconds
116     */
117    execute: function(renderContext, time) {
118 
119       // Update the collision model
120       this.updateModel();
121 
122       // If the host object needs to know about collisions...
123       if (this.getHostObject().onCollide)
124       {
125          // Get the PCL and check for collisions
126          var pcl = this.collisionModel.getPCL(this.getHostObject().getPosition());
127          var status = ColliderComponent.CONTINUE;
128          EngineSupport.forEach(pcl, function(obj) {
129             if (status == ColliderComponent.CONTINUE)
130             {
131                status = this.getHostObject().onCollide(obj);
132             }
133          }, this);
134       }
135    }
136 
137 }, /** @scope ColliderComponent.prototype */{ // Statics
138 
139    /**
140     * Get the class name of this object
141     *
142     * @type String
143     */
144    getClassName: function() {
145       return "ColliderComponent";
146    },
147 
148    /**
149     * When <tt>onCollide</tt> is called on the host object, it should
150     * return <tt>ColliderComponent.CONTINUE</tt> if it hasn't handled
151     * the collision, or if it wishes to be notified about other potential
152     * collisions.
153     * @type Number
154     */
155    CONTINUE: 0,
156 
157    /**
158     * When <tt>onCollide</tt> is called on the host object, it should
159     * return <tt>ColliderComponent.STOP</tt> if no more collisions
160     * should be reported.
161     * @type Number
162     */
163    STOP: 1
164 
165 });
166 
167 return ColliderComponent;
168 
169 });
