 32 // Includes
 33 Engine.include("/components/component.logic.js");
 34 
 35 Engine.initObject("NotifierComponent", "LogicComponent", function() {
 36 
 37 /**
 38  * @class A component which notifies objects when an action occurs.
 39  * @extends LogicComponent
 40  */
 41 var NotifierComponent = LogicComponent.extend(/** @scope NotifierComponent.prototype */{
 42 
 43    notifyLists: null,
 44 
 45    /**
 46     * @constructor
 47     * @memberOf NotifierComponent
 48     */
 49    constructor: function(name, priority) {
 50       this.base(name, priority || 1.0);
 51       this.notifyLists = {};
 52    },
 53 
 54    release: function() {
 55       this.base();
 56       this.notifyLists = null;
 57    },
 58 
 59    /**
 60     * Add a recipient function for the event type specified.  The function
 61     * will be passed the event object as its only parameter.
 62     *
 63     * @param type {String} The type name of the event list.
 64     * @param fn {Function} The function to call when the event triggers.
 65     * @memberOf NotifierComponent
 66     */
 67    addRecipient: function(type, thisObj, fn) {
 68       if (this.notifyLists[type] == null) {
 69          this.notifyLists[type] = [];
 70       }
 71       this.notifyLists[type].push({parent: thisObj, func: fn});
 72    },
 73 
 74    /**
 75     * Remove a recipient function for the event type specified.
 76     *
 77     * @param type {String} The type name of the event list.
 78     * @param fn {Function} The function to remove from the notification list.
 79     * @memberOf NotifierComponent
 80     */
 81    removeRecipient: function(type, thisObj, fn) {
 82       var o = {parent: thisObj, func: fn};
 83       EngineSupport.arrayRemove(this.notifyLists[type], o);
 84    },
 85 
 86    /**
 87     * Run through the list of recipients functions for the
 88     * event type specified.  Optimized for speed if the list
 89     * is large.
 90     * @memberOf NotifierComponent
 91     */
 92    notifyRecipients: function(type, eventObj) {
 93       if (this.notifyLists[type] == null)
 94       {
 95          // No handlers for this type
 96          return;
 97       }
 98 
 99       // Using Duff's device with loop inversion
100       var p = 0;
101       var s = null;
102       switch(this.notifyLists[type].length & 0x3)
103       {
104          case 3:
105             s = this.notifyLists[type][p++];
106             s.func.call(s.parent, eventObj);
107          case 2:
108             s = this.notifyLists[type][p++];
109             s.func.call(s.parent, eventObj);
110          case 1:
111             s = this.notifyLists[type][p++];
112             s.func.call(s.parent, eventObj);
113       }
114 
115       if (p < this.notifyLists[type].length)
116       {
117          do
118          {
119             s = this.notifyLists[type][p++];
120             s.func.call(s.parent, eventObj);
121 
122             s = this.notifyLists[type][p++];
123             s.func.call(s.parent, eventObj);
124 
125             s = this.notifyLists[type][p++];
126             s.func.call(s.parent, eventObj);
127 
128             s = this.notifyLists[type][p++];
129             s.func.call(s.parent, eventObj);
130 
131          } while (p < this.notifyLists[type].length);
132       }
133    }
134 }, {
135    /**
136     * Get the class name of this object
137     *
138     * @type String
139     * @memberOf NotifierComponent
140     */
141    getClassName: function() {
142       return "NotifierComponent";
143    }
144 });
145 
146 return NotifierComponent;
147 
148 });
