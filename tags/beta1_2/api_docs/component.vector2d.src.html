 36 Engine.initObject("Vector2DComponent", "RenderComponent", function() {
 37 
 38 /**
 39  * @class A render component that renders its contents from a set of points.
 40  * @extends BaseComponent
 41  */
 42 var Vector2DComponent = RenderComponent.extend(/** @scope Vector2DComponent.prototype */{
 43 
 44    strokeStyle: "#ffffff",     // Default to white lines
 45 
 46    lineWidth: 1,
 47 
 48    fillStyle: null,          // Default to none
 49 
 50    points: null,
 51 
 52    fullBox: null,
 53 
 54    closedManifold: null,
 55 
 56    /**
 57     * @constructor
 58     * @memberOf Vector2DComponent
 59     */
 60    constructor: function(name, priority) {
 61       this.base(name, priority || 0.1);
 62       this.closedManifold = true;
 63    },
 64 
 65    release: function() {
 66       this.base();
 67       this.strokeStyle = "#ffffff";
 68       this.lineWidth = 1;
 69       this.fillStyle = null;
 70       this.points = null;
 71       this.fullBox = null;
 72       this.closedManifold = null;
 73    },
 74 
 75    /**
 76     * Calculate the bounding box from the set of
 77     * points which comprise the shape to be rendered.
 78     * @private
 79     */
 80    calculateBoundingBox: function() {
 81       var x1 = 0;
 82       var x2 = 0;
 83       var y1 = 0;
 84       var y2 = 0;
 85       for (var p = 0; p < this.points.length; p++)
 86       {
 87          var pt = this.points[p];
 88 
 89          if (pt.x < x1)
 90          {
 91             x1 = pt.x;
 92          }
 93          if (pt.x > x2)
 94          {
 95             x2 = pt.x;
 96          }
 97          if (pt.y < y1)
 98          {
 99             y1 = pt.y;
100          }
101          if (pt.y > y2)
102          {
103             y2 = pt.y;
104          }
105       }
106 
107       var bbox = new Rectangle2D(x1, y1, Math.abs(x1) + x2, Math.abs(y1) + y2);
108       this.getHostObject().setBoundingBox(bbox);
109 
110       // Figure out longest axis
111       if (bbox.len_x() > bbox.len_y)
112       {
113          this.fullBox = new Rectangle2D(x1,x1,Math.abs(x1) + x2,Math.abs(x1) + x2);
114       }
115       else
116       {
117          this.fullBox = new Rectangle2D(y1,y1,Math.abs(y1) + y2,Math.abs(y1) + y2);
118       }
119    },
120 
121    /**
122     * Set the points which comprise the shape of the object to
123     * be rendered to the context.
124     *
125     * @param pointArray {Array} An array of <tt>Point2D</tt> instances
126     */
127    setPoints: function(pointArray) {
128       this.points = pointArray;
129       this.renderState = null;
130       this.calculateBoundingBox();
131    },
132 
133    /**
134     * Set the color of the lines to be drawn for this shape.
135     *
136     * @param strokeStyle {String} The HTML color of the stroke (lines) of the shape
137     */
138    setLineStyle: function(strokeStyle) {
139       this.strokeStyle = strokeStyle;
140    },
141 
142    /**
143     * Returns the line style that will be used to draw this shape.
144     * @type String
145     */
146    getLineStyle: function() {
147       return this.strokeStyle;
148    },
149 
150    /**
151     * Set the width of lines used to draw this shape.
152     *
153     * @param lineWidth {Number} The width of lines in the shape
154     */
155    setLineWidth: function(lineWidth) {
156       this.lineWidth = lineWidth;
157    },
158 
159    /**
160     * Returns the width of the lines used to draw the shape.
161     * @type Number
162     */
163    getLineWidth: function() {
164       return this.lineWidth;
165    },
166 
167    /**
168     * Set the color used to fill the shape.
169     *
170     * @param fillStyle {String} The HTML color used to fill the shape.
171     */
172    setFillStyle: function(fillStyle) {
173       this.fillStyle = fillStyle;
174    },
175 
176    /**
177     * @memberOf Vector2DComponent
178     */
179    getFillStyle: function() {
180       return this.fillStyle;
181    },
182 
183    /**
184     * Set whether or not we draw a polygon or polyline.  <tt>true</tt>
185     * to draw a polygon (the path formed by the points is a closed loop.
186     *
187     * @param closed {Boolean}
188     */
189    setClosed: function(closed) {
190       this.closedManifold = closed;
191    },
192 
193    /**
194     * Draw the shape, defined by the points, to the rendering context
195     * using the specified line style and fill style.
196     *
197     * @param renderContext {RenderContext} The context to render to
198     * @param time {Number} The engine time in milliseconds
199     */
200    execute: function(renderContext, time) {
201       if (!(this.points && this.base(renderContext, time)))
202       {
203          return;
204       }
205 
206       // Set the stroke and fill styles
207       if (this.getLineStyle() != null)
208       {
209          renderContext.setLineStyle(this.strokeStyle);
210       }
211 
212       renderContext.setLineWidth(this.lineWidth);
213 
214       if (this.getFillStyle() != null)
215       {
216          renderContext.setFillStyle(this.fillStyle);
217       }
218 
219       // Render out the points
220       if (this.closedManifold)
221       {
222          renderContext.drawPolygon(this.points);
223       }
224       else
225       {
226          renderContext.drawPolyline(this.points);
227       }
228 
229       if (this.fillStyle)
230       {
231          renderContext.drawFilledPolygon(this.points);
232       }
233    }
234 }, {
235    /**
236     * Get the class name of this object
237     *
238     * @type String
239     * @memberOf Vector2DComponent
240     */
241    getClassName: function() {
242       return "Vector2DComponent";
243    }
244 });
245 
246 return Vector2DComponent;
247 
248 });
