 38 /**
 39  * @class A <tt>canvas</tt> element represented within the engine.  A canvas
 40  * is a 2D context which can render lines, images, and polygons.  Transformations
 41  * can be saved and restored, allowing for complex, stacked transformations.
 42  *
 43  * @extends RenderContext2D
 44  */
 45 var CanvasContext = RenderContext2D.extend(/** @scope CanvasContext.prototype */{
 46

 53    /**
 54     * Create an instance of a 2D rendering context using the canvas element.
 55     *
 56     * @param name {String} The name of this context.  Default: CanvasContext
 57     * @param width {Number} The width (in pixels) of the canvas context.
 58     * @param height {Number} The height (in pixels) of the canvas context.
 59     * @constructor
 60     */
 61    constructor: function(name, width, height) {
 62       Assert((width != null && height != null), "Width and height must be specified in CanvasContext");
 63 
 64       this.setWidth(width);
 65       this.setHeight(height);
 66       var canvas;
 67       var worldScale = this.getWorldScale();
 68       // Create the canvas element
 69       canvas = document.createElement("canvas");
 70       if (jQuery.browser.msie) {
 71          canvas.style.width = this.width * worldScale;
 72          canvas.style.height = this.height * worldScale;
 73       } else {
 74          canvas.width = this.width * worldScale;
 75          canvas.height = this.height * worldScale;
 76       }
 77 
 78       this.setViewport(Rectangle2D.create(0, 0, this.width, this.height));
 79       this.base(name || "CanvasContext", canvas);
 80 
 81       canvas.id = this.getId();
 82    },
 83 
 84    afterAdd: function(parent) {
 85    },
 86 
 87    release: function() {
 88       this.base();
 89       this.context2D = null;
 90       this.mouseHandler = false;
 91    },
 92 
 93    setWorldScale: function(scaleX, scaleY) {
 94       this.base(scaleX, scaleY);
 95 
 96       scaleY = scaleY ? scaleY : scaleX;
 97       // Adjust the element accordingly
 98       $(this.getElement())
 99          .attr("width", this.getWidth() * scaleX)
100          .attr("height", this.getHeight() * scaleY);
101 
102       this.setViewport(Rectangle2D.create(0, 0, this.getWidth(), this.getHeight()));
103    },

139     * Reset the context, clearing it and preparing it for drawing.

141    reset: function(rect) {
142       var cRect = (rect != null ? rect : this.getViewport());

343    /**
344     * Draw an image on the context.
345     *
346     * @param rect {Rectangle2D} The rectangle that specifies the position and
347     *             dimensions of the image rectangle.
348     * @param image {Object} The image to draw onto the context
349     * @param [srcRect] {Rectangle2D} <i>[optional]</i> The source rectangle within the image, if
350     *                <tt>null</tt> the entire image is used
351     */
352    drawImage: function(rect, image, srcRect) {
353 		var d = rect.get();
354       if (srcRect) {
355          var s = srcRect.get();
356          this.get2DContext().drawImage(image,
357             s.x, s.y, s.w, s.h, d.x, d.y, d.w, d.h);
358       } else {
359          this.get2DContext().drawImage(image, d.x, d.y, d.w, d.h);
360       }
361       this.base(rect, image);
362    },

364    /**
365     * Capture an image from the context.
366     *
367     * @param rect {Rectangle2D} The area to capture
368     * @returns Image data capture
369     * @type ImageData
370     */
371    getImage: function(rect) {
372       this.base();
373 
374       // Clamp the rectangle to be within the bounds of the context
375       var p = rect.getTopLeft();
376       var tr = new Point2D((p.x < 0 ? 0 : (p.x > this.getWidth() ? this.getWidth() - 1 : p.x)),
377                            (p.y < 0 ? 0 : (p.y > this.getHeight() ? this.getHeight() - 1 : p.y)));
378       var d = rect.getDims();
379       var r = p.x + d.x;
380       var b = p.y + d.y;
381       var wh = new Point2D((r > this.getWidth() ? this.getWidth() - tr.x : (r < 0 ? 1 : d.x)),
382                            (b > this.getHeight() ? this.getHeight() - tr.y : (b < 0 ? 1 : d.y)));
383 
384 
385       return this.get2DContext().getImageData(tr.x, tr.y, wh.x, wh.y);
386    },

406     * @param point {Point2D} The poisition at which to draw the image

409       var x = (point.x < 0 ? 0 : (point.x > this.getWidth() ? this.getWidth() - 1 : point.x));

412       {
413          this.get2DContext().putImageData(imageData, x, y);
414       }
