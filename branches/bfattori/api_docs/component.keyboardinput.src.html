 31 // Includes
 32 Engine.include("/engine/engine.events.js");
 33 Engine.include("/components/component.input.js");
 34 
 35 Engine.initObject("KeyboardInputComponent", "InputComponent", function() {
 36 
 37 /**
 38  * @class A component which responds to keyboard events and notifies
 39  * its host object by calling one of three methods.  The host object
 40  * should implement any of the following methods to receive the corresponding event:
 41  * <ul>
 42  * <li>onKeyDown - A key was pressed down</li>
 43  * <li>onKeyUp - A key was released</li>
 44  * <li>onKeyPress - A key was pressed and released</li>
 45  * </ul>
 46  *
 47  * @extends InputComponent
 48  */
 49 var KeyboardInputComponent = InputComponent.extend(/** @scope KeyboardInputComponent.prototype */{
 50 
 51    /**
 52     * Create an instance of a keyboard input component.
 53     *
 54     * @param name {String} The unique name of the component.
 55     * @param priority {Number} The priority of the component among other input components.
 56     * @constructor
 57     */
 58    constructor: function(name, priority) {
 59       this.base(name, priority);
 60 
 61       var ctx = Engine.getDefaultContext();
 62       var self = this;
 63 
 64       // Add the event handlers
 65 		ctx.addEvent(this, "keydown", function(evt) {
 66 			self._keyDownListener(evt);
 67 		});
 68 		ctx.addEvent(this, "keyup", function(evt) {
 69 			self._keyUpListener(evt);
 70 		});
 71 		ctx.addEvent(this, "keypress", function(evt) {
 72 			self._keyPressListener(evt);
 73 		});
 74    },
 75 
 76    /**
 77     * Destroy this instance and remove all references.
 78     */
 79    destroy: function() {
 80       var ctx = Engine.getDefaultContext();
 81 
 82       // Clean up event handlers
 83 		ctx.removeEvent(this, "keydown");
 84 		ctx.removeEvent(this, "keyup");
 85 		ctx.removeEvent(this, "keypress");
 86       this.base();
 87    },
 88 
 89    /**
 90     * @private
 91     */
 92    _keyDownListener: function(eventObj) {
 93       if (this.getHostObject().onKeyDown)
 94       {
 95          this.getHostObject().onKeyDown(eventObj);
 96       }
 97    },
 98 
 99    /**
100     * @private
101     */
102    _keyUpListener: function(eventObj) {
103       if (this.getHostObject().onKeyUp)
104       {
105          this.getHostObject().onKeyUp(eventObj);
106       }
107    },
108 
109    /**
110     * @private
111     */
112    _keyPressListener: function(eventObj) {
113       if (this.getHostObject().onKeyPress)
114       {
115          this.getHostObject().onKeyPress(eventObj);
116       }
117    }
118 
119 }, {
120    /**
121     * Get the class name of this object
122     *
123     * @type String
124     * @memberOf KeyboardInputComponent
125     */
126    getClassName: function() {
127       return "KeyboardInputComponent";
128    }
129 });
130 
131 return KeyboardInputComponent;
132 
133 });
