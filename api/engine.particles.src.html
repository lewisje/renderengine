 41 /**
 42  * @class Base particle class.  A particle only needs to implement the
 43  *        <tt>draw()</tt> method. The remainder of the functionality is
 44  *        handled by this abstract class.
 45  *
 46  * @param lifetime {Number} The life of the particle, in milliseconds
 47  * @extends PooledObject
 48  * @constructor
 49  * @description Create a particle
 50  */
 51 var Particle = PooledObject.extend(/** @scope Particle.prototype */{
 52 
 53    life: 0,
 54    engine: null,
 55    birth: 0,
 56    dead: false,
 57    pos: null,
 58 
 59    /**
 60     * @private
 61     */
 62    constructor: function(lifetime) {
 63       this.base("Particle");
 64       this.life = lifetime;
 65       this.birth = 0;
 66       this.dead = false;
 67       this.pos = Point2D.create(0,0);
 68    },
 69    
 70    /**
 71     * Destroy the particle
 72     */
 73    destroy: function() {
 74    	this.pos.destroy();
 75    	this.base();
 76    },
 77 
 78    /**
 79     * Release the object back into the pool.
 80     */
 81    release: function() {
 82       this.base();
 83       this.life = 0;
 84       this.engine = null;
 85       this.birth = 0;
 86       this.dead = true;
 87       this.pos = null;
 88    },
 89 
 90    /**
 91     * Initializes the object within the <tt>ParticleEngine</tt>
 92     * @param pEngine {ParticleEngine} The particle engine which owns the particle
 93     * @param time {Number} The world time when the particle was created
 94     * @private
 95     */
 96    init: function(pEngine, time) {
 97       this.engine = pEngine;
 98       this.life += time;
 99       this.birth = time;
100       this.dead = false;
101    },
102 
103 	/**
104 	 * Get the current position of the particle
105 	 * @return {Point2D}
106 	 */
107 	getPosition: function() {
108 		return this.pos;
109 	},
110 	
111 	/**
112 	 * Set the X and Y world coordinates of the particle
113 	 * @param x {Number} X world coordinate
114 	 * @param y {Number} Y world coordinate
115 	 */
116 	setPosition: function(x, y) {
117 		this.pos.set(x,y);
118 	},
119 
120    /**
121     * Update the particle in the render context, calling its draw method.
122     * @param renderContext {RenderContext} The context where the particle is drawn
123     * @param time {Number} The world time, in milliseconds
124     */
125    update: function(renderContext, time) {
126       if (time < this.life &&
127       		renderContext.getViewport().containsPoint(this.getPosition())) {
128 	   	// if the particle is still alive, and it isn't outside the viewport
129          this.draw(renderContext, time);
130          return true;
131       } else {
132          return false;
133       }
134    },
135 
136    /**
137     * Get the time-to-live for the particle (when it will expire)
138     * @return {Number} milliseconds
139     */
140    getTTL: function() {
141       return this.life;
142    },
143 
144    /**
145     * Get the time at which the particle was created
146     * @return {Number} milliseconds
147     */
148    getBirth: function() {
149       return this.birth;
150    },
151 
152 	
153 
154    /**
155     * [ABSTRACT] Draw the particle
156     * @param renderContext {RenderContext} The context to render the particle to
157     * @param time {Number} The world time, in milliseconds
158     */
159    draw: function(renderContext, time) {
160       // ABSTRACT
161    }

178 Engine.initObject("ParticleEngine", "BaseObject", function() {
179 
180 /**
181  * @class The particle engine is a system for updating and expiring
182  *        particles within a game environment.  This is registered with the
183  *        render context so it will be updated at regular intervals.  The maximum
184  *        number of supported particles can be configured, but defaults to 120.
185  *        <p/>
186  *        Particles should be simple objects which don't need to perform many
187  *        calculations before being drawn.  All particles are rendered in world
188  *        coordinates to speed up processing.
189  *        </p>
190  *        It is possible to run multiple particle engines within a render context,
191  *        but it might be non-productive to do so.
192  *
193  * @extends BaseObject
194  * @constructor
195  * @description Create a particle engine
196  */
197 var ParticleEngine = BaseObject.extend(/** @scope ParticleEngine.prototype */{
198 
199    particles: null,
200    liveParticles: 0,
201    lastTime: 0,
202    maximum: 0,
203    force: 0,
204 
205    /**
206     * @private
207     */
208    constructor: function() {
209       this.base("ParticleEngine");
210       this.particles = Container.create();
211       this.maximum = ParticleEngine.MAX_PARTICLES;
212       this.liveParticles = 0;
213    },
214 
215    /**
216     * @private
217     */
218    destroy: function() {
219 		this.particles.cleanUp();
220 		this.particles.destroy();
221       this.base();
222    }, 
223 
224    /**
225     * Releases the object back into the pool.
226     */
227    release: function() {
228       this.base();
229       this.particles = null,
230       this.lastTime = 0;
231       this.maximum = 0;
232       this.liveParticles = 0;
233    },
234 
235    /**
236     * Add a group of particles at one time.  This reduces the number of calls
237     * to {@link #addParticle} which resorts the array of particles each time.
238     * @param particles {Container} A container of particles to add at one time
239     */
240    addParticles: function(particles) {
241 		if (!Container.isInstance(particles)) {
242 			// If the particles are an Array, convert to a Container first
243 			var oldP = particles;
244 			particles = Container.create();
245 			particles.addAll(oldP);
246 		}
247 
248       // If the new particles exceed the size of the engine's
249       // maximum, truncate the remainder
250       if (particles.size() > this.maximum) {
251          var discard = particles.reduce(this.maximum);
252 			discard.cleanUp();
253 			discard.destroy();
254       }
255       
256       // Initialize all of the new particles
257       for (var i = particles.iterator(); i.hasNext(); ) {
258          i.next().init(this, this.lastTime);
259       }
260 		i.destroy();
261       
262       // The maximum number of particles to animate
263       var total = this.liveParticles + particles.size();
264       if (total > this.maximum) {
265          total = this.maximum;
266       }
267       
268 		// If we can fit the entire set of particles without overflowing,
269 		// add all the particles and be done.
270 		if (particles.size() <= this.maximum - this.liveParticles) {
271 			this.particles.addAll(particles);
272 		} else {
273 			// There isn't enough space to put all of the particles into
274 			// the container.  So, we'll only add what we can.
275 			var maxLeft = this.maximum - total;
276 			var easySet = particles.subset(0, maxLeft);
277 			this.particles.addAll(easySet);
278 			easySet.destroy();
279 		}
280 		particles.destroy();
281 		this.liveParticles = this.particles.size();
282    },
283 
284    /**
285     * Add a single particle to the engine.  If many particles are being
286     * added at one time, use {@link #addParticles} instead to add a
287     * {@link Container} of particles.
288     *
289     * @param particle {Particle} A particle to animate
290     */
291    addParticle: function(particle) {
292 		if (this.particles.size() < this.maximum) {
293          particle.init(this, this.lastTime);
294 			this.particles.add(particle);
295 			this.liveParticles = this.particles.size();
296 		} else {
297 			// nowhere to put it
298 			particle.destroy();
299 		}
300    },
301    
302    /**
303     * Set the absolute maximum number of particles the engine will allow.
304     * @param maximum {Number} The maximum particles the particle engine allows
305     */
306    setMaximum: function(maximum) {
307       var oldMax = this.maximum;
308       this.maximum = maximum;
309       
310       // Kill off particles if the size is reduced
311       if (this.maximum < oldMax) {
312          var discard = this.particles.reduce(this.maximum);
313 			discard.cleanUp();
314 			discard.destroy();
315       }
316    },
317    
318    /**
319     * Get the maximum number of particles allowed in the particle engine.
320     * @return {Number}
321     */
322    getMaximum: function() {
323       return this.maximum;
324    },
325    
326    /**
327     * Update a particle, removing it and nulling its reference
328     * if it is dead.  Only live particles are updated
329     * @private
330     */
331    runParticle: function(particle, renderContext, time) {
332       if (!particle.update(renderContext, time)) {
333 			this.particles.remove(particle);
334          particle.destroy();
335       }
336    },
337 
338    /**
339     * Update the particles within the render context, and for the specified time.
340     *
341     * @param renderContext {RenderContext} The context the particles will be rendered within.
342     * @param time {Number} The global time within the engine.
343     */
344    update: function(renderContext, time) {
345       var p = 1;
346       this.lastTime = time;
347 
348       Engine.addMetric("particles", this.liveParticles, false, "#");
349       
350       // If there are no live particles, don't do anything
351       if (this.liveParticles == 0) {
352          return;
353       }
354 
355       renderContext.pushTransform();
356 
357 		for (var itr = this.particles.iterator(); itr.hasNext(); ) {
358 			this.runParticle(itr.next(), renderContext, time);
359 		}
360 		itr.destroy();
361       
362       renderContext.popTransform();
363       this.liveParticles = this.particles.size();
364    },
365 
366    /**
367     * Get the properties object for the particle engine
368     * @return {Object}
369     */
370    getProperties: function() {
371       var self = this;
372       var prop = this.base(self);
373       return $.extend(prop, {
374          "Count" : [function() { return self.particles.size(); },
375                     null, false]
376       });
377    }
378 
379 
380 }, /** @scope ParticleEngine.prototype */{
381    /**
382     * Get the class name of this object
383     *
384     * @return {String} "ParticleEngine"
385     */
386    getClassName: function() {
387       return "ParticleEngine";
388    },
389    
390    /**
391     * Default maximum number of particles in the system. To change
392     * the value, see {@link ParticleEngine#setMaximum}
393     * @type Number
394     */
395    MAX_PARTICLES: 120
396 
397 });
398 
399 return ParticleEngine;
