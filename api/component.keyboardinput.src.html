 34 // Includes
 35 Engine.include("/engine/engine.events.js");
 36 Engine.include("/components/component.input.js");
 37 
 38 Engine.initObject("KeyboardInputComponent", "InputComponent", function() {
 39 
 40 /**
 41  * @class A component which responds to keyboard events and notifies
 42  * its {@link HostObject} by calling one of three methods.  The <tt>HostObject</tt>
 43  * should implement any of the following methods to receive the corresponding event:
 44  * <ul>
 45  * <li><tt>onKeyDown()</tt> - A key was pressed down</li>
 46  * <li><tt>onKeyUp()</tt> - A key was released</li>
 47  * <li><tt>onKeyPress()</tt> - A key was pressed and released</li>
 48  * </ul>
 49  * Each function should take up to six arguments.  The first argument is the character
 50  * code, a number which represents the key that was pressed. The second argument is the
 51  * <tt>keyCode</tt>, a number which represents special keys that were pressed, such as
 52  * the arrow keys and function keys.  See the
 53  * {@link EventEngine} for key codes of the non-alphabetic keys. The third argument
 54  * is a boolean indicating if the Control key was pressed.  The fourth argument is a
 55  * boolean indicating if the Alt key was pressed.  The fifth argument is a boolean
 56  * indicating if the Shift key was pressed.  The sixth and final argument is the
 57  * actual event object itself.
 58  *
 59  * @param name {String} The unique name of the component.
 60  * @param priority {Number} The priority of the component among other input components.
 61  * @extends InputComponent
 62  * @constructor
 63  * @description Create an instance of a keyboard input component. 
 64  */
 65 var KeyboardInputComponent = InputComponent.extend(/** @scope KeyboardInputComponent.prototype */{
 66 
 67    /**
 68     * @private
 69     */
 70    constructor: function(name, priority) {
 71       this.base(name, priority);
 72 
 73       var ctx = Engine.getDefaultContext();
 74       var self = this;
 75 
 76       // Add the event handlers
 77       ctx.addEvent(this, "keydown", function(evt) {
 78          return self._keyDownListener(evt);
 79       });
 80       ctx.addEvent(this, "keyup", function(evt) {
 81          return self._keyUpListener(evt);
 82       });
 83       ctx.addEvent(this, "keypress", function(evt) {
 84          return self._keyPressListener(evt);
 85       });
 86    },
 87 
 88    /**
 89     * Destroy this instance and remove all references.
 90     * @private
 91     */
 92    destroy: function() {
 93       var ctx = Engine.getDefaultContext();
 94 
 95       // Clean up event handlers
 96       ctx.removeEvent(this, "keydown");
 97       ctx.removeEvent(this, "keyup");
 98       ctx.removeEvent(this, "keypress");
 99       this.base();
100    },
101 
102    /**
103     * @private
104     */
105    playEvent: function(e) {
106       var evt = document.createEvent("KeyboardEvent");
107       evt.initKeyEvent(e.type, true, false, null, e.ctrlKey, false, e.shiftKey, false, e.keyCode, 0);
108       this.getHostObject().getRenderContext().getSurface().dispatchEvent(evt);
109    },
110 
111    /**
112     * @private
113     */
114    _keyDownListener: function(eventObj) {
115       if (this.getHostObject().onKeyDown) {
116          this.record(eventObj,KeyboardInputComponent.RECORD_PART);
117          return this.getHostObject().onKeyDown(eventObj.which, eventObj.keyCode, eventObj.ctrlKey, eventObj.altKey, eventObj.shiftKey, eventObj);
118       }
119    },
120 
121    /**
122     * @private
123     */
124    _keyUpListener: function(eventObj) {
125       if (this.getHostObject().onKeyUp) {
126          this.record(eventObj,KeyboardInputComponent.RECORD_PART);
127          return this.getHostObject().onKeyUp(eventObj.which, eventObj.keyCode, eventObj.ctrlKey, eventObj.altKey, eventObj.shiftKey, eventObj);
128       }
129    },
130 
131    /**
132     * @private
133     */
134    _keyPressListener: function(eventObj) {
135       if (this.getHostObject().onKeyPress) {
136          this.record(eventObj,KeyboardInputComponent.RECORD_PART);
137          return this.getHostObject().onKeyPress(eventObj.which, eventObj.keyCode, eventObj.ctrlKey, eventObj.altKey, eventObj.shiftKey, eventObj);
138       }
139    }
140 
141 }, /** @scope KeyboardInputComponent.prototype */{
142    /**
143     * Get the class name of this object
144     *
145     * @return {String} "KeyboardInputComponent"
146     */
147    getClassName: function() {
148       return "KeyboardInputComponent";
149    },
150    
151    RECORD_PART: ["shiftKey","ctrlKey","altKey","keyCode"]
152 });
153 
154 return KeyboardInputComponent;
155 
156 });
