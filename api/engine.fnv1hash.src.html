 32 /*
 33  * A family of fast hash functions, originally created by Glenn Fowler, Phong Vo,
 34  * and improved by Landon Curt Noll.  Implemented in Javascript by Brett Fattori.
 35  * 
 36  * <p>FNV1 hashes are designed to be fast while maintaining a low collision rate.
 37  * The FNV1 speed allows one to quickly hash lots of data while maintaining a
 38  * reasonable collision rate. The high dispersion of the FNV1 hashes makes them
 39  * well suited for hashing nearly identical strings such as URLs, hostnames,
 40  * filenames, text, IP addresses, etc.</p>
 41  * 
 42  * <p>FNV1a is a variant of FNV1, which is slightly better suited for hashing
 43  * short values (< 4 octets).</p>
 44  * 
 45  * <p>This is a straightforward port of the public domain C version,
 46  * written by Landon Curt Noll (one of the authors), available from
 47  * <a href="http://www.isthe.com/chongo/tech/comp/fnv/">his website</a>.</p>
 48  * 
 49  * <p>The usage pattern is as follows: to compute the initial hash value
 50  * you call one of the <code>init(...)</code> methods. After that you may
 51  * update the hash zero or more times with additional values using the
 52  * <code>update(...)</code> methods. When you are done, you can retrieve the
 53  * final hash value with {@link #getHash()}.</p>
 54  * <p>Individual instances of FNV1 are reusable after you call one of
 55  * the <code>init(...)</code> methods. However, these implementations are NOT
 56  * synchronized, so proper care should be taken when using this hash in a multi-threaded
 57  * environment.</p>
 58  * 
 59  * @author Brett Fattori <bfattori AT gmail DOT com>
 60  * @private
 61  */
 62 var FNV1 = function(algo) {
 63 
 64    var hash = 0;
 65 
 66    // Create the linkage to the hashing algorithm
 67    var fnv = algo.fnv;
 68    var getHashValue = algo.getHash || function(h) { return Number(h).toString(16);};
 69    var INIT = algo.INIT;
 70 
 71    // Convert a string to an array of bytes
 72    function getBytes(s) {
 73       var buf = [];
 74       for (var i = 0; i < s.length; i++) {
 75          buf.push(s.charCodeAt(i));
 76       }
 77       return buf;
 78    }
 79 
 80    return {
 81 
 82       /**
 83        * Initialize this hash instance. Any previous state is reset, and the new
 84        * hash value is computed.
 85        * 
 86        * @param buf byte buffer from which to compute the hash
 87        * @param offset starting position in the buffer
 88        * @param len number of bytes after the starting position
 89        */
 90       init: function(buf, offset, len) {
 91          if (typeof buf == "string") {
 92             buf = getBytes(buf);
 93             offset = 0;
 94             len = buf.length;
 95          }
 96          hash = fnv(buf, offset, len, INIT);
 97       },
 98 
 99       /**
100        * Update the hash value. Repeated calls to this method update the hash
101        * value accordingly, and they are equivalent to calling the <code>init(...)</code>
102        * method once with a concatenated value of all parameters.
103        * @param s see (@link #init(String)}
104        */
105       update: function(buf, offset, len) {
106          if (typeof buf == "string") {
107             buf = getBytes(buf);
108             offset = 0;
109             len = buf.length;
110          }
111          hash = fnv(buf, offset, len, hash);
112       },
113 
114       /**
115        * Retrieve the hash value
116        * @return hash value
117        */
118       getHash: function() {
119          return getHashValue(hash);
120       }
121    };
122 };
123 
124 /**
125  * Implementation of FNV1 - a fast hash function.
126  * 
127  * <p>This implementation uses 32-bit operations, and the values returned from
128  * {@link #getHash()} are limited to the lower 32 bits.</p>
129  * 
130  * @author Andrzej Bialecki <ab@getopt.org>
131  */
132 var FNV132 = (function() {
133    return {
134       fnv: function(buf, offset, len, seed) {
135          for (var i = offset; i < offset + len; i++) {
136             seed += (seed << 1) + (seed << 4) + (seed << 7) + (seed << 8) + (seed << 24);
137             seed ^= buf[i];
138          }
139          return seed;
140       },
141       
142       getHash: function(hash) {
143          return Number(hash & 0x00000000ffffffff).toString(16);
144       },
145       
146       INIT: 0x811c9dc5
147    };
148 })();
149 
150 /**
151  * Implementation of FNV1a - a fast hash function. The FNV1a variant provides a
152  * slightly better dispersion for short (< 4 bytes) values than plain FNV1.
153  * 
154  * <p>This implementation uses 32-bit operations, and the values returned from
155  * {@link #getHash()} are limited to the lower 32 bits.</p>
156  * 
157  * @author Andrzej Bialecki <ab@getopt.org>
158  */
159 var FNV1a32 = (function() {
160    return {
161       fnv: function(buf, offset, len, seed) {
162          for (var i = offset; i < offset + len; i++) {
163             seed ^= buf[i];
164             seed += (seed << 1) + (seed << 4) + (seed << 7) + (seed << 8) + (seed << 24);
165          }
166          return seed;
167       },
168       
169       getHash: function(hash) {
170          return Number(hash & 0x00000000ffffffff).toString(16);
171       },
172 
173       INIT: 0x811c9dc5
174    };
175 })();

180 Engine.initObject("FNV1Hash", "PooledObject", function() {
181 
182    /**
183     * @class A class for creating a single hash, or an evolving hash. Single hashes reset each time a new
184     *        source string is provided, whereas the evolving hash will build upon previous hashes.  The
185     *        hash for a string will always be the same.  Hashing a set of strings, in order, will always
186     *        result in the same evolved hash.  Uses the {@link FNV1a32} hashing routine by default.
187     *
188     * @param [hashRoutine=FNV1a32] {FNV1} The hash routine to use.
189     * @extends PooledObject
190     */
191    var FNV1Hash = PooledObject.extend(/** @scope FNV1Hash.prototype */{
192 
193       fnv132: null,
194       
195       gotten: null,
196 
197       constructor: function(hashRoutine) {
198          this.base("FNV1Hash");
199          this.fnv132 = new FNV1(hashRoutine || FNV1a32);
200          this.gotten = false;
201       },
202 
203       /**
204        * Release the hash back into the pool for later use.
205        */
206       release: function() {
207          this.base();
208          this.fnv132 = null;
209          this.gotten = null;
210       },
211       
212       /**
213        * Get a one-time hash for the provided string.  To instead update the hash
214        * use the {@link #updateHash} to continually evolve the hash with new data.
215        *
216        * @param str {String} The value to get the hash for
217        * @return {String} A hexadecimal hash of the string input
218        */
219       getHash: function(str) {
220          this.gotten = true;
221          this.fnv132.init(str);
222          return this.fnv132.getHash();
223       },
224       
225       /**
226        * Update the existing hash with more data.
227        *
228        * @param str {String} The value to get the hash for
229        * @return {String} A hexadecimal hash of the string input
230        */
231       updateHash: function(str) {
232          if (this.gotten) {
233             this.fnv132.update(str);
234             return this.fnv132.getHash();
235          } else {
236             return this.getHash(str);
237          }
238       }
239       
240    }, { /** @scope FNV1Hash.prototype */
241    
242       getClassName: function() {
243          return "FNV1Hash";
244       }
245    });
246    
247    return FNV1Hash;
248    
249 });
