 34 // Includes
 35 Engine.include("/engine/engine.container.js");
 36 Engine.include("/components/component.logic.js");
 37 
 38 Engine.initObject("HostComponent", "LogicComponent", function() {
 39 
 40 /**
 41  * @class A component that can execute host objects.  Allows embedding
 42  *        of multiple objects into one object.
 43  *
 44  * @param name {String} The name of the component
 45  * @param [priority=1.0] {Number} The priority of this component
 46  * @extends LogicComponent
 47  * @constructor
 48  * @description Creates a <tt>HostComponent</tt> which can contain {@link HostObject HostObjects}.
 49  *              This allows a component to embed other hosts within it.  Each time the
 50  *              component is executed, each host will be given a chance to update as well.
 51  */
 52 var HostComponent = LogicComponent.extend(/** @scope HostComponent.prototype */{
 53 
 54    objects: null,
 55 
 56    /**
 57     * @private
 58     */
 59    constructor: function(name, priority) {
 60       this.base(name, priority || 1.0);
 61       this.objects = new HashContainer();
 62    },
 63 
 64    /**
 65     * Releases the component back into the object pool.  See {@link PooledObject#release}
 66     * for more information.
 67     */
 68    release: function() {
 69       this.base();
 70       this.objects = null;
 71    },
 72 
 73    /**
 74     * Destroys all of the hosts contained within this component.
 75     * @private
 76     */
 77    destroy: function() {
 78       this.objects.destroy();
 79       this.base();
 80    },
 81 
 82    /**
 83     * Add a {@link HostObject} to the component to be processed when
 84     * this component is executed.  Objects will be updated in the order in
 85     * which they are added.
 86     *
 87     * @param name {String} A unique name to refer to the object by
 88     * @param obj {HostObject} The host object reference
 89     */
 90    add: function(name, obj) {
 91       Assert((obj instanceof HostObject), "You can only add HostObjects to a HostComponent");
 92       this.objects.add(name.toUpperCase(), obj);
 93    },
 94 
 95    /**
 96     * Retrieve the {@link HostObject} that is associated with the
 97     * given name from the component.
 98     *
 99     * @param name {String} The unique name of the object
100     * @return {HostObject}
101     */
102    get: function(name) {
103       return this.objects.get(name.toUpperCase());
104    },
105 
106    /**
107     * Remove the host object from the component.
108     *
109     * @param obj {HostObject} The host object reference
110     * @return {HostObject} The object which was removed
111     */
112    remove: function(obj) {
113       return this.objects.remove(obj);
114    },
115 
116    /**
117     * Update each of the host objects within this component.  The order
118     * in which hosts are updated is equivalent to the order in which
119     * the objects were added.
120     *
121     * @param renderContext {RenderContext} The rendering context
122     * @param time {Number} The engine time in milliseconds
123     */
124    execute: function(renderContext, time) {
125       var objs = this.objects.getObjects();
126       for (var c in objs) {
127 
128          // Make sure the host object's render context matches 
129          // this component's host object's context
130          if (objs[c].getRenderContext() == null) {
131             objs[c].setRenderContext(renderContext);
132             Console.info(this.getHostObject().getId() + "[" + this.getName() + "]: SetRenderContext '" + renderContext.getId() + "'");
133          }
134 
135          objs[c].update(renderContext, time);
136       }
137    }
138 }, /** @scope HostComponent.prototype */{
139    /**
140     * Get the class name of this object
141     *
142     * @return {String} "HostComponent"
143     */
144    getClassName: function() {
145       return "HostComponent";
146    }
147 });
148 
149 return HostComponent;
150 
151 });
