 34 Engine.include("/rendercontexts/context.htmlelement.js");

 36 Engine.initObject("HTMLDivContext", "HTMLElementContext", function() {
 37 
 38 /**
 39  * @class A simple extension of the {@link HTMLElementContext} which uses a DIV
 40  * element to represent the context.  This is just a convenience method.
 41  *
 42  * @extends HTMLElementContext
 43  */
 44 var HTMLDivContext = HTMLElementContext.extend(/** @scope HTMLDivContext.prototype */{
 45 
 46    /**
 47     * Create an instance of a document rendering context.  This context
 48     * represents the HTML document body.  Theoretically, only one of these
 49     * contexts should ever be created.
 50     * @constructor
 51     */
 52    constructor: function(name, contextWidth, contextHeight) {
 53       var ctx = $("<div>").css({
 54          width: contextWidth,
 55          height: contextHeight,
 56          position: "absolute",
 57 			overflow: "hidden"
 58       });
 59       this.base(name || "HTMLDivContext", ctx);
 60 		this.setViewport(Rectangle2D.create(0, 0, contextWidth, contextHeight));
 61    }
 62    
 63 }, /** @scope HTMLDivContext.prototype */{
 64 
 65    /**
 66     * Get the class name of this object
 67     *
 68     * @return {String} The string "HTMLDivContext"
 69     */
 70    getClassName: function() {
 71       return "HTMLDivContext";
 72    }
 73 });
 74 
 75 return HTMLDivContext;
 76 
 77 });
