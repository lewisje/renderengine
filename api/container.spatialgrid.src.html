container.spatialgrid.js


Summary

A simple collision model which divides a finite space up into a coarse grid to assist in quickly finding objects within that space.
Class Summary
GridNode A single node within a SpatialGrid.
SpatialGrid A structure which divides a finite space up into a coarse grid to perform "broad phase" collision determinations within the space.

 99 Engine.initObject("SpatialGrid", "SpatialContainer", function() {
100 
101 /**
102  * @class A structure which divides a finite space up into a coarse grid to 
103  * 		 perform "broad phase" collision determinations within the space.  
104  * 		 After the PCL (potential collision list) is built, a "narrow phase" 
105  * 		 collision model would need to be employed to determine accurate collision 
106  * 		 response.  Using AABB overlapping for simple true/false determinations is
107  * 		 one method.  Another method would be to use something like GJK to determine
108  * 		 by how much two objects' convex hulls are overlapped.
109  * 		 <p/>
110  * 		 A spatial grid is defined by the size of the space and the number of
111  * 		 divisions within that space.  A smaller PCL will result from a larger
112  * 		 number of divisions, but the amount of data required to store the cells
113  * 		 also increases.  Also, larger numbers of divisions means that as objects
114  * 		 move, the determination of which cell the object is within increases as 
115  * 		 well.
116  *
117  * @constructor
118  * @description Create an instance of a spatial grid model
119  * @param width {Number} The width of the area
120  * @param height {Number} The height of the area
121  * @param divisions {Number} The number of divisions along both axis
122  * @extends SpatialContainer
123  */
124 var SpatialGrid = SpatialContainer.extend(/** @scope SpatialGrid.prototype */{
125 
126    divisions: 1,
127 
128    xLocator: 1,
129    yLocator: 1,
130    
131    accuracy: 0,
132 
133    /**
134     * @private
135     */
136    constructor: function(width, height, divisions) {
137       this.base("SpatialGrid", width, height);
138 
139       // Divide the space up into a grid
140       var gX = Math.floor(width / divisions);
141       var gY = Math.floor(height / divisions);
142       this.divisions = divisions;
143       this.xLocator = 1 / gX;
144       this.yLocator = 1 / gY;
145 		this.accuracy = SpatialGrid.HIGH_ACCURACY;
146 
147       var grid = [];
148       this.setRoot(grid);
149 
150       for (var y = 0; y < this.divisions; y++)
151       {
152          for (var x = 0; x < this.divisions; x++)
153          {
154             var rect = new Rectangle2D(x * gX, y * gY, gX, gY);
155             grid[x + (y * this.divisions)] = new GridNode(rect);
156          }
157       }
158    },
159 
160    /**
161     * Releases the object back into the object pool.  See {@link PooledObject#release}
162     * for more information.
163     */
164    release: function() {
165       this.base();
166       this.divisions = 1;
167       this.xLocator = 1;
168       this.yLocator = 1;
169 		this.accuracy = 0;
170    },
171 
172    /**
173     * Set the accuracy of the collision checks to either {@link SpatialGrid#GOOD_ACCURACY} or
174     * {@link SpatialGrid#BEST_ACCURACY}.
175     * 
176     * @param accuracy {Number} The level of accuracy during PCL generation
177     */
178    setAccuracy: function(accuracy) {
179       this.accuracy = (accuracy > SpatialGrid.BEST_ACCURACY || accuracy < SpatialGrid.GOOD_ACCURACY) ? 
180          SpatialGrid.BEST_ACCURACY : accuracy;
181    },
182    
183    /**
184     * Get the accuracy level of collision checks.
185     * @return {Number} The accuracy level
186     */
187    getAccuracy: function() {
188       return this.accuracy;
189    },
190 
191    /**
192     * Find the node that contains the specified point.
193     *
194     * @param point {Point2D} The point to locate the node for
195     * @return {SpatialNode}
196     */
197    findNodePoint: function(point) {
198       return this.getRoot()[Math.floor(point.x * this.xLocator) + (Math.floor(point.y * this.yLocator) * this.divisions)];
199    },
200 
201    /**
202     * Get the node within the grid.
203     * @param x {Number} The virtual X coordinate in our grid
204     * @param y {Number} The virtual Y coordinate in our grid
205     * @return {Number}
206     * @private
207     */
208    getNode: function(x, y) {
209 		// Normalize X and Y within the bounds of the grid
210 		x = x < 0 ? 0 : (x > this.divisions - 1 ? this.divisions - 1 : x);
211 		y = y < 0 ? 0 : (y > this.divisions - 1 ? this.divisions - 1 : y);
212       return this.getRoot()[x + (y * this.divisions)];
213    },
214 
215    /**
216     * Get the list of objects with respect to the point given.  Objects will
217     * be returned from the nodes that make up the grid node containing
218     * the point, and the following adjacent nodes:
219     * <ul>
220     * <li><b>Good Accuracy</b> - Just the node containing the point (G)</li>
221     * <li><b>Best Accuracy</b> - The four polar nodes around the center (G, B)</li>
222     * <li><b>High Accuracy</b> - The eight nodes around the center (G, B, H)</li>
223     * </ul>
224     * For example, if you had a 3x3 grid with the object in the center node, the nodes 
225     * marked below would be included in the result set:
226     * <pre>
227     *  +---+---+---+
228     *  | H | B | H |
229     *  +---+---+---+
230     *  | B | G | B |
231     *  +---+---+---+
232     *  | H | B | H |
233     *  +---+---+---+
234     * </pre>
235     *
236     * @param point {Point2D} The point to begin the search at.
237     * @return {Container} A container of objects found that could be collision targets
238     */
239    getPCL: function(point) {
240 
241 		var pcl = this.base(point);
242 		pcl.clear();
243 
244       // The origin node
245       var x = Math.floor(point.x * this.xLocator);
246       var y = Math.floor(point.y * this.yLocator);
247 
248       // build the node set
249       var nodes = [], n;
250 		
251 		// Start with GOOD_ACCURACY
252       nodes.push(this.getNode(x, y));
253      
254       // if our borders cross the margin, we can drop up to two nodes
255       if (this.accuracy >= SpatialGrid.BEST_ACCURACY) {
256          if (x > 0) { n = this.getNode(x - 1, y); if (n.getCount() != 0) nodes.push(n); }
257          if (x < this.divisions) { n = this.getNode(x + 1, y); if (n.getCount() != 0) nodes.push(n); }
258          if (y > 0) { n = this.getNode(x, y - 1); if (n.getCount() != 0) nodes.push(n); }
259          if (y < this.divisions) { n = this.getNode(x, y + 1); if (n.getCount() != 0) nodes.push(n); }
260       }
261 		
262 		// For highest number of checks, we'll include all eight surrounding nodes
263 		if (this.accuracy == SpatialGrid.HIGH_ACCURACY) {
264 			if (x > 0 && y > 0) { n = this.getNode(x - 1, y - 1); if (n.getCount() != 0) nodes.push(n); }
265 			if (x < this.divisions && y < this.divisions) { n = this.getNode(x + 1, y + 1); if (n.getCount() != 0) nodes.push(n); }
266 			if (x > 0 && y < this.divisions) { n = this.getNode(x - 1, y + 1); if (n.getCount() != 0) nodes.push(n); }
267 			if (x < this.divisions && y > 0) { n = this.getNode(x + 1, y - 1); if (n.getCount() != 0) nodes.push(n); }
268 		}
269 
270       for (var d = 0; d < nodes.length; d++) {
271 			if (!nodes[d]) {
272 				debugger;
273 			}
274 			var objs = nodes[d].getObjects();
275          pcl.addAll(objs);
276 			objs.destroy();
277       }
278       return pcl;
279    },
280    
281    /**
282     * Returns all objects within every node of the spatial grid.
283     * @return {Container} A container with all objects in the spatial grid
284     */
285    getObjects: function() {
286       var objs = this.base();
287       EngineSupport.forEach(this.getRoot(), function(node) {
288          objs.addAll(node.getObjects());
289       });
290       return objs;
291    }
292 
293 }, /** @scope SpatialGrid.prototype */{
294    /**
295     * Get the class name of this object
296     *
297     * @return {String} "SpatialGrid"
298     */
299    getClassName: function() {
300       return "SpatialGrid";
301    },
302    
303    /**
304     * Collision checks are limited to the exact node where the
305     * object being tested resides.
306     * @type {Number}
307     */
308    GOOD_ACCURACY: 0,
309    
310    /**
311     * Collision checks are performed in the node where the object
312     * being tested resides, and in the four surrounding polar nodes.
313     * @type {Number}
314     * @deprecated
315     */
316    BEST_ACCURACY: 1,
317 
318    /**
319     * Collision checks are performed in the node where the object
320     * being tested resides, and in the four surrounding polar nodes.
321     * @type {Number}
322     */
323 	BETTER_ACCURACY: 1,
324 	
325    /**
326     * Collision checks are performed in the node where the object
327     * being tested resides, and in the eight surrounding nodes.
328     * @type {Number}
329     */
330 	HIGH_ACCURACY: 2
331 });
332 
333 return SpatialGrid;
334 
335 });
