container.spatialgrid.js


Summary

A simple collision model which divides a finite space up into a coarse grid to assist in quickly finding objects within that space.
Class Summary
GridNode A single node within a SpatialGrid.
SpatialGrid A structure which divides a finite space up into a more coarse grid to assist in quickly finding objects within that space.

 42 /**
 43  * @class A single node within a SpatialGrid.  When the collision model is
 44  * 		 updated, the nodes within the grid will be updated to reflect the
 45  * 		 objects within it.  A node defines a single rectangle within the
 46  * 		 entire {@link SpatialGrid}

 82 });

 88 Engine.initObject("SpatialGrid", "SpatialContainer", function() {
 89 
 90 /**
 91  * @class A structure which divides a finite space up into a more
 92  *        coarse grid to assist in quickly finding objects within that
 93  *        space.
 94  *
 95  * @constructor
 96  * @description Create an instance of a spatial grid model
 97  * @param width {Number} The width of the area
 98  * @param height {Number} The height of the area
 99  * @param divisions {Number} The number of divisions along both axis
100  * @extends SpatialContainer
101  */
102 var SpatialGrid = SpatialContainer.extend(/** @scope SpatialGrid.prototype */{
103 
104    divisions: 1,
105 
106    xLocator: 1,
107    yLocator: 1,
108 
109 	/**
110 	 * @private
111 	 */
112    constructor: function(width, height, divisions) {
113       this.base("SpatialGrid", width, height);
114 
115       // Divide the space up into a grid
116       var gX = Math.floor(width / divisions);
117       var gY = Math.floor(height / divisions);
118       this.divisions = divisions;
119       this.xLocator = 1 / gX;
120       this.yLocator = 1 / gY;
121 
122       var grid = [];
123       this.setRoot(grid);
124 
125       for (var y = 0; y < this.divisions; y++)
126       {
127          for (var x = 0; x < this.divisions; x++)
128          {
129             var rect = new Rectangle2D(x * gX, y * gY, gX, gY);
130             grid[x + (y * this.divisions)] = new GridNode(rect);
131          }
132       }
133    },
134 
135    /**
136     * Releases the object back into the object pool.  See {@link PooledObject#release}
137     * for more information.
138     */
139    release: function() {
140       this.base();
141       this.divisions = 1;
142       this.xLocator = 1;
143       this.yLocator = 1;
144    },
145 
146    /**
147     * Find the node that contains the specified point.
148     *
149     * @param point {Point2D} The point to locate the node for
150     * @return {SpatialNode}
151     */
152    findNodePoint: function(point) {
153       return this.getRoot()[Math.floor(point.x * this.xLocator) + (Math.floor(point.y * this.yLocator) * this.divisions)];
154    },
155 
156    /**
157     * Get the node within the grid.
158     * @param x {Number} The virtual X coordinate in our grid
159     * @param y {Number} The virtual Y coordinate in our grid
160     * @return {Number}
161     * @private
162     */
163    getNode: function(x, y) {
164       return this.getRoot()[x + (y * this.divisions)];
165    },
166 
167    /**
168     * Get the list of objects with respect to the point given.  Objects will
169     * be returned from the cross that makes up the grid node containing
170     * the point, and the four adjacent points.  For example, if you had a
171     * 3x3 grid with the object in the center node, the nodes marked with
172     * asterisks below would be included in the result set:
173     * <pre>
174     *  +---+---+---+
175     *  |   | * |   |
176     *  +---+---+---+
177     *  | * | * | * |
178     *  +---+---+---+
179     *  |   | * |   |
180     *  +---+---+---+
181     * </pre>
182     *
183     * @param point {Point2D} The point to begin the search at.
184     * @return {Array} An array of objects found within the cross
185     */
186    getPCL: function(point) {
187 
188       // We'll build our list from the 5 node cross section
189       var x = Math.floor(point.x * this.xLocator);
190       var y = Math.floor(point.y * this.yLocator);
191 
192       // if our borders cross the margin, we can drop up to two nodes
193       var nodes = [];
194       nodes.push(this.getNode(x, y));
195       if (x > 0) { nodes.push(this.getNode(x - 1, y)); }
196       if (x < this.divisions) { nodes.push(this.getNode(x + 1, y)); }
197       if (y > 0) { nodes.push(this.getNode(x, y - 1)); }
198       if (y < this.divisions) { nodes.push(this.getNode(x, y + 1)); }
199 
200       var o = [];
201       if (nodes.length > 0)
202       {
203          for (var n = 0; n < nodes.length; n++)
204          {
205             if (nodes[n])
206             {
207                o = o.concat(nodes[n].getObjects());
208             }
209          }
210       }
211       return o;
212    }
213 
214 }, /** @scope SpatialGrid.prototype */{
215    /**
216     * Get the class name of this object
217     *
218     * @return {String} "SpatialGrid"
219     */
220    getClassName: function() {
221       return "SpatialGrid";
222    }
223 });
224 
225 return SpatialGrid;
226 
227 });
