 92       if ($("div.metric-button.minimize").length > 0) {

111    /**
112     * Render the metrics window
113     * @private
114     */
115    renderMetrics: function() {
116 
117       if (this.showMetricsWindow && !this.metricDisplay) {
118          this.metricDisplay = $("<div/>").addClass("metrics");
119          this.metricDisplay.append(this.metricButton("run", function() { Engine.run(); }));
120          this.metricDisplay.append(this.metricButton("pause", function() { Engine.pause(); }));
121          this.metricDisplay.append(this.metricButton("shutdown", function() { Engine.shutdown(); }));
122 
123          this.metricDisplay.append(this.metricButton("close", function() { Engine.hideMetrics(); }));
124          this.metricDisplay.append(this.metricButton("minimize", function() { Engine.manMetrics(); }));
125 
126          this.metricDisplay.append($("<div class='items'/>"));
127          this.metricDisplay.appendTo($("body"));
128       }
129       
130       if ((this.showMetricsWindow || this.showMetricsProfile) && this.lastMetricSample-- == 0)
131       {
132          // Add some metrics to assist the developer
133          Engine.addMetric("FPS", this.getFPS(), false, "#");
134          Engine.addMetric("aFPS", this.getActualFPS(), true, "#");
135          Engine.addMetric("availTime", this.fpsClock, false, "#ms");
136          Engine.addMetric("frameGenTime", Engine.frameTime, true, "#ms");
137          Engine.addMetric("engineLoad", Math.floor(this.getEngineLoad() * 100), true, "#%");
138          Engine.addMetric("visibleObj", Engine.vObj, false, "#");
139          Engine.addMetric("droppedFrames", Engine.droppedFrames, false, "#");
140          Engine.addMetric("upTime", Math.floor((Engine.worldTime - Engine.upTime)/1000), false, "# sec");
141 
142          this.updateMetrics();
143          this.lastMetricSample = this.metricSampleRate;
144       }
145       
146       if (this.showMetricsProfile && EngineSupport.sysInfo().browser == "msie" &&
147 			 parseFloat(EngineSupport.sysInfo().version) < 9) {
148          // Profiler not supported in IE
149          this.showMetricsProfile = false;
150       }
151       
152       if (this.showMetricsProfile && !this.profileDisplay) {
153          this.profileDisplay = $("<canvas width='150' height='100'/>").addClass("engine-profile");
154          this.profileDisplay.appendTo($("body"));
155          this.profileDisplay[0].getContext('2d').save();
156       }
157    },

171    /**
172     * Add a metric to the game engine that can be displayed
173     * while it is running.  If smoothing is selected, a 3 point
174     * running average will be used to smooth out jitters in the
175     * value that is shown.  For the <tt>fmt</tt> argument,
176     * you can provide a string which contains the pound sign "#"
177     * that will be used to determine where the calculated value will
178     * occur in the formatted string.
179     *
180     * @param metricName {String} The name of the metric to track
181     * @param value {String/Number} The value of the metric.
182     * @param smoothing {Boolean} <tt>true</tt> to use 3 point average smoothing
183     * @param fmt {String} The way the value should be formatted in the display (e.g. "#ms")
184     * @memberOf Engine
185     */
186    addMetric: function(metricName, value, smoothing, fmt) {
187       if (smoothing) {
188          var vals = this.metrics[metricName] ? this.metrics[metricName].values : [];
189          if (vals.length == 0) {
190             // Init
191             vals.push(value);
192             vals.push(value);
193             vals.push(value);
194          }
195          vals.shift();
196          vals.push(value);
197          var v = Math.floor((vals[0] + vals[1] + vals[2]) * 0.33);
198          this.metrics[metricName] = { val: (fmt ? fmt.replace("#", v) : v), values: vals, act: v };
199       } else {
200          this.metrics[metricName] = { val: (fmt ? fmt.replace("#", value) : value), act: value };
201       }
202    },

215    /**
216     * Updates the display of the metrics window.
217     * @private
218     * @memberOf Engine
219     */
220    updateMetrics: function() {
221       var h = "", ctx;
222       if (this.showMetricsProfile) {
223          ctx = this.profileDisplay[0].getContext('2d');
224          ctx.save();
225          ctx.translate(147, 0);
226       }
227 
228       for (var m in this.metrics)
229       {
230 			if (this.showMetricsWindow) {
231 	         h += m + ": " + this.metrics[m].val + "<br/>";
232 			}
233          if (this.showMetricsProfile) {
234             switch (m) {
235                case "engineLoad": this.drawProfilePoint("#ffff00", this.metrics[m].act); break;
236                //case "frameGenTime": this.drawProfilePoint("#ff8888", this.metrics[m].act); break;
237                case "visibleObj": this.drawProfilePoint("#339933", this.metrics[m].act); break;
238                case "poolLoad" : this.drawProfilePoint("#a0a0ff", this.metrics[m].act); break;
239             }
240          }
241       }
242 		if (this.showMetricsWindow) {
243 			$(".items", this.metricDisplay).html(h);
244 		}
245 	   if (this.showMetricsProfile) {
246          ctx.restore();
247          this.moveProfiler();
248       }
249    },
250 
251    drawProfilePoint: function(color, val) {
252       var ctx = this.profileDisplay[0].getContext('2d');
253       ctx.strokeStyle = color
254       try {
255          if (!isNaN(val)) {
256             ctx.beginPath();
257             ctx.moveTo(0, this.profiles[color] || 100);
258             ctx.lineTo(1, (100 - val < 1 ? 1 : 100 - val));
259             ctx.closePath();
260             ctx.stroke();
261             this.profiles[color] = (100 - val < 1 ? 1 : 100 - val);
262          }
263       } catch(ex) {
264          
265       }
266    },
267    
268    moveProfiler: function() {
269       var ctx = this.profileDisplay[0].getContext('2d');
270       var imgData = ctx.getImageData(1,0,149,100);
271       ctx.save();
272       ctx.translate(-1,0);
273       ctx.putImageData(imgData, 0, 0);
274       ctx.restore();
275    },
276 
277    /**
278     * Run the metrics display.
279     * @private
280     * @memberOf Engine
281     */
282    doMetrics: function() { 
283       // Output any metrics
284       if (Engine.showMetricsWindow || Engine.showMetricsProfile) {
285          Engine.renderMetrics();
286       } else if (!Engine.showMetricsWindow && Engine.metricDisplay) {
287          Engine.metricDisplay.remove();
288          Engine.metricDisplay = null;
289       }
290    }
