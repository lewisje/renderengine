Summary

An extension of the 2D render context using the SVG (Scalable Vector Graphics) as a render context.

 41 var SVGContext = RenderContext2D.extend({
 42 
 43    context2D: null,
 44 
 45    transformStack: null,
 46 
 47    currentTransform: null,
 48 
 49    /**
 50     * Create an instance of a 2D rendering context using the canvas element.
 51     *
 52     * @param contextName {String} The name of this context.  Default: CanvasContext
 53     * @param width {Number} The width (in pixels) of the canvas context.
 54     * @param height {Number} The height (in pixels) of the canvas context.
 55     */
 56    constructor: function(width, height) {
 57       Assert((width != null && height != null), "Width and height must be specified in CanvasContext");
 58 
 59       this.setWidth(width);
 60       this.setHeight(height);
 61 
 62       // Create the canvas element
 63       var svg = document.createElement("svg");
 64       svg.xmlns = "http://www.w3.org/2000/svg";
 65       svg.version = "1.1";
 66       svg.width = this.width + "px";
 67       svg.height = this.height + "px";
 68       svg.id = this.getId();
 69 
 70       this.transformStack = new SVGTransformList();
 71       this.transformStack.initialize();
 72 
 73       this.currentTransform = new SVGTransform();
 74 
 75       this.base("SVGContext", svg);
 76    },
 77 
 78    /**
 79     * Push a transform state onto the stack.
 80     */
 81    pushTransform: function() {
 82       this.base();
 83       this.transformList.appendItem(this.currentTransform);
 84    },
 85 
 86    /**
 87     * Pop a transform state off the stack.
 88     */
 89    popTransform: function() {
 90       this.base();
 91       this.currentTransform = this.transformList.removeItem(this.transformList.numberOfItems - 1);
 92    },
 93 
 94    /**
 95     * Get the current transform

153    _arc: function(point, radiusX, startAngle, endAngle) {
154       this.startPath();

165       this.fillPath();
166       this.base(point, radiusX, startAngle, endAngle);
167    },
168 
169    _poly: function(pointArray) {
170       this.startPath();
171       this.moveTo(pointArray[0]);
172       var p = 1;
173 
174       // Using Duff's device with loop inversion
175       switch((pointArray.length - 1) & 0x3)
176       {
177          case 3:
178             this.lineTo(pointArray[p++]);
179          case 2:
180             this.lineTo(pointArray[p++]);
181          case 1:
182             this.lineTo(pointArray[p++]);
183       }
184 
185       if (p < pointArray.length)
186       {
187          do
188          {
189             this.lineTo(pointArray[p++]);
190             this.lineTo(pointArray[p++]);
191             this.lineTo(pointArray[p++]);
192             this.lineTo(pointArray[p++]);
193          } while (p < pointArray.length);
194       }
195 
196       this.endPath();
197    },
198 
199    /**
200     * Creates a render list which will make inline calls to the
201     * line drawing methods instead of looping over them.  Logically
202     * this method returns a function which will draw the polygon.
203     *
204     * @param pointArray {Array} An array of Point2D objects
205     * @type Function
206     */
207    buildRenderList: function(pointArray) {
208       var f = "arguments.callee.ctx.startPath(); arguments.callee.ctx.moveTo(arguments.callee.ptArr[0]);";
209       for (var p = 1; p < pointArray.length; p++)
210       {
211          f += "arguments.callee.ctx.lineTo(arguments.callee.ptArr[" + p + "]);";
212       }
213       f += "arguments.callee.ctx.endPath();arguments.callee.ctx.strokePath();";
214       var _fastPoly = new Function(f);
215       _fastPoly.ctx = this;
216       _fastPoly.ptArr = pointArray;
217       _fastPoly.isRenderList = true;
218       return _fastPoly;
219    },
220 
221    drawPolygon: function(pointArray) {
222       if (pointArray.isRenderList)
223       {
224          pointArray();
225          return;
226       }
227       this._poly(pointArray);
228       this.strokePath();
229       this.base(pointArray);
230    },
231 
232    drawFilledPolygon: function(pointArray) {
233       if (pointArray.isRenderList)
234       {
235          pointArray();
236       this.fillPath();
237          return;
238       }
239       this._poly(pointArray);
240       this.fillPath();
241       this.base(pointArray);
242    },
243 
244    drawLine: function(point1, point2) {
245       this.startPath();
246       this.moveTo(point1.x, point1.y);
247       this.lineTo(point2.x, point2.y);
248       this.endPath();
249       this.strokePath();
250       this.base(point1, point2);
251    },

256    },
