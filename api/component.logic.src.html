 33 // Includes
 34 Engine.include("/components/component.base.js");
 35 
 36 Engine.initObject("LogicComponent", "BaseComponent", function() {
 37 
 38 /**
 39  * @class Logic components are sort of a catch-all of components.  They aren't
 40  *        any one of the specific types, so they fall under the type of LOGIC.
 41  *        Logic components are in the middle of the importance scale, so they
 42  *        are processed after input and transformations, but before collision and
 43  *        rendering.  This makes them ideal for additional processing, such as the
 44  *        {@link HostComponent}.
 45  *
 46  * @param name {String} The name of the component
 47  * @param [priority=1.0] {Number} The priority of the component
 48  * @extends BaseComponent
 49  * @constructor
 50  * @description Creates a logic component.
 51  */
 52 var LogicComponent = BaseComponent.extend(/** @scope LogicComponent.prototype */{
 53 
 54    /**
 55     * @private
 56     */
 57    constructor: function(name, priority) {
 58       this.base(name, BaseComponent.TYPE_LOGIC, priority || 1.0);
 59    }
 60 }, /** @scope LogicComponent.prototype */{
 61    /**
 62     * Get the class name of this object
 63     * @return {String} "LogicComponent"
 64     */
 65    getClassName: function() {
 66       return "LogicComponent";
 67    }
 68 });
 69 
 70 return LogicComponent;
 71 
 72 });
