loader.bitmapfont.js


Summary

An extension of the image resource loader for handling bitmap fonts.
Class Summary
BitmapFontLoader Loads bitmap fonts and makes them available to the system.

 34 // Includes
 35 Engine.include("/resourceloaders/loader.image.js");
 36 
 37 Engine.initObject("BitmapFontLoader", "ImageLoader", function() {
 38 
 39 /**
 40  * @class Loads bitmap fonts and makes them available to the system.
 41  *
 42  * @constructor
 43  * @param name {String=BitmapFontLoader} The name of the resource loader
 44  * @extends ImageLoader
 45  */
 46 var BitmapFontLoader = ImageLoader.extend(/** @scope BitmapFontLoader.prototype */{
 47 
 48    fonts: null,
 49 
 50    /**
 51     * @private
 52     */
 53    constructor: function(name) {
 54       this.base(name || "BitmapFontLoader");
 55       this.fonts = {};
 56    },
 57 
 58    /**
 59     * Load a font resource from a URL.
 60     *
 61     * @param name {String} The name of the resource
 62     * @param url {String} The URL where the resource is located
 63     */
 64    load: function(name, url, info) {
 65 
 66       // The bitmap is in the same path
 67       var path = Engine.getEnginePath() + "/fonts/";
 68 
 69       if (url)
 70       {
 71          var thisObj = this;
 72 
 73          // Get the file from the server
 74 			Engine.loadJSON(path + url, function(fontInfo, status) {
 75 				if (status == 200) {
 76 					Console.log("Acquired font info: ", fontInfo);
 77 	            thisObj.load(name, null, fontInfo);
 78 				} else {
 79 					Console.error("Error loading font for name '" + name + "', due to: " + status);
 80 				}
 81          });
 82       }
 83       else
 84       {
 85          Console.log("Loading font: " + name + " @ " + path + info.bitmapImage);
 86 
 87          // Store the font info
 88          this.fonts[name] = info;
 89 
 90          // Load the bitmap file
 91          this.base(name, path + info.bitmapImage, info.width, info.height);
 92       }
 93    },
 94 
 95    /**
 96     * Get the font with the specified name from the cache.  The
 97     * object returned contains the bitmap as <tt>image</tt> and
 98     * the font definition as <tt>info</tt>.
 99     *
100     * @param name {String} The name of the object to retrieve
101     * @return {Object} The object representing the named font
102     */
103    get: function(name) {
104       var bitmap = this.base(name);
105       var font = {
106          image: bitmap,
107          info: this.fonts[name]
108       };
109       return font;
110    },
111 
112    /**
113     * The name of the resource this loader will get.
114     * @return {String} The string "bitmap font"
115     */
116    getResourceType: function() {
117       return "bitmap font";
118    }
119 }, { /** @scope BitmapFontLoader.protoype */
120    /**
121     * Get the class name of this object
122     * @return {String} The string "BitmapFontLoader"
123     */
124    getClassName: function() {
125       return "BitmapFontLoader";
126    }
127 });
128 
129 return BitmapFontLoader;
130 
131 });
