 34 Engine.initObject("MathObject", "PooledObject", function() {
 35 
 36 /**
 37  * @class The base object class which represents a math object within
 38  * the engine.  All math objects should extend from this class mainly due to
 39  * the fact that the engine can switch between pooling the object and running
 40  * transiently.
 41  * <p/>
 42  * Google's Chrome browser seems to operate better with transient objects while
 43  * other browsers appear to run better with pooled objects.
 44  * <p/>
 45  * 
 46  * @param name {String} The name of the object
 47  * @extends PooledObject
 48  * @constructor
 49  * @description Create a math object. 
 50  */
 51 var MathObject = PooledObject.extend(/** @scope MathObject.prototype */{
 52 
 53    constructor: function(name) {
 54       if (!Engine.options.transientMathObject) {
 55          this.base(name);
 56       }
 57    },
 58 
 59    /**
 60     * Destroy this object instance (remove it from the Engine).  The object's release
 61     * method is called after destruction so it will be returned to the pool of objects 
 62     * to be used again.
 63     */
 64    destroy: function() {
 65       if (!Engine.options.transientMathObject) {
 66          this.base();
 67       }
 68    }
 69 
 70  }, /** @scope BaseObject.prototype */{
 71 
 72    /**
 73     * Similar to a constructor, all pooled objects implement this method.
 74     * The <tt>create()</tt> method will either create a new instance, if no object of the object's
 75     * class exists within the pool, or will reuse an existing pooled instance of
 76     * the object.  Either way, the constructor for the object instance is called so that
 77     * instance creation can be maintained in the constructor.
 78     * <p/>
 79     * Usage: <tt>var obj = [ObjectClass].create(arg1, arg2, arg3...);</tt>
 80     *
 81     * @memberOf PooledObject
 82     */
 83    create: function() {
 84       if (Engine.options.transientMathObject) {
 85          return new this(arguments[0],arguments[1],arguments[2],arguments[3],arguments[4],
 86                          arguments[5],arguments[6],arguments[7],arguments[8],arguments[9],
 87                          arguments[10],arguments[11],arguments[12],arguments[13],arguments[14]);
 88       } else {
 89          return PooledObject.create.apply(this, arguments);
 90       }
 91    },
 92 
 93    /**
 94     * Get the class name of this object
 95     *
 96     * @return {String} "BaseObject"
 97     */
 98    getClassName: function() {
 99       return "MathObject";
100    }
101 	
102 });
103 
104 return MathObject;

110 /**
111  * @class A 2D point class with helpful methods for manipulation
112  *
113  * @param x {Point2D|Number} If this arg is a Point2D, its values will be
114  *                           copied into the new point.
115  * @param y {Number} The Y coordinate of the point.  Only required if X
116  *                   was a number.
117  * @constructor
118  * @description Create a new 2D point.
119  * @extends MathObject
120  */
121 var Point2D = MathObject.extend(/** @scope Point2D.prototype */{
122 
123    _vec: null,
124 
125    x: 0,
126    y: 0,
127 
128    /**
129     * @private
130     */
131    constructor: function(x, y) {
132       this.base("Point2D");
133 		this._vec = $V([0,0]);
134       this.set(x, y);
135    },
136 
137    release: function() {
138       this.base();
139       this.x = 0;
140       this.y = 0;
141       this._vec.setElements([0,0]);
142    },
143 
144    /**
145     * Private method used to update underlying object.
146     * @private
147     */
148    upd: function() {
149       this.x = this._vec.e(1); this.y = this._vec.e(2);
150    },
151 
152 	/**
153 	 * Returns a simplified version of a Point2D.  The simplified version is
154 	 * an array with two elements: X, Y.
155 	 * @return {Array}
156 	 */
157 	simplify: function() {
158 		return [this._vec.e(1), this._vec.e(2)];
159 	},
160 
161    /**
162     * Returns <tt>true</tt> if this point is equal to the specified point.
163     *
164     * @param point {Point2D} The point to compare to
165     * @return {Boolean} <tt>true</tt> if the two points are equal
166     */
167    equals: function(point) {
168 		return this._vec.eql(point._vec);
169    },
170 
171    /**
172     * Set the position of a 2D point.
173     *
174     * @param x {Point2D|Number|Array} If this arg is a Point2D, its values will be
175     *                           copied into the new point.
176     * @param y {Number} The Y coordinate of the point.  Only required if X
177     *                   was a number.
178     */
179    set: function(x, y) {
180 		if (x.length && x.splice && x.shift) {
181 			// An array
182 			this._vec.setElements([x[0],x[1]]);
183 		} else if (Point2D.isInstance(x)) {
184          this._vec = x._vec.dup();
185       } else {
186          AssertWarn((y != null), "Undefined Y value for point initialized to zero.");
187          this._vec.setElements([x, y]);
188       }
189       this.upd();
190       return this;
191    },
192    
193    /**
194     * Get the elements of this point as an object with elements X and Y.
195     * @return {Object}
196     */
197    get: function() {
198       return { x: this._vec.e(1), y: this._vec.e(2) };
199    },
200 
201    /**
202     * Set the X coordinate.
203     *
204     * @param x {Number} The X coordinate
205     */
206    setX: function(x) {
207       this._vec.setElements([x, this._vec.e(2)]);
208       this.upd();
209    },
210 
211    /**
212     * Set the Y coordinate.
213     *
214     * @param y {Number} The Y coordinate
215     */
216    setY: function(y) {
217       this._vec.setElements([this._vec.e(1), y]);
218       this.upd();
219    },
220 
221    /**
222     * A method that mutates this point by adding the point to it.
223     *
224     * @param point {Point2D} A point
225     * @return {Point2D} This point
226     */
227    add: function(point) {
228       this._vec = this._vec.add(point._vec);
229       this.upd();
230       return this;
231    },
232 
233    /**
234     * A mutator method that adds the scalar value to each component of this point.
235     * @param scalar {Number} A number
236     * @return {Point2D} This point
237     */
238    addScalar: function(scalar) {
239       this._vec = this._vec.map(function(x) { return x + scalar; });
240       this.upd();
241       return this;
242    },
243 
244    /**
245     * A mutator method that subtracts the specified point from this point.
246     * @param point {Point2D} a point
247     * @return {Point2D} This point
248     */
249    sub: function(point) {
250       this._vec = this._vec.subtract(point._vec);
251       this.upd();
252       return this;
253    },
254 
255    /**
256     * A mutator method that multiplies the components of this point with another.
257     * @param point {Point2D} A point
258     * @return {Point2D} This point
259     */
260    convolve: function(point) {
261       this._vec = this._vec.map(function(x, i) { return x * (i == 1 ? point.x : point.y); });
262       this.upd();
263       return this;
264    },
265 
266    /**
267     * A mutator method that divides the components of this point by another.  The point 
268     * cannot contain zeros for its components.
269     * @param point {Point2D} A point
270     * @return {Point2D} This point
271     */
272    convolveInverse: function(point) {
273       Assert((point.x != 0 && point.y != 0), "Division by zero in Point.convolveInverse");
274       this._vec = this._vec.map(function(x, i) { return x / (i == 1 ? point.x : point.y); });
275       this.upd();
276       return this;
277    },
278 
279    /**
280     * A mutator methor that multiplies the components of this point by a scalar value.
281     * @param scalar {Number} A number
282     * @return {Point2D} This point
283     */
284    mul: function(scalar) {
285       this._vec = this._vec.map(function(x) { return x * scalar; });
286       this.upd();
287       return this;
288    },
289 
290    /**
291     * A mutator method that divides the components of this point by a scalar value.
292     * @param scalar {Number} A number - cannot be zero
293     * @return {Point2D} This point
294     */
295    div: function(scalar) {
296       Assert((scalar != 0), "Division by zero in Point.divScalar");
297 
298       this._vec = this._vec.map(function(x) { return x / scalar; });
299       this.upd();
300       return this;
301    },
302 
303    /**
304     * A mutator method that negates this point, inversing it's components.
305     * @return {Point2D} This point
306     */
307    neg: function() {
308       this._vec.setElements([ -this._vec.e(1), -this._vec.e(2) ]);
309       this.upd();
310       return this;
311    },
312 
313    /**
314     * Returns true if the point is the zero point.
315     * @return {Boolean} <tt>true</tt> if the point's elements are both zero.
316     */
317    isZero: function() {
318       return this._vec.eql(Vector.Zero);
319    },
320    
321    /**
322     * Returns the distance between this and another point.
323     * @param point {Point2D} The point to compare against
324     * @return {Number} The distance between the two points
325     */
326    dist: function(point) {
327       return this._vec.distanceFrom(point._vec);
328    },
329 	
330 	/**
331 	 * Project this point from 2 dimensions to 3 dimensions, using one of three projection
332 	 * types: {@link Math2D.ISOMETRIC_PROJECTION}  <i>(default)</i>, {@link Math2D.DIMETRIC_SIDE_PROJECTION}, or
333 	 * {@link Math2D.DIMETRIC_TOP_PROJECTION}.
334 	 * <p/>
335 	 * Reference: http://www.compuphase.com/axometr.htm
336 	 *
337 	 * @param height {Number} The height of the ground.  We must use a particular height to
338 	 * 		extrapolate our 3D coordinates from.  If the ground is considered level, this can remain zero.
339 	 * @param projectionType {Number} One of the three projection types in {@link Math2D}
340 	 * @return {Point3D} This point, projected into 3 dimensions
341 	 */
342 	project: function(height, projectionType) {
343 		height = height || 0;
344 		projectionType = projectionType || Math2D.ISOMETRIC_PROJECTION;
345 		var pt = Point3D.create(0,0,0), j = this.get();
346 		switch (projectionType) {
347 			case Math2D.ISOMETRIC_PROJECTION:
348 				pt.set(0.5 * j.x + j.y - height, -(0.5 * j.x) + j.y - height, height); break;
349 			case Math2D.DIMETRIC_SIDE_PROJECTION:
350 				pt.set(j.x + (2 * (j.y - height)), 4 * j.y - height, height); break;
351 			case Math2D.DIMETRIC_TOP_PROJECTION:
352 				pt.set(j.x - ((j.y - height) /2 ), 2 * (j.y - height), height); break;		
353 		}
354 		return pt;		
355 	},
356 
357    /**
358     * Returns a printable version of this object fixed to two decimal places.
359     * @return {String} Formatted as "x,y"
360     */
361    toString: function() {
362       return Number(this._vec.e(1)).toFixed(2) + "," + Number(this._vec.e(2)).toFixed(2);
363    }
364 
365 }, { /** @scope Point2D.prototype */
366 
367    /**
368     * Return the classname of the this object
369     * @return {String} "Point2D"
370     */
371    getClassName: function() {
372       return "Point2D";
373    }
374 });
375 
376 /**
377  * The "zero" point. This point should not be modified.
378  */
379 Point2D.ZERO = Point2D.create(0,0);
380 
381 return Point2D;
382
