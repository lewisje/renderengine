Summary

 37 Engine.initObject("Iterator", "PooledObject", function() {
 38 
 39 /**
 40  * @class Create an iterator over a {@link Container} instance. An
 41  * iterator is a convenient object to traverse the list of elements
 42  * within the container.  The simplest way to traverse the list is
 43  * as follows:
 44  * <pre>
 45  * for (var itr = Iterator.create(containerObj); itr.hasNext(); ) {
 46  *    // Get the next object in the container
 47  *    var o = itr.next();
 48  *    
 49  *    // Do something with the object
 50  *    o.doSomething();
 51  * }
 52  * 
 53  * // Destroy the iterator when done
 54  * itr.destroy();
 55  * </pre>
 56  * The last step is important so that you're not creating a lot
 57  * of objects, especially if the iterator is used repeatedly.
 58  * Since the iterator is a pooled object, it will be reused.
 59  *
 60  * @param container {Container} The container to iterate over.
 61  * @constructor
 62  * @extends PooledObject
 63  * @description Create an iterator over a collection
 64  */
 65 var Iterator = PooledObject.extend(/** @scope Iterator.prototype */{
 66 
 67    idx: 0,
 68 
 69    c: null,
 70 
 71    objs: null,
 72 
 73    /**
 74     * @private
 75     */
 76    constructor: function(container) {
 77       this.base("Iterator");
 78       this.idx = 0;
 79       this.c = container;
 80 
 81       // Duplicate the elements in the container
 82       this.objs = new Array().concat(container.getObjects());
 83    },
 84 
 85    /**
 86     * Release the object back into the object pool.
 87     */
 88    release: function() {
 89       this.base();
 90       this.idx = 0;
 91       this.c = null;
 92       this.objs = null;
 93    },
 94 
 95    /**
 96     * Reverse the order of the elements in the container (non-destructive) before
 97     * iterating over them.  You cannot call this method after you have called {@link #next}.
 98     */
 99    reverse: function() {
100       Assert(this.idx == 0, "Cannot reverse Iterator after calling next()");
101       this.objs.reverse();
102    },
103 
104    /**
105     * Get the next element from the iterator.
106     * @return {Object} The next element in the iterator
107     * @throws {Error} An error if called when no more elements are available
108     */
109    next: function() {
110       if (this.idx < this.c.size())
111       {
112          return this.objs[this.idx++];
113       }
114       throw new Error("Index out of range");
115    },
116 
117    /**
118     * Returns <tt>true</tt> if the iterator has more elements.
119     * @return {Boolean}
120     */
121    hasNext: function() {
122       return (this.idx < this.c.size());
123    }
124 
125 }, /** @scope Iterator.prototype */{ 
126    /**
127     * Get the class name of this object
128     *
129     * @return {String} "Iterator"
130     */
131    getClassName: function() {
132       return "Iterator";
133    }
134 });
135 
136 return Iterator;
137

140 Engine.initObject("Container", "BaseObject", function() {
141 
142 /**
143  * @class A container is a logical collection of objects.  A container
144  *        is responsible for maintaining the list of objects within it.
145  *        When a container is destroyed, all objects within the container
146  *        are destroyed with it.
147  *
148  * @param containerName {String} The name of the container
149  * @extends BaseObject
150  * @constructor
151  * @description Create a container.
152  */
153 var Container = BaseObject.extend(/** @scope Container.prototype */{
154 
155    objects: null,
156 
157    /**
158     * @private
159     */
160    constructor: function(containerName) {
161       this.base(containerName || "Container");
162       this.objects = [];
163    },
164 
165    /**
166     * Release the object back into the object pool.
167     */
168    release: function() {
169       this.base();
170       this.objects = null;
171    },
172 
173    /**
174     * Destroy all objects contained within this object.  Calls the
175     * <tt>destroy()</tt> method on each object, giving them a chance
176     * to perform clean up operations.
177     */
178    destroy: function() {
179       this.cleanUp();
180       this.base();
181    },
182 
183    /**
184     * Returns the count of the number of objects within the
185     * container.
186     *
187     * @return {Number} The number of objects in the container
188     */
189    size: function() {
190       return this.objects.length;
191    },
192 
193    /**
194     * Add an object to the container.
195     *
196     * @param obj {Object} The object to add to the container.
197     */
198    add: function(obj) {
199       this.objects.push(obj);
200       if (obj.getId) {
201          Console.log("Added ", obj.getId(), "[", obj, "] to ", this.getId(), "[", this, "]");
202       }
203    },
204 
205    /**
206     * Insert an object into the container at the given index. Asserts if the
207     * index is out of bounds for the container.  The index must be greater than
208     * or equal to zero, and less than or equal to the size of the container minus one.
209     * 
210     * @param index {Number} The index to insert the object at.
211     * @param obj {Object} The object to insert into the container
212     */
213    insert: function(index, obj) {
214       Assert(!(index < 0 || index > this.objects.length - 1), "Index out of range when inserting object!");
215       this.objects.splice(index, 0, obj);
216    },
217    
218    /**
219     * Replaces the given object with the new object.  If the old object is
220     * not found, no action is performed.
221     * 
222     * @param oldObj {Object} The object to replace
223     * @param newObj {Object} The object to put in place
224     * @return {Object} The object which was replaced
225     */
226    replace: function(oldObj, newObj) {
227       return this.replaceAt(EngineSupport.indexOf(this.objects, oldObj), newObj);      
228    },
229    
230    /**
231     * Replaces the object at the given index, returning the object that was there
232     * previously. Asserts if the index is out of bounds for the container.  The index 
233     * must be greater than or equal to zero, and less than or equal to the size of the 
234     * container minus one.
235     * 
236     * @param index {Number} The index at which to replace the object
237     * @param obj {Object} The object to put in place
238     * @return {Object} The object which was replaced
239     */
240    replaceAt: function(index, obj) {
241       Assert(!(index < 0 || index > this.objects.length - 1), "Index out of range when inserting object!");
242       return this.objects.splice(index, 1, obj);      
243    },
244    
245    /**
246     * Remove an object from the container.  The object is
247     * not destroyed when it is removed from the container.
248     *
249     * @param obj {Object} The object to remove from the container.
250     */
251    remove: function(obj) {
252       if (obj.getId) {
253          Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
254       }
255       EngineSupport.arrayRemove(this.objects, obj);
256    },
257 
258    /**
259     * Remove an object from the container at the specified index.
260     * The object is not destroyed when it is removed.
261     *
262     * @param idx {Number} An index between zero and the size of the container minus 1.
263     * @return {Object} The object removed from the container.
264     */
265    removeAtIndex: function(idx) {
266       Assert((idx >= 0 && idx < this.size()), "Index of out range in Container");
267 
268       var obj = this.objects[idx];
269 
270       Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
271       this.objects.splice(idx, 1);
272 
273       return obj;
274    },
275 
276    /**
277     * Get the object at the index specified. If the container has been
278     * sorted, objects might not be in the position you'd expect.
279     *
280     * @param idx {Number} The index of the object to get
281     * @return {Object} The object at the index within the container
282     * @throws {Error} Index out of bounds if the index is out of the list of objects
283     */
284    get: function(idx) {
285       if (idx < 0 || idx > this.size()) {
286          throw new Error("Index out of bounds");
287       }
288       
289       return this.objects[idx];
290    },
291 
292    /**
293     * Remove all objects from the container.  None of the objects are
294     * destroyed, only removed from this container.
295     */
296    clear: function() {
297       this.objects.length = 0;
298       this.objects = [];
299    },
300 
301    /**
302     * Remove and destroy all objects from the container.
303     */
304    cleanUp: function() {
305       while(this.objects.length > 0)
306       {
307          var o = this.objects.shift();
308          o.destroy();
309       }
310       this.clear();
311    },
312 
313    /**
314     * Get the array of all objects within this container.  If a filtering
315     * function is provided, only objects matching the filter will be
316     * returned from the object collection.
317     * <p/>
318     * The filter function needs to return <tt>true</tt> for each element
319     * that should be contained in the filtered set.  The function will be
320     * passed the following arguments:
321     * <ul>
322     * <li>element - The array element being operated upon</li>
323     * <li>index - The index of the element in the array</li>
324     * <li>array - The entire array of elements in the container</li>
325     * </ul>
326     * Say you wanted to filter a host objects components based on a
327     * particular type.  You might do something like the following:
328     * <pre>
329     * var logicComponents = host.getObjects(function(el, idx) {
330     *    if (el.getType() == BaseComponent.TYPE_LOGIC) {
331     *       return true;
332     *    }
333     * });
334     * </pre>
335     *
336     * @param filterFn {Function} A function to filter the set of
337     *                 elements with.  If you pass <tt>null</tt> the
338     *                 entire set of objects will be returned.
339     * @return {Array} The array of filtered objects
340     */
341    getObjects: function(filterFn) {
342       if (filterFn) {
343          return EngineSupport.filter(this.objects, filterFn);
344       } else {
345          return this.objects;
346       }
347    },
348 
349    /**
350     * Sort the objects within the container, using the provided function.
351     * The function will be provided object A and B.  If the result of the
352     * function is less than zero, A will be sorted before B.  If the result is zero,
353     * A and B retain their order.  If the result is greater than zero, A will
354     * be sorted after B.
355     *
356     * @param [fn] {Function} The function to sort with. If <tt>null</tt> the objects
357     *          will be sorted in "natural" order.
358     */
359    sort: function(fn) {
360       Console.log("Sorting ", this.getName(), "[" + this.getId() + "]");
361       this.objects.sort(fn);
362    },
363 
364    /**
365     * Returns a property object with accessor methods to modify the property.
366     * @return {Object} The properties object
367     */
368    getProperties: function() {
369       var self = this;
370       var prop = this.base(self);
371       return $.extend(prop, {
372          "Contains"  : [function() { return self.objects.length; },
373                         null, false]
374       });
375    },
376 
377    /**
378     * Serializes a container to XML.
379     * @return {String} The XML string
380     */
381    toString: function(indent) {
382       indent = indent ? indent : "";
383       var props = this.getProperties();
384       var xml = indent + "<" + this.constructor.getClassName();
385       for (var p in props) {
386          // If the value should be serialized, call it's getter
387          if (props[p][2]) {
388             xml += " " + p + "=\"" + props[p][0]().toString() + "\"";
389          }
390       }
391       xml += ">\n";
392 
393       // Dump children
394       for (var o in this.objects) {
395          xml += this.objects[o].toString(indent + "   ");
396       }
397 
398       // Closing tag
399       xml += indent + "</" + this.constructor.getClassName() + ">\n";
400       return xml;
401    }
402 
403 }, /** @scope Container.prototype */{
404    /**
405     * Get the class name of this object
406     *
407     * @return {String} "Container"
408     */
409    getClassName: function() {
410       return "Container";
411    }
412 });
413 
414 return Container;

418 Engine.initObject("HashContainer", "Container", function() {
419 
420 /**
421  * @class A hash container is a logical collection of objects.  A hash container
422  *        is a container with a backing object for faster lookups.  Objects within
423  *        the container must have unique names. When a container is destroyed, all
424  *        objects within the container are destroyed with it.
425  *
426  * @param containerName {String} The name of the container. Default: Container
427  * @extends Container
428  * @constructor
429  * @description Create a hashed container object.
430  */
431 var HashContainer = Container.extend(/** @scope HashContainer.prototype */{
432 
433    objHash: null,
434 
435    /**
436     * @private
437     */
438    constructor: function(containerName) {
439       this.base(containerName || "HashContainer");
440       this.objHash = {};
441    },
442 
443    /**
444     * Release the object back into the object pool.
445     */
446    release: function() {
447       this.base();
448       this.objHash = null;
449    },
450 
451    /**
452     * Returns <tt>true</tt> if the object name is already in
453     * the hash.
454     *
455     * @param name {String} The name of the hash to check
456     * @return {Boolean}
457     */
458    isInHash: function(key) {
459       return (this.objHash["_" + String(key)] != null);
460    },
461 
462    /**
463     * Add an object to the container.
464     *
465     * @param key {String} The name of the object to store.  Names must be unique
466     *                      or the object with that name will be overwritten.
467     * @param obj {BaseObject} The object to add to the container.
468     */
469    add: function(key, obj) {
470       AssertWarn(!this.isInHash(key), "Object already exists within hash!");
471 
472       if (this.isInHash(key)) {
473          // Remove the old one first
474          this.removeHash(key);
475       }
476 
477       // Some keys weren't being accepted (like "MOVE") so added
478       // an underscore to prevent keyword collisions
479       this.objHash["_" + String(key)] = obj;
480       this.base(obj);
481    },
482 
483    /**
484     * Remove an object from the container.  The object is
485     * not destroyed when it is removed from the container.
486     *
487     * @param obj {BaseObject} The object to remove from the container.
488     */
489    remove: function(obj) {
490       for (var o in this.objHash)
491       {
492          if (this.objHash[o] === obj)
493          {
494             this.removeHash(o);
495             break;
496          }
497       }
498 
499       this.base(obj);
500    },
501 
502    /**
503     * Remove the object with the given key name from the container.
504     *
505     * @param name {String} The object to remove
506     * @return {Object} The object removed
507     */
508    removeHash: function(key) {
509       var obj = this.objHash["_" + String(key)];
510       EngineSupport.arrayRemove(this.objects, this.objHash["_" + String(key)]);
511       delete this.objHash["_" + String(key)];
512       return obj;
513    },
514 
515    /**
516     * Remove an object from the container at the specified index.
517     * The object is not destroyed when it is removed.
518     *
519     * @param idx {Number} An index between zero and the size of the container minus 1.
520     * @return {Object} The object removed from the container.
521     */
522    removeAtIndex: function(idx) {
523       var obj = this.base(idx);
524       for (var o in this.objHash) {
525          if (this.objHash[o] === obj) {
526             this.removeHash(o);
527             break;
528          }
529       }
530 
531       return obj;
532    },
533 
534    /**
535     * If a number is provided, the request will be passed to the
536     * base object, otherwise a name is assumed and the hash will
537     * be retrieved.
538     *
539     * @param idx {Number|String} The index or hash of the object to get
540     * @return {Object}
541     */
542    get: function(idx) {
543       if (typeof idx == 'string') {
544          return this.objHash["_" + String(idx)];
545       } else {
546          return this.base(idx);
547       }
548    },
549 
550    /**
551     * Remove all objects from the container.  None of the objects are
552     * destroyed.
553     */
554    clear: function() {
555       this.base();
556       this.objHash = {};
557    },
558 
559    /**
560     * Cleans up the references to the objects (destroys them) within
561     * the container.
562     */
563    cleanUp: function() {
564       this.base();
565    }
566 
567 }, /** @scope HashContainer.prototype */{
568    /**
569     * Get the class name of this object
570     *
571     * @return {String} "HashContainer"
572     */
573    getClassName: function() {
574       return "HashContainer";
575    }
576 });
577 
578 return HashContainer;
