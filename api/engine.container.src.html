 37 Engine.initObject("Iterator", "PooledObject", function() {
 38 
 39 /**
 40  * @class Create an iterator over a {@link Container} instance. An
 41  * iterator is a convenient object to traverse the list of objects
 42  * within the container.  If the backing <code>Container</code> is
 43  * modified, the <code>Iterator</code> will reflect these changes.
 44  * <p/>
 45  * The simplest way to traverse the list is as follows:
 46  * <pre>
 47  * for (var itr = Iterator.create(containerObj); itr.hasNext(); ) {
 48  *    // Get the next object in the container
 49  *    var o = itr.next();
 50  *    
 51  *    // Do something with the object
 52  *    o.doSomething();
 53  * }
 54  * 
 55  * // Destroy the iterator when done
 56  * itr.destroy();
 57  * </pre>
 58  * The last step is important so that you're not creating a lot
 59  * of objects, especially if the iterator is used repeatedly.
 60  * Since the iterator is a pooled object, it will be reused.
 61  *
 62  * @param container {Container} The container to iterate over.
 63  * @constructor
 64  * @extends PooledObject
 65  * @description Create an iterator over a collection
 66  */
 67 var Iterator = PooledObject.extend(/** @scope Iterator.prototype */{
 68 
 69    c: null,
 70 	p: null,
 71 	r: false,
 72 	logicalNext: null,
 73 
 74    /**
 75     * @private
 76     */
 77    constructor: function(container) {
 78       this.base("Iterator");
 79       this.c = container;
 80 		this.p = container._head;
 81 		this.r = false;
 82 		this.logicalNext = null;
 83    },
 84 
 85    /**
 86     * Release the object back into the object pool.
 87     */
 88    release: function() {
 89       this.base();
 90       this.c = null;
 91 		this.p = null;
 92 		this.r = false;
 93 		this.logicalNext = null;
 94    },
 95 
 96    /**
 97     * Reverse the order of the elements in the container (non-destructive) before
 98     * iterating over them.  You cannot call this method after you have called {@link #next}.
 99     */
100    reverse: function() {
101       Assert(this.p === this.c._head, "Cannot reverse Iterator after calling next()");
102       this.r = true;
103 		this.p = this.c._tail;
104    },
105 
106    /**
107     * Get the next element from the iterator.
108     * @return {Object} The next element in the iterator
109     * @throws {Error} An error if called when no more elements are available
110     */
111    next: function() {
112    
113 		/* pragma:DEBUG_START */
114 		try {
115 			Profiler.enter("Iterator.next()");
116 		/* pragma:DEBUG_END */
117 
118 			if (this.c.isDestroyed()) {
119 				throw new Error("Invalid iterator over destroyed container!");
120 			}
121 
122 			// Get the next and move the pointer
123 			var o = this.p.ptr;
124 			this.p = (this.r ? this.p.prev : this.p.next);
125 
126 			while (o != null && o.isDestroyed()) {
127 				// See about the next in line
128 				o = this.p ? this.p.ptr : null;
129 				this.p = (o && this.r ? this.p.prev : this.p.next);
130 			}
131 
132 			if (o != null) {
133 				return o;
134 			} else {
135 				throw new Error("Index out of range");
136 			}
137 
138 		/* pragma:DEBUG_START */
139 		} finally {
140 			Profiler.exit();
141 		}
142 		/* pragma:DEBUG_END */
143    },
144 
145    /**
146     * Returns <tt>true</tt> if the iterator has more elements.
147     * @return {Boolean}
148     */
149    hasNext: function() {
150 		var o = null;
151 		// If the container hasn't been destroyed
152 		if (this.c && !this.c.isDestroyed()) {
153 			var o = this.p;
154 			while (o != null && o.ptr != null && o.ptr.isDestroyed()) {
155 				// Skip destroyed objects
156 				o = (this.r ? o.prev : o.next);
157 			}
158 		}
159       return o != null;
160    }

163    /**
164     * Get the class name of this object
165     *
166     * @return {String} "Iterator"
167     */
168    getClassName: function() {
169       return "Iterator";
170    }
171 });

173 return Iterator;

179 /**
180  * @class A container is a logical collection of objects.  A container
181  *        is responsible for maintaining the list of objects within it.
182  *        When a container is destroyed, none of the objects within the container
183  *        are destroyed with it.  If the objects must be destroyed, call
184  *        {@link #cleanUp}.  A container is a doubly linked list of objects
185  *        to all for objects to be added and removed without disrupting the
186  *        structure of the list.
187  *
188  * @param containerName {String} The name of the container
189  * @extends BaseObject
190  * @constructor
191  * @description Create a container.
192  */
193 var Container = BaseObject.extend(/** @scope Container.prototype */{
194 
195    _head: null,
196 	_tail: null,
197 	sz: 0,
198 
199    /**
200     * @private
201     */
202    constructor: function(containerName) {
203       this.base(containerName || "Container");
204       this._head = null;
205 		this._tail = null;
206 		this.sz = 0;
207    },
208 
209    /**
210     * Release the object back into the object pool.
211     */
212    release: function() {
213       this.base();
214       this.clear();
215    },
216 
217    /**
218     * Clears the container, however, all of the objects within the container
219     * are not destroyed automatically.  This is to prevent the early clean up of
220     * objects which are being used by a transient container.
221     */
222    destroy: function() {
223       this.base();
224    },
225 
226    /**
227     * Returns the count of the number of objects within the
228     * container.
229     *
230     * @return {Number} The number of objects in the container
231     */
232    size: function() {
233       return this.sz;
234    },
235 
236 	/**
237 	 * Create a new node for the list
238 	 * @param obj {Object} The object
239 	 * @private
240 	 */
241 	_new: function(obj) {
242 		var o = {
243 			prev: null,
244 			next: null,
245 			ptr: obj
246 		};
247 		return o;
248 	},
249 	
250 	/**
251 	 * Find the list node at the given index.  No bounds checking
252 	 * is performed with this function.
253 	 * @param idx {Number} The index where the item exists
254 	 * @param offset {Object} The object to start at, "head" if null
255 	 * @private
256 	 */
257 	_find: function(idx, offset) {
258 		var n = offset || this._head, c = idx;
259 		while ( n != null && c-- > 0) {
260 			n = n.next;
261 		}
262 		return (c > 0 ? null : n);
263 	},
264 	
265 	/**
266 	 * Look through the list to find the given object.  If the object is
267 	 * found, the  list node is returned.  If no object is found, the method
268 	 * returns <code>null</code>.
269 	 * 
270 	 * @param obj {Object} The object to find
271 	 * @private
272 	 */
273 	_peek: function(obj) {
274 		var n = this._head;
275 		while (n != null) {
276 			if (n.ptr === obj) {
277 				return n;
278 			}
279 			n = n.next;
280 		}
281 		return null;
282 	},
283 
284    /**
285     * Add an object to the container.
286     *
287     * @param obj {Object} The object to add to the container.
288     */
289    add: function(obj) {
290 		var n = this._new(obj);
291 		if (this._head == null && this._tail == null) {
292 			this._head = n;
293 			this._tail = n;
294 		} else {
295 			this._tail.next = n;
296 			n.prev = this._tail;
297 			this._tail = n;
298 		}
299 
300       if (obj.getId) {
301          Console.log("Added ", obj.getId(), "[", obj, "] to ", this.getId(), "[", this, "]");
302       }
303 		this.sz++;
304 		
305 		// Return the list node that was added
306 		return n;
307    },
308 	
309 	/**
310 	 * Add all of the objects in the container or array to this container, at the end
311 	 * of this container.  If "arr" is a container, the head of "arr" is joined to the
312 	 * tail of this, resulting in a very fast operation.  Because this method, when
313 	 * performed on a container, is just joining the two lists, no duplication of
314 	 * elements from the container is performed.  As such, removing elements from the
315 	 * new container will affect this container as well.
316 	 * 
317 	 * @param arr {Container|Array} A container or array of objects
318 	 */
319 	addAll: function(arr) {
320 		if (Container.isInstance(arr)) {
321 			for (var i = arr.iterator(); i.hasNext(); ) {
322 				this.add(i.next());
323 			}
324 			i.destroy();
325 		} else {
326 			for (var i in arr) {
327 				this.add(arr[i]);
328 			}
329 		}
330 	},
331 
332 	/**
333 	 * Clone this container, returning a new container which points to all of the
334 	 * objects in this container.
335 	 * @return {Container} A new container with all of the objects from the current container
336 	 */
337 	clone: function() {
338 		var c = Container.create();
339 		c.addAll(this.getAll());
340 		return c;
341 	},
342 	
343 	/**
344 	 * Concatenate a container or array to the end of this container, 
345 	 * returning a new container with all of the elements.  The
346 	 * array or container will be <i>deep copied</i> and appended to this
347 	 * container.  While the actual pointers to the objects aren't deep copied,
348 	 * one can be assured that modifying the array or container structure being
349 	 * appended will not affect either container.
350 	 *
351 	 * @param arr {Container|Array} A container or array of objects
352 	 * @return {Container} A new container with all objects from both
353 	 */
354 	concat: function(arr) {
355 		if (Container.isInstance(arr)) {
356 			arr = arr.getAll();
357 		}
358 		var c = this.clone();
359 		c.addAll(arr);
360 		return c;
361 	},

363    /**
364     * Insert an object into the container at the given index. Asserts if the
365     * index is out of bounds for the container.  The index must be greater than
366     * or equal to zero, and less than or equal to the size of the container minus one.
367     * The effect is to extend the length of the container by one.
368     * 
369     * @param index {Number} The index to insert the object at.
370     * @param obj {Object} The object to insert into the container
371     */
372    insert: function(index, obj) {
373       Assert(!(index < 0 || index > this.size()), "Index out of range when inserting object!");
374 		var o = this._find(index);
375 		var n = this._new(obj);
376 		
377 		n.prev = o.prev;
378 		n.prev.next = n;
379 		n.next = o;
380 		o.prev = n;
381 		this.sz++;
382 		
383 		// Return the list node that was inserted
384 		return n;
385    },
386    
387    /**
388     * Replaces the given object with the new object.  If the old object is
389     * not found, no action is performed.
390     * 
391     * @param oldObj {Object} The object to replace
392     * @param newObj {Object} The object to put in place
393     * @return {Object} The object which was replaced
394     */
395    replace: function(oldObj, newObj) {
396 		var o = this._peek(oldObj), r = null;
397 		if (o.ptr != null) {
398 			r = o.ptr;
399 			o.ptr = newObj;		
400 		}
401       return r;      
402    },
403    
404    /**
405     * Replaces the object at the given index, returning the object that was there
406     * previously. Asserts if the index is out of bounds for the container.  The index 
407     * must be greater than or equal to zero, and less than or equal to the size of the 
408     * container minus one.
409     * 
410     * @param index {Number} The index at which to replace the object
411     * @param obj {Object} The object to put in place
412     * @return {Object} The object which was replaced
413     */
414    replaceAt: function(index, obj) {
415       Assert(!(index < 0 || index > this.size()), "Index out of range when inserting object!");
416 		var o = this._find(index);
417 		var r = o.ptr;
418 		o.ptr = obj;
419       return r;      
420    },
421    
422    /**
423     * Remove an object from the container.  The object is
424     * not destroyed when it is removed from the container.
425     *
426     * @param obj {Object} The object to remove from the container.
427     * @return {Object} The object that was removed
428     */
429    remove: function(obj) {
430 		var o = this._peek(obj);
431 		//AssertWarn(o != null, "Removing object from collection which is not in collection");
432 		
433 		if (o != null) {
434 			if (o === this._head && o === this._tail) {
435 				this.clear();
436 				this.sz = 0;
437 				return;	
438 			}
439 
440 			if (o === this._head) {
441 				this._head = o.next;
442 				if (this._head == null) {
443 					this.clear();
444 					this.sz = 0;
445 					return;
446 				}
447 			}
448 			
449 			if (o === this._tail) {
450 				this._tail = o.prev;
451 			}
452 
453 			if (o.next) o.next.prev = o.prev;
454 			if (o.prev) o.prev.next = o.next;
455 			o.prev = o.next = null;
456 			this.sz--;
457 
458 	      if (obj.getId) {
459 	         Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
460 	      }
461 
462 			return o.ptr;		
463 		}
464 		return null;
465    },
466 
467    /**
468     * Remove an object from the container at the specified index.
469     * The object is not destroyed when it is removed.
470     *
471     * @param idx {Number} An index between zero and the size of the container minus 1.
472     * @return {Object} The object removed from the container.
473     */
474    removeAtIndex: function(idx) {
475       Assert((idx >= 0 && idx < this.size()), "Index of out range in Container");
476 
477 		var o = this._find(idx);
478 		if (o === this._head) {
479 			this._head = o.next;
480 		}
481 		if (o === this.tail) {
482 			this._tail = o.prev;
483 		}
484 		if (o.next) o.next.prev = o.prev;
485 		if (o.prev) o.prev.next = o.next;
486 		o.prev = o.next = null;
487 		var r = o.ptr;
488 		
489       Console.log("Removed ", r.getId(), "[", r, "] from ", this.getId(), "[", this, "]");
490 		this.sz--;
491       return r;
492    },

494 	/**
495 	 * Reduce the container so that it's length is the specified length.  If <code>length</code>
496 	 * is larger than the size of this container, no operation is performed.  Setting <code>length</code>
497 	 * to zero is effectively the same as calling {@link #clear}.  Objects which would logically
498 	 * fall after <code>length</code> are not automatically destroyed.
499 	 * 
500 	 * @param length {Number} The maximum number of elements
501 	 * @return {Container} The subset of elements being removed
502 	 */
503 	reduce: function(length) {
504 		if (length > this.size()) {
505 			return Container.create();
506 		}
507 		var a = this.getAll();
508 		var sub = this.subset(length, a.length, a);
509 		if (length == 0) {
510 			return sub;
511 		}
512 
513 		a.length = length;
514 		this.clear();
515 		for (var i in a) {
516 			this.add(a[i]);
517 		}
518 		return sub;		
519 	},

521 	/**
522 	 * A new <code>Container</code> which is a subset of the current container
523 	 * from the starting index (inclusive) to the ending index (exclusive).  Modifications
524 	 * made to the objects in the subset will affect this container's objects.
525 	 *  
526 	 * @param start {Number} The starting index in the container
527 	 * @param end {Number} The engine index in the container
528 	 * @return {Container} A subset of the container.
529 	 */
530 	subset: function(start, end, b) {
531 		var a = b || this.getAll();
532 		var c = Container.create();
533 		for (var i = start; i < end; i++) {
534 			c.add(a[i]);
535 		}
536 		return c;			
537 	},

539    /**
540     * Get the object at the index specified. If the container has been
541     * sorted, objects might not be in the position you'd expect.
542     *
543     * @param idx {Number} The index of the object to get
544     * @return {Object} The object at the index within the container
545     * @throws {Error} Index out of bounds if the index is out of the list of objects
546     */
547    get: function(idx) {
548       if (idx < 0 || idx > this.size()) {
549          throw new Error("Index out of bounds");
550       }
551 		return this._find(idx).ptr;
552    },

555 	 * Get an array of all of the objects in this container.
556 	 * @return {Array} An array of all of the objects in the container
557 	 */
558 	getAll: function() {
559 		var a = [], i = this.iterator();
560 		while (i.hasNext()) {
561 			a.push(i.next());
562 		}
563 		i.destroy();
564 		return a;
565 	},

567 	/**
568 	 * For each object in the container, the function will be executed.
569 	 * The function takes one argument: the object being processed.
570 	 * Unless otherwise specified, <code>this</code> refers to the container.
571 	 * 
572 	 * @param fn {Function} The function to execute for each object
573 	 * @param [thisp] {Object} The object to use as <code>this</code> inside the function
574 	 */
575 	forEach: function(fn, thisp) {
576 		var itr = this.iterator();
577 		while (itr.hasNext()) {
578 			fn.call(thisp || this, itr.next());
579 		}
580 		itr.destroy();
581 	},

583 	/**
584 	 * Filters the container with the function, returning a new <code>Container</code>
585 	 * with the objects that pass the test in the function.  If the object should be
586 	 * included in the new <code>Container</code>, the function should return <code>true</code>.
587 	 * The function takes one argument: the object being processed.
588 	 * Unless otherwise specified, <code>this</code> refers to the container.
589 	 * 
590 	 * @param fn {Function} The function to execute for each object
591 	 * @param [thisp] {Object} The object to use as <code>this</code> inside the function
592 	 * @return {Container}
593 	 */
594 	filter: function(fn, thisp) {
595 		var arr = EngineSupport.filter(this.getAll(), fn, thisp || this);
596 		var c = Container.create();
597 		c.addAll(arr);
598 		return c;		
599 	},

602     * Remove all objects from the container.  None of the objects are

605    clear: function() {
606 		this._head = null;

611    /**
612     * Remove and destroy all objects from the container.
613     */
614    cleanUp: function() {
615 		var a = this.getAll(), h;
616 		while (a.length > 0) {
617 			h = a.shift();
618 			if (h.destroy) {
619 				h.destroy();
620 			}
621 		}
622       this.clear();
623    },

625    /**
626     * Get the array of all objects within this container.  If a filtering
627     * function is provided, only objects matching the filter will be
628     * returned from the object collection.
629     * <p/>
630     * The filter function needs to return <tt>true</tt> for each element
631     * that should be contained in the filtered set.  The function will be
632     * passed the following arguments:
633     * <ul>
634     * <li>element - The array element being operated upon</li>
635     * <li>index - The index of the element in the array</li>
636     * <li>array - The entire array of elements in the container</li>
637     * </ul>
638     * Say you wanted to filter a host objects components based on a
639     * particular type.  You might do something like the following:
640     * <pre>
641     * var logicComponents = host.getObjects(function(el, idx) {
642     *    if (el.getType() == BaseComponent.TYPE_LOGIC) {
643     *       return true;
644     *    }
645     * });
646     * </pre>
647     *
648     * @param filterFn {Function} A function to filter the set of
649     *                 elements with.  If you pass <tt>null</tt> the
650     *                 entire set of objects will be returned.
651     * @return {Array} The array of filtered objects
652     */
653    getObjects: function(filterFn) {
654 		var a = this.getAll();
655       if (filterFn) {
656          return EngineSupport.filter(a, filterFn);
657       } else {
658          return a;
659       }
660    },

662    /**
663     * Sort the objects within the container, using the provided function.
664     * The function will be provided object A and B.  If the result of the
665     * function is less than zero, A will be sorted before B.  If the result is zero,
666     * A and B retain their order.  If the result is greater than zero, A will
667     * be sorted after B.
668     *
669     * @param [fn] {Function} The function to sort with. If <tt>null</tt> the objects
670     *          will be sorted in "natural" order.
671     */
672    sort: function(fn) {
673       Console.log("Sorting ", this.getName(), "[" + this.getId() + "]");
674 		var a = this.getAll().sort(fn);
675 		
676 		// Relink
677 		this._head = this._new(a[0]);
678 		var p=this._head;
679 		for (var i = 1; i < a.length; i++) {
680 			var n = this._new(a[i]);
681 			p.next = n;
682 			n.prev = p;
683 			p = n;
684 		}
685 		this._tail = p;
686 		this.sz = a.length;
687    },

963 /**
964  * @class An implementation of a RedBlackTree data structure.  RedBlackTree has
965  *			 a worst-case time of O(log n) which is fast enough to quickly locate
966  *			 objects in the tree.  Duplicates are not allowed in a RedBlackTree.
967  *			 <p/>
968  *			 Items added to a RedBlackTree must implement a <code>compareTo(t)</code>
969  *			 method which returns a Number.  Zero if the value of the item is equal to
970  *			 "t", -1 if the value is less than t, 1 if the value is greater than "t".
971  *			 <p/>
972  *			 References:
973  *			   http://www.java-tips.org/java-se-tips/java.lang/red-black-tree-implementation-in-java.html<br/>
974  *				http://www.eternallyconfuzzled.com/tuts/datastructures/jsw_tut_rbtree.aspx
975  *
976  * @param name {String} The name of the container. Default: RBTree
977  * @extends BaseObject
978  * @constructor
979  * @description Create a red-black tree object.
980  * @private
981  */
982 var RedBlackTree = BaseObject.extend({
983 
984 	nullNode:null,
985 	current:null,
986 	parent:null,
987 	grand:null,
988 	great:null,
989 	header:null,
990 
991 	constructor: function(name) {
992 		this.base(name || "RBTree");
993 		this.header = new RedBlackNode(null);
994 		this.header.left = this.header.right = nullRBNode;
995 
996 		this.nullNode = new RedBlackNode(null);
997 		this.nullNode.left = this.nullNode.right = this.nullNode;
998 	},
999 	
1000 	/**
1001     * Insert into the tree.  <code>item</code> must implement the <code>compareTo(t)</code>
1002     * method, otherwise insertion will fail with an assertion.
1003     *
1004     * @param item {Object} the item to insert.
1005     */	
1006    insert: function(item) {
1007    	Assert(item.compareTo, "Items added to RedBlackTree must implement compareTo(t) method");
1008    	this.current = this.parent = this.grand = this.header;
1009    	this.nullNode.element = item;
1010    	
1011    	while(RedBlackTree.compare(item, this.current) != 0) {
1012    		this.great = this.grand; this.grand = this.parent; this.parent = this.current;
1013    		this.current = RedBlackTree.compare(item, this.current) < 0 ?
1014    			this.current.left : this.current.right;
1015    			
1016    		// Check if two red children; fix if so
1017    		if (this.current.left.color == RedBlackNode.RED &&
1018    			 this.current.right.color == RedBlackNode.RED) {
1019    			 	this.handleReorient(item);
1020    		}
1021    	}
1022    	
1023    	// Insertion fails if item already present
1024    	Assert(this.current == this.nullNode, "RedBlackTree duplication exception: " + item.toString());
1025    	this.current = new RedBlackNode(item, this.nullNode, this.nullNode);
1026    	
1027    	// Attach to parent
1028    	if (RedBlackTree.compare(item, this.parent) < 0) {
1029    		this.parent.left = this.current;
1030    	} else {
1031    		this.parent.right = this.current;
1032    	}
1033    	
1034    	this.handleReorient(item);
1035    },

1153    			 (RedBlackTree.compare(item, this.parent) < 0)) {

1165     * Called by handleReorient.

1167     * @param parent the parent of the root of the rotated subtree.
1168     * @return the root of the rotated subtree.

1174    			RedBlackTree.rotateWithLeftChild(prnt.left) :
1175    			RedBlackTree.rotateWithRightChild(prnt.left);
1176    	} else {
1177    		return prnt.right = RedBlackTree.compare(item, prnt.right) < 0 ?
1178    			RedBlackTree.rotateWithLeftChild(prnt.right) :
1179    			RedBlackTree.rotateWithRightChild(prnt.right);
1180    	}
1181    }
