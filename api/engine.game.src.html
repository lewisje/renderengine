 44    constructor: null,

 52    },

 62    /**
 63     * Get the display name of the game.
 64     * @memberOf Game
 65     * @static
 66     */
 67    getName: function() {
 68       return "Game";
 69    },
 70 
 71    /**
 72     * Load a script relative to this game.  Scripts cannot be specified
 73     * with an absolute URL.
 74     *
 75     * @param scriptSource {String} The relative path to the script to load.
 76     * @memberOf Game
 77     * @static
 78     */
 79    load: function(scriptSource) {
 80       Assert((scriptSource.indexOf("http") == -1), "Game scripts can only be loaded relative to the game's path");
 81       Engine.loadScript( (scriptSource.charAt(0) != "/" ? "./" : ".") + scriptSource );
 82    },
 83 
 84 	/**
 85 	 * Load a script, relative to the game engine.  Scripts cannot be loaded
 86 	 * with an absolute URL.
 87 	 * 
 88 	 * @param scriptPath {String} The relative path of the script to load.
 89 	 */
 90    loadEngineScript: function(scriptPath) {
 91       Engine.loadNow(scriptPath, Game.scriptLoaded);
 92    },
 93 
 94    scriptLoaded: function(scriptPath) {
 95    },
 96 
 97    /**
 98     * Abstracted to keep games separate from Engine
 99     */
100    setQueueCallback: function(cb) {
101       Engine.setQueueCallback(cb);
102    },
103 
104    /**
105     * Get the number of players this game supports.
106     *
107     * @type Number
108     * @memberOf Game
109     * @static
110     */
111    getPlayers: function() {
112       return 1;
113    },
114 
115    getGamePath: function() {
116       var loc = window.location;
117       var path = loc.protocol + "//" + loc.host + loc.pathname.substring(0,loc.pathname.lastIndexOf("/"));
118       path += (path.charAt(path.length - 1) == "/" ? "" : "/");
119       return path;
120    },
121 
122    getFilePath: function(fileName) {
123       var loc = window.location;
124       if (fileName.indexOf(loc.protocol) != -1 && fileName.indexOf(loc.host) == -1) {
125          throw new Error("File cannot be located on another server!");
126       }
127 
128       if (fileName.indexOf(loc.protocol) == -1) {
129          return this.getGamePath() + fileName;
130       } else {
131          return file;
132       }
133    }
