 37 /**
 38  * @class A static object for uniformly handling events within all browser
 39  *        platforms.  The event engine is an abstraction of the jQuery event
 40  *        system.  Methods are provided for adding and removing events in
 41  *        a programmatic way.  Additionally the engine has key codes for
 42  *        common keys which aren't part of the letters or numbers.
 43  *        <p/>
 44  *        While the engine provides a low-level way to attach events, when
 45  *        working with game objects methods are provided to manage events
 46  *        better than with the engine itself.
 47  *        <p/>
 48  *        Adding an event:
 49  * <pre>
 50  * var t = $(".myElement");
 51  * EventEngine.setHandler(t, "click", function(event) {
 52  *    MyObject.evtHandler();
 53  * });
 54  * </pre>
 55  * @see BaseObject#addEvent
 56  * @see BaseObject#removeEvent
 57  */
 58 var EventEngine = Base.extend(/** @scope EventEngine.prototype */{
 59 
 60    constructor: null,
 61 
 62    /**
 63     * Set an event handler on a target.  The handler function will
 64     * be called whenever the event occurs.
 65     *
 66     * @param target {String/jQuery} The target for the event.  This should either be a
 67     *                               CSS selector, or a jQuery object.
 68     * @param [data] {Array} Optional data to pass to the handler when it is invoked.
 69     * @param name {String} The event to handle.  ie: "click" or "mouseover"
 70     * @param handler {Function} The handler function to assign to the target
 71     */
 72    setHandler: function(target, data, name, handler) {
 73       if (typeof data == "string")
 74       {
 75          handler = name;
 76          name = data;
 77          data = null;
 78       }
 79 
 80       if (target == document.body) {
 81          target = document;
 82       }
 83 
 84       jQuery(target).bind(name, data || handler, handler);
 85    },
 86 
 87    /**
 88     * Clear an event handler that was previously assigned to the target.  If no
 89     * specific handler is assigned, all event handlers will be removed from the target.
 90     *
 91     * @param target {String/jQuery} The target for the event.  This should either be a
 92     *                               CSS selector, or a jQuery object.
 93     * @param name {String} The event to handle.  ie: "click" or "mouseover"
 94     * @param handler {Function} The handler function to unassign from the target
 95     */
 96    clearHandler: function(target, name, handler) {
 97       if (target == document.body) {
 98          target = document;
 99       }
100       jQuery(target).unbind(name, handler);
101    },
102    
103    /**
104     * Get the key code for the provided character.  The value returned
105     * will be for the lowercase key value, unless the second argument is
106     * set to <code>true</code> which will return the exact key code for the
107     * provided character.
108     * @param charStr {String} A single character to get the key code for
109     * @param [literal] {Boolean} <code>true</code> to return the literal code without
110     *        first converting the character to lower case.
111     * @return {Number} The key code for the given character
112     */
113    keyCodeForChar: function(charStr, literal) {
114       return (literal ? charStr : charStr.toUpperCase()).charCodeAt(0);
115    },
