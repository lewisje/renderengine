component.billboard2d.js


Summary

A render component which will render the contents of a generated image until the contents are updated.
Class Summary
Billboard2DComponent The billboard component renders the contents of an image which was generated by a linked render component.

 37 Engine.initObject("Billboard2DComponent", "RenderComponent", function() {
 38 
 39 /**
 40  * @class The billboard component renders the contents of an image which
 41  *        was generated by a linked render component.  When the contents
 42  *        of the linked component are re-rendered, the contents of the
 43  *        image are updated.  The best usage of this component is for infrequently
 44  *        changing vector drawn objects.  For example:
 45  *        <pre>
 46  *  constructor: function(size, position, pWidth, pHeight) {
 47  *     this.base("Spaceroid");
 48  *
 49  *     // Add components to move and draw the asteroid
 50  *     this.add(Mover2DComponent.create("move"));
 51  *     this.add(Billboard2DComponent.create("draw", Vector2DComponent.create("vector")));
 52  *     this.add(ColliderComponent.create("collider", Spaceroids.collisionModel));
 53  *        </pre>
 54  *        Accessing the <tt>Vectory2DComponent</tt> within the <tt>Billboard2DComponent</tt>
 55  *        is as simple as calling {@link #getComponent}.  If the contents of the linked
 56  *        component are updated, you will need to call {@link #regenerate} to recreate the
 57  *        billboard image.
 58  *
 59  *
 60  * @param name {String} The name of the component
 61  * @param priority {Number} The priority of the component between 0.0 and 1.0
 62  * @constructor
 63  * @extends RenderComponent
 64  * @description Creates a 2d billboard component.
 65  */
 66 var Billboard2DComponent = RenderComponent.extend(/** @scope Billboard2DComponent.prototype */{
 67 
 68    billboard: null,
 69    mode: null,
 70    renderComponent: null,
 71 	hostRect: null,
 72 
 73    /**
 74     * @private
 75     */
 76    constructor: function(name, renderComponent, priority) {
 77 		Assert(RenderComponent.isInstance(renderComponent) ||
 78 				 AbstractTextRenderer.isInstance(renderComponent), "Attempt to assign a non-render component to a billboard component");
 79       this.base(name, priority || 0.1);
 80       this.billboard = $("<img src='' width='1' height='1'/>");
 81       this.mode = Billboard2DComponent.REDRAW;
 82       this.renderComponent = renderComponent;
 83 		
 84       // All billboard components share the same temporary context
 85       if (Billboard2DComponent.tempContext == null) {
 86          // TODO: At some point it would be good to mimic the
 87          // context to which the component is rendering 
 88          Billboard2DComponent.tempContext = CanvasContext.create("billboardTemp", 800, 800);
 89          if (typeof FlashCanvas != "undefined") {
 90             FlashCanvas.initElement(Billboard2DComponent.tempContext.getSurface());
 91          }
 92       }
 93    },
 94 
 95 	/**
 96 	 * @private
 97 	 */
 98 	destroy: function() {
 99 		this.renderComponent.destroy();
100 		this.base();
101 	},
102 
103    /**
104     * Releases the component back into the object pool. See {@link PooledObject#release}
105     * for more information.
106     */
107    release: function() {
108       this.base();
109       this.billboard = null;
110       this.mode = null;
111       this.renderComponent = null;
112    },
113 
114    /**
115     * Establishes the link between this component and its host object.
116     * When you assign components to a host object, it will call this method
117     * so that each component can refer to its host object, the same way
118     * a host object can refer to a component with {@link HostObject#getComponent}.
119     *
120     * @param hostObject {HostObject} The object which hosts this component
121     */
122    setHostObject: function(hostObject) {
123       this.renderComponent.setHostObject(hostObject);
124       this.base(hostObject);
125    },
126 
127 	/**
128 	 * Call this method when the linked render component has been updated
129 	 * to force the billboard to be redrawn.
130 	 */
131    regenerate: function() {
132       this.mode = Billboard2DComponent.REDRAW;
133 		this.hostRect = null;
134    },
135 
136 	/**
137 	 * Get the linked render component.
138 	 * @return {RenderComponent}
139 	 */
140    getComponent: function() {
141       return this.renderComponent;
142    },
143 
144    /**
145     * Draws the contents of the billboard to the render context.  This
146     * component operates in one of two modes.  When the contents of the
147     * subclassed component are redrawing, a temporary render context is created
148     * to which the component renders.  The second mode is where the contents
149     * of the context from the first mode are rendered instead of performing
150     * all of the operations required to render the component.  This component
151     * is only good if the contents don't change often.
152     *
153     * @param renderContext {RenderContext} The rendering context
154     * @param time {Number} The engine time in milliseconds
155     */
156    execute: function(renderContext, time) {
157       if (!this.base(renderContext, time)) {
158          return;
159       }
160       
161 		if (Engine.options.hardwareAccel && Engine.options.billboards) {
162 		  	// Get the host objects bounding box
163 			var hostBox = this.getHostObject().getBoundingBox().get();
164 			
165 			if (this.mode == Billboard2DComponent.REDRAW) {
166 				// Provide a temporary context which is used to render the contents into
167 				Assert((Billboard2DComponent.tempContext != null), "Billboard2DComponent temporary context is not defined!");
168 				
169 				// Clear the temporary context and render the associated component to it
170 				Billboard2DComponent.tempContext.getElement().width = hostBox.w;
171 				Billboard2DComponent.tempContext.getElement().height = hostBox.h + 1;
172 				Billboard2DComponent.tempContext.reset();
173 				var p = Point2D.create(Math.abs(hostBox.x), Math.abs(hostBox.y) + 1);
174 				Billboard2DComponent.tempContext.setPosition(p);
175 				p.destroy();
176 				this.renderComponent.execute(Billboard2DComponent.tempContext, time);
177 				
178 				// Extract the rendered image
179 				this.billboard.attr("src", Billboard2DComponent.tempContext.getDataURL()).attr("width", hostBox.w).attr("height", hostBox.h + 1);
180 				this.mode = Billboard2DComponent.NORMAL;
181 			}
182 			
183 			// Render the billboard.  If the bounding box's origin is negative in
184 			// either X or Y, we'll need to move the transformation there before rendering the object
185 			this.transformOrigin(renderContext, true);
186 			if (hostBox.x < 0 || hostBox.y < 0) {
187 				renderContext.pushTransform();
188 				renderContext.setPosition(this.getHostObject().getBoundingBox().getTopLeft());
189 			}
190 			if (!this.hostRect) {
191 				this.hostRect = Rectangle2D.create(0, 0, hostBox.w, hostBox.h);
192 			}
193 			try {
194 				renderContext.drawImage(this.hostRect, this.billboard[0]);
195 			} 
196 			catch (ex) {
197 			// TODO: Find a better way to perform this operation since try/catch is SLOW
198 			// It appears that Firefox might not have a full image rendered, so calling
199 			// drawImage fails with a component exception.  To abate this possible issue,
200 			// we try the call and catch the failure...	
201 			}
202 			if (hostBox.x < 0 || hostBox.y < 0) {
203 				renderContext.popTransform();
204 			}
205 			this.transformOrigin(renderContext, false);
206 		} else {
207 			this.renderComponent.execute(renderContext, time);
208 		}
209 			
210       // Debug the collision node
211       if (Engine.getDebugMode())
212       {
213          renderContext.setLineStyle("blue");
214          renderContext.drawRectangle(this.getHostObject().getBoundingBox());
215       }
216    }
217 
218 }, /** @scope Billboard2DComponent.prototype */{ 
219 
220    /**
221     * Get the class name of this object
222     *
223     * @return {String} "RenderComponent"
224     */
225    getClassName: function() {
226       return "Billboard2DComponent";
227    },
228 
229    /**
230     * The component will render to a temporary context from which the
231     * actual content will be rendered.
232     * @type {Number}
233     */
234    REDRAW: 0,
235 
236    /**
237     * The component will render the contents of the billboard.
238     * @type {Number}
239     */
240    NORMAL: 1,
241    
242    /**
243     * A temporary context to which all billboards will render their
244     * bitmaps.
245     * @private
246     */
247    tempContext: null
248 
249 });
250 
251 return Billboard2DComponent;
252
