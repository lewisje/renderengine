engine.timers.js


Summary

A collection of timer objects.
Class Summary
Interval An extension of {@link Timer} that wraps the window.setInterval method.
MultiTimeout An extension of {@link Timeout} that will repeat the specified number of times before destroying itself.
OneShotTimeout An extension of {@link Timeout} which is a one-shot timer that cannot be restarted and will self-destroy after it completes its interval.
OneShotTrigger An extension of {@link OneShotTimeout} which is a one-shot timer that triggers a callback, at regular intervals, until the timer has expired.
Timeout An extension of {@link Timer} that wraps the window.setTimeout method.
Timer The base abstract class for all timer objects.

 36 Engine.initObject("Timer", "BaseObject", function() {
 37 
 38 /**
 39  * @class The base abstract class for all timer objects.
 40  *
 41  * @param name {String} The name of the timer
 42  * @param interval {Number} The interval for the timer, in milliseconds
 43  * @param callback {Function} The function to call when the interval is reached
 44  *
 45  * @extends BaseObject
 46  * @constructor
 47  * @description Create a timer object
 48  */
 49 var Timer = BaseObject.extend(/** @scope Timer.prototype */{
 50 
 51    timer: null,
 52 
 53    running: false,
 54    paused: false,
 55 
 56    /**
 57     * @private
 58     */
 59    constructor: function(name, interval, callback) {
 60       callback = name instanceof Number ? interval : callback;
 61       interval = name instanceof Number ? name : interval;
 62       name = name instanceof Number ? "Timer" : name;
 63 
 64       Assert((typeof callback == "function"), "Callback must be a function in Timer");
 65 
 66       this.base(name);
 67       this.interval = interval;
 68       this.callback = callback;
 69       
 70       // The engine needs to know about this timer
 71       Engine.addTimer(this.getId(), this);
 72       
 73       this.restart();
 74    },
 75 
 76    /**
 77     * @private
 78     */
 79    release: function() {
 80       this.base();
 81       this.timer = null;
 82       this.running = false;
 83       this.paused = false;
 84    },
 85 
 86    /**
 87     * Stop the timer and remove it from the system
 88     */
 89    destroy: function() {
 90       // The engine needs to remove this timer
 91       Engine.removeTimer(this.getId());
 92 
 93       this.timer = null;
 94       this.base();
 95    },
 96 
 97    /**
 98     * Get the underlying system timer object.
 99     */
100    getTimer: function() {
101       return this.timer;
102    },
103 
104    /**
105     * Set the underlying system timer object.
106     *
107     * @param timer {Object} The timer object
108     */
109    setTimer: function(timer) {
110       this.timer = timer;
111    },
112 
113    /**
114     * Returns <tt>true</tt> if the timer is currently running.
115     * @return {Boolean} <tt>true</tt> if the timer is running
116     */
117    isRunning: function() {
118       return this.running;
119    },
120 
121    /**
122     * Cancel the timer.
123     */
124    cancel: function() {
125       this.timer = null;
126       this.running = false;
127    },
128 
129    /**
130     * Pause the timer.  In the case where a timer was already processing,
131     * a restart would begin the timing process again with the full time
132     * allocated to the timer.  In the case of multi-timers (ones that retrigger
133     * a callback, or restart automatically a number of times) only the remaining
134     * iterations will be processed.
135     */
136    pause: function() {
137       this.cancel();
138       this.paused = true;
139    },
140 
141    /**
142     * Cancel the running timer and restart it.
143     */
144    restart: function() {
145       this.cancel();
146       this.running = true;
147       this.paused = false;
148    },
149 
150    /**
151     * Set the callback function for this timer.  If the timer is
152     * currently running, it will be restarted.
153     *
154     * @param callback {Function} A function object to call
155     */
156    setCallback: function(callback) {
157       Assert((typeof callback == "function"), "Callback must be a function in Timer.setCallback");
158       this.callback = callback;
159       if (this.isRunning)
160       {
161          this.restart();
162       }
163    },
164 
165    /**
166     * Get the callback function for this timer.  When the callback is called,
167     * the scope of the function will be the {@link Timer} itself.
168     * @return {Function} The callback function
169     */
170    getCallback: function() {
171       var timerFn = function() {
172          arguments.callee.cb.call(arguments.callee.timer);  
173       };
174       timerFn.cb = this.callback;
175       timerFn.timer = this;
176       return timerFn;
177    },
178 
179    /**
180     * Set the interval of this timer.  If the timer is running, it
181     * will be cancelled.
182     *
183     * @param interval {Number} The interval of this timer, in milliseconds
184     */
185    setInterval: function(interval) {
186       this.cancel();
187       this.interval = interval;
188    },
189 
190    /**
191     * Get the interval of this timer, in milliseconds.
192     * @return {Number} The interval
193     */
194    getInterval: function() {
195       return this.interval;
196    }
197 }, /** @scope Timer.prototype */{ 
198    /**
199     * Get the class name of this object
200     * @return {String} "Timer"
201     */
202    getClassName: function() {
203       return "Timer";
204    }
205 });
206 
207 return Timer;
208 
209 });

213 /**
214  * @class An extension of {@link Timer} that wraps the <tt>window.setTimeout</tt> method.
215  *
216  * @param name {String} The name of the timer

224 
225    /**
226     * Cancel this timeout timer.
227     */
228    cancel: function() {
229       window.clearTimeout(this.getTimer());
230       this.base();
231    },
232    
233    /**
234     * Cancel and destroy the timeout
235     */
236    destroy: function() {
237       this.cancel();
238       this.base();
239    },
240 
241    /**
242     * Restart this timeout timer
243     */
244    restart: function() {
245       this.setTimer(window.setTimeout(this.getCallback(), this.getInterval()));
246    }
247 }, /** @scope Timeout.prototype */{ 
248    /**
249     * Get the class name of this object

261 Engine.initObject("OneShotTimeout", "Timeout", function() {
262 
263 /**
264  * @class An extension of {@link Timeout} which is a one-shot timer that cannot 
265  *        be restarted and will self-destroy after it completes its interval.  Within
266  *        the callback, <tt>this</tt> refers to the <tt>Timer</tt> object itself.
267  *
268  * @param name {String} The name of the timer
269  * @param interval {Number} The interval for the timer, in milliseconds
270  * @param callback {Function} The function to call when the interval is reached
271  * @constructor
272  * @extends Timeout
273  * @description Create a one-shot timeout
274  */
275 var OneShotTimeout = Timeout.extend(/** @scope OneShotTimeout.prototype */{
276 
277    /**
278     * @private
279     */
280    constructor: function(name, interval, callback) {
281 
282       var cb = function() {
283          this.destroy();
284          arguments.callee.cbFn.call(this);
285       };
286       cb.cbFn = callback;
287 
288       this.base(name, interval, cb);
289    },
290 
291    /**
292     * This timer cannot be restarted.
293     * @private
294     */
295    restart: function() {
296       if (!this.paused && this.running)
297       {
298          return;
299       }
300 
301       this.base();
302    }
303 }, /** @scope OneShotTimeout.prototype */{ 
304 
305    /**
306     * Get the class name of this object
307     * @return {String} "OneShotTimeout"
308     */
309    getClassName: function() {
310       return "OneShotTimeout";
311    }
312 });
313 
314 return OneShotTimeout;
315 
316 });

318 Engine.initObject("OneShotTrigger", "OneShotTimeout", function() {
319 
320 /**
321  * @class An extension of {@link OneShotTimeout} which is a one-shot timer that triggers a callback, 
322  *        at regular intervals, until the timer has expired.  When the timer expires, the 
323  *        trigger will automatically destroy itself.  Within the callbacks, <tt>this</tt>
324  *        refers to the <tt>Timer</tt> object itself.
325  *
326  * @param name {String} The name of the timer
327  * @param interval {Number} The full interval for the timer, in milliseconds
328  * @param callback {Function} The function to call when the full interval is reached
329  * @param triggerInterval {Number} The interval between triggers, in milliseconds
330  * @param triggerCallback {Function} The function to call for each trigger interval
331  * @extends OneShotTimeout
332  * @constructor
333  * @description Create a one-shot triggering timeout
334  */
335 var OneShotTrigger = OneShotTimeout.extend(/** @scope OneShotTimeout.prototype */{
336 
337    constructor: function(name, interval, callback, triggerInterval, triggerCallback) {
338       var doneFn = function() {
339          var aC = arguments.callee;
340          aC.intv.destroy();
341          aC.cb.call(this);
342       };
343       // Create an Interval internally
344       doneFn.intv = Interval.create(name + "_trigger", triggerInterval, triggerCallback);
345       doneFn.cb = callback;
346 
347       this.base(name, interval, doneFn);
348    }
349 }, /** @scope OneShotTrigger.prototype */{ 
350 
351    /**
352     * Get the class name of this object
353     * @return {String} "OneShotTrigger"
354     */
355    getClassName: function() {
356       return "OneShotTrigger";
357    }
358 });
359 
360 return OneShotTrigger;

364 Engine.initObject("MultiTimeout", "Timeout", function() {
365 
366 /**
367  * @class An extension of {@link Timeout} that will repeat the specified number of times before
368  *        destroying itself.  The callback will be triggered with the
369  *        repetition number as the only argument.  Within the callback, <tt>this</tt>
370  *        refers to the <tt>Timer</tt> object itself.
371  *
372  * @param name {String} The name of the timer
373  * @param reps {Number} The number of repetitions to restart the timer automatically
374  * @param interval {Number} The interval for the timer, in milliseconds
375  * @param callback {Function} The function to call when the interval is reached
376  * @extends Timeout
377  * @constructor
378  * @description Creat a multi-timeout triggering time
379  */
380 var MultiTimeout = Timeout.extend(/** @scope MultiTimeout.prototype */{
381 
382    constructor: function(name, reps, interval, callback) {
383 
384       var cb = function() {
385          var aC = arguments.callee;
386          if (aC.reps-- > 0) {
387             aC.cbFn.call(this, aC.totalReps);
388             aC.totalReps++;
389             this.restart();
390          } else {
391             this.destroy();
392          }
393       };
394       cb.cbFn = callback;
395       cb.reps = reps;
396       cb.totalReps = 0;
397 
398       this.base(name, interval, cb);
399    }
400    
401 }, /** @scope MultiTimeout.prototype */{ 
402 
403    /**
404     * Get the class name of this object
405     * @return {String} "MultiTimeout"
406     */
407    getClassName: function() {
408       return "MultiTimeout";
409    }
410 });
411 
412 return MultiTimeout;
413
