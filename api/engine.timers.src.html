engine.timers.js


Summary

A collection of timer objects.
Class Summary
Interval A timer that wraps the window.setInterval method.
MultiTimeout A timer that will repeat the specified number of times before destroying itself.
OneShotTimeout A one-shot timer that cannot be restarted and will self-destroy after it completes its interval.
OneShotTrigger A one-shot timer that triggers a callback, at regular intervals, until the timer has expired.
Timeout A timer that wraps the window.setTimeout method.
Timer The base abstract class for all timers.

 36 Engine.initObject("Timer", "BaseObject", function() {
 37 
 38 /**
 39  * @class The base abstract class for all timers.
 40  *
 41  * @param name {String} The name of the timer
 42  * @param interval {Number} The interval for the timer, in milliseconds
 43  * @param callback {Function} The function to call when the interval is reached
 44  *
 45  * @extends BaseObject
 46  * @constructor
 47  * @description Create a timer object
 48  */
 49 var Timer = BaseObject.extend(/** @scope Timer.prototype */{
 50 
 51    timer: null,
 52 
 53    running: false,
 54 
 55    /**
 56     * @private
 57     */
 58    constructor: function(name, interval, callback) {
 59       callback = name instanceof Number ? interval : callback;
 60       interval = name instanceof Number ? name : interval;
 61       name = name instanceof Number ? "Timer" : name;
 62 
 63       Assert((typeof callback == "function"), "Callback must be a function in Timer");
 64 
 65       this.base(name);
 66       this.interval = interval;
 67       this.callback = callback;
 68       this.restart();
 69    },
 70 
 71    /**
 72     * @private
 73     */
 74    release: function() {
 75       this.base();
 76       this.timer = null;
 77       this.running = false;
 78    },
 79 
 80    /**
 81     * Stop the timer and remove it from the system
 82     */
 83    destroy: function() {
 84       this.timer = null;
 85       this.base();
 86    },
 87 
 88    /**
 89     * Get the underlying system timer object.
 90     */
 91    getTimer: function() {
 92       return this.timer;
 93    },
 94 
 95    /**
 96     * Set the underlying system timer object.
 97     *
 98     * @param timer {Object} The timer object
 99     */
100    setTimer: function(timer) {
101       this.timer = timer;
102    },
103 
104    /**
105     * Returns <tt>true</tt> if the timer is currently running.
106     * @return {Boolean} <tt>true</tt> if the timer is running
107     */
108    isRunning: function() {
109       return this.running;
110    },
111 
112    /**
113     * Cancel the timer.
114     */
115    cancel: function() {
116       this.timer = null;
117       this.running = false;
118    },
119 
120    /**
121     * Cancel the running timer and restart it.
122     */
123    restart: function() {
124       this.cancel();
125       this.running = true;
126    },
127 
128    /**
129     * Set the callback function for this timer.  If the timer is
130     * currently running, it will be restarted.
131     *
132     * @param callback {Function} A function object to call
133     */
134    setCallback: function(callback) {
135       Assert((typeof callback == "function"), "Callback must be a function in Timer.setCallback");
136       this.callback = callback;
137       if (this.isRunning)
138       {
139          this.restart();
140       }
141    },
142 
143    /**
144     * Get the callback function for this timer.
145     * @return {Function} The callback function
146     */
147    getCallback: function() {
148       return this.callback;
149    },
150 
151    /**
152     * Set the interval of this timer.  If the timer is running, it
153     * will be cancelled.
154     *
155     * @param interval {Number} The interval of this timer, in milliseconds
156     */
157    setInterval: function(interval) {
158       this.cancel();
159       this.interval = interval;
160    },
161 
162    /**
163     * Get the interval of this timer, in milliseconds.
164     * @return {Number} The interval
165     */
166    getInterval: function() {
167       return this.interval;
168    }
169 }, /** @scope Timer.prototype */{ 
170    /**
171     * Get the class name of this object
172     * @return {String} "Timer"
173     */
174    getClassName: function() {
175       return "Timer";
176    }
177 });
178 
179 return Timer;
180 
181 });

233 Engine.initObject("OneShotTimeout", "Timeout", function() {
234 
235 /**
236  * @class A one-shot timer that cannot be restarted and will
237  *        self-destroy after it completes its interval..
238  *
239  * @param name {String} The name of the timer
240  * @param interval {Number} The interval for the timer, in milliseconds
241  * @param callback {Function} The function to call when the interval is reached
242  * @constructor
243  * @extends Timeout
244  * @description Create a one-shot timeout
245  */
246 var OneShotTimeout = Timeout.extend(/** @scope OneShotTimeout.prototype */{
247 
248    /**
249     * @private
250     */
251    constructor: function(name, interval, callback) {
252 
253       var cb = function() {
254          var aC = arguments.callee;
255          aC.timer.destroy();
256          aC.cbFn();
257       };
258       cb.cbFn = callback;
259       cb.timer = this;
260 
261       this.base(name, interval, cb);
262    },
263 
264    /**
265     * This timer cannot be restarted.
266     * @private
267     */
268    restart: function() {
269       if (this.running)
270       {
271          return;
272       }
273 
274       this.base();
275    }
276 }, /** @scope OneShotTimeout.prototype */{ 
277 
278    /**
279     * Get the class name of this object
280     * @return {String} "OneShotTimeout"
281     */
282    getClassName: function() {
283       return "OneShotTimeout";
284    }
285 });
286 
287 return OneShotTimeout;

291 Engine.initObject("OneShotTrigger", "OneShotTimeout", function() {
292 
293 /**
294  * @class A one-shot timer that triggers a callback, at regular intervals,
295  *        until the timer has expired.  When the timer expires, the trigger
296  *        will automatically destroy itself.
297  *
298  * @param name {String} The name of the timer
299  * @param interval {Number} The interval for the timer, in milliseconds
300  * @param callback {Function} The function to call when the interval is reached
301  * @param triggerInterval {Number} The interval between triggers, in milliseconds
302  * @param triggerCallback {Function} The function to call for each trigger interval
303  * @extends OneShotTimeout
304  * @constructor
305  * @description Create a one-shot triggering timeout
306  */
307 var OneShotTrigger = OneShotTimeout.extend(/** @scope OneShotTimeout.prototype */{
308 
309    constructor: function(name, interval, callback, triggerInterval, triggerCallback) {
310       var doneFn = function() {
311          var aC = arguments.callee;
312          aC.intv.destroy();
313          aC.cb();
314       };
315       // Create an Interval internally
316       doneFn.intv = Interval.create(name + "_trigger", triggerInterval, triggerCallback);
317       doneFn.cb = callback;
318 
319       this.base(name, interval, doneFn);
320    }
321 }, /** @scope OneShotTrigger.prototype */{ 
322 
323    /**
324     * Get the class name of this object
325     * @return {String} "OneShotTrigger"
326     */
327    getClassName: function() {
328       return "OneShotTrigger";
329    }
330 });
331 
332 return OneShotTrigger;

336 Engine.initObject("MultiTimeout", "Timeout", function() {
337 
338 /**
339  * @class A timer that will repeat the specified number of times before
340  *        destroying itself.  The callback will be triggered with the
341  *        repetition number as the only argument.
342  *
343  * @param name {String} The name of the timer
344  * @param reps {Number} The number of repetitions to restart the timer automatically
345  * @param interval {Number} The interval for the timer, in milliseconds
346  * @param callback {Function} The function to call when the interval is reached
347  * @extends Timeout
348  * @constructor
349  * @description Creat a multi-timeout triggering time
350  */
351 var MultiTimeout = Timeout.extend(/** @scope MultiTimeout.prototype */{
352 
353    constructor: function(name, reps, interval, callback) {
354 
355       var cb = function() {
356          var aC = arguments.callee;
357          if (aC.reps-- > 0) {
358             aC.cbFn(aC.reps);
359             aC.timer.restart();
360          } else {
361             aC.timer.destroy();
362          }
363       };
364       cb.cbFn = callback;
365       cb.timer = this;
366       cb.reps = reps;
367 
368       this.base(name, interval, cb);
369    }
370    
371 }, /** @scope MultiTimeout.prototype */{ 
372 
373    /**
374     * Get the class name of this object
375     * @return {String} "MultiTimeout"
376     */
377    getClassName: function() {
378       return "MultiTimeout";
379    }
380 });
381 
382 return MultiTimeout;
