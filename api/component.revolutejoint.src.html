 41 /**
 42  * @class A revolute joint which allows two bodies to revolve around a common
 43  * 		 anchor point in a {@link Simulation}.  
 44  *
 45  * @param name {String} Name of the component
 46  * @param body1 {BaseBodyComponent} The first body for the joint
 47  * @param body2 {BaseBodyComponent} The second body for the joint
 48  * @param anchor {Point2D} A point, in world coordinates relative to the two 
 49  * 	bodies, to use as the joint's anchor point
 50  *
 51  * @extends BaseJointComponent
 52  * @constructor
 53  * @description Creates a revolute joint between two physical bodies.
 54  */

 56 
 57 	anchor: null,

 59    /**
 60     * @private
 61     */
 62 	constructor: function(name, body1, body2, anchor) {
 63 		var jointDef = new b2RevoluteJointDef();
 64 		
 65 		this.anchor = Point2D.create(anchor);
 66 		this.base(name || "RevoluteJoint", body1, body2, jointDef);	
 67 	},
 68 
 69 	/**
 70 	 * When simulation starts offset the anchor point by the position of rigid body 1 (the "from" body).
 71 	 * @private
 72 	 */
 73 	startSimulation: function() {
 74 		if (!this.getSimulation()) {
 75 			
 76 			var jA = Point2D.create(this.anchor);
 77 			jA.add(this.getBody1().getPosition());
 78 			jA.add(this.getBody1().getLocalOrigin());
 79 			var a = jA.get();
 80 	
 81 			this.getJointDef().anchorPoint.Set(a.x, a.y);
 82 			jA.destroy();
 83 		}		
 84 		
 85 		this.base();
 86 	},
 87 	
 88 	/**
 89 	 * Get the upper limiting angle, in degrees, through which the joint can rotate.
 90 	 * @return {Number}
 91 	 */
 92 	getUpperLimitAngle: function() {
 93 		return Math2D.radToDeg(this.getJointDef().upperAngle);
 94 	},
 95 	
 96 	/**
 97 	 * Set the upper limiting angle through which the joint can rotate.
 98 	 * 
 99 	 * @param angle {Number} An angle in degrees
100 	 */
101 	setUpperLimitAngle: function(angle) {
102 		this.getJointDef().upperAngle = Math2D.degToRad(angle);
103 		this.getJointDef().enableLimit = true;	
104 	},
105 	
106 	/**
107 	 * Get the lower limiting angle, in degrees, through which the joint can rotate.
108 	 * @return {Number}
109 	 */
110 	getLowerLimitAngle: function() {
111 		return Math2D.radToDeg(this.getJointDef().lowerAngle);	
112 	},
113 	
114 	/**
115 	 * Set the upper limiting angle through which the joint can rotate.
116 	 * 
117 	 * @param angle {Number} An angle in degrees
118 	 */
119 	setLowerLimitAngle: function(angle) {
120 		this.getJointDef().lowerAngle = Math2D.degToRad(angle);	
121 		this.getJointDef().enableLimit = true;	
122 	},
123 	
124 	/**
125 	 * During simulation, this returns the current angle of the joint
126 	 * in degrees.  Outside of simulation it will always return zero.
127 	 * @return {Number}
128 	 */
129 	getJointAngle: function() {
130 		if (this.simulation) {
131 			return Math2D.radToDeg(this.getJoint().GetJointAngle());
132 		} else {
133 			return 0;
134 		}
135 	},
136 	
137 	/**
138 	 * Get the torque which will be applied when the joint is used as a motor.
139 	 * @return {Number} 
140 	 */
141 	getMotorTorque: function() {
142 		if (this.simulation) {
143 			return this.getJoint().GetMotorTorque(1 / Engine.getFPS());
144 		} else {
145 			this.getJointDef().motorTorque;	
146 		}
147 	},
148 	
149 	/**
150 	 * Set the torque which is applied via the motor, or to resist forces applied to it.  You can
151 	 * use this value to simulate joint friction by setting the motor speed to zero
152 	 * and applying a small amount of torque.  During simulation, the torque is applied directly
153 	 * to the joint.
154 	 * 
155 	 * @param torque {Number} The amount of torque to apply
156 	 */
157 	setMotorTorque: function(torque) {
158 		if (this.simulation) {
159 			// Apply directly to the joint
160 			this.getJoint().SetMotorTorque(torque);
161 		} else {
162 			// Apply to the joint definition
163 			this.getJointDef().motorTorque = torque;
164 			this.getJointDef().enableMotor = true;	
165 		}
