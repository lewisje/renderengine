component.basebody.js


Summary

The base component type for all physical bodies which can be used in a {@link Simulation}.
Class Summary
BaseBodyComponent The base rigid body component which initializes rigid bodies for use in a {@link Simulation}.

 40 Engine.initObject("BaseBodyComponent", "Transform2DComponent", function() {

 42 /**
 43  * @class The base rigid body component which initializes rigid bodies
 44  * 		 for use in a {@link Simulation}.  
 45  *
 46  * @param name {String} Name of the component
 47  * @param shapeDef {b2ShapeDef} The shape definition. Either {@link b2CircleDef}, {@link b2BoxDef}, or
 48  * 			{@link b2PolyDef}.
 49  *
 50  * @extends Transform2DComponent
 51  * @constructor
 52  * @description All physical body components should extend from this component type
 53  * 				 to inherit such values as density and friction, and gain access to position and rotation.
 54  */
 55 var BaseBodyComponent = Transform2DComponent.extend(/** @scope BaseBodyComponent.prototype */{
 56 
 57 	bodyDef: null,
 58 	shapeDef: null,
 59 	simulation: null,
 60 	body: null,
 61 	rotVec: null,
 62 	bodyPos: null,
 63 	renderComponent: null,
 64 	origin: null,
 65 
 66    /**
 67     * @private
 68     */
 69 	constructor: function(name, shapeDef) {
 70 		this.base(name || "BaseBody");	
 71 
 72 		this.shapeDef = shapeDef;
 73 		this.shapeDef.restitution = BaseBodyComponent.DEFAULT_RESTITUTION;
 74 		this.shapeDef.density = BaseBodyComponent.DEFAULT_DENSITY;
 75 		this.shapeDef.friction = BaseBodyComponent.DEFAULT_FRICTION;
 76 		this.simulation = null;
 77 		this.rotVec = Vector2D.create(0,0);
 78 		this.bodyPos = Point2D.create(0,0);
 79 		this.origin = Point2D.create(0,0);
 80 	},
 81 
 82 	destroy: function() {
 83 		if (this.renderComponent != null) {
 84 			this.renderComponent.destroy();
 85 		}
 86 		
 87 		this.rotVec.destroy();
 88 		this.bodyPos.destroy();
 89 		this.origin.destroy();
 90 		
 91 		this.base();
 92 	},
 93 	
 94 	release: function() {
 95 		this.base();
 96 
 97 		this.rotVec = null;
 98 		this.bodyPos = null;
 99 		this.origin = null;
100 	},
101 
102 	/**
103 	 * Start simulating the body.  If the body isn't a part of the simulation,
104 	 * it is added and simulation occurs.  Position and rotation will be updated.
105 	 */
106 	startSimulation: function() {
107 		if (!this.simulation) {
108 			this.simulation = this.getHostObject().getSimulation();
109 			this.body = this.simulation.addBody(this.getBodyDef());
110 		}
111 	},
112 	
113 	/**
114 	 * Stop simulating the body.  If the body is a part of a simulation,
115 	 * it is removed and simulation stops.  The position and rotation of
116 	 * the body will not be updated.
117 	 */
118 	stopSimulation: function() {
119 		if (this.simulation) {
120 			this.simulation.removeBody(this.getBody());			
121 			this.simulation = null;
122 		}
123 	},
124 
125 	/**
126 	 * Set the associated render component for this body.  This is typically used by the
127 	 * {@link PhysicsActor} to link a body to a renderer so that each body can have an
128 	 * associated renderer applied.
129 	 * 
130 	 * @param renderComponent {RenderComponent} The render component to associate with this body
131 	 */
132 	setRenderComponent: function(renderComponent) {
133 		this.renderComponent = renderComponent;
134 		if (renderComponent != null) {
135 			this.renderComponent.setHostObject(this.getHostObject());
136 		}
137 	},
138 	
139 	/**
140 	 * Get the associated render component for this body.
141 	 * @return {RenderComponent} or <code>null</code>
142 	 */
143 	getRenderComponent: function() {
144 		return this.renderComponent;
145 	},
146 	
147 	/**
148 	 * Set the origin of the rigid body.  By default, the origin is the top left corner of
149 	 * the bounding box for the body.  Most times the origin should be set to the center
150 	 * of the body.
151 	 * 
152 	 * @param x {Number|Point2D} The X coordinate or a <tt>Point2D</tt>
153 	 * @param y {Number} The Y coordinate or <tt>null</tt> if X is a <tt>Point2D</tt>
154 	 */
155 	setLocalOrigin: function(x, y) {
156 		this.origin.set(x, y);
157 	},
158 	
159 	/**
160 	 * Get the local origin of the body.
161 	 * @return {Point2D}
162 	 */
163 	getLocalOrigin: function() {
164 		return this.origin;
165 	},
166 	
167 	/**
168 	 * Get the center of the body.
169 	 * @return {Point2D}
170 	 */
171 	getCenter: function() {
172 		return Point2D.create(this.getPosition()).add(this.getLocalOrigin());
173 	},
174 	
175 	/**
176 	 * [ABSTRACT] Get a box which bounds the body.
177 	 * @return {Rectangle2D}
178 	 */
179 	getBoundingBox: function() {
180 		return Rectangle2D.create(0,0,1,1);
181 	},
182 	
183 	/**
184 	 * Get the Box2d shape definition object.
185 	 * @return {b2ShapeDef}
186 	 */
187 	getShapeDef: function() {
188 		return this.shapeDef;
189 	},
190 	
191 	/**
192 	 * Get the Box2d body definition object.
193 	 * @return {b2BodyDef}
194 	 */
195 	getBodyDef: function() {
196 		if (this.bodyDef == null) {
197 			this.bodyDef = new b2BodyDef();
198 			this.bodyDef.AddShape(this.shapeDef);
199 		}
200 		return this.bodyDef;
201 	},
202 	
203 	/**
204 	 * Get the Box2d body object which was added to the simulation.
205 	 * @return {b2Body}
206 	 */
207 	getBody: function() {
208 		return this.body;
209 	},
210 	
211 	/**
212 	 * Set the resitution (bounciness) of the body.  The value should be between
213 	 * zero and one.  Values higher than one are accepted, but produce objects which
214 	 * are unrealistically bouncy.
215 	 * 
216 	 * @param restitution {Number} A value between 0.0 and 1.0
217 	 */
218 	setRestitution: function(restitution) {
219 		this.shapeDef.resitution = restitution
220 	},
221 	
222 	/**
223 	 * Get the resitution (bounciness) value for the body.
224 	 * @return {Number}
225 	 */
226 	getRestitution: function() {
227 		return this.shapeDef.resitution;
228 	},
229 	
230 	/**
231 	 * Set the density of the body.
232 	 * 
233 	 * @param density {Number} The density of the body
234 	 */
235 	setDensity: function(density) {
236 		this.shapeDef.density = density;
237 	},
238 	
239 	/**
240 	 * Get the density of the body.
241 	 * @return {Number}
242 	 */
243 	getDensity: function() {
244 		return this.shapeDef.density;
245 	},
246 	
247 	/**
248 	 * Set the friction of the body.  Lower values slide easily across other bodies.
249 	 * Higher values will cause a body to stop moving as it slides across other bodies.
250 	 * However, even a body which has high friction will keep sliding across a body
251 	 * with no friction.
252 	 * 
253 	 * @param friction {Number} The friction of the body
254 	 */
255 	setFriction: function(friction) {
256 		this.shapeDef.friction = friction;
257 	},
258 	
259 	/**
260 	 * Get the friction of the body.
261 	 * @return {Number}
262 	 */
263 	getFriction: function() {
264 		return this.shapeDef.friction;
265 	},
266 	
267 	/**
268 	 * Set the initial position of the body.  Once a body is in motion, updating
269 	 * its position should be avoided since it doesn't fit with physical simulation.
270 	 * To change an object's position, try applying forces or impulses to the body.
271 	 * 
272 	 * @param point {Point2D} The initial position of the body
273 	 */
274 	setPosition: function(point) {
275 		var pos = point.get();
276 		this.getBodyDef().position.Set(pos.x, pos.y);
277 	},
278 	
279 	/**
280 	 * Get the position of the body during simulation.  This value is updated
281 	 * as the simulation is stepped.
282 	 * @return {Point2D}
283 	 */
284 	getPosition: function() {
285 		if (this.simulation) {
286 			this.bodyPos.set(this.body.m_position.x, this.body.m_position.y);
287 		} else {
288 			this.bodyPos.set(this.getBodyDef().position.x, this.getBodyDef().position.y);
289 		}
290 		return this.bodyPos;	
291 	},
292 	
293 	/**
294 	 * Get the rotation of the body during simulation.  This value is updated
295 	 * as the simulation is stepped.
296 	 * @return {Number}
297 	 */
298 	getRotation: function() {
299 		var rC = this.getBody().GetRotationMatrix().col1;
300 		this.rotVec.set(rC.x, rC.y);
301 		var cV = rC.x > 0 ? BaseBodyComponent.UP_VECTOR : BaseBodyComponent.DOWN_VECTOR;
302 		var md = rC.x > 0 ? 0 : 180;
303 		var ab = this.rotVec.angleBetween(cV) - md;
304 		return ab - 90;
305 	},
306 	
307 	/**
308 	 * Apply a force to the body.  Forces are comprised of a force vector and
309 	 * a position.  The force vector is the direction in which the force is
310 	 * moving, while the position is where on the body the force is acting.
311 	 * Forces act upon a body from world coordinates.
312 	 * 
313 	 * @param forceVector {Vector2D} The force vector
314 	 * @param position {Point2D} The position where the force is acting upon the body
315 	 */
316 	applyForce: function(forceVector, position) {
317 		this.getBody().WakeUp();
318 		var f = forceVector.get();
319 		var d = position.get();
320 		var fv = new b2Vec2(f.x, f.y);
321 		var dv = new b2Vec2(d.x, d.y);
322 		this.getBody().ApplyForce(fv, dv);	
323 	},
324 	
325 	/**
326 	 * Apply an impulse to the body.  Impulses are comprised of a force vector and
327 	 * a position.  The impulse vector is the direction of the impulse, while the position
328 	 * is where on the body the impulse will be applied.
329 	 * Impulses act upon a body locally, adjusting its velocity.
330 	 * 
331 	 * @param impulseVector {Vector2D} The impulse vectory
332 	 * @param position {Point2D} the position where the impulse is originating from in the body
333 	 */
334 	applyImpulse: function(impulseVector, position) {
335 		this.getBody().WakeUp();
336 		var i = impulseVector.get();
337 		var d = position.get();
338 		var iv = new b2Vec2(i.x, i.y);
339 		var dv = new b2Vec2(d.x, d.y);
340 		this.getBody().ApplyImpulse(iv, dv);
341 	},
342 	
343 	/**
344 	 * Apply torque to the body.
345 	 * 
346 	 * @param torque {Number} The amount of torque to apply to the body
347 	 */
348 	applyTorque: function(torque) {
349 		this.getBody().WakeUp();
350 		this.getBody().ApplyTorque(torque);
351 	},
352 	
353 	/**
354 	 * Get the computed mass of the body.
355 	 * @return {Number} The mass of the body
356 	 */
357 	getMass: function() {
358 		return this.getBody().getMass();
359 	},
360 	
361 	/**
362 	 * Returns <code>true</code> if the body is static.  A body is static if it
363 	 * isn't updated part of the simulation during contacts.
364 	 * @return {Boolean}
365 	 */
366 	isStatic: function() {
367 		return this.getBody().IsStatic();	
368 	},
369 	
370 	/**
371 	 * Returns <code>true</code> if the body is sleeping.  A body is sleeping if it
372 	 * has settled to the point where no movement is being calculated.  If you want
373 	 * to perform an action upon a body, other than applying force, torque, or impulses,
374 	 * you must call {@link #wakeUp}.
375 	 * @return {Boolean}
376 	 */
377 	isSleeping: function() {
378 		return this.getBody().IsSleeping();	
379 	},
380 	
381 	/**
382 	 * Returns <code>true</code> if the body is frozen.  A frozen body can still be
383 	 * collided with, but will not, itself, be updated.  Once a body is frozen, it cannot
384 	 * be moved again.  See {@link #freeze}.
385 	 * @return {Boolean}
386 	 */
387 	isFrozen: function() {
388 		return this.getBody().IsFrozen();	
389 	},
390 	
391 	/**
392 	 * Wake up a body, adding it back into the collection of bodies being simulated.
393 	 * Acting upon a body without first waking it up will do nothing.
394 	 */
395 	wakeUp: function() {
396 		this.getBody().WakeUp();
397 	},
398 	
399 	/**
400 	 * Freeze a body, permanently taking it out of simulation, except for collisions.
401 	 */
402 	freeze: function() {
403 		this.getBody().Freeze();	
404 	}
405 	
406 	
407 }, { /** @scope BaseBodyComponent.prototype */
408 
409    /**
410     * Get the class name of this object
411     *
412     * @return {String} "BaseBodyComponent"
413     */
414    getClassName: function() {
415       return "BaseBodyComponent";
416    },
417 	
418 	/**
419 	 * The default restitution (bounciness) of a body
420 	 */
421 	DEFAULT_RESTITUTION: 0.48,
422 	
423 	/**
424 	 * The default density of a body
425 	 */
426 	DEFAULT_DENSITY: 1.0,
427 	
428 	/**
429 	 * The default friction of a body
430 	 */
431 	DEFAULT_FRICTION: 0,
432 	
433 	/**
434 	 * A simple up (0, -1) vector to calculate rotation
435 	 * @private
436 	 */
437 	UP_VECTOR: new Vector2D(0, -1),
438 	DOWN_VECTOR: new Vector2D(0, 1)
439    
440 });
441 
442 return BaseBodyComponent;
443
