 39 Engine.initObject("SpriteActor", "Object2D", function() {
 40 
 41 /**
 42  * @class A <tt>SpriteActor</tt> is an actor object within a game, whose renderer
 43  * 		 is a 2D sprite.  It can have actions assigned to it, which are triggered
 44  * 		 by either a controlling player or generated by the program.
 45  * @param name {String} The name of the object
 46  * @extends Object2D
 47  * @constructor
 48  * @description Create a sprite actor
 49  */
 50 var SpriteActor = Object2D.extend({
 51 
 52    editing: false,
 53 
 54    sprite: null,
 55 
 56    velocityVec: null,
 57 
 58    constructor: function(name) {
 59       this.base(name || "Actor");
 60 
 61       this.editing = false;
 62 
 63       // Add components to move and draw the player
 64       this.add(SpriteComponent.create("draw"));
 65       this.add(Transform2DComponent.create("move"));
 66    },
 67 
 68    getProperties: function() {
 69       var self = this;
 70       var prop = this.base(self);
 71       return $.extend(prop, {
 72          "Sprite" :     [function() { return self.sprite.getName(); },
 73                          function(i) { self.setSprite(SpriteTest.spriteLoader.getSprite("smbtiles", i)); }, true]
 74       });
 75    },
 76 
 77    /**
 78     * Update the player within the rendering context.  This draws
 79     * the shape to the context, after updating the transform of the
 80     * object.  If the player is thrusting, draw the thrust flame
 81     * under the ship.
 82     *
 83     * @param renderContext {RenderContext} The rendering context
 84     * @param time {Number} The engine time in milliseconds
 85     */
 86    update: function(renderContext, time) {
 87       renderContext.pushTransform();
 88       this.base(renderContext, time);
 89 
 90       if (this.editing) {
 91          renderContext.setLineStyle("white");
 92          renderContext.setLineWidth(2);
 93          renderContext.drawRectangle(this.getSprite().getBoundingBox());
 94       }
 95 
 96       renderContext.popTransform();
 97    },
 98 
 99    setSprite: function(sprite) {
100       this.sprite = sprite;
101       this.setBoundingBox(sprite.getBoundingBox());
102       this.getComponent("draw").setSprite(sprite);
103    },
104 
105    getSprite: function() {
106       return this.sprite;
107    },
108 
109    /**
110     * Get the position of the ship from the mover component.
111     * @type Point2D
112     */
113    getPosition: function() {
114       return this.getComponent("move").getPosition();
115    },
116 
117    getRenderPosition: function() {
118       return this.getComponent("move").getRenderPosition();
119    },
120 
121    /**
122     * Set, or initialize, the position of the mover component
123     *
124     * @param point {Point2D} The position to draw the ship in the playfield
125     */
126    setPosition: function(point) {
127       this.base(point);
128       this.getComponent("move").setPosition(point);
129    },
130 
131    getScale: function() {
132       return this.getComponent("move").getScale();
133    },
134 
135    setScale: function(s) {
136       this.getComponent("move").setScale(s);
137    },
138 
139    getRotation: function() {
140       return this.getComponent("move").getRotation();
141    },
142 
143    setRotation: function(r) {
144       this.getComponent("move").setRotation(r);
145    },
146 
147    /**
148     * Set up the player object on the playfield.  The width and
149     * heigh of the playfield are used to determine the center point
150     * where the player starts.
151     *
152     * @param pWidth {Number} The width of the playfield in pixels
153     * @param pHeight {Number} The height of the playfield in pixels
154     */
155    setup: function(pWidth, pHeight) {
156    },
157 
158    setEditing: function(state) {
159       this.editing = state;
160    },
161 
162    isEditable: function() {
163       return true;
164    }
165 
166 }, { // Static
167 
168    /**
169     * Get the class name of this object
170     * @return The string <tt>SpriteActor</tt>
171     * @type String
172     */
173    getClassName: function() {
174       return "SpriteActor";
175    }
176 });
177 
178 return SpriteActor;
179 
180 });
