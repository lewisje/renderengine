 92 /**
 93  * @class A quadtree which divides up a space into smaller and smaller
 94  *        divisions to quickly locate objects within that space.
 95  *
 96  * @param width {Number} The width of the area
 97  * @param height {Number} The height of the area
 98  * @param divisions {Number} The number of divisions determines the smallest node size
 99  * @extends SpatialContainer
100  */
101 var Quadtree = SpatialContainer.extend(/** @scope QuadTree.prototype */{
102 
103    /*
104     * Build a quadtree which can be used to quickly locate objects
105     * within a defined space.
106     */
107    constructor: function(width, height, divisions) {
108 
109       this.base("QuadTree", width, height);
110       var smWidth = Math.floor(width / divisions);
111 
112       function subdivNode(node, rect) {
113          var hW = Math.floor(width / 2);
114          var hH = Math.floor(height / 2);
115 
116          if (hW < smWidth)
117          {
118             return;
119          }
120 
121          // This node
122          node = new QuadtreeNode(rect);
123          var curTopLeft = rect.getTopLeft();
124 
125          // Top-left
126          var n;
127          var r = new Rectangle2D(curTopLeft.x, curTopLeft.y, hW, hH);
128          node.getQuads()[0] = subdivNode({}, r);
129 
130          // Top-right
131          r = new Rectangle2D(curTopLeft.x + hW, curTopLeft.y, hW, hH);
132          node.getQuads()[1] = subdivNode({}, r);
133 
134          // Bottom-left
135          r = new Rectangle2D(curTopLeft.x, curTopLeft.y + hH, hW, hH);
136          node.getQuads()[2] = subdivNode({}, r);
137 
138          // Bottom-right
139          r = new Rectangle2D(curTopLeft.x + hW, curTopLeft.y + hH, hW, hH);
140          node.getQuads()[3] = subdivNode({}, r);
141 
142          return node;
143       }
144 
145       // Build the spatial quadtree
146       subdivNode(this.getRoot(), new Rectangle2D(0, 0, width, height));
147    },
148 
149    /**
150     * Find the quadtree node which contains the point specified.
151     *
152     * @param point {Point2D} The point to locate
153     * @param node {QuadtreeNode} The node to start the search at
154     * @type QuadtreeNode
155     */
156    findNodePoint: function(point, node) {
157       var fNode = node;
158       if (node.rect.containsPoint(point) && node.getQuads()[0] != null)
159       {
160          // Check to see if any of the quads contain the point
161          var p = 0;
162          while (fNode == node && p < 4)
163          {
164             fNode = this.findNodePoint(point, node.getQuads()[p++]);
165          }
166       }
167       return fNode;
168    },
169 
170    /**
171     * Find the smallest quadtree node which contains the rectangle specified.
172     *
173     * @param rect {Rectangle2D} The rectangle to locate
174     * @param node {QuadtreeNode} The node to start the search at
175     * @type QuadtreeNode
176     */
177    findNodeRect: function(rect, node) {
178       var fNode = node;
179       if (node.rect.containsRect(rect) && node.getQuads()[0] != null)
180       {
181          // Check to see if any of the quads contain the rect

183          while (fNode == node && p < 4)
184          {
185             fNode = this.findNodeRect(rect, node.getQuads()[p++]);
186          }
187       }
188       return fNode;
189    },
