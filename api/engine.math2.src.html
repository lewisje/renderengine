 34 /**
 35  * @class A static class which provides methods for generating random integers
 36  * 		 and floats between 0 and 1.  The class also provides a way to seed the
 37  * 		 random number generator for repeatable results.
 38  * 
 39  * @static
 40  */
 41 var Math2 = Base.extend(/** @scope Math2.prototype */{
 42 	constructor: null,
 43 	
 44 	state: 1,
 45 	m: 0x100000000, // 2**32;
 46 	a: 1103515245,
 47 	c: 12345,
 48 	
 49 	/**
 50 	 * Seed the random number generator with a known number.  This
 51 	 * ensures that random numbers occur in a known sequence.
 52 	 * 
 53 	 * @param seed {Number} An integer to seed the number generator with
 54 	 */
 55 	seed: function(seed) {
 56 		// LCG using GCC's constants
 57 		this.state = seed ? seed : Math.floor(Math.random() * (this.m-1));
 58 	},
 59 	
 60 	/**
 61 	 * Returns a random integer between 0 and 4,294,967,296.
 62 	 * @return {Number} An integer between 0 and 2^32
 63 	 */
 64 	randomInt: function() {
 65 		this.state = (this.a * this.state + this.c) % this.m;
 66 		return this.state;
 67 	},
 68 	
 69 	/**
 70 	 * Returns a pseudo-random number between 0 (inclusive) and 1 (exclusive)
 71 	 * @return {Number} A number between 0 and 1
 72 	 */
 73 	random: function() {
 74 		// returns in range [0,1]
 75 		return this.randomInt() / (this.m - 1);
 76 	},
 77 	
 78 	/**
 79 	 * Return a random value within the <tt>low</tt> to <tt>hight</tt> range,
 80 	 * optionally as an integer value only.
 81 	 *
 82 	 * @param low {Number} The low part of the range
 83 	 * @param high {Number} The high part of the range
 84 	 * @param [whole] {Boolean} Return whole values only
 85 	 * @return {Number}
 86 	 */
 87 	randomRange: function(low, high, whole) {
 88 		var v = low + (this.random() * high);
 89 		return (whole ? Math.floor(v) : v);
 90 	},
 91 	
 92 	/**
 93 	 * Parse a binary string into a number.
 94 	 * 
 95 	 * @param bin {String} Binary string to parse
 96 	 * @return {Number}
 97 	 */
 98 	parseBin: function(bin) {
 99 		if (!isNaN(bin)) {
100 			return parseInt(bin, 2);
101 		}
102 	},
103 	
104 	/**
105 	 * Converts a number to a hexidecimal string, prefixed by "0x".
106 	 *
107 	 * @param num {Number} The number to convert
108 	 * @return {String}
109 	 */
110 	toHex: function(num) {
111 		if (!isNaN(num)) {
112 			return ("0x" + num.toString(16));
113 		}
114 	},
115 	
116 	/**
117 	 * Converts a number to a binary string.
118 	 *
119 	 * @param num {Number} The number to convert
120 	 * @return {String}
121 	 */
122 	toBinary: function(num) {
123 		if (!isNaN(num)) {
124 			return num.toString(2);
125 		}
126 	}
127 });
