 37 Engine.initObject("ImageLoader", "RemoteLoader", function() {
 38 
 39 /**
 40  * @class Loads images and stores the reference to those images.  Images
 41  *        are stored on the client-side in a simple cache for faster re-use.
 42  *        When loading images, you assign a name to the image.  This would allow
 43  *        you to re-use the image without having to load it again for another
 44  *        purpose.
 45  *        <p/>
 46  *        Loading images is fairly simple.  You only need to create an instance
 47  *        of an image loader (multiple images can be loaded by the same resource
 48  *        loader) and then use it to load the images:
 49  <pre>
 50           this.imageLoader = ImageLoader.create();
 51           
 52           // Load an image
 53           this.imageLoader.load("keys", this.getFilePath("resources/fingerboard.png"), 220, 171);
 54  </pre>
 55  *        In the example above, <tt>this</tt> refers to a {@link Game} object which
 56  *        implements the {@link Game#getFilePath getFilePath()} method which is
 57  *        used to get a path relative to where the game is located on the server.
 58  *
 59  * @constructor
 60  * @param name {String=ImageLoader} The name of the resource loader
 61  * @extends RemoteLoader
 62  */
 63 var ImageLoader = RemoteLoader.extend(/** @scope ImageLoader.prototype */{
 64 
 65    /**
 66     * @private
 67     */
 68    constructor: function(name) {
 69       this.base(name || "ImageLoader");
 70    },
 71 
 72    /**
 73     * Load an image resource from a URL.  Images are cached within the page
 74     * in an invisible object for fast retrieval.
 75     *
 76     * @param name {String} The name of the resource
 77     * @param url {String} The URL where the resource is located
 78     * @param width {Number} The width of this resource, in pixels
 79     * @param height {Number} The height of this resource, in pixels
 80     */
 81    load: function(name, url, width, height) {
 82 
 83       // Create the area if it doesn't exist which will
 84       // be used to load the images from their URL
 85       if (this.getElement() == null)
 86       {
 87          var div = jQuery("<div/>")
 88             .css({ background: "black", display: "none" });
 89 
 90          this.setElement(div[0]);
 91 
 92          // Add it to the default context so it can be cleaned up
 93          Engine.getDefaultContext().add(this);
 94       }
 95 
 96       // Create an image element
 97       var imageInfo = null;
 98       if (url != null) {
 99          imageInfo = this.loadImageResource(name, url, width, height);
100       }
101 
102       this.base(name, imageInfo);
103    },
104 
105    /**
106     * Lazy loads an image resource when the information for it becomes available.  It
107     * is best to specify the width and height of the resource, but it isn't necessary
108     * to load the image.
109     *
110     * @param name {String} The name of the resource
111     * @param url {String} The URL where the resource is located
112     * @param width {Number} The width of this resource, in pixels, or <tt>null</tt>
113     * @param height {Number} The height of this resource, in pixels, or <tt>null</tt>
114     * @return {HTMLImage} The image loaded
115     */
116    loadImageResource: function(name, url, width, height) {
117       var image = null;
118       if (width && height) {
119          image = $("<img/>").attr("src", url).attr("width", width).attr("height", height);
120       } else {
121          image = $("<img/>").attr("src", url);
122       }
123       
124       var thisObj = this;
125       if (!$.browser.Wii) {
126          image.bind("load", function() {
127             thisObj.setReady(name, true);
128          });
129       } else {
130          // Calculate an approximate wait time based on dimensions
131          OneShotTimeout.create("readyImg", (width * height) * ImageLoader.loadAdjust, function() {
132             thisObj.setReady(name, true); 
133          });
134       }
135 
136       // Append it to the container so it can load the image
137       $(this.getElement()).append(image);
138       var info = {
139          width: width,
140          height: height,
141          image: image
142       };
143       return info;
144    },
145 
146    /**
147     * Get the image from the resource stored with the specified name, or <tt>null</tt>
148     * if no such image exists.
149     *
150     * @param name {String} The name of the image resource
151     * @return {HTMLImage} The image
152     */
153    get: function(name) {
154       var imgInfo = this.base(name);
155       return imgInfo ? imgInfo.image[0] : null;
156    },
157 	
158 	/**
159 	 * Get an {@link Image} object from the resource which represents the image, or <tt>null</tt>
160 	 * if no such image exists.
161 	 * @param name {String} The name of the image resource
162 	 * @return {Image}
163 	 */
164 	getImage: function(name) {
165 		return Image.create("Image", name, this); 	
166 	},
167    
168    /**
169     * Get the dimensions of an image from the resource stored with 
170     * the specified name, or <tt>null</tt> if no such image exists.
171     *
172     * @param name {String} The name of the image resource
173     * @return {Point2D} A point which represents the width and height of the image
174     */
175    getDimensions: function(name) {
176       var imgInfo = this.getCachedObjects()[name] ? this.getCachedObjects()[name].data : null;
177       return imgInfo ? Point2D.create(imgInfo.width, imgInfo.height) : null;
178    },
179 
180    /**
181     * The name of the resource this loader will get.
182     * @return {String} The string "image"
183     */
184    getResourceType: function() {
185       return "image";
186    }
187 }, /** @scope ImageLoader.prototype */{
188    /**
189     * Get the class name of this object
190     * @return {String} The string "ImageLoader"
191     */
192    getClassName: function() {
193       return "ImageLoader";
194    },
195    
196    /**
197     * The ratio by which to scale image load times when loading on the Wii
198     */
199    loadAdjust: 0.05
200 });
201 
202 return ImageLoader;
203 
204 });

206 Engine.initObject("Image", "PooledObject", function() {
207 
208 var Image = PooledObject.extend(/** @scope Image.prototype */{
209 
210 	image: null,
211 
212    /**
213     * @private
214     */
215    constructor: function(name, imageName, imageLoader) {
216       this.base(name || "Image");
217 		this.image = imageLoader.get(imageName);
218 		var dims = imageLoader.getDimensions(imageName);
219       this.bbox = Rectangle2D.create(0, 0, dims.x, dims.y);
220    },
221 	
222    /**
223     * @private
224     */
225    release: function() {
226       this.base();
227       this.image = null;
228       this.bbox = null;
229    },
230 
231    /**
232     * Get the bounding box for the image.
233     * @return {Rectangle2D} The bounding box which contains the entire image
234     */
235    getBoundingBox: function() {
236       return this.bbox;
237    },
238 
239 	/**
240 	 * Get the HTML image object which contains the image.
241 	 * @return {HTMLImage}
242 	 */
243 	getImage: function() {
244 		return this.image;
245 	}		
246 	
247 },{
248 
249    /**
250     * Gets the class name of this object.
251     * @return {String} The string "Image"
252     */
253    getClassName: function() {
254       return "Image";
255    }	
256 	
257 });
258 
259 return Image;
260
