 34 // Includes
 35 Engine.include("/components/component.base.js");
 36 
 37 Engine.initObject("ColliderComponent", "BaseComponent", function() {
 38 
 39 /**
 40  * @class A collider component is responsible for handling potential collisions by 
 41  *        updating the associated collision model and checking for possible collisions.
 42  *
 43  * @param name {String} Name of the component
 44  * @param collisionModel {SpatialCollection} The collision model
 45  * @param priority {Number} Between 0.0 and 1.0, with 1.0 being highest
 46  *
 47  * @extends BaseComponent
 48  * @constructor
 49  * @description Creates a collider component which tests the collision model for
 50  *              potential collisions. Each frame, the component will update a potential
 51  *              collision list (PCL) for the host object, using its current position
 52  *              obtained from {@link Object2D#getPosition}. Each object which meets
 53  *              certain criteria will be passed to an <tt>onCollide()<tt> method which
 54  *              must be implemented by the host object.
 55  *              <p/>
 56  *              The event handler will be passed the time the collision was detected
 57  *              as its first argument, and the potential collision object as the second.
 58  *              The host must determine if the collision is valid for itself, and then
 59  *              return a value which indicates whether the component should contine to
 60  *              check for collisions, or if it should stop.
 61  *              <p/>
 62  *              If the <tt>onCollide()</tt> method is not implemented on the host, no 
 63  *              collision events will be passed.
 64  */
 65 var ColliderComponent = BaseComponent.extend(/** @scope ColliderComponent.prototype */{
 66 
 67    collisionModel: null,
 68    
 69    objectType: null,
 70 
 71    /**
 72     * @private
 73     */
 74    constructor: function(name, collisionModel, priority) {
 75       this.base(name, BaseComponent.TYPE_COLLIDER, priority || 1.0);
 76       this.collisionModel = collisionModel;
 77       this.objectType = null;
 78    },
 79 
 80    /**
 81     * Releases the component back into the pool for reuse.  See {@link PooledObject#release}
 82     * for more information.
 83     */
 84    release: function() {
 85       this.base();
 86       this.collisionModel = null;
 87       this.objectType = null;
 88    },
 89 
 90    /**
 91     * Get the collision model being used by this component.
 92     * @return {SpatialContainer} The collision model
 93     */
 94    getCollisionModel: function() {
 95       return this.collisionModel;
 96    },
 97    
 98    /**
 99     * Get the object type that this collider component will respond to.  If
100     * the value is <tt>null</tt>, all objects are potential collision objects.
101     * @return {BaseObject} The only object type to collide with, or <tt>null</tt> for any object
102     */
103    getObjectType: function() {
104       return this.objectType;
105    },
106    
107    /**
108     * Set the object type that this component will respond to.  Setting this to <tt>null</tt>
109     * will trigger a potential collision when <i>any object</i> comes into possible contact 
110     * with the component's host based on the collision model.  If the object isn't of this type,
111     * no collision tests will be performed.  This allows the developer to fine tune which
112     * object the collision component is responsible for.  As such, multiple collision components
113     * could be used to handle different types of collisions.
114     *
115     * @param objType {BaseObject} The object type to check for
116     */
117    setObjectType: function(objType) {
118       this.objectType = objType;
119    },
120 
121    /**
122     * Update the collision model that this component was initialized with.
123     * Creates an object called <tt>ModelData</tt> on the host object which contains
124     * information relative to the collision model.  This data contains:
125     * <ul>
126     * <li>lastNode - The last node the object was reported within.
127     * </ul>
128     * This data is used by the collider components to determine where to test for
129     * collisions, and whether or not the PCL should be rebuilt.
130     */
131    updateModel: function() {
132 
133       // Get the model data for the object
134       var obj = this.getHostObject();
135       if (!obj.ModelData)
136       {
137          obj.ModelData = { lastNode: null };
138       }
139 
140       if ( obj.ModelData.lastNode && obj.ModelData.lastNode.getRect().containsPoint(obj.getPosition()) )
141       {
142          // The object is within the same node
143          return;
144       }
145 
146       // Find the node that contains the object
147       var aNode = this.getCollisionModel().findNodePoint(obj.getPosition());
148       if (aNode != null)
149       {
150          if (obj.ModelData.lastNode && (obj.ModelData.lastNode.getIndex() != aNode.getIndex()))
151          {
152             obj.ModelData.lastNode.removeObject(obj);
153             aNode.addObject(obj);
154          }
155          obj.ModelData.lastNode = aNode;
156       }
157    },
158 
159    /**
160     * Updates the object within the collision model and determines if
161     * the host object should to be alerted whenever a potential collision
162     * has occurred.  If a potential collision occurs, an array (referred to
163     * as a Potentail Collision List, or PCL) will be created which
164     * contains objects that might be colliding with the host object.  It
165     * is up to the host object to make the final determination that a
166     * collision has occurred.
167     * <p/>
168     * The list of objects within the PCL will be passed to the <tt>onCollide()</tt>
169     * method (if declared) on the host object.  If a collision occurred and was
170     * handled, the <tt>onCollide()</tt> method should return {@link CollisionComponent#STOP},
171     * otherwise, it should return {@link CollisionComponent#CONTINUE} to continue
172     * checking objects from the PCL against the host object.
173     *
174     * @param renderContext {RenderContext} The render context for the component
175     * @param time {Number} The current engine time in milliseconds
176     */
177    execute: function(renderContext, time) {
178 
179       var host = this.getHostObject();
180 
181       // Update the collision model
182       this.updateModel();
183 
184       // If the host object needs to know about collisions...
185       if (host.onCollide)
186       {
187          // Get the PCL and check for collisions
188          var pcl = this.getCollisionModel().getPCL(host.getPosition());
189          var status = ColliderComponent.CONTINUE;
190          EngineSupport.forEach(pcl, function(obj) {
191             if (this.getHostObject() != obj && 
192                 (this.getObjectType() == null || obj instanceof this.getObjectType()) &&
193                 status == ColliderComponent.CONTINUE)
194             {
195                status = this.testCollision(time, obj);
196             }
197          }, this);
198       }
199    },
200    
201    /**
202     * Call the host object's <tt>onCollide()</tt> method, passing the time of the collision
203     * and the potential collision object.  The return value should either tell the collision
204     * tests to continue, or to stop.
205     * <p/>
206     * For the base collider component the collision test is up to the host object to determine.
207     *
208     * @param time {Number} The engine time (in milliseconds) when the potential collision occurred
209     * @param collisionObj {HostObject} The host object with which the collision potentially occurs
210     * @return {Number} A status indicating whether to continue checking, or to stop
211     */
212    testCollision: function(time, collisionObj) {
213       return this.getHostObject().onCollide(collisionObj, time);
214    }
215 
216 }, /** @scope ColliderComponent.prototype */{ // Statics
217 
218    /**
219     * Get the class name of this object
220     *
221     * @return {String} "ColliderComponent"
222     */
223    getClassName: function() {
224       return "ColliderComponent";
225    },
226 
227    /**
228     * When <tt>onCollide()</tt> is called on the host object, it should
229     * return this value if it hasn't handled the collision, or if it wishes to be 
230     * notified about other potential collisions.
231     * @type {Number}
232     */
233    CONTINUE: 0,
234 
235    /**
236     * When <tt>onCollide()</tt> is called on the host object, it should
237     * return this if no more collisions should be reported.
238     * @type {Number}
239     */
240    STOP: 1
241 
242 });
243 
244 return ColliderComponent;
245 
246 });
