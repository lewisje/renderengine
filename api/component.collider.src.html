Class Summary
ColliderComponent Creates a collider component which tests the collision model for potential collisions.

 35 Engine.include("/components/component.base.js");

 37 Engine.initObject("ColliderComponent", "BaseComponent", function() {
 38 
 39 /**
 40  * @class Creates a collider component which tests the collision model for
 41  *              potential collisions. Each frame, the component will update a potential
 42  *              collision list (PCL) for the host object, using its current position
 43  *              obtained from {@link Object2D#getPosition}. Each object which meets
 44  *              certain criteria will be passed to an <tt>onCollide()</tt> method which
 45  *              must be implemented by the host object.
 46  *              <p/>
 47  *              The event handler will be passed the potential collision object
 48  *              as its first argument, the time the collision was detected as the second,
 49  *              and the target's collision mask as the third.
 50  *              The host must determine if the collision is valid for itself, and then
 51  *              return a value which indicates whether the component should contine to
 52  *              check for collisions, or if it should stop.
 53  *              <p/>
 54  *              If the <tt>onCollide()</tt> method is not implemented on the host, no 
 55  *              collision events will be passed.
 56  *              <p/>
 57  *              Additionally, the host can implement <tt>onCollideEnd()</tt> to be notified
 58  *              when collisions have stopped.  The time the collisions stopped will be the
 59  *              only argument.
 60  *
 61  * @param name {String} Name of the component
 62  * @param collisionModel {SpatialCollection} The collision model
 63  * @param priority {Number} Between 0.0 and 1.0, with 1.0 being highest
 64  *
 65  * @extends BaseComponent
 66  * @constructor
 67  * @description A collider component is responsible for handling potential collisions by 
 68  *        updating the associated collision model and checking for possible collisions.
 69  */
 70 var ColliderComponent = BaseComponent.extend(/** @scope ColliderComponent.prototype */{
 71 
 72    collisionModel: null,
 73 	collideSelf: false,
 74 
 75    /**
 76     * @private
 77     */
 78    constructor: function(name, collisionModel, priority) {
 79       this.base(name, BaseComponent.TYPE_COLLIDER, priority || 1.0);
 80       this.collisionModel = collisionModel;
 81 		this.collideSelf = false;
 82    },
 83 
 84 	/**
 85     * Establishes the link between this component and its host object.
 86     * When you assign components to a host object, it will call this method
 87     * so that each component can refer to its host object, the same way
 88     * a host object can refer to a component with {@link HostObject#getComponent}.
 89     *
 90     * @param hostObject {HostObject} The object which hosts this component
 91 	 */
 92 	setHostObject: function(hostObj) {
 93 		this.base(hostObj);
 94 		this.setCollisionMask(0x7FFFFFFF);
 95 	},
 96 
 97    /**
 98     * Releases the component back into the pool for reuse.  See {@link PooledObject#release}
 99     * for more information.
100     */
101    release: function() {
102       this.base();
103       this.collisionModel = null;
104 		this.collideSelf = false;
105    },
106 
107    /**
108     * Get the collision model being used by this component.
109     * @return {SpatialContainer} The collision model
110     */
111    getCollisionModel: function() {
112       return this.collisionModel;
113    },
114 	
115 	/**
116 	 * Set whether or not an object can collide with itself.  By default,
117 	 * this is <tt>false</tt>.
118 	 * @param state {Boolean} <tt>true</tt> if an object can collide with itself
119 	 */
120 	setCollideSelf: function(state) {
121 		this.collideSelf = state;
122 	},
123 	
124 	/**
125 	 * Returns <tt>true</tt> if an object can collide with itself.  Default: <tt>false</tt>
126 	 * @return {Boolean}
127 	 */
128 	getCollideSelf: function() {
129 		return this.collideSelf;
130 	},
131 	
132 	/**
133 	 * Returns a set of flags which can result in a collision between two objects.
134 	 * The flags are bits within a 31-bit Integer that correspond to possible collisions.
135 	 * @return {Number}
136 	 */
137 	getCollisionMask: function() {
138 		return this.collisionModel.getObjectSpatialData(this.getHostObject(), "collisionMask");	
139 	},
140    
141    /**
142     * Get the object type that this collider component will respond to.  If
143     * the value is <tt>null</tt>, all objects are potential collision objects.
144     * @return {BaseObject} The only object type to collide with, or <tt>null</tt> for any object
145     * @deprecated see {@link #getCollisionFlags}
146     */
147    getObjectType: function() {
148       return null;
149    },
150    
151 	/**
152 	 * Collision masks allow objects to be considered colliding or not depending on ANDing
153 	 * the results.  The flags occupy the lowest 31 bits, so there can be a number of 
154 	 * combinations which result in a collision.
155 	 * 
156 	 * @param collisionMask {Number} A 31-bit integer
157 	 */
158 	setCollisionMask: function(collisionMask) {
159 		this.collisionModel.setObjectSpatialData(this.getHostObject(), "collisionMask", collisionMask);
160 	},
161 	
162    /**
163     * Set the object type that this component will respond to.  Setting this to <tt>null</tt>
164     * will trigger a potential collision when <i>any object</i> comes into possible contact 
165     * with the component's host based on the collision model.  If the object isn't of this type,
166     * no collision tests will be performed.  This allows the developer to fine tune which
167     * object the collision component is responsible for.  As such, multiple collision components
168     * could be used to handle different types of collisions.
169     *
170     * @param objType {BaseObject} The object type to check for
171     * @deprecated see {@link #setCollisionFlags}
172     */
173    setObjectType: function(objType) {
174    },
175 
176    /**
177     * Update the collision model that this component was initialized with.
178     * As objects move about the world, the objects will move to different
179     * areas (or nodes) within the collision model.  It is necessary to
180     * update this model frequently so collisions can be determined.
181     */
182    updateModel: function() {
183       var obj = this.getHostObject();
184 		this.getCollisionModel().addObject(obj, obj.getPosition());
185    },
186 
187 	/**
188 	 * Get the collision node the host object is within, or <tt>null</tt> if it
189 	 * is not within a node.
190 	 * @return {SpatialNode}
191 	 */
192    getSpatialNode: function() {
193 		return this.collisionModel.getObjectSpatialData(this.getHostObject(), "lastNode");
194    },
195 
196    /**
197     * Updates the object within the collision model and determines if
198     * the host object should to be alerted whenever a potential collision
199     * has occurred.  If a potential collision occurs, an array (referred to
200     * as a Potential Collision List, or PCL) will be created which
201     * contains objects that might be colliding with the host object.  It
202     * is up to the host object to make the final determination that a
203     * collision has occurred.  If no collisions have occurred, that will be reported
204     * as well.
205     * <p/>
206     * The list of objects within the PCL will be passed to the <tt>onCollide()</tt>
207     * method (if declared) on the host object.  If a collision occurred and was
208     * handled, the <tt>onCollide()</tt> method should return {@link CollisionComponent#STOP},
209     * otherwise, it should return {@link CollisionComponent#CONTINUE} to continue
210     * checking objects from the PCL against the host object.
211     *
212     * @param renderContext {RenderContext} The render context for the component
213     * @param time {Number} The current engine time in milliseconds
214     */
215    execute: function(renderContext, time) {
216 
217       var host = this.getHostObject();
218 
219       // Update the collision model
220       this.updateModel();
221 
222       // If the host object needs to know about collisions...
223 		var pcl = null;
224       if (host.onCollide)
225       {
226          // Get the PCL and check for collisions
227          pcl = this.getCollisionModel().getPCL(host.getPosition());
228          var status = ColliderComponent.CONTINUE;
229 			pcl.forEach(function(obj) {
230 				var hostMask = this.collisionModel.getObjectSpatialData(host, "collisionMask");
231 				var targetMask = this.collisionModel.getObjectSpatialData(obj, "collisionMask");
232             if ((hostMask & targetMask) <= hostMask && 
233 					  status == ColliderComponent.CONTINUE) {
234 
235                status = this.testCollision(time, obj, hostMask, targetMask); 
236             }
237          }, this);
238       }
239 		
240 		if (host.onCollideEnd) {
241 			if (!pcl || (pcl && (this.collideSelf || pcl.size() > 1))) {
242 				host.onCollideEnd(time);
243 			}
244 		}
245    },
246    
247    /**
248     * Call the host object's <tt>onCollide()</tt> method, passing the time of the collision,
249     * the potential collision object, and the collision object's flags.  The return value should 
250     * indicate if the collision tests should continue or stop.
251     * <p/>
252     * If the host object implements <tt>onCollideEnd()</tt> and the we're not colliding with
253     * ourself or anything else, the method is triggered on the host.
254     * <p/>
255     * For <tt>ColliderComponent</tt> the collision test is up to the host object to determine.
256     *
257     * @param time {Number} The engine time (in milliseconds) when the potential collision occurred
258     * @param collisionObj {HostObject} The host object with which the collision potentially occurs
259     * @param hostMask {Number} The collision mask for the host object
260     * @param targetMask {Number} The collision mask for <tt>collisionObj</tt>
261     * @return {Number} A status indicating whether to continue checking, or to stop
262     */
263    testCollision: function(time, collisionObj, hostMask, targetMask) {
264 		if (!(this.collideSelf && this === collisionObj)) {
265       	return this.getHostObject().onCollide(collisionObj, time, targetMask);
266 		}
267    }
268 
269 }, /** @scope ColliderComponent.prototype */{ // Statics
270 
271    /**
272     * Get the class name of this object
273     *
274     * @return {String} "ColliderComponent"
275     */
276    getClassName: function() {
277       return "ColliderComponent";
278    },
279 
280    /**
281     * When <tt>onCollide()</tt> is called on the host object, it should
282     * return this value if it hasn't handled the collision, or if it wishes to be 
283     * notified about other potential collisions.
284     * @type {Number}
285     */
286    CONTINUE: 0,
287 
288    /**
289     * When <tt>onCollide()</tt> is called on the host object, it should
290     * return this if no more collisions should be reported.
291     * @type {Number}
292     */
293    STOP: 1
294 
295 });
296 
297 return ColliderComponent;
298 
299 });
