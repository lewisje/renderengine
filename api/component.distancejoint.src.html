 34 // Includes
 35 Engine.include("/engine/engine.math2d.js");
 36 Engine.include("/components/component.basejoint.js");
 37 Engine.include("/physics/dynamics/joints/b2DistanceJointDef.js");
 38 
 39 Engine.initObject("DistanceJointComponent", "BaseJointComponent", function() {
 40 
 41 /**
 42  * @class A distance joint which maintains constant distance between two bodies
 43  * 		 in a {@link Simulation}.  
 44  *
 45  * @param name {String} Name of the component
 46  * @param body1 {BaseBodyComponent} The first body for the joint
 47  * @param body2 {BaseBodyComponent} The second body for the joint
 48  *
 49  * @extends BaseJointComponent
 50  * @constructor
 51  * @description Creates a distance joint between two physical bodies.  The distance can
 52  * 				 be softened by adjusting the frequency and the damping ratio of the joint.
 53  * 				 Rotation is not limited by this joint.  The anchors for the joint are the
 54  * 				 rigid body center's.
 55  */
 56 var DistanceJointComponent = BaseJointComponent.extend(/** @scope DistanceJointComponent.prototype */{
 57 
 58    /**
 59     * @private
 60     */
 61 	constructor: function(name, body1, body2) {
 62 		var jointDef = new b2DistanceJointDef();
 63 		this.base(name || "DistanceJoint", body1, body2, jointDef);	
 64 	},
 65 	
 66 	/**
 67 	 * When simulation starts set the anchor points to the position of each rigid body.
 68 	 * @private
 69 	 */
 70 	startSimulation: function() {
 71 		if (!this.getSimulation()) {
 72 			var a1 = Point2D.create(this.getBody1().getPosition());
 73 			var a2 = Point2D.create(this.getBody2().getPosition());
 74 			a1.add(this.getBody1().getLocalOrigin());
 75 			a2.add(this.getBody2().getLocalOrigin());
 76 	
 77 			this.getJointDef().anchorPoint1.Set(a1.get().x, a1.get().y);
 78 			this.getJointDef().anchorPoint2.Set(a2.get().x, a2.get().y);
 79 			a1.destroy();
 80 			a2.destroy();
 81 		}
 82 		
 83 		this.base();
 84 	},
 85 	
 86 	/**
 87 	 * Set the frequency which is used to determine joint softness.  According to 
 88 	 * Box2d documentation the frequency should be less than half of the time step
 89 	 * used for the simulation.  In the engine, the frequency of the time step is
 90 	 * the frame rate.
 91 	 * 
 92 	 * @param hz {Number} The frequency in Hertz.
 93 	 */
 94 	setFrequency: function(hz) {
 95 		this.getJointDef().frequencyHz = hz;
 96 	},
 97 	
 98 	/**
 99 	 * Get the frequency from the joint definition.
100 	 * @return {Number}
101 	 */
102 	getFrequency: function() {
103 		return this.getJointDef().frequencyHz;
104 	},
105 	
106 	/**
107 	 * Set the damping ratio which is used to determine joint softness.  The value
108 	 * should be between 0.0 and 1.0, with 1.0 being extremely rigid.
109 	 * 
110 	 * @param dampingRatio {Number} A value between 0.0 and 1.0
111 	 */
112 	setDampingRatio: function(dampingRatio) {
113 		this.getJointDef().dampingRatio = dampingRatio;
114 	},
115 	
116 	/**
117 	 * Get the damping ratio from the joint definition.
118 	 * @return {Number}
119 	 */
120 	getDampingRatio: function() {
121 		return this.getJointDef().dampingRatio;
122 	}
123 		
124 }, { /** @scope DistanceJointComponent.prototype */
125 
126    /**
127     * Get the class name of this object
128     *
129     * @return {String} "DistanceJointComponent"
130     */
131    getClassName: function() {
132       return "DistanceJointComponent";
133    }   
134 });
135 
136 return DistanceJointComponent;
137 
138 });
