 34 // Includes
 35 Engine.include("/engine/engine.math2d.js");
 36 Engine.include("/components/component.render.js");
 37 
 38 Engine.initObject("SpriteComponent", "RenderComponent", function() {
 39 
 40 /**
 41  * @class A render component that renders its contents from a {@link Sprite}.  Sprites
 42  *        are 2d graphics which are either a single frame (static) or multiple frames
 43  *        (dynamic).  The sprite's descriptor will define that for the component.
 44  *
 45  * @param name {String} The component name
 46  * @param [priority=0.1] {Number} The render priority
 47  * @param sprite {Sprite} The sprite to render
 48  * @extends RenderComponent
 49  * @constructor
 50  * @description Create a sprite component.
 51  */
 52 var SpriteComponent = RenderComponent.extend(/** @scope SpriteComponent.prototype */{
 53 
 54    currentSprite: null,
 55 
 56    /**
 57     * @private
 58     */
 59    constructor: function(name, priority, sprite) {
 60       if (priority instanceof Sprite) {
 61          sprite = priority;
 62          priority = 0.1;
 63       }
 64       this.base(name, priority);
 65       this.currentSprite = sprite;
 66    },
 67 
 68    /**
 69     * Releases the component back into the object pool. See {@link PooledObject#release} for
 70     * more information.
 71     */
 72    release: function() {
 73       this.base();
 74       this.currentSprite = null;
 75    },
 76 
 77    /**
 78     * Calculate the bounding box from the set of
 79     * points which comprise the shape to be rendered.
 80     * @private
 81     */
 82    calculateBoundingBox: function() {
 83       return this.currentSprite.getBoundingBox();
 84     },
 85 
 86    /**
 87     * Set the sprite the component will render.
 88     *
 89     * @param sprite {Sprite} The sprite to render
 90     */
 91    setSprite: function(sprite) {
 92       this.currentSprite = sprite;
 93       
 94       if (this.getHostObject().jQ()) {
 95          this.getHostObject().jQ().css({
 96             width: sprite.getBoundingBox().len_x(),
 97             height: sprite.getBoundingBox().len_y(),
 98             background: "url('" + sprite.getSourceImage().src + "') no-repeat"
 99          });
100       }
101    },
102 
103    /**
104     * Get the sprite the component is rendering.
105     *
106     * @return {Sprite} A <tt>Sprite</tt> instance
107     */
108    getSprite: function() {
109       return this.currentSprite;
110    },
111 
112    /**
113     * Draw the sprite to the render context.  The frame, for animated
114     * sprites, will be automatically determined based on the current
115     * time passed as the second argument.
116     *
117     * @param renderContext {RenderContext} The context to render to
118     * @param time {Number} The engine time in milliseconds
119     */
120    execute: function(renderContext, time) {
121 
122       if (!this.base(renderContext, time))
123       {
124          return;
125       }
126 
127       if (this.currentSprite) {
128          renderContext.drawSprite(this.currentSprite, time, this.getHostObject());
129       }
130    }
131 }, /** @scope SpriteComponent.prototype */{ 
132    /**
133     * Get the class name of this object
134     *
135     * @return {String} "SpriteComponent"
136     */
137    getClassName: function() {
138       return "SpriteComponent";
139    }
140 });
141 
142 return SpriteComponent;
143 
144 });
