component.circlecollider.js


Summary

A collision component which determines collisions via bounding circle comparisons.
Class Summary
CircleColliderComponent An extension of the {@link ColliderComponent} which will check if the object's are colliding based on a bounding circle.

 36 Engine.include("/components/component.collider.js");

 38 Engine.initObject("CircleColliderComponent", "CircleColliderComponent", function() {
 39 
 40 /**
 41  * @class An extension of the {@link ColliderComponent} which will check if the
 42  *        object's are colliding based on a bounding circle.  
 43  *
 44  * @param name {String} Name of the component
 45  * @param collisionModel {SpatialCollection} The collision model
 46  * @param priority {Number} Between 0.0 and 1.0, with 1.0 being highest
 47  *
 48  * @extends ColliderComponent
 49  * @constructor
 50  * @description For this component to function, a movement vector must be available on the 
 51  *        host object by implementing the {@link Object2D#getVelocity} method which returns 
 52  *        a {@link Vector2D} instance.  Additionally the host object and collision object 
 53  *        must implement the {@link Object2D#getCircle} method which returns a 
 54  *        {@link Circle2D} instance.
 55  */
 56 var CircleColliderComponent = ColliderComponent.extend(/** @scope CircleColliderComponent.prototype */{
 57 
 58    /**
 59     * Call the host object's <tt>onCollide()</tt> method, passing the time of the collision
 60     * and the potential collision object.  The return value should either tell the collision
 61     * tests to continue, or to stop.
 62     * <p/>
 63     * A circular bounding area collision must occur to trigger the <tt>onCollide()</tt> method.
 64     *
 65     * @param time {Number} The engine time (in milliseconds) when the potential collision occurred
 66     * @param collisionObj {HostObject} The host object with which the collision potentially occurs
 67     * @return {Number} A status indicating whether to continue checking, or to stop
 68     */
 69    testCollision: function(time, collisionObj) {
 70       
 71       var host = this.getHostObject();
 72       
 73       // Check for the required methods and see if a collision will occur
 74       if (host.getCircle && collisionObj.getCircle && host.getVelocity &&
 75           this.isPotentialCollision(collisionObj)) {
 76 
 77          return this.base(collisionObj, time);
 78       }
 79       
 80       return ColliderComponent.CONTINUE;
 81    },
 82    
 83    /**
 84     * Check for a potential collision between the host object's circle and the
 85     * collision object's circle.  Returns <tt>true</tt> if a collision will occur.
 86     * @private
 87     */
 88    isPotentialCollision: function(collisionObj) {
 89       
 90       var host = this.getHostObject();
 91       
 92       // Early out test
 93       var dist = collisionObj.getCircle().getCenter().dist(host.getCircle().getCenter());
 94       var sumRad = collisionObj.getCircle().getRadius() + host.getCircle().getRadius();
 95       dist -= sumRad;
 96       if (host.getVelocity().len() < dist) {
 97          // No collision possible
 98          return false;
 99       }
100 
101       var norm = Vector2D.create(host.getVelocity()).normalize();
102 
103       // Find C, the vector from the center of the moving
104       // circle A to the center of B
105       var c = Vector2D.create(collisionObj.getCircle().getCenter().sub(host.getCircle().getCenter()));
106       var dot = norm.dot(c);
107 
108       // Another early escape: Make sure that A is moving
109       // towards B! If the dot product between the movevec and
110       // B.center - A.center is less that or equal to 0,
111       // A isn't isn't moving towards B
112       if (dot <= 0) {
113         return false;
114       }
115 
116       var lenC = c.len();
117       var f = (lenC * lenC) - (dot * dot);
118 
119       // Escape test: if the closest that A will get to B
120       // is more than the sum of their radii, there's no
121       // way they are going collide
122       var sumRad2 = sumRad * sumRad;
123       if (f >= sumRad2) {
124         return false;
125       }
126 
127       // We now have F and sumRadii, two sides of a right triangle.
128       // Use these to find the third side, sqrt(T)
129       var t = sumRad2 - f;
130 
131       // If there is no such right triangle with sides length of
132       // sumRadii and sqrt(f), T will probably be less than 0.
133       // Better to check now than perform a square root of a
134       // negative number.
135       if (t < 0) {
136         return false;
137       }
138 
139       // Therefore the distance the circle has to travel along
140       // movevec is D - sqrt(T)
141       var distance = dot - Math.sqrt(t);
142 
143       // Get the magnitude of the movement vector
144       var mag = host.getVelocity().len();
145 
146       // Finally, make sure that the distance A has to move
147       // to touch B is not greater than the magnitude of the
148       // movement vector.
149       if (mag < distance) {
150         return false;
151       }
152 
153       // Set the length of the movevec so that the circles will just
154       // touch
155       //var moveVec = host.getVelocity().normalize();
156       //movevec = moveVec.mul(distance);
157 
158       return true;         
159    }
160 
161 }, { /** @scope CircleColliderComponent.prototype */
162 
163    /**
164     * Get the class name of this object
165     *
166     * @return {String}
167     */
168    getClassName: function() {
169       return "CircleColliderComponent";
170    }
171    
172 });
173 
174 return CircleColliderComponent;
