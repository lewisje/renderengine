component.circlecollider.js


Summary

A collision component which determines collisions via bounding circle comparisons.
Class Summary
CircleColliderComponent An extension of the {@link ColliderComponent} which will check if the object's are colliding based on a bounding circle.

 38 Engine.initObject("CircleColliderComponent", "CircleColliderComponent", function() {
 39 
 40 /**
 41  * @class An extension of the {@link ColliderComponent} which will check if the
 42  *        object's are colliding based on a bounding circle.  If the bounding circle method
 43  *			 isn't available, the bounding box will be used to approximate a bounding circle.
 44  *
 45  * @param name {String} Name of the component
 46  * @param collisionModel {SpatialCollection} The collision model
 47  * @param priority {Number} Between 0.0 and 1.0, with 1.0 being highest
 48  *
 49  * @extends ColliderComponent
 50  * @constructor
 51  * @description Creates a collider component for circle-circle collision testing.  Each object
 52  *              must implement either the {@link Object2D#getWorldBox} or
 53  *              {@link Object2D#getCircle} method and return a world-oriented bounding box or
 54  *              circle, respectively.
 55  */
 56 var CircleColliderComponent = ColliderComponent.extend(/** @scope CircleColliderComponent.prototype */{
 57 
 58 	hasMethods: null,
 59 
 60    /**
 61     * Releases the component back into the pool for reuse.  See {@link PooledObject#release}
 62     * for more information.
 63     */
 64    release: function() {
 65       this.base();
 66 		this.hasMethod = null;		
 67 	},
 68 
 69 	/**
 70     * Establishes the link between this component and its host object.
 71     * When you assign components to a host object, it will call this method
 72     * so that each component can refer to its host object, the same way
 73     * a host object can refer to a component with {@link HostObject#getComponent}.
 74     *
 75     * @param hostObject {HostObject} The object which hosts this component
 76 	 */
 77 	setHostObject: function(hostObj) {
 78 		this.base(hostObj);
 79 		this.hasMethods = [hostObj.getCircle != undefined, hostObj.getWorldBox != undefined]; // getCircle, getWorldBox
 80 		/* pragma:DEBUG_START */
 81 		AssertWarn(this.hasMethods[0] || this.hasMethods[1], "Object " + hostObj.toString() + " does not have getCircle() or getWorldBox() methods");
 82 		/* pragma:DEBUG_END */
 83 	},
 84 
 85    /**
 86     * Call the host object's <tt>onCollide()</tt> method, passing the time of the collision,
 87     * the potential collision object, and the host and target masks.  The return value should 
 88     * either tell the collision tests to continue, or to stop.
 89     * <p/>
 90     * A circular bounding area collision must occur to trigger the <tt>onCollide()</tt> method.
 91     *
 92     * @param time {Number} The engine time (in milliseconds) when the potential collision occurred
 93     * @param collisionObj {HostObject} The host object with which the collision potentially occurs
 94     * @param hostMask {Number} The collision mask for the host object
 95     * @param targetMask {Number} The collision mask for <tt>collisionObj</tt>
 96     * @return {Number} A status indicating whether to continue checking, or to stop
 97     */
 98    testCollision: function(time, collisionObj, hostMask, targetMask) {
 99       
100       var host = this.getHostObject();
101       var hCircle = Circle2D.create(0,0,1);
102 		var oCircle = Circle2D.create(0,0,1);
103 		var wBox, r;
104 		
105 		// Check for easy methods
106 		if (this.hasMethods[0]) {
107 			hCircle.set(host.getCircle());
108 		} else {
109 			// Approximate a circle with the world box
110 			wBox = this.hasMethods[1] ? host.getWorldBox() : null;
111 			if (wBox == null) {
112 				hCircle.destroy();
113 				oCircle.destroy();
114 				return ColliderComponent.CONTINUE;	// Can't perform check
115 			}
116 			r = wBox.get();
117 			hCircle.set(wBox.getCenter(), r.w > r.h ? r.w / 2 : r.h / 2); 
118 		}
119 		
120 		if (collisionObj.getCircle) {
121 			oCircle.set(collisionObj.getCircle());
122 		} else {
123 			// Approximate a circle with the world box
124 			wBox = collisionObj.getWorldBox ? collisionObj.getWorldBox() : null;
125 			if (wBox == null) {
126 				hCircle.destroy();
127 				oCircle.destroy();
128 				return ColliderComponent.CONTINUE;	// Can't perform check
129 			}
130 			r = wBox.get();
131 			oCircle.set(wBox.getCenter(), r.w > r.h ? r.w / 2 : r.h / 2); 
132 		}
133 		
134       // See if a collision will occur
135       if (hCircle.isIntersecting(oCircle)) {
136       	hCircle.destroy();
137       	oCircle.destroy();
138          return this.base(time, collisionObj, hostMask, targetMask);
139       }
140       
141 		hCircle.destroy();
142 		oCircle.destroy();
143       return ColliderComponent.CONTINUE;
144    }
145 	
146 }, { /** @scope CircleColliderComponent.prototype */
147 
148    /**
149     * Get the class name of this object
150     *
151     * @return {String}
152     */
153    getClassName: function() {
154       return "CircleColliderComponent";
155    }
156    
157 });
158 
159 return CircleColliderComponent;
160 
161 });
