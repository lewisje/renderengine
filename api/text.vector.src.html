 41  * @class A text renderer which draws text with simple vectors.  This type of text

 75    /**
 76     * Calculate the bounding box for the text and set it on the host object.
 77     * @private
 78     */
 79    calculateBoundingBox: function() {
 80 
 81       var x1 = 0;
 82       var x2 = 0;
 83       var y1 = 0;
 84       var y2 = 0;
 85       for (var p = 0; p < this.rText.length; p++)
 86       {
 87          var pt = this.rText[p];
 88          if (pt) {
 89             if (pt.x < x1)
 90             {
 91                x1 = pt.x;
 92             }
 93             if (pt.x > x2)
 94             {
 95                x2 = pt.x;
 96             }
 97             if (pt.y < y1)
 98             {
 99                y1 = pt.y;
100             }
101             if (pt.y > y2)
102             {
103                y2 = pt.y;
104             }
105          }
106       }
107 
108       this.getHostObject().setBoundingBox(new Rectangle2D(x1 * this.getSize(), y1 * this.getSize(), (Math.abs(x1) + x2 + 2) * this.getSize(), (Math.abs(y1) + y2 + 2) * this.getSize()));
109    },
110 
111    /**
112     * Set the scaling of the text
113     * @param size {Number}
114     */
115    setSize: function(size) {
116       this.base(size);
117       this.calculateBoundingBox();
118    },

120    /**
121     * Set the text to render.
122     *
123     * @param text {String} The text to vectorize
124     */
125    setText: function(text) {
126       // We only have uppercase letters
127       text = String(text).toUpperCase();
128       this.base(text);
129 		
130 		if (this.rText.length > 0) {
131 			for (var r in this.rText) {
132 				if (this.rText[r])
133 					this.rText[r].destroy();
134 			}
135 		}
136 		
137       this.rText = [];
138       var spacing = 11.5;
139 
140       // Replace special chars
141       text = text.replace(/©/g,"a").replace(/®/g,"b");
142 
143       var lCount = text.length;
144       var align = this.getTextAlignment();
145       var letter = (align == AbstractTextRenderer.ALIGN_RIGHT ? text.length - 1 : 0);
146       var kern = new Point2D((align == AbstractTextRenderer.ALIGN_RIGHT ? -spacing : spacing), 0);
147 		var lineHeight = this.getSize() * 5;
148 		var y = 0;
149       //var space = new Point2D((align == AbstractTextRenderer.ALIGN_RIGHT ? -spacing : spacing) * 0.07, 0);
150 
151 
152       // Vectorize the text
153       var pc = new Point2D(0,y);
154       while (lCount-- > 0)
155       {
156          var ltr = [];
157 			var chr = text.charCodeAt(letter);
158 			if (chr == 10) {
159 				// Support multi-line text
160 				y += (this.getSize() * 10) + this.getLineSpacing();
161 				pc.set(0, y);
162 			} else {
163 	         var glyph = VectorText.charSet[chr - 32];
164 	         if (glyph.length == 0)
165 	         {
166 	            pc.add(kern);
167 	         }
168 	         else
169 	         {
170 	            for (var p = 0; p < glyph.length; p++)
171 	            {
172 	               if (glyph[p] != null)
173 	               {
174 	                  this.rText.push(Point2D.create(glyph[p][0], glyph[p][1]).add(pc));
175 	               }
176 	               else
177 	               {
178 	                  this.rText.push(null);
179 	               }
180 	            }
181 	            this.rText.push(null);
182 	            pc.add(kern);
183 	         }
184 			}
185          letter += (align == AbstractTextRenderer.ALIGN_RIGHT ? -1 : 1);
186       }
187 		pc.destroy();
188       this.calculateBoundingBox();
189    },
190 
191    /**
192     * @private
193     */
194    execute: function(renderContext, time) {
195 
196       if (this.rText.length == 0)
197       {
198          return;
199       }
200 
201       renderContext.pushTransform();
202       renderContext.setScale(this.getSize());
203       // Set the stroke and fill styles
204       if (this.getColor() != null)
205       {
206          renderContext.setLineStyle(this.getColor());
207       }
208 
209       renderContext.setLineWidth(this.getTextWeight());
210       renderContext.drawPolyline(this.rText);
211       renderContext.popTransform();
212    }
