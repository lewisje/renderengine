 37 // Includes
 38 Engine.include("/engine/engine.resourceloader.js");
 39 
 40 Engine.initObject("RemoteFileLoader", "ResourceLoader", function() {
 41 
 42 /**
 43  * @class Loads a text or XML file from the specified URL.  It is possible to check for the
 44  *        existence of a file with this type of loader.
 45  *
 46  * @constructor
 47  * @param [name=RemoteFileLoader] {String} The name of the resource loader
 48  * @extends ResourceLoader
 49  * @private
 50  */
 51 var RemoteFileLoader = ResourceLoader.extend(/** @scope RemoteFileLoader.prototype */{
 52 
 53    pending: null,
 54 
 55    /**
 56     * private
 57     */
 58    constructor: function(name) {
 59       this.base(name || "RemoteFileLoader");
 60       this.pending = null;
 61    },
 62    
 63    release: function() {
 64       this.base();
 65       this.pending = null;
 66    },
 67 
 68    /**
 69     * Helper method to get the remote data and handle incoming
 70     * returns.
 71     * @param remoteURL {String} The URL to retrieve.  Must be relative.
 72     * @param remoteType {String} The type of data: "xml", "text"
 73     * @param [sync] {Boolean} <tt>true</tt> to make blocking request
 74     * @private
 75     */
 76    getFile: function(cacheRec, sync) {
 77       Assert(remoteURL.indexOf("http") == -1, "URL must exist relative to this server");
 78       
 79       // Make the request. We'll handle the result ourselves
 80       var self = this;
 81       var xhr = $.ajax({
 82                type: "GET",
 83                url: cacheRec.url,
 84                data: null,
 85                async: !sync,
 86                success: function(data) {
 87                   cacheRec.data = data;
 88                   cacheRec.status = RemoteFileLoader.STATUS_OK;
 89                   self.success(cacheRec);
 90                },
 91                complete: function(xhr, status) {
 92                   if (status != "success") {
 93                      cacheRec.status = RemoteFileLoader.STATUS_FAIL;
 94                   }
 95                },
 96                dataType: cacheRec.type
 97             });
 98             
 99       return cacheRec;
100    },
101    
102    /**
103     * Check for the existence of a file at the url provided in a
104     * synchronous manner.  This will cause the program's execution to
105     * wait until the file can be verified.
106     * 
107     * @param remoteUrl {String} The URL where the file is located.
108     * @param type {String} The type of data to check
109     * @return <tt>true</tt> if the file exists
110     */
111    exists: function(remoteUrl, type) {
112       this.pending = this.getCacheRecord(remoteURL, type);
113       var r = this.getFile(this.pending, true);
114       return (r.status == RemoteFileLoader.STATUS_OK);
115    },
116 
117    load: function(name, remoteURL, remoteType, sync) {
118       var cacheRec = null;
119       if (this.pending.url == remoteURL) {
120          cacheRec = this.pending;
121          this.pending = null;
122       } else {
123          cacheRec = this.getCacheRecord(remoteURL, remoteType);
124       }
125       cacheRec.name = name;
126       
127       this.base(name, cacheRec, false);
128       this.getFile(cacheRec, sync);
129    },
130 
131    success: function(cacheRec) {
132       if (cacheRec.name) {
133          this.set(name, cacheRec);
134          this.setReady(name, true);
135       }
136    },
137    
138    getCacheRecord: function(remoteURL, remoteType) {
139       return {
140          url: remoteURL,
141          status: RemoteFileLoader.STATUS_INIT,
142          type: remoteType,
143          data: null,
144          xhr: null
145       };
146    },
147 
148    getData: function(name) {
149       var d = this.get(name);
150       return d ? d.data : null;
151    },
152 
153    /**
154     * The name of the resource this loader will get.
155     * @returns {String} The string "remote file"
156     */
157    getResourceType: function() {
158       return "remote file";
159    }
160 
161 }, /** @scope RemoteFileLoader.prototype */{
162    /**
163     * Get the class name of this object.
164     * @return {String} The string "RemoteFileLoader"
165     */
166    getClassName: function() {
167       return "RemoteFileLoader";
168    },
169    
170    /** Request initializating **/
171    STATUS_INIT: -1,
172    
173    /** Request succeeded **/
174    STATUS_OK: 1,
175    
176    /** There was some sort of error while loading the file **/
177    STATUS_FAIL: 0
178    
179 });
180 
181 return RemoteFileLoader;
182 
183 });
