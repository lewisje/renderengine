engine.pooledobject.js


Summary

An object that has functionality to assist in keeping memory usage down and to minimize the effect of the JavaScript garbage collector.
Class Summary
PooledObject Pooled objects are created as needed, and reused from a static pool of all objects, organized by classname.

 37 /**
 38  * @class Pooled objects are created as needed, and reused from a static pool
 39  *        of all objects, organized by classname.  When an object is created, if one 
 40  *        is not available in the pool, a new object is created.  When the object 
 41  *        is destroyed, it is returned to the pool so it can be used again.  This has the
 42  *        effect of minimizing the requirement for garbage collection, reducing
 43  *        cycles needed to clean up dead objects.
 44  *
 45  * @param name {String} The name of the object within the engine.
 46  * @constructor
 47  * @description Create an instance of this object, assigning a global identifies to it.
 48  */
 49 var PooledObject = Base.extend(/** @scope PooledObject.prototype */{
 50 
 51    // The Id assigned by the engine
 52    id: -1,
 53 
 54    // The name of the object
 55    name: "",
 56 	
 57 	_destroyed: false,

 61    /**
 62     * @private
 63     */
 64    constructor: function(name) {
 65 		this._destroyed = false;
 66       this.name = name;
 67       this.id = Engine.create(this);
 68    },
 69 
 70    /**
 71     * When a pooled object is destroyed, its <tt>release()</tt> method will be called
 72     * so it has a chance to can clean up instance variables before being put back into 
 73     * the pool for reuse. The variables should be returned to an "uninitialized" state.
 74     */
 75    release: function() {
 76       this.name = "";
 77       this.id = -1;
 78    },
 79 
 80    /**
 81     * Destroy this object instance (remove it from the Engine).  The object's release
 82     * method is called after destruction so it will be returned to the pool of objects 
 83     * to be used again.
 84     */
 85    destroy: function() {
 86       // Clean up the engine reference to this object
 87       Engine.destroy(this);
 88 
 89 		this._destroyed = true;
 90       PooledObject.returnToPool(this);
 91       Engine.addMetric("poolLoad", Math.floor((PooledObject.poolSize / PooledObject.poolNew) * 100), false, "#%");
 92 
 93       // Reset any variables on the object after putting
 94       // it back in the pool.
 95       this.release();
 96    },

 98    /**
 99     * Get the managed Id of this object within the Engine.
100     *
101     * @return {String}

103    getId: function() {
104       return this.id;
105    },

107    /**
108     * Get the original name this object was created with.
109     *
110     * @return {String} The name used when creating this object

114    },

116    /**
117     * Returns an object that assigns getter and setter methods
118     * for exposed properties of an object.

122       var self = this;
123       var prop = {};
124       return $.extend(prop, {

136    /**

140    toXML: function(indent) {
141       indent = indent ? indent : "";
142       var props = this.getProperties();
143       var xml = indent + "<" + this.constructor.getClassName();
144       for (var p in props) {
145          // If the value should be serialized, call it's getter
146          if (props[p][2]) {
147             xml += " " + p.toLowerCase() + "=\"" + props[p][0]().toString() + "\"";
148          }
149       }
150 
151       xml += "/>\n";
152       return xml;

235    /**
236     * Similar to a constructor, all pooled objects implement this method.
237     * The <tt>create()</tt> method will either create a new instance, if no object of the object's
238     * class exists within the pool, or will reuse an existing pooled instance of
239     * the object.  Either way, the constructor for the object instance is called so that
240     * instance creation can be maintained in the constructor.
241     * <p/>
242     * Usage: <tt>var obj = [ObjectClass].create(arg1, arg2, arg3...);</tt>
243     *
244     * @memberOf PooledObject
245     */
246    create: function() {
247       // Check the pool for the object type
248       if (PooledObject.objectPool[this.getClassName()] && PooledObject.objectPool[this.getClassName()].length != 0) {
249          PooledObject.poolSize--;
250          Engine.addMetric("poolLoad", Math.floor((PooledObject.poolSize / PooledObject.poolNew) * 100), false, "#%");
251          var obj = PooledObject.objectPool[this.getClassName()].shift();
252          obj.constructor.apply(obj, arguments);
253 
254          /* pragma:DEBUG_START */ 
255          PooledObject.classPool[this.getClassName()][1]++;
256          PooledObject.classPool[this.getClassName()][2]--;
257          /* pragma:DEBUG_END */
258 
259          return obj;
260       } else {
261          PooledObject.poolNew++;
262          Engine.addMetric("poolNew", PooledObject.poolNew, false, "#");
263          
264          /* pragma:DEBUG_START */
265          if (PooledObject.classPool[this.getClassName()]) {
266             PooledObject.classPool[this.getClassName()][0]++;
267          } else {
268             // 0: new, 1: in use, 2: pooled
269             PooledObject.classPool[this.getClassName()] = [1,0,0];
270          }
271          /* pragma:DEBUG_END */
272          
273          // TODO: Any more than 15 arguments and construction will fail!
274          return new this(arguments[0],arguments[1],arguments[2],arguments[3],arguments[4],
275                          arguments[5],arguments[6],arguments[7],arguments[8],arguments[9],
276                          arguments[10],arguments[11],arguments[12],arguments[13],arguments[14]);
277       }
278    },
279 
280    /**
281     * During object destruction, it needs to return itself to the
282     * pool so it can be used again.  Instead of creating new instances
283     * of each object, we utilize a pooled object so we don't need the
284     * garbage collector to be invoked for removed objects.
285     * @private
286     */
287    returnToPool: function(obj) {
288       // If there is no pool for objects of this type, create one
289       if (!PooledObject.objectPool[obj.constructor.getClassName()]) {
290          PooledObject.objectPool[obj.constructor.getClassName()] = [];
291       }
292 
293       // Push this object into the pool
294       PooledObject.poolSize++;
295       PooledObject.objectPool[obj.constructor.getClassName()].push(obj);
296 
297       /* pragma:DEBUG_START */
298       if (PooledObject.classPool[obj.constructor.getClassName()][1] != 0) {
299          PooledObject.classPool[obj.constructor.getClassName()][1]--;
300       }
301       PooledObject.classPool[obj.constructor.getClassName()][2]++;
302       /* pragma:DEBUG_END */
303          
304 
305       Engine.addMetric("pooledObjects", PooledObject.poolSize, false, "#");
306    },
