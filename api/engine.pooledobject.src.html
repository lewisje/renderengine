engine.pooledobject.js


Summary

An object that has functionality to assist in keeping memory usage down and to minimize the effect of the JavaScript garbage collector.
Class Summary
PooledObject Pooled objects are created as needed, and reused from a static pool of all objects, organized by class.

 37 /**
 38  * @class Pooled objects are created as needed, and reused from a static pool
 39  *        of all objects, organized by class.  When an object is created, if one 
 40  *        is not available in the pool, a new object is created.  When the object 
 41  *        is destroyed, it is returned to the pool so it can be used again.  This has the
 42  *        effect of minimizing the requirement of garbage collection, reducing
 43  *        cycles needed to clean up dead objects.
 44  *
 45  * @param name {String} The name of the object from which the Id will be generated.
 46  * @constructor
 47  * @description Create an instance of this object, assigning a name to it.  An
 48  *              object reference will be maintained by the {@link Engine} class,
 49  *              which gives the object final responsibility for making sure the
 50  *              object can be destroyed.
 51  */
 52 var PooledObject = Base.extend(/** @scope PooledObject.prototype */{
 53 
 54    // The Id assigned by the engine
 55    id: -1,
 56 
 57    // The name of the object
 58    name: "",
 59 
 60    /**
 61     * @private
 62     */
 63    constructor: function(name) {
 64       this.name = name;
 65       this.id = Engine.create(this);
 66    },
 67 
 68    /**
 69     * When a pooled object is destroyed, its <tt>release()</tt> method will be called
 70     * so it has a chance to can clean up instance variables before being put back into 
 71     * the pool for reuse. The variables should be returned to an "uninitialized" state.
 72     */
 73    release: function() {
 74       this.name = "";
 75       this.id = -1;
 76    },
 77 
 78    /**
 79     * Destroy this object instance (remove it from the Engine).  The object's release
 80     * method is called after destruction so it will be returned to the pool of objects 
 81     * to be used again.
 82     */
 83    destroy: function() {
 84 
 85       PooledObject.returnToPool(this);
 86 
 87       // Clean up the engine reference to this object
 88       Engine.destroy(this);
 89 
 90       // Reset any variables on the object before putting
 91       // it back in the pool.
 92       this.release();
 93    },
 94 
 95    /**
 96     * Get the managed Id of this object within the Engine.
 97     *
 98     * @return {String}
 99     */
100    getId: function() {
101       return this.id;
102    },
103 
104    /**
105     * Get the original name this object was created with.
106     *
107     * @return {String} The name used when creating this object
108     */
109    getName: function() {
110       return this.name;
111    },
112 
113    /**
114     * Returns an object that assigns getter and setter methods
115     * for exposed properties of an object.
116     * @return {Object} An object which contains getter and setter methods.
117     */
118    getProperties: function() {
119       var self = this;
120       var prop = {};
121       return $.extend(prop, {
122          "Id"           : [function() { return self.getId(); },
123                            null, false],
124          "ObjectName"   : [function() { return self.getName(); },
125                            function(i) { self.setName(i); }, true]
126       });
127    },
128 
129    /**
130     * Serialize the object to XML.
131     * @return {String}
132     */
133    toString: function(indent) {
134       indent = indent ? indent : "";
135       var props = this.getProperties();
136       var xml = indent + "<" + this.constructor.getClassName();
137       for (var p in props) {
138          // If the value should be serialized, call it's getter
139          if (props[p][2]) {
140             xml += " " + p + "=\"" + props[p][0]().toString() + "\"";
141          }
142       }
143 
144       xml += "/>\n";
145       return xml;
146    }

174    /**
175     * Similar to a constructor, all pooled objects implement this method.
176     * The <tt>create()</tt> method will either create a new instance, if no object of the object's
177     * class exists within the pool, or will reuse an existing pooled instance of
178     * the object.  Either way, the constructor for the object instance is called so that
179     * instance creation can be maintained in the constructor.
180     * <p/>
181     * Usage: <tt>var obj = [ObjectClass].create(arg1, arg2, arg3...);</tt>
182     *
183     * @memberOf PooledObject
184     */
185    create: function() {
186       // Check the pool for the object type
187       if (PooledObject.objectPool[this.getClassName()] && PooledObject.objectPool[this.getClassName()].length != 0) {
188          PooledObject.poolSize--;
189          Engine.addMetric("poolUse", Math.floor((PooledObject.poolSize / PooledObject.poolNew) * 100), false, "#%");
190          var obj = PooledObject.objectPool[this.getClassName()].shift();
191          obj.constructor.apply(obj, arguments);
192          return obj;
193       } else {
194          PooledObject.poolNew++;
195          Engine.addMetric("newObjs", PooledObject.poolNew, false, "#");
196          // TODO: Any more than 15 arguments and construction will fail!
197          return new this(arguments[0],arguments[1],arguments[2],arguments[3],arguments[4],
198 								 arguments[5],arguments[6],arguments[7],arguments[8],arguments[9],
199 								 arguments[10],arguments[11],arguments[12],arguments[13],arguments[14]);
200       }
201    },
