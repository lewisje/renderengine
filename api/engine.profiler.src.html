 41 /**
 42  * @class A static JavaScript implementation of a simple profiler.
 43  * @static
 44  */
 45 var Profiler = {
 46 	profileStack: [],
 47 	allProfiles: {},
 48 	profiles: [],
 49 	running: false
 50 };
 51 
 52 /**
 53  * Start the profiler.
 54  * @memberOf Profiler
 55  */
 56 Profiler.start = function() {
 57 	Profiler.resetProfiles();
 58 	Profiler.running = true;
 59 };
 60 
 61 /**
 62  * Stop the profiler, dumping whatever was being profiled.
 63  * @memberOf Profiler
 64  */
 65 Profiler.stop = function() {
 66 	Profiler.dump();
 67 	Profiler.running = false;
 68 };

 70 /**
 71  * Add a profile monitor to the stack of running profiles.
 72  * @param prof {String} The name of the profile
 73  * @memberOf Profiler
 74  */
 75 Profiler.enter = function(prof) {
 76 	if (!Profiler.running) { return; }
 77 	var profile = Profiler.allProfiles[prof];
 78 	if (profile == null) {
 79 		// Create a monitor
 80 		profile = Profiler.allProfiles[prof] = {
 81 			name: prof,
 82 			startMS: now(),
 83 			execs: 0,
 84 			totalMS: 0,
 85 			instances: 1,
 86 			pushed: false
 87 		};
 88 	} else {
 89 		profile.startMS = profile.instances == 0 ? now() : profile.startMS;
 90 		profile.instances++;
 91 	}
 92 	Profiler.profileStack.push(profile);
 93 };
 94 
 95 /**
 96  * For every "enter", there needs to be a matching "exit" to
 97  * tell the profiler to stop timing the contained code.  Note
 98  * that "exit" doesn't take any parameters.  It is necessary that
 99  * you properly balance your profile stack.  Too many "exit" calls
100  * will result in a stack underflow. Missing calls to "exit" will
101  * result in a stack overflow.
102  * @memberOf Profiler
103  */
104 Profiler.exit = function() {
105 	if (!Profiler.running) { return; }
106 	if (Profiler.profileStack.length == 0) {
107 		var msg = "Profile stack underflow";
108 		if (typeof console !== "undefined") { console.error(msg); }
109 		throw(msg);
110 	}
111 
112 	var profile = Profiler.profileStack.pop();
113 	profile.endMS = new Date();
114 	profile.execs++;
115 	profile.instances--;
116 	profile.totalMS += profile.instances == 0 ? (profile.endMS.getTime() - profile.startMS.getTime()) : 0;
117 	if (!profile.pushed) {
118 		// If we haven't remembered it, do that now
119 		profile.pushed = true;
120 		Profiler.profiles.push(profile);
121 	}
122 };

124 /**
125  * Reset any currently running profiles and clear the stack.
126  * @memberOf Profiler
127  */
128 Profiler.resetProfiles = function() {
129 	Profiler.profileStack = [];
130 	Profiler.allProfiles = {};
131 	Profiler.profiles = [];
132 };
