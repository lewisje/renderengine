loader.remote.js


Summary

A resource loader for handling remote objects located on the server.
Class Summary
RemoteLoader Loads JSON objects from a specified URL.

 34 // Includes
 35 Engine.include("/engine/engine.resourceloader.js");
 36 
 37 Engine.initObject("RemoteLoader", "ResourceLoader", function() {
 38 
 39 /**
 40  * @class Loads JSON objects from a specified URL.
 41  *
 42  * @constructor
 43  * @param name {String=RemoteLoader} The name of the resource loader
 44  * @extends ResourceLoader
 45  */
 46 var RemoteLoader = ResourceLoader.extend(/** @scope RemoteLoader.prototype */{
 47 
 48    /**
 49     * private
 50     */
 51    constructor: function(name) {
 52       this.base(name || "RemoteLoader");
 53    },
 54 
 55    /**
 56     * Performs a synchronous check for a file on the server.  While this approach will
 57     * work in most cases, there is the possibility that the server will become unavailable
 58     * before the request is made.  In this case, the application will hang until the
 59     * request is satisfied (which may be never).
 60     *
 61     * @param url {String} The URL to check
 62     * @return {Boolean} <tt>true</tt> if the file exists on the server or is in
 63     *          the cache.
 64     */
 65    exists: function(url) {
 66       var stat = jQuery.ajax({
 67          type: "GET",
 68          url: url,
 69          async: false,
 70          dataType: "text"
 71       }).status;      
 72       
 73       // If it returns OK or Cache not modified...
 74       return (stat == RemoteLoader.STATUS_OK || stat == RemoteLoader.STATUS_CACHED);
 75    },
 76 
 77    /**
 78     * The name of the resource this loader will get.
 79     * @returns {String} The string "remote"
 80     */
 81    getResourceType: function() {
 82       return "remote";
 83    }
 84 
 85 }, /** @scope RemoteLoader.prototype */{
 86    /**
 87     * Get the class name of this object.
 88     * @return {String} The string "SpriteLoader"
 89     */
 90    getClassName: function() {
 91       return "RemoteLoader";
 92    },
 93 	
 94 	/**
 95 	 * Transmit status ok
 96 	 */
 97 	STATUS_OK: 200,
 98 	
 99 	/**
100 	 * Transmit status - Cached
101 	 */
102 	STATUS_CACHED: 304,
103 	
104 	/**
105 	 * Transmit status - Not found
106 	 */
107 	STATUS_NOT_FOUND: 404,
108 	
109 	/**
110 	 * Transmit status - Server error
111 	 */
112 	STATUS_SERVER_ERROR: 500
113 });
114 
115 return RemoteLoader;
116 
117 });
