engine.resourceloader.js


Summary

The base class for all resource loaders. It has the functionality for managing a local cache of loaded objects.
Class Summary
ResourceLoader A resource loader is a generalized interface used by all resource loaders.

 34 // Includes
 35 Engine.include("/engine/engine.baseobject.js");
 36 
 37 Engine.initObject("ResourceLoader", "BaseObject", function() {
 38 
 39 /**
 40  * @class  A resource loader is a generalized interface used by all resource
 41  *         loaders.  It is designed to provide a common set of routines for
 42  *         loading resources (fonts, images, game data, etc...) from some
 43  *         location.  Additionally, objects are cached by this base class,
 44  *         although some classes make use of other methods to enhance the
 45  *         caching, such as the {@link ImageLoader} class.
 46  *
 47  * @param [name=ResourceLoader] {String} The name of the resource loader.
 48  * @constructor
 49  * @extends BaseObject
 50  * @description Create a resource loader
 51  */
 52 var ResourceLoader = BaseObject.extend(/** @scope ResourceLoader.prototype */{
 53 
 54    cache: null,
 55 
 56    length: 0,
 57 
 58    /**
 59     * @private
 60     */
 61    constructor: function(name) {
 62       this.base(name || "ResourceLoader");
 63       this.cache = {};
 64    },
 65 
 66    /**
 67     * @private
 68     */
 69    release: function() {
 70       this.base();
 71       this.cache = null;
 72       this.length = 0;
 73    },
 74 
 75    /**
 76     * @private
 77     */
 78    destroy: function() {
 79       this.clear();
 80       this.base();
 81    },
 82 
 83    /**
 84     * Load an object via this resource loader, and add it to the cache.
 85     *
 86     * @param name {String} The name to refer to the loaded object
 87     * @param data {Object} The data to store in the cache
 88     * @param isReady {Boolean} A flag that states whether or not a resource
 89     *                          is ready to use.
 90     */
 91    load: function(name, data, isReady) {
 92       var obj = { data: data, ready: isReady || false};
 93       this.cache[name] = obj;
 94       this.length++;
 95       Console.log("Loading " + this.getResourceType() + ": " + name);
 96    },
 97 
 98    /**
 99     * Set the "ready" state of the resource.  When a resource has been completely
100     * loaded, set the resource "ready" state to <tt>true</tt> to allow objects
101     * waiting for those resources to utilize them.
102     *
103     * @param name {String} The name of the resource
104     * @param isReady {Boolean} <tt>true</tt> to set the resource to "ready for use"
105     */
106    setReady: function(name, isReady) {
107       this.cache[name].isReady = isReady;
108       if (isReady) {
109          Console.log(this.getResourceType() + " " + name + " ready...");
110       }
111    },
112 
113    /**
114     * Check to see if a named resource is "ready for use".
115     * @param name {String} The name of the resource to check ready status for,
116     *             or <tt>null</tt> for all resources in loader.
117     * @return {Boolean} <tt>true</tt> if the resource is loaded and ready to use
118     */
119    isReady: function(name) {
120       if (name) {
121          return this.cache[name] ? this.cache[name].isReady : false;
122       } else {
123          // Check the status of all loader elements
124          var rList = this.getResources();
125          if (rList.length == 0) {
126             // Early out, no resources to load
127             return true;
128          }
129          for (var r in rList) {
130             if (!this.isReady(rList[r])) {
131                return false;
132             }
133          }
134          return true;
135       }
136    },
137 
138    /**
139     * Unload an object from this resource loader.  Removes the object
140     * from the cache.
141     *
142     * @param name {String} The name of the object to remove
143     */
144    unload: function(name) {
145       if (this.cache[name].data.destroy) {
146          // Make sure that cached objects have a chance to clean up
147          this.cache[name].data.destroy();
148       }
149 
150       this.cache[name] = null;
151       delete this.cache[name];
152       this.length--;
153    },
154 
155    /**
156     * Get the object with the specified name from the cache.
157     *
158     * @param name {String} The name of the object to retrieve
159     * @return {Object} The object stored within the cache
160     */
161    get: function(name) {
162       if (this.cache[name]) {
163          return this.cache[name].data;
164       } else {
165          return null;
166       }
167    },
168    
169    /**
170     * Set the data associated with the name.  The ready state is set
171     * to <tt>false</tt>, so it will be up to the developer to call
172     * {@link #setReady} on the object if the object is truly ready for use.
173     * @param name {String} The name of the cache record
174     * @param data {Object} Data to store
175     */
176    set: function(name, data) {
177       var obj = { data: data, ready: false};
178       this.cache[name] = obj;
179    },
180 
181    /**
182     * Returns the cache.  You should not manipulate the cache directly.
183     * instead, call methods to update the cache.
184     * @return {Object} The cache
185     */
186    getCachedObjects: function() {
187       return this.cache;
188    },
189 
190    /**
191     * Clear the objects contained in the cache.
192     */
193    clear: function() {
194       for (var o in this.cache) {
195          this.cache[o] = null;
196       }
197 
198       this.cache = {};
199       this.length = 0;
200    },
201 
202    /**
203     * Get the names of all the resources available in this resource loader.
204     * @return {Array} An array of resource names
205     */
206    getResources: function() {
207       var n = [];
208       for (var i in this.cache) {
209          n.push(i);
210       }
211       return n;
212    },
213 
214 	/**
215 	 * Exports an object which contains references to all of the resources
216 	 * which have been loaded by this resource loader.  This method is optional
217 	 * and may not be implemented directly by the loader.
218 	 */
219 	exportAll: function() {
220 		return {};
221 	},
222 
223    /**
224     * The name of the resource this loader will get.
225     * @return {String} The string "default"
226     */
227    getResourceType: function() {
228       return "default";
229    }
230 }, /** @scope ResourceLoader.prototype */{ 
231 
232    /**
233     * Get the class name of this object
234     * @return {String} "ResourceLoader"
235     */
236    getClassName: function() {
237       return "ResourceLoader";
238    }
239 
240 });
241 
242 return ResourceLoader;
243 
244 });
