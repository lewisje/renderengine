Summary

A native context font renderer. Uses the context's font rendering mechanism to generate textual output.

 34 // Includes
 35 Engine.include("/textrender/text.abstractrender.js");

 37 Engine.initObject("ContextText", "AbstractTextRenderer", function() {
 38 
 39 /**
 40  * @class A text renderer which draws text from a bitmap font file.
 41  *
 42  * @constructor
 43  * @extends AbstractTextRenderer
 44  */
 45 var ContextText = AbstractTextRenderer.extend(/** @scope ContextText.prototype */{
 46 
 47 	/**
 48 	 * @private
 49 	 */
 50    constructor: function() {
 51       this.base();
 52 		this.tInit();
 53    },
 54 
 55 	/**
 56 	 * Initialize some basics
 57 	 */
 58 	tInit: function() {
 59 		this.setTextAlignment(RenderContext2D.FONT_ALIGN_LEFT);
 60 		this.setTextWeight(RenderContext2D.FONT_WEIGHT_NORMAL);
 61 		this.setTextFont("sans-serif");
 62 		this.setTextStyle(RenderContext2D.FONT_STYLE_NORMAL);
 63 	},
 64 
 65 	/**
 66 	 * @private
 67 	 */
 68    release: function() {
 69       this.base();
 70 		this.tInit();
 71    },
 72 
 73    /**
 74     * Calculate the bounding box for the text and set it on the host object.
 75     * @private
 76     */
 77    calculateBoundingBox: function() {
 78       return;
 79    },
 80 
 81    /**
 82     * @private
 83     */
 84    execute: function(renderContext, time) {
 85 
 86       if (this.getText().length == 0)
 87       {
 88          return;
 89       }
 90 
 91 		renderContext.setFontStyle(this.getTextStyle());
 92 		renderContext.setFontAlign(this.getTextAlignment());
 93 		renderContext.setFontWeight(this.getTextWeight());
 94 		renderContext.setFont(this.getTextFont());
 95 		
 96 		renderContext.setFillStyle(this.getColor());
 97       renderContext.drawText(Point2D.ZERO, this.getText());
 98    }
 99 	
100 }, /** @scope BitmapText.prototype */{
101    /**
102     * Get the class name of this object
103     * @return {String} The string "BitmapText"
104     */
105    getClassName: function() {
106       return "ContextText";
107    }
108 });
109 
110 return ContextText;
111 
112 });
