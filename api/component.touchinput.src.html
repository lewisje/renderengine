component.touchinput.js


Summary

An extension of the input component to handle touch inputs from devices which support them.
Class Summary
Touch An object which represents a touch event generated by a {@link TouchInputComponent}.
TouchInputComponent A component which responds to touch events and notifies its {@link HostObject} by calling one of four methods.

 38 Engine.initObject("TouchInputComponent", "InputComponent", function() {
 39 
 40 /**
 41  * @class A component which responds to touch events and notifies
 42  * its {@link HostObject} by calling one of four methods.  The <tt>HostObject</tt>
 43  * should implement any of the following methods to receive the corresponding event:
 44  * <ul>
 45  * <li><tt>onTouchStart()</tt> - A touch event started</li>
 46  * <li><tt>onTouchEnd()</tt> - A touch event ended</li>
 47  * <li><tt>onTouchMove()</tt> - A movement occurred after a touch event started</li>
 48  * <li><tt>onTouchCancel()</tt> - A touch event was cancelled</li>
 49  * </ul>
 50  * Each function should take up to two arguments.  The first argument is an array of
 51  * {@link Touch} objects which represent each touch that occurred in the event.  Some
 52  * platforms support multi-touch, so each touch will be represented in the array.  The 
 53  * second argument is the actual event object itself.
 54  *
 55  * @param name {String} The unique name of the component.
 56  * @param [passThru] {Boolean} set to <tt>true</tt> to pass the event to the device 
 57  * @param [priority] {Number} The priority of the component among other input components.
 58  * @extends InputComponent
 59  * @constructor
 60  * @description Create an instance of a touch input component. 
 61  */
 62 var TouchInputComponent = InputComponent.extend(/** @scope TouchInputComponent.prototype */{
 63 
 64    /**
 65     * @private
 66     */
 67    constructor: function(name, passThru, priority) {
 68 		passThru = (typeof passThru == "number" ? false : passThru);
 69 		priority = (typeof passThru == "number" ? passThru : null);
 70       this.base(name, priority);
 71 
 72       var ctx = Engine.getDefaultContext();
 73       var self = this;
 74 
 75       // Add the event handlers
 76       ctx.addEvent(this, "touchstart", function(evt) {
 77 			if (!passThru) {
 78 	         evt.preventDefault();
 79 			}
 80          return self._touchStartListener(evt);
 81       });
 82       ctx.addEvent(this, "touchend", function(evt) {
 83 			if (!passThru) {
 84 	         evt.preventDefault();
 85 			}
 86          return self._touchEndListener(evt);
 87       });
 88       ctx.addEvent(this, "touchmove", function(evt) {
 89 			if (!passThru) {
 90 	         evt.preventDefault();
 91 			}
 92          return self._touchMoveListener(evt);
 93       });
 94       ctx.addEvent(this, "touchcancel", function(evt) {
 95 			if (!passThru) {
 96 	         evt.preventDefault();
 97 			}
 98          return self._touchCancelListener(evt);
 99       });
100    },
101 
102    /**
103     * Destroy this instance and remove all references.
104     * @private
105     */
106    destroy: function() {
107       var ctx = Engine.getDefaultContext();
108 
109       // Clean up event handlers
110       ctx.removeEvent(this, "touchstart");
111       ctx.removeEvent(this, "touchend");
112       ctx.removeEvent(this, "touchmove");
113       ctx.removeEvent(this, "touchcancel");
114       this.base();
115    },
116 
117    /**
118     * Process the touches and pass an array of touch objects to be handled by the
119     * host object.
120     * @private
121     */
122    processTouches: function(eventObj) {
123       var touches = [];
124 		if (eventObj.touches) {
125 	      for (var i = 0; i < eventObj.touches.length; i++) {
126 	         touches.push(new Touch(eventObj.touches[i]));
127 	      }
128 		}
129       return touches;
130    },
131 
132    /**
133     * @private
134     */
135    _touchStartListener: function(eventObj) {
136       if (this.getHostObject().onTouchStart) {
137          return this.getHostObject().onTouchStart(this.processTouches(eventObj.originalEvent), eventObj);
138       }
139    },
140 
141    /**
142     * @private
143     */
144    _touchEndListener: function(eventObj) {
145       if (this.getHostObject().onTouchEnd) {
146          return this.getHostObject().onTouchEnd(this.processTouches(eventObj.originalEvent), eventObj);
147       }
148    },
149 
150    /**
151     * @private
152     */
153    _touchMoveListener: function(eventObj) {
154       if (this.getHostObject().onTouchMove) {
155          return this.getHostObject().onTouchMove(this.processTouches(eventObj.originalEvent), eventObj);
156       }
157    },
158 
159    /**
160     * @private
161     */
162    _touchCancelListener: function(eventObj) {
163       if (this.getHostObject().onTouchCancel) {
164          return this.getHostObject().onTouchCancel(this.processTouches(eventObj.originalEvent), eventObj);
165       }
166    }
167 
168 }, /** @scope TouchInputComponent.prototype */{
169    /**
170     * Get the class name of this object
171     *
172     * @return {String} "TouchInputComponent"
173     */
174    getClassName: function() {
175       return "TouchInputComponent";
176    },
177    
178    RECORD_PART: ["shiftKey","ctrlKey","altKey","keyCode"]
179 });
180 
181 return TouchInputComponent;
182 
183 });

186 Engine.initObject("Touch", null, function() {
187 
188 
189    /**
190     * @class An object which represents a touch event generated by a {@link TouchInputComponent}.
191     *        Since touches are handled differently across platforms, this class aims to
192     *        normalize what is returned and allow a game to handle the input without concern
193     *        for platform.
194     *
195     * @param touch {Event} The touch from the event object
196     * @extends InputComponent
197     * @constructor
198     * @description Create a touch object.
199     */
200    var Touch = Base.extend(/** @scope Touch.prototype */{
201 
202       touchX: null,
203       touchY: null,
204 
205       /**
206        * @private
207        */
208       constructor: function(touch) {
209          this.touchX = touch.pageX;
210          this.touchY = touch.pageY;
211       },
212 
213       /**
214        * Get the X coordinate of the touch
215        * @return {Number} The X coordinate of the touch
216        */
217       getX: function() {
218          return this.touchX;
219       },
220 
221       /**
222        * Get the Y coordinate of the touch
223        * @return {Number} The Y coordinate of the touch
224        */
225       getY: function() {
226          return this.touchY;
227       },
228 
229       /**
230        * Get a {@link Point2D} which represents the location of the touch
231        * @return {Point2D} The coordinates of the touch
232        */
233       get: function() {
234          return Point2D.create(this.touchX, this.touchY);
235       }
236 
237    }, /** @scope TouchInputComponent.prototype */{
238       /**
239        * Get the class name of this object
240        *
241        * @return {String} "TouchInputComponent"
242        */
243       getClassName: function() {
244          return "Touch";
245       }
246    });
247 
248    return Touch;
249 
250 });
