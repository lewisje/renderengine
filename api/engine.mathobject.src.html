 33 // Includes
 34 Engine.include("/engine/engine.pooledobject.js");
 35 
 36 Engine.initObject("MathObject", "PooledObject", function() {
 37 
 38 /**
 39  * @class The base object class which represents a math object within
 40  * the engine.  All math objects should extend from this class mainly due to
 41  * the fact that the engine can switch between pooling the object and running
 42  * transiently.
 43  * <p/>
 44  * Google's Chrome browser seems to operate better with transient objects while
 45  * other browsers appear to run better with pooled objects.
 46  * <p/>
 47  * 
 48  * @param name {String} The name of the object
 49  * @extends PooledObject
 50  * @constructor
 51  * @description Create a math object. 
 52  */
 53 var MathObject = PooledObject.extend(/** @scope MathObject.prototype */{
 54 
 55    constructor: function(name) {
 56       if (!MathObject.isTransient) {
 57          this.base(name);
 58       }
 59    },
 60 
 61    /**
 62     * Destroy this object instance (remove it from the Engine).  The object's release
 63     * method is called after destruction so it will be returned to the pool of objects 
 64     * to be used again.
 65     */
 66    destroy: function() {
 67       if (!MathObject.isTransient) {
 68          this.base();
 69       }
 70    }
 71 
 72  }, /** @scope BaseObject.prototype */{
 73 
 74    /**
 75     * Similar to a constructor, all pooled objects implement this method.
 76     * The <tt>create()</tt> method will either create a new instance, if no object of the object's
 77     * class exists within the pool, or will reuse an existing pooled instance of
 78     * the object.  Either way, the constructor for the object instance is called so that
 79     * instance creation can be maintained in the constructor.
 80     * <p/>
 81     * Usage: <tt>var obj = [ObjectClass].create(arg1, arg2, arg3...);</tt>
 82     *
 83     * @memberOf PooledObject
 84     */
 85    create: function() {
 86       if (MathObject.isTransient) {
 87          return new this(arguments[0],arguments[1],arguments[2],arguments[3],arguments[4],
 88                          arguments[5],arguments[6],arguments[7],arguments[8],arguments[9],
 89                          arguments[10],arguments[11],arguments[12],arguments[13],arguments[14]);
 90       } else {
 91          return PooledObject.create.apply(this, arguments);
 92       }
 93    },
 94 
 95    /**
 96     * Get the class name of this object
 97     *
 98     * @return {String} "BaseObject"
 99     */
100    getClassName: function() {
101       return "MathObject";
102    },
103    
104    // Transient math objects aren't pooled
105    isTransient: false
106 
107 });
108 
109 // On certain browsers, we want to let math objects die and not pool them
110 switch (EngineSupport.sysInfo().browser) {
111    case "chrome" : MathObject.isTransient = true;
112                    break;
113    case "mozilla" : if (EngineSupport.sysInfo().version.indexOf("1.9") == 0) MathObject.isTransient = true;
114                    break;
115 }
116 
117 return MathObject;
118 
119 });
