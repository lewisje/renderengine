engine.spatialcontainer.js


Summary

Spatial containers maintain a collection of objects and can report on potential objects within a defined space of that container.
Class Summary
SpatialContainer An abstract class to represent spatial containers.
SpatialNode A single node within a spatial container.

 47 /**
 48  * @class A single node within a spatial container.  Has an index for fast node
 49  *        comparisons, and a list of objects contained within the node.

 55 
 56    idx: 0,
 57    objects: null,

 75    /**
 76     * Get a Container which is all objects within this node.

 83 	/**

 91    /**
 92     * Add an object to this node.

107    },

109    /**

135 Engine.initObject("SpatialContainer", "BaseObject", function() {
136 
137 /**
138  * @class An abstract class to represent spatial containers.  Spatial containers
139  *        contain game-world objects and can report on potential objects within a defined
140  *        space of that container.
141  *
142  * @param name {String} The name of the container
143  * @param width {Number} The width of the container
144  * @param height {Number} The height of the container
145  * @extends BaseObject
146  * @constructor
147  * @description Create a spatial container
148  */
149 var SpatialContainer = BaseObject.extend(/** @scope SpatialContainer.prototype */{
150 
151    root: null,
152    width: 0,
153    height: 0,
154 	pcl: null,
155 
156    /**
157     * @private
158     */
159    constructor: function(name, width, height) {
160       this.base(name || "SpatialContainer");
161       this.width = width;
162       this.height = height;
163 		this.pcl = Container.create();
164    },
165 
166    /**
167     * @private
168     */
169    release: function() {
170       this.base();
171       this.root = null;
172       this.width = 0;
173       this.height = 0;
174 		this.pcl = null;
175    },
176 
177    /**
178     * Get the width of the container.
179     * @return {Number} The width
180     */
181    getWidth: function() {
182       return this.width;
183    },
184 
185    /**
186     * Get the height of the container.
187     * @return {Number} The height
188     */
189    getHeight: function() {
190       return this.height;
191    },
192 
193    /**
194     * Get the root of the container.
195     * @return {Object} The root
196     */
197    getRoot: function() {
198       return this.root;
199    },
200 
201    /**
202     * Set the root of the container.
203     *
204     * @param root {Object} The root object of this container
205     */
206    setRoot: function(root) {
207       this.root = root;
208    },
209 
210    /**
211     * [ABSTRACT] Find the node that contains the specified point.
212     *
213     * @param point {Point2D} The point to locate the node for
214     * @return {SpatialNode}
215     */
216    findNodePoint: function(point) {
217    	return null;
218    },
219    
220    /**
221     * Add an object to the node which corresponds to the position
222     * provided.  Adding an object at a specific point will remove it from whatever
223     * node it was last in.
224     *
225     * @param obj {BaseObject} The object to add to the collision model
226     * @param point {Point2D} The world position where the object is
227     */
228    addObject: function(obj, point) {
229    	// See if the object is already in a node and remove it
230    	var oldNode = this.getObjectSpatialData(obj, "lastNode");
231    	if (oldNode != null) {
232    		if (!oldNode.contains(point)) {
233    			// The node is no longer in the same node
234    			oldNode.removeObject(obj);
235    		} else {
236    			// The object hasn't left the node
237    			return;
238    		}
239    	}
240    	
241    	// Find the node by position and add the object to it
242    	var node = this.findNodePoint(point);
243    	if (node != null) {
244 			node.addObject(obj);
245 
246 			// Update the collision data on the object
247 			this.setObjectSpatialData(obj, "lastNode", node); 
248 		}
249    },
250    
251 	/**
252 	 * Get the spatial model data for the object.  If <tt>key</tt> is provided, only the
253 	 * data for <tt>key</tt> will be returned.  If the data has not yet been assigned,
254 	 * an empty object will be created to contain the data.
255 	 * 
256 	 * @param obj {BaseObject} The object which has the data
257 	 * @param [key] {String} Optional key which contains the data, or <tt>null</tt> for the
258 	 * 	entire data model. 
259 	 */
260 	getObjectSpatialData: function(obj, key) {
261 		var mData = obj.getObjectDataModel(SpatialContainer.DATA_MODEL);
262 		if (mData == null) {
263 			obj.setObjectDataModel(SpatialContainer.DATA_MODEL, {});
264 			mData = obj.getObjectDataModel(SpatialContainer.DATA_MODEL);
265 		}
266 		return key ? mData[key] : mData;
267 	},
268 	
269 	/**
270 	 * Set a key, within the object's spatial data model, to a specific value.
271 	 * 
272 	 * @param obj {BaseObject} The object to receive the data
273 	 * @param key {String} The key to set the data for
274 	 * @param value {Object} The value to assign to the key
275 	 */
276 	setObjectSpatialData: function(obj, key, value) {
277 		var mData = this.getObjectSpatialData(obj);
278 		mData[key] = value;
279 	},
280 	
281 	/**
282 	 * Clear all of the spatial container model data.
283 	 * 
284 	 * @param obj {BaseObject} The object which has the data model
285 	 */
286 	clearObjectSpatialData: function(obj) {
287 		obj.setObjectDataMode(SpatialContainer.DATA_MODEL, null);
288 	},
289 	
290    /**
291     * Remove an object from the collision model.  This is done so collisions are
292     * no longer checked against this object.
293     *
294     * @param obj {BaseObject} The object to remove
295     */
296    removeObject: function(obj) {
297 		var oldNode = this.getObjectSpatialData(obj, "lastNode");
298 		if (oldNode != null) {
299 			oldNode.removeObject(obj);
300 		}
301 		obj[SpatialContainer.DATA_MODEL] = null;
302    },
303    
304    /**
305     * Returns a potential collision list of objects that are contained
306     * within the defined sub-space of the container.
307     *
308     * @param point {Point2D} The point to build with
309     * @return {Container} The PCL
310     */
311    getPCL: function(point) {
312       return this.pcl;
313    },
314 
315    /**
316     * Returns all objects within the spatial container.
317     * @return {Container} A container of all objects in the container
318     */
319    getObjects: function() {
320       return Container.create();
321    },
322    
323    /**
324     * Returns all objects within the spatial container of a particular
325     * class type.
326     * @return {Array} An array of objects in the container, filtered by class
327     */
328    getObjectsOfType: function(clazz) {
329       return this.getObjects().filter(function(obj) {
330          return clazz.isInstance(obj);
331       }, this);
332    }
333 
334 }, /** @scope SpatialContainer.prototype */{ 
335    /**
336     * Get the class name of this object
337     *
338     * @return {String} "SpatialContainer"
339     */
340    getClassName: function() {
341       return "SpatialContainer";
342    },
343 	
344 	/**
345 	 * @private
346 	 */
347 	DATA_MODEL: "SpatialContainer"
348 
349 });
350 
351 return SpatialContainer;
352 
353 });
