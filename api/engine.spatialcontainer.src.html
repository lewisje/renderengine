engine.spatialcontainer.js


Summary

Spatial containers maintain a collection of objects and can report on potential objects within a defined space of that container.
Class Summary
SpatialContainer An abstract class to represent spatial containers.
SpatialNode A single node within a spatial container.

136 Engine.initObject("SpatialContainer", "BaseObject", function() {
137 
138 /**
139  * @class An abstract class to represent spatial containers.  Spatial containers
140  *        contain game-world objects and can report on potential objects within a defined
141  *        space of that container.
142  *
143  * @param name {String} The name of the container
144  * @param width {Number} The width of the container
145  * @param height {Number} The height of the container
146  * @extends BaseObject
147  * @constructor
148  * @description Create a spatial container
149  */
150 var SpatialContainer = BaseObject.extend(/** @scope SpatialContainer.prototype */{
151 
152    root: null,
153 
154    width: 0,
155 
156    height: 0,
157 	
158 	pcl: null,
159 
160    /**
161     * @private
162     */
163    constructor: function(name, width, height) {
164       this.base(name || "SpatialContainer");
165       this.width = width;
166       this.height = height;
167 		this.pcl = Container.create();
168    },
169 
170    /**
171     * @private
172     */
173    release: function() {
174       this.base();
175       this.root = null;
176       this.width = 0;
177       this.height = 0;
178 		this.pcl = null;
179    },
180 
181    /**
182     * Get the width of the container.
183     * @return {Number} The width
184     */
185    getWidth: function() {
186       return this.width;
187    },
188 
189    /**
190     * Get the height of the container.
191     * @return {Number} The height
192     */
193    getHeight: function() {
194       return this.height;
195    },
196 
197    /**
198     * Get the root of the container.
199     * @return {Object} The root
200     */
201    getRoot: function() {
202       return this.root;
203    },
204 
205    /**
206     * Set the root of the container.
207     *
208     * @param root {Object} The root object of this container
209     */
210    setRoot: function(root) {
211       this.root = root;
212    },
213 
214    /**
215     * [ABSTRACT] Find the node that contains the specified point.
216     *
217     * @param point {Point2D} The point to locate the node for
218     * @return {SpatialNode}
219     */
220    findNodePoint: function(point) {
221    	return null;
222    },
223    
224    /**
225     * Add an object to the node which corresponds to the position
226     * provided.  Adding an object at a specific point will remove it from whatever
227     * node it was last in.
228     *
229     * @param obj {BaseObject} The object to add to the collision model
230     * @param point {Point2D} The world position where the object is
231     */
232    addObject: function(obj, point) {
233    	// See if the object is already in a node and remove it
234    	var oldNode = obj._COLLISION_MODEL_DATA && obj._COLLISION_MODEL_DATA.lastNode ? obj._COLLISION_MODEL_DATA.lastNode : null;
235    	if (oldNode != null) {
236    		if (!oldNode.contains(point)) {
237    			// The node is no longer in the same node
238    			oldNode.removeObject(obj);
239    		} else {
240    			// The object hasn't left the node
241    			return;
242    		}
243    	}
244    	
245    	// Find the node by position and add the object to it
246    	var node = this.findNodePoint(point);
247    	if (node != null) {
248 			node.addObject(obj);
249 
250 			// Update the collision data on the object
251 			obj._COLLISION_MODEL_DATA = { 
252 				lastNode: node 
253 			};
254 		}
255    },
256    
257    /**
258     * Remove an object from the collision model.  This is done so collisions are
259     * no longer checked against this object.
260     *
261     * @param obj {BaseObject} The object to remove
262     */
263    removeObject: function(obj) {
264 		var oldNode = obj._COLLISION_MODEL_DATA && obj._COLLISION_MODEL_DATA.lastNode ? obj._COLLISION_MODEL_DATA.lastNode : null;
265 		if (oldNode != null) {
266 			oldNode.removeObject(obj);
267 		}
268 		obj._COLLISION_MODEL_DATA = null;
269    },
270    
271    /**
272     * Returns a potential collision list of objects that are contained
273     * within the defined sub-space of the container.
274     *
275     * @param point {Point2D} The point to build with
276     * @return {Container} The PCL
277     */
278    getPCL: function(point) {
279       return this.pcl;
280    },
281 
282    /**
283     * Returns all objects within the spatial container.
284     * @return {Container} A container of all objects in the container
285     */
286    getObjects: function() {
287       return Container.create();
288    },
289    
290    /**
291     * Returns all objects within the spatial container of a particular
292     * class type.
293     * @return {Array} An array of objects in the container, filtered by class
294     */
295    getObjectsOfType: function(clazz) {
296       return this.getObjects().filter(function(obj) {
297          return clazz.isInstance(obj);
298       }, this);
299    }
300 
301 }, /** @scope SpatialContainer.prototype */{ 
302    /**
303     * Get the class name of this object
304     *
305     * @return {String} "SpatialContainer"
306     */
307    getClassName: function() {
308       return "SpatialContainer";
309    }
310 
311 });
312 
313 return SpatialContainer;
314
