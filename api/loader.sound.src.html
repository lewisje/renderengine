loader.sound.js


Summary

A resource loader for MP3 sounds and a sound class for working with the loaded sounds via SoundManager 2.
Class Summary
Sound Represents a sound object that is abstracted from the sound system.
SoundLoader Loads sounds and stores the reference to them.

 40 Engine.initObject("SoundLoader", "RemoteLoader", function() {

 42 /**
 43  * @class Loads sounds and stores the reference to them.  Provides a simple
 44  *        way to play and stop the sounds that have been loaded.
 45  *        <p/>
 46  *        Sounds must be 44.1KHz and have a bitrate of 192k to play correctly.
 47  *
 48  * @constructor
 49  * @param name {String=SoundLoader} The name of the resource loader
 50  * @extends RemoteLoader
 51  */
 52 var SoundLoader = RemoteLoader.extend(/** @scope SoundLoader.prototype */{
 53 
 54    init: false,
 55 
 56    queuedSounds: null,
 57 
 58    checkReady: null,
 59 
 60    soundManager: null,
 61 
 62    queueingSounds: true,
 63 
 64    loadingSounds: 0,
 65 
 66    /**
 67     * @private
 68     */
 69    constructor: function(name) {
 70       this.base(name || "SoundLoader");
 71       this.init = false;
 72       this.queuedSounds = [];
 73       this.queueingSounds = true;
 74       this.loadingSounds = 0;
 75 
 76       if (typeof SoundManager != "undefined") {
 77 
 78          // Create a link to the object
 79          this.soundManager = window.soundManager;
 80          this.soundManager.debugMode = false;
 81 
 82          // directory where SM2 .SWFs live
 83          this.soundManager.url = Engine.getEnginePath() + '/libs/';
 84 			
 85 			if (GetSwfVer() != null) {
 86 				// Detect the version of flash available.  If 9 or higher, use 9
 87 				var hasReqestedVersion = DetectFlashVer(9, 0, 0);
 88 				if (hasReqestedVersion) {
 89 					this.soundManager.flashVersion = 9;
 90 				} else {
 91 					this.soundManager.flashVersion = 8;
 92 				}
 93 
 94 	         // Debugging enabled?
 95 	         this.soundManager.debugMode = EngineSupport.checkBooleanParam("debugSound");
 96 	
 97 	         var self = this;
 98 	
 99 	         this.soundManager.onload = function() {
100 	            Engine.soundsEnabled = true;
101 	            Console.warn("SoundManager loaded successfully");
102 	            self.queueingSounds = false
103 	            self.loadQueuedSounds();
104 	         };
105 	
106 	         this.soundManager.onerror = function() {
107 	            Engine.soundsEnabled = false;
108 	            Console.warn("SoundManager not loaded");
109 	            self.queueingSounds = false;
110 	            self.loadQueuedSounds();
111 	         };
112 	
113 	         if (Engine.getEnginePath().indexOf("file:") == 0) {
114 	            this.soundManager.sandbox.type = "localWithFile";
115 	         }
116 	
117 	         this.soundManager.go();
118 
119 			} else {
120 				// Flash not installed
121 				Engine.soundsEnabled = false;
122 				this.queueingSounds = false;
123 			}
124 
125       } else {
126 			this.queueingSounds = false;
127          Engine.soundsEnabled = false;
128       }
129    },
130 
131    destroy: function() {
132       // Stop the sound manager
133       if (Engine.isSoundEnabled()) {
134          this.soundManager.destruct();
135       }
136    },
137 
138    /**
139     * Load a sound resource from a URL. If the sound system does not initialize, for whatever
140     * reason, you can still call the sound's methods.
141     *
142     * @param name {String} The name of the resource
143     * @param url {String} The URL where the resource is located
144     */
145    load: function(name, url) {
146 
147       if (this.queueingSounds) {
148          this.queuedSounds.push({n: name, u: url});
149          return;
150       }
151 
152       var soundObj = null;
153 
154       if (Engine.isSoundEnabled()) {
155 
156          // Only MP3 files are supported
157          Assert(url.indexOf(".mp3") > 0, "Only MP3 sound format is supported!");
158 
159          // Create the sound object
160          var sound = this.soundManager.createSound({
161             "id": name,
162             "url": url,
163             "autoPlay": false,
164             "autoLoad": true,
165             "volume": 50
166          });
167          soundObj = Sound.create(name, sound);
168 
169          // We'll need to periodically check a sound's "readyState" for success
170          // to know when the sound is ready for usage.
171          this.loadingSounds++;
172          if (!this.checkReady) {
173             var self = this;
174             this.checkReady = window.setTimeout(function() {
175                var sounds = self.getResources();
176                for (var s in sounds) {
177                   if (!self.isReady(sounds[s]) && self.get(sounds[s]).smSound.readyState == SoundLoader.LOAD_SUCCESS) {
178                      self.setReady(sounds[s], true);
179                      self.loadingSounds--;
180                   } else if (self.get(sounds[s]).smSound.readyState == SoundLoader.LOAD_ERROR) {
181                      Console.error("An error occurred loading the sound ", sounds[s]);
182                   }
183                }
184 
185                if (self.loadingSounds != 0) {
186                   // There are still sounds loading
187                   self.checkReady = window.setTimeout(arguments.callee, 500);
188                } else {
189                   window.clearTimeout(self.checkReady);
190                   self.checkReady = null;
191                }
192             }, 500);
193          }
194 
195          if (!this.init) {
196             Engine.getDefaultContext().add(this);
197             this.init = true;
198          }
199       } else {
200          soundObj = Sound.create(name, null);
201       }
202 
203       this.base(name, soundObj);
204    },
205 
206    loadQueuedSounds: function() {
207       for (var s in this.queuedSounds) {
208          this.load(this.queuedSounds[s].n, this.queuedSounds[s].u);
209       }
210       this.queuedSounds = null;
211    },
212 
213    /**
214     * Unload a sound, calling the proper methods in SoundManager2.
215     *
216     * @param sound {String} The name of the sound to unload
217     */
218    unload: function(sound) {
219       var s = this.get(sound).destroy();
220       this.base(sound);
221    },
222 
223    /**
224     * Creates a {@link Sound} object representing the named sound.
225     *
226     * @param resource {String} A loaded sound resource
227     * @param sound {String} The name of the sound from the resource
228     * @return {Sound} A {@link Sound} instance
229     */
230    getSound: function(sound) {
231       return this.get(sound);
232    },
233 
234    /**
235     * The name of the resource this loader will get.
236     * @return {String} The string "sound"
237     */
238    getResourceType: function() {
239       return "sound";
240    }
241 }, /** @scope SoundResourceLoader.prototype */{
242    /**
243     * Get the class name of this object
244     * @return {String} The string "SoundResourceLoader"
245     */
246    getClassName: function() {
247       return "SoundLoader";
248    },
249 
250    /**
251     * Indicates that a sound is loading.
252     * @type Number
253     */
254    LOAD_LOADING: 1,
255 
256    /**
257     * Indicates an error state during sound loading.
258     * @type Number
259     */
260    LOAD_ERROR: 2,
261 
262    /**
263     * Indicates a successful sound load.
264     * @type Number
265     */
266    LOAD_SUCCESS: 3
267 });
268 
269 return SoundLoader;

273 Engine.initObject("Sound", "PooledObject", function() {
274 
275 /**
276  * @class Represents a sound object that is abstracted from the sound system.
277  *        In the current case, sound is loaded via SoundManager2 which is implemented
278  *        with a flash bridge.  If the sound system does not initialize, for whatever
279  *        reason, you can still call the sound's methods.
280  *
281  * @constructor
282  * @param name {String} The name of the sound
283  * @param soundObj {SMSound} A sound object returned from SoundManager2
284  * @extends PooledObject
285  */
286 var Sound = PooledObject.extend(/** @scope Sound.prototype */{
287 
288    smSound: null,
289 
290    volume: -1,
291 
292    paused: false,
293 
294    pan: -1,
295 
296    muted: false,
297 
298    /**
299     * @private
300     */
301    constructor: function(name, soundObj) {
302       this.smSound = soundObj;
303       this.volume = 50;
304       this.paused = false;
305       this.pan = 0;
306       this.muted = false;
307       return this.base(name);
308    },
309 
310    /**
311     * Destroy the sound object
312     */
313    destroy: function() {
314       this.smSound.unload();
315       this.base();
316    },
317 
318    /**
319     * @private
320     */
321    release: function() {
322       this.base();
323       this.volume = -1;
324       this.pan = -1;
325       this.paused = false;
326       this.muted = false;
327       this.smSound = null;
328    },
329 
330    /**
331     * Play the sound.  If the volume is specified, it will set volume of the
332     * sound before playing.  If the sound was paused, it will be resumed.
333     *
334     * @param volume {Number} <i>[optional]</i> An integer between 0 (muted) and 100 (full volume)
335     */
336    play: function(volume) {
337       if (this.paused) {
338          this.resume();
339          return;
340       }
341 
342       if (volume && volume != this.getVolume()) {
343          this.setVolume(volume);
344       }
345 
346       if (Engine.isSoundEnabled()) {
347          this.smSound.play();
348       }
349    },
350 
351    /**
352     * If the sound is playing, stop the sound and reset it to the beginning.
353     */
354    stop: function() {
355       this.paused = false;
356       if (Engine.isSoundEnabled()) {
357          this.smSound.stop();
358       }
359    },
360 
361    /**
362     * If the sound is playing, pause the sound.
363     */
364    pause: function() {
365       this.paused = true;
366       if (Engine.isSoundEnabled()) {
367          this.smSound.pause();
368       }
369    },
370 
371    /**
372     * Returns <tt>true</tt> if the sound is currently paused.
373     * @return {Boolean} <tt>true</tt> if the sound is paused
374     */
375    isPaused: function() {
376       return this.paused;
377    },
378 
379    /**
380     * If the sound is paused, it will resume playing the sound.
381     */
382    resume: function() {
383       this.paused = false;
384       if (Engine.isSoundEnabled()) {
385          this.smSound.resume();
386       }
387    },
388 
389    /**
390     * Mute the sound (set its volume to zero).
391     */
392    mute: function() {
393       this.muted = true;
394       if (Engine.isSoundEnabled()) {
395          this.smSound.mute();
396       }
397    },
398 
399    /**
400     * Unmute the sound (reset its volume to what it was before muting).
401     */
402    unmute: function() {
403       if (!this.muted) {
404          return;
405       }
406 
407       this.muted = false;
408       if (Engine.isSoundEnabled()) {
409          this.smSound.unmute();
410       }
411    },
412 
413    /**
414     * Set the volume of the sound to an integer between 0 (muted) and 100 (full volume).
415     *
416     * @param volume {Number} The volume of the sound
417     */
418    setVolume: function(volume) {
419       if (isNaN(volume)) {
420          return;
421       }
422 
423       // clamp it
424       volume = (volume < 0 ? 0 : volume > 100 ? 100 : volume);
425       this.volume = volume;
426       if (Engine.isSoundEnabled()) {
427          this.smSound.setVolume(volume);
428       }
429    },
430 
431    /**
432     * Get the volume the sound is playing at.
433     * @return {Number} An integer between 0 and 100
434     */
435    getVolume: function() {
436       return this.volume;
437    },
438 
439    /**
440     * Set the pan of the sound, with -100 being full left and 100 being full right.
441     *
442     * @param pan {Number} An integer between -100 and 100, with 0 being center.
443     */
444    setPan: function(pan) {
445       this.pan = pan;
446       if (Engine.isSoundEnabled()) {
447          this.smSound.setPan(pan);
448       }
449    },
450 
451    /**
452     * Get the pan of the sound, with -100 being full left and 100 being full right.
453     * @return {Number} An integer between -100 and 100
454     */
455    getPan: function() {
456       return this.pan;
457    },
458 
459    /**
460     * Set the sound offset in milliseconds.
461     *
462     * @param millisecondOffset {Number} The offset into the sound to play from
463     */
464    setPosition: function(millisecondOffset) {
465       this.position = millisecondOffset;
466       if (Engine.isSoundEnabled()) {
467          this.smSound.setPosition(millisecondOffset);
468       }
469    },
470 
471    /**
472     * Get the position of the sound, in milliseconds, from the start of the sound.
473     * @return {Number} The millisecond offset into the sound
474     */
475    getLastPosition: function() {
476       if (Engine.isSoundEnabled()) {
477          this.position = this.smSound.position;
478       }
479       return this.position;
480    },
481 
482    /**
483     * Get the total size, in bytes, of the sound.  If the sound engine is not
484     * initialized, returns 0.
485     * @return {Number} The size of the sound, in bytes
486     */
487    getSizeBytes: function() {
488       if (Engine.isSoundEnabled()) {
489          return this.smSound.bytesTotal;
490       }
491       return 0;
492    },
493 
494    /**
495     * Get the length of the sound, in milliseconds.  If the sound hasn't fully loaded,
496     * it will be the number of milliseconds currently loaded.  Due to the nature of
497     * Variable Bitrate (VBR) sounds, this number may be inaccurate.
498     * @return {Number} The length of the sound, in milliseconds
499     */
500    getDuration: function() {
501       if (Engine.isSoundEnabled()) {
502          return this.smSound.duration;
503       }
504       return 0;
505    },
506 
507    getReadyState: function() {
508       if (Engine.isSoundEnabled()) {
509          return this.smSound.readyState;
510       }
511       return 0;
512    }
513 
514 }, /** @scope Sound.prototype */{
515    /**
516     * Gets the class name of this object.
517     * @return {String} The string "Sound"
518     */
519    getClassName: function() {
520       return "Sound";
521    }
522 });
523 
524 return Sound;
525
