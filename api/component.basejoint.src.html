Summary

The base component type for all physical joints which can be used in a {@link Simulation}.
Class Summary
BaseJointComponent The base physical joint component which initializes physical joints for use in a {@link Simulation}.

 40 Engine.initObject("BaseJointComponent", "LogicComponent", function() {
 41 
 42 /**
 43  * @class The base physical joint component which initializes physical joints
 44  * 		 for use in a {@link Simulation}.  
 45  *
 46  * @param name {String} Name of the component
 47  * @param shapeDef {b2ShapeDef} The shape definition. Either {@link b2CircleDef}, {@link b2BoxDef}, or
 48  * 			{@link b2PolyDef}.
 49  *
 50  * @extends LogicComponent
 51  * @constructor
 52  * @description All physical joint components should extend from this component type.
 53  */
 54 var BaseJointComponent = LogicComponent.extend(/** @scope BaseJointComponent.prototype */{
 55 
 56 	jointDef: null,
 57 	simulation: null,
 58 	collideBodies: false,
 59 	joint: null,
 60 	body1: null,
 61 	body2: null,
 62 
 63    /**
 64     * @private
 65     */
 66 	constructor: function(name, body1, body2, jointDef) {
 67 		Assert(BaseBodyComponent.isInstance(body1), "Body 1 for joint is not a BaseBodyComponent");
 68 		Assert(BaseBodyComponent.isInstance(body2), "Body 2 for joint is not a BaseBodyComponent");
 69 
 70 		this.base(name || "BaseJoint");	
 71 
 72 		this.jointDef = jointDef;
 73 		this.simulation = null;
 74 		this.body1 = body1;
 75 		this.body2 = body2;
 76 		
 77 		this.collideBodies = false;
 78 	},
 79 
 80 	/**
 81 	 * Start simulating the joint.  If the joint isn't a part of the simulation,
 82 	 * it is added and simulation occurs.
 83 	 */
 84 	startSimulation: function() {
 85 		if (!this.simulation) {
 86 			this.simulation = this.getHostObject().getSimulation();
 87 			this.getJointDef().body1 = this.body1.getBody();
 88 			this.getJointDef().body2 = this.body2.getBody();
 89 			this.getJointDef().collideConnected = this.getCollideBodies();
 90 
 91 			this.joint = this.simulation.addJoint(this.getJointDef());
 92 		}
 93 	},
 94 	
 95 	/**
 96 	 * Stop simulating the body.  If the body is a part of a simulation,
 97 	 * it is removed and simulation stops.  The position and rotation of
 98 	 * the body will not be updated.
 99 	 */
100 	stopSimulation: function() {
101 		if (this.simulation) {
102 			this.simulation.removeJoint(this.getJoint());			
103 			this.simulation = null;
104 		}
105 	},
106 	
107 	getSimulation: function() {
108 		return this.simulation;
109 	},
110 	
111 	/**
112 	 * Get the Box2d joint definition object.
113 	 * @return {b2JointDef}
114 	 */
115 	getJointDef: function() {
116 		return this.jointDef;
117 	},
118 
119 	/**
120 	 * Get the Box2d joint object.
121 	 * @return {b2Joint}
122 	 */
123 	getJoint: function() {
124 		return this.joint;
125 	},
126 
127 	/**
128 	 * Get the component which corresponds to body 1 of the joint.
129 	 * @return {BaseBodyComponent}
130 	 */
131 	getBody1: function() {
132 		return this.body1;
133 	},
134 
135 	/**
136 	 * Get the component which corresponds to body 2 of the joint.
137 	 * @return {BaseBodyComponent}
138 	 */
139 	getBody2: function() {
140 		return this.body2;
141 	},
142 	
143 	/**
144 	 * Returns <code>true</code> if the two bodies will check for collisions
145 	 * with eachother during the simulation.  The default is <code>false</code>.
146 	 * @return {Boolean}
147 	 */
148 	getCollideBodies: function() {
149 		return this.collideBodies;
150 	},
151 	
152 	/**
153 	 * Set a flag which determines if the two bodies will collide with eachother
154 	 * during simulation.  After the simulation has started, you cannot change this
155 	 * flag without first stopping simulation on the bodies and joint.
156 	 * 
157 	 * @param state {Boolean} Set to <code>true</code> to enable collisions between the two
158 	 * 	bodies.  Default: <code>false</code>
159 	 */
160 	setCollideBodies: function(state) {
161 		this.collideBodies = state;		
162 	}
163 	
164 	/* pragma:DEBUG_START */
165 	/**
166 	 * Adds joint debugging
167 	 * @private
168 	 */	
169 	,execute: function(renderContext, time) {
170 		this.base(renderContext, time);
171 		if (Engine.getDebugMode()) {
172 			renderContext.pushTransform();
173 			renderContext.setLineStyle("red");
174 			var b1p = Point2D.create(this.getBody1().getPosition());
175 			var b2p = Point2D.create(this.getBody2().getPosition());
176 			renderContext.drawLine(b1p, b2p);
177 			b1p.destroy();
178 			b2p.destroy();
179 			
180 			/*// Draw the anchor point
181 			var a = Point2D.create(this.anchor);
182 			a.add(this.getBody1().getPosition());
183 			renderContext.setFillStyle("green");
184 			renderContext.drawFilledArc(a, 8, 0, 360);
185 			a.destroy();
186 			*/
187 			renderContext.popTransform();
188 		}	
189 	}
190 	/* pragma:DEBUG_END */
191 	
192 }, { /** @scope BaseJointComponent.prototype */
193 
194    /**
195     * Get the class name of this object
196     *
197     * @return {String} "BaseJointComponent"
198     */
199    getClassName: function() {
200       return "BaseJointComponent";
201    }   
202 });
203 
204 return BaseJointComponent;
205 
206 });
