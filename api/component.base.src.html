component.base.js


Summary

The base class from which all components extend. A component is a single part of the functionality used by a HostObject.

 40 /**
 41  * @class All components extend from this object class.  A component is one
 42  *        part of an operating whole object (a {@link HostObject}) which is
 43  *        responsible for only a portion of the overall operation of an in-
 44  *        game object.  Components are broken down into five major categories:
 45  *        <ul>
 46  *          <li><b>TYPE_INPUT</b> - Input from controllers (keyboard, mouse, etc.)</li>
 47  *          <li><b>TYPE_TRANSFORM</b> - Performs transformations on the host object</li>
 48  *          <li><b>TYPE_LOGIC</b> - Handles logical operations that are not related to
 49  *              input and collision</li>
 50  *          <li><b>TYPE_COLLIDER</b> - Determines what this object is possibly colliding
 51  *              with, and reports those collisions via callbacks to the host.</li>
 52  *          <li><b>TYPE_RENDER</b> - Performs some sort of rendering operation to the context</li>
 53  *        </ul>
 54  *        Components are executed in the order listed.  First, all inputs are
 55  *        checked, then logic is performed.  Logic may be internal to a host object
 56  *        itself, but some components perform an object-centric type of logic that
 57  *        can be reused.  Next, collisions are checked.  And finally, rendering can
 58  *        occur.
 59  *        <p/>
 60  *        Within each component type set, components can be prioritized so that
 61  *        one component will execute before others.  Such an ordering allows for
 62  *        multiple components of each type to perform their tasks in an order
 63  *        that the host defines.
 64  *
 65  *
 66  * @extends BaseObject
 67  * @constructor
 68  * @description Create a new instance of a component, setting the name, type, and
 69  *              update priority of this component compared to all other components
 70  *              within the host.
 71  * @param name {String} The name of the component
 72  * @param type {Number} The type of the component
 73  * @param priority {Number} A value between 0.0 and 1.0.  Default: 0.5
 74  */
 75 var BaseComponent = BaseObject.extend(/** @scope BaseComponent.prototype */{
 76 
 77    priority: 0,
 78 
 79    type: -1,
 80 
 81    host: null,
 82 
 83    /**
 84     * @private 
 85     */
 86    constructor: function(name, type, priority) {
 87       Assert((name != null), "You must assign a name to every Component.");
 88       name = name.toUpperCase();
 89 
 90       Assert((type != null && (type >= BaseComponent.TYPE_INPUT && type <= BaseComponent.TYPE_RENDERING)),
 91              "You must specify a type for BaseComponent");
 92 
 93       this.type = type;
 94 
 95       Assert((priority != null && (priority >= 0.0 && priority <= 1.0)),
 96              "Priority must be between 0.0 and 1.0 for BaseComponent");
 97 
 98       this.priority = priority || 0.5;
 99       this.base(name);
100    },
101 
102    /**
103     * Releases the object back into the object pool.  See {@link PooledObject#release}
104     * for more information.
105     */
106    release: function() {
107       this.base();
108       this.priority = 0;
109       this.type = -1;
110       this.host = null;
111    },
112 
113    /**
114     * Establishes the link between this component and its host object.
115     * When you assign components to a host object, it will call this method
116     * so that each component can refer to its host object, the same way
117     * a host object can refer to a component with {@link HostObject#getComponent}.
118     *
119     * @param hostObject {HostObject} The object which hosts this component
120     */
121    setHostObject: function(hostObject) {
122       this.host = hostObject;
123    },
124 
125    /**
126     * Gets the host object this component is a part of.  When the component was
127     * assigned to a host object, the host object will have set itself as the container
128     * via {@link #setHostObject}.
129     *
130     * @return {HostObject}
131     */
132    getHostObject: function() {
133       return this.host;
134    },
135 
136    /**
137     * Get the type of this component.  The value will be one of:
138     * {@link #TYPE_INPUT}, {@link #TYPE_TRANSFORM}, {@link #TYPE_LOGIC},
139     * {@link #TYPE_COLLIDER}, or {@link #TYPE_RENDERING}
140     *
141     * @type Number
142     */
143    getType: function() {
144       return this.type;
145    },
146 
147    /**
148     * Set the execution priority of this component with
149     * 1.0 being the highest priority and 0.0 being the lowest.  Components
150     * within a host object are sorted by type, and then priority.  As such,
151     * two components with the same type will be sorted by priority with the
152     * higer value executing before the lower value.  This allows you to layer
153     * components like the {@link #TYPE_RENDER} componenent so that one effect
154     * is drawn before another.
155     *
156     * @param priority {Number} A value between 0.0 and 1.0
157     */
158    setPriority: function(priority) {
159       this.priority = priority;
160       this.getHost().sort();
161    },
162 
163    /**
164     * Returns the priority of this component.
165     *
166     * @return {Number} A value between 0.0 and 1.0
167     */
168    getPriority: function() {
169       return this.priority;
170    },
171 
172    /**
173     * [ABSTRACT] This method is called by the host object to run the component, 
174     * updating its state.  Not all components will need an execute
175     * method.  However, it is important to include one if you need to 
176     * update the state of the component each engine cycle.
177     *
178     * @param renderContext {RenderContext} The context the component will render within.
179     * @param time {Number} The global engine time
180     */
181    execute: function(renderContext, time) {
182       // ABSTRACT METHOD DECLARATION
183    },
184 
185    /**
186     * Get the type of the component as a string.
187     * @return {String}
188     */
189    getTypeString: function() {
190       var ts = "";
191       switch (this.getType()) {
192          case BaseComponent.TYPE_INPUT: ts = "TYPE_INPUT"; break;
193          case BaseComponent.TYPE_TRANSFORM: ts = "TYPE_TRANSFORM"; break;
194          case BaseComponent.TYPE_LOGIC: ts = "TYPE_LOGIC"; break;
195          case BaseComponent.TYPE_COLLIDER: ts = "TYPE_COLLIDER"; break;
196          case BaseComponent.TYPE_RENDERING: ts = "TYPE_RENDERING"; break;
197          default: ts = "TYPE_UNKNOWN";
198       }
199 
200       return ts;
201    }
202 
203 
204 }, /** @scope BaseComponent.prototype */{
205 
206    /**
207     * Get the class name of this object
208     *
209     * @return {String} The string "BaseComponent"
210     */
211    getClassName: function() {
212       return "BaseComponent";
213    },
214 
215    /**
216     * The constant value for INPUT components.
217     * @type {Number}
218     */
219    TYPE_INPUT:          1,
220 
221    /**
222     * The constant value for TRANSFORM (movement) components.
223     * @type {Number}
224     */
225    TYPE_TRANSFORM:      2,
226 
227    /**
228     * The constant value for LOGIC components.
229     * @type {Number}
230     */
231    TYPE_LOGIC:          3,
232 
233    /**
234     * The constant value for COLLIDER components.
235     * @type {Number}
236     */
237    TYPE_COLLIDER:       4,
238 
239    /**
240     * The constant value for RENDERING components.
241     * @type {Number}
242     */
243    TYPE_RENDERING:      5
244 });
245 
246 return BaseComponent;
