127    /**
128     * Check to see if a line intersects another
129     *
130     * @param p1 {Point2D} Start of line 1
131     * @param p2 {Point2D} End of line 1
132     * @param p3 {Point2D} Start of line 2
133     * @param p4 {Point2D} End of line 2
134     * @return {Boolean} <tt>true</tt> if the lines intersect
135     */
136    lineLineCollision: function(p1, p2, p3, p4) {
137       var d = ((p4.y - p3.y) * (p2.x - p1.x)) - ((p4.x - p3.x) * (p2.y - p1.y));
138       var n1 = ((p4.x - p3.x) * (p1.y - p3.y)) - ((p4.y - p3.y) * (p1.x - p3.x));
139       var n2 = ((p2.x - p1.x) * (p1.y - p3.y)) - ((p2.y - p1.y) * (p1.x - p3.x));
140 
141       if ( d == 0.0 )
142       {
143          if ( n1 == 0.0 && n2 == 0.0 )
144          {
145             return false;  //COINCIDENT;
146          }
147          return false;   // PARALLEL;
148       }
149       var ua = n1 / d;
150       var ub = n2 / d;
151 
152       return (ua >= 0.0 && ua <= 1.0 && ub >= 0.0 && ub <= 1.0);
153    },

203    /**
204     * A static method used to calculate a direction vector
205     * from a heading angle.
206     *
207     * @param origin {Point2D} The origin of the shape
208     * @param baseVec {Vector2D} The base vector
209     * @param angle {Number} The rotation in degrees
210     * @return {Vector2D} The direction vector
211     */
212    getDirectionVector: function(origin, baseVec, angle) {
213       var r = Math2D.degToRad(angle);
214 
215       var x = Math.cos(r) * baseVec.x - Math.sin(r) * baseVec.y;
216       var y = Math.sin(r) * baseVec.x + Math.cos(r) * baseVec.y;
217 
218       var v = Vector2D.create(x, y).sub(origin);
219       return v.normalize();
220    },

262 	/**
263 	 * Calculate an approximate 2D convex hull for the given array of points.
264 	 * <p/>
265 	 * Copyright 2001, softSurfer (www.softsurfer.com)
266 	 * This code may be freely used and modified for any purpose
267 	 * providing that this copyright notice is included with it.
268 	 * SoftSurfer makes no warranty for this code, and cannot be held
269 	 * liable for any real or imagined damage resulting from its use.
270 	 * Users of this code must verify correctness for their application.
271 	 * 
272 	 * @param points {Array} An array of {@link Point2D} instances
273 	 * @param k {Number} The approximation accuracy (larger = more accurate)
274 	 * @return {Array} An array of {@link Point2D} which contains the 
275 	 * 	approximate hull of the given points
276 	 */ 
277 	convexHull: function(points, k) {
278 	
279 		var NONE = -1;
280 		var minmin=0, minmax=0,
281 			 maxmin=0, maxmax=0,
282 			 xmin = points[0].get().x,  xmax = points[0].get().x,
283 			 cP, bot=0, top=(-1), n = points.length,  // indices for bottom and top of the stack
284 	 		 hull = [];
285 			 
286 		// Get the points with (1) min-max x-coord, and (2) min-max y-coord
287 		for ( i=1; i < n; i++) {
288 			cP = points[i].get();
289 			if (cP.x <= xmin) {
290 				if (cP.x < xmin) {        // new xmin
291 					xmin = cP.x;
292 					minmin = minmax = i;
293 				} else {                      // another xmin
294 					if (cP.y < points[minmin].get().y)
295 						minmin = i;
296 					else if (cP.y > points[minmax].get().y)
297 						minmax = i;
298 				}
299 			}
300 			
301 			if (cP.x >= xmax) {
302 				if (cP.x > xmax) {        // new xmax
303 					xmax = cP.x;
304 					maxmin = maxmax = i;
305 				} else {                      // another xmax
306 					if (cP.y < points[maxmin].get().y)
307 						maxmin = i;
308 					else if (cP.y > points[maxmax].get().y)
309 						maxmax = i;
310 				}
311 			}
312 		}
313 		
314 		if (xmin == xmax) {      // degenerate case: all x-coords == xmin
315 			hull[++top] = points[minmin];           // a point, or
316 			if (minmax != minmin)           // a nontrivial segment
317 				hull[++top] = points[minmax];
318 			return hull;                   // one or two points
319 		}
320 		
321 		// Next, get the max and min points in the k range bins
322 		var bin = new Bin(k+2);   // first allocate the bins
323 		bin.B[0].min = minmin;         bin.B[0].max = minmax;        // set bin 0
324 		bin.B[k+1].min = maxmin;       bin.B[k+1].max = maxmax;      // set bin k+1
325 		for (var b = 1; b <= k; b++) { // initially nothing is in the other bins
326 			bin.B[b].min = bin.B[b].max = NONE;
327 		}
328 		
329 		for (var b, i=0; i < n; i++) {
330 			var cPP = points[i]; 
331 			cP = cPP.get();
332 			if (cP.x == xmin || cP.x == xmax) // already have bins 0 and k+1 
333 				continue;
334 			
335 			// check if a lower or upper point
336 			if (Math2D.pointLeftOfLine(points[minmin], points[maxmin], cPP) < 0) {  // below lower line
337 				b = (k * (cP.x - xmin) / (xmax - xmin) ) + 1;  // bin #
338 				if (bin.B[b].min == NONE)       // no min point in this range
339 					bin.B[b].min = i;           // first min
340 				else if (cP.y < points[bin.B[b].min].get().y)
341 					bin.B[b].min = i;           // new min
342 				continue;
343 			}
344 			
345 			if (Math2D.pointLeftOfLine(points[minmax], points[maxmax], cPP) > 0) {  // above upper line
346 				b = (k * (cP.x - xmin) / (xmax - xmin) ) + 1;  // bin #
347 				if (bin.B[b].max == NONE)       // no max point in this range
348 					bin.B[b].max = i;           // first max
349 				else if (cP.y > points[bin.B[b].max].get().y)
350 					bin.B[b].max = i;           // new max
351 				continue;
352 			}
353 		}
354 		
355 		// Now, use the chain algorithm to get the lower and upper hulls
356 		// the output array hull[] will be used as the stack
357 		// First, compute the lower hull on the stack hull[]
358 		for (var i = 0; i <= k+1; ++i) {
359 			if (bin.B[i].min == NONE)  // no min point in this range
360 				continue;
361 			
362 			var cPP = points[bin.B[i].min];    // select the current min point
363 			cP = cPP.get();
364 
365 			while (top > 0) {        // there are at least 2 points on the stack
366 				// test if current point is left of the line at the stack top
367 				if (Math2D.pointLeftOfLine(hull[top-1], hull[top], cPP) > 0)
368 					break;         // cP is a new hull vertex
369 				else
370 					top--;         // pop top point off stack
371 			}
372 			hull[++top] = cPP;        // push current point onto stack
373 		}
374 		
375 		// Next, compute the upper hull on the stack H above the bottom hull
376 		if (maxmax != maxmin)      // if distinct xmax points
377 			hull[++top] = points[maxmax];  // push maxmax point onto stack
378 			
379 		bot = top;                 // the bottom point of the upper hull stack
380 		for (var i = k; i >= 0; --i) {
381 			if (bin.B[i].max == NONE)  // no max point in this range
382 				continue;
383 			
384 			var cPP = points[bin.B[i].max];   // select the current max point
385 			cP = cPP.get();
386 			
387 			while (top > bot) {      // at least 2 points on the upper stack
388 				// test if current point is left of the line at the stack top
389 				if (Math2D.pointLeftOfLine(hull[top-1], hull[top], cPP) > 0)
390 					break;         // current point is a new hull vertex
391 				else
392 					top--;         // pop top point off stack
393 			}
394 			hull[++top] = cPP;        // push current point onto stack
395 		}
396 		if (minmax != minmin)
397 			hull[++top] = points[minmin];  // push joining endpoint onto stack
398 		
399 		bin = null;                  // free bins before returning
400 		return hull;              // # of points on the stack
401 	},

403 	/**
404 	 * Determine the Minkowski Difference of two convex hulls.  Useful for
405 	 * calculating collision response.
406 	 * 
407 	 * @param hullA {Array} An array of {@link Point2D}
408 	 * @param hullB {Array} An array of {@link Point2D}
409 	 * @return {Array} An array of {@link Point2D} which are the Minkowski Difference of
410 	 * 	the two hulls.
411 	 */
412 	minkDiff: function(hullA, hullB) {
413 		var cP = 0, minkDiff = new Array(hullA.length * hullB.length);
414 		for (var a in hullA) {
415 			for (var b in hullB) {
416 				var ha = hullA[a].get(), hb = hullB[b].get(),
417 					 pt = Point2D.create(hb.x - ha.x, hb.y - ha.y);
418 				minkDiff[cP++] = pt;	
419 			}
420 		}
421 		return minkDiff;	
422 	},

424    /**
425     * Helper method to determine if one circle will collide with another circle
426     * based on its direction of movement.  The circle's centers should be in
427     * world coordinates.
428     * 
429     * @param circle {Circle2D} The first circle
430     * @param velocity {Vector2D} The first circle's velocity vector
431     * @param targetCircle {Circle2D} The second circle
432     * @return {Vector2D} The vector which keeps the two circles from overlapping, 
433     * 	or <tt>null</tt> if they cannot overlap.
434     */
435    circleCircleCollision: function(circle, velocity, targetCircle) {
436       
437       // Early out test
438       var dist = targetCircle.getCenter().dist(circle.getCenter());
439       var sumRad = targetCircle.getRadius() + circle.getRadius();
440       dist -= sumRad;
441       if (velocity.len() < dist) {
442          // No collision possible
443          return null;
444       }
445 
446       var norm = Vector2D.create(velocity).normalize();
447 
448       // Find C, the vector from the center of the moving
449       // circle A to the center of B
450       var c = Vector2D.create(targetCircle.getCenter().sub(circle.getCenter()));
451       var dot = norm.dot(c);
452 
453       // Another early escape: Make sure that A is moving
454       // towards B! If the dot product between the movevec and
455       // B.center - A.center is less that or equal to 0,
456       // A isn't isn't moving towards B
457       if (dot <= 0) {
458       	norm.destroy();
459       	c.destroy();
460       	return null;
461       }
462 
463       var lenC = c.len();
464       var f = (lenC * lenC) - (dot * dot);
465 
466       // Escape test: if the closest that A will get to B
467       // is more than the sum of their radii, there's no
468       // way they are going collide
469       var sumRad2 = sumRad * sumRad;
470       if (f >= sumRad2) {
471       	norm.destroy();
472       	c.destroy();
473       	return null;
474       }
475 
476       // We now have F and sumRadii, two sides of a right triangle.
477       // Use these to find the third side, sqrt(T)
478       var t = sumRad2 - f;
479 
480       // If there is no such right triangle with sides length of
481       // sumRadii and sqrt(f), T will probably be less than 0.
482       // Better to check now than perform a square root of a
483       // negative number.
484       if (t < 0) {
485       	norm.destroy();
486       	c.destroy();
487       	return null;
488       }
489 
490       // Therefore the distance the circle has to travel along
491       // movevec is D - sqrt(T)
492       var distance = dot - Math.sqrt(t);
493 
494       // Get the magnitude of the movement vector
495       var mag = velocity.len();
496 
497       // Finally, make sure that the distance A has to move
498       // to touch B is not greater than the magnitude of the
499       // movement vector.
500       if (mag < distance) {
501       	norm.destroy();
502       	c.destroy();
503       	return null;
504       }
505 
506       // Set the length of the vector which causes the circles 
507       // to just touch
508       var moveVec = Vector2D.create(norm.mul(distance));
509 		norm.destroy();
510 		c.destroy();
511       
512       return moveVec;         
513    }
