105    /**
106     * Check to see if a line intersects another
107     *
108     * @param p1 {Point2D} Start of line 1
109     * @param p2 {Point2D} End of line 1
110     * @param p3 {Point2D} Start of line 2
111     * @param p4 {Point2D} End of line 2
112     * @return {Boolean} <tt>true</tt> if the lines intersect
113     */
114    lineLineCollision: function(p1, p2, p3, p4) {
115       var d = ((p4.y - p3.y) * (p2.x - p1.x)) - ((p4.x - p3.x) * (p2.y - p1.y));
116       var n1 = ((p4.x - p3.x) * (p1.y - p3.y)) - ((p4.y - p3.y) * (p1.x - p3.x));
117       var n2 = ((p2.x - p1.x) * (p1.y - p3.y)) - ((p2.y - p1.y) * (p1.x - p3.x));
118 
119       if ( d == 0.0 )
120       {
121          if ( n1 == 0.0 && n2 == 0.0 )
122          {
123             return false;  //COINCIDENT;
124          }
125          return false;   // PARALLEL;
126       }
127       var ua = n1 / d;
128       var ub = n2 / d;
129 
130       return (ua >= 0.0 && ua <= 1.0 && ub >= 0.0 && ub <= 1.0);
131    },

149    /*
150     * Test to see if a line intersects a Rectangle.
151     *
152     * @param p1 {Point2D} The start of the line
153     * @param p2 {Point2D} The end of the line
154     * @param rect {Rectangle} The box to test against
155     * @return <tt>true</tt> if the line intersects the box
156     * @type Boolean
157    lineBoxCollision: function(p1, p2, rect)
158    {
159       if (Math2D.boxPointCollision(rect, p1) &&
160           Math2D.boxPointCollision(rect, p2))
161       {
162          // line inside
163          return true;
164       }
165 
166       // check each line for intersection
167       var topLeft = rect.getTopLeft();
168       var bottomRight = rect.getBottomRight();
169       var topRight = new Point2D(rect.x, rect.y).add(new Point2D(rect.width, 0));
170       var bottomLeft = new Point2D(rect.x, rect.y).add(new Point2D(0, rect.height));
171 
172       if (Math2D.lineLineCollision(p1, p2, topLeft, topRight)) return true;
173       if (Math2D.lineLineCollision(p1, p2, topRight, bottomRight)) return true;
174       if (Math2D.lineLineCollision(p1, p2, bottomRight, bottomLeft)) return true;
175       if (Math2D.lineLineCollision(p1, p2, bottomLeft, topLeft)) return true;
176 
177       return false;

181    /**
182     * A static method used to calculate a direction vector
183     * from a heading angle.
184     *
185     * @param origin {Point2D} The origin of the shape
186     * @param baseVec {Vector2D} The base vector
187     * @param angle {Number} The rotation in degrees
188     * @return {Vector2D} The direction vector
189     */
190    getDirectionVector: function(origin, baseVec, angle) {
191       var r = Math2D.degToRad(angle);
192 
193       var x = Math.cos(r) * baseVec.x - Math.sin(r) * baseVec.y;
194       var y = Math.sin(r) * baseVec.x + Math.cos(r) * baseVec.y;
195 
196       var v = Vector2D.create(x, y).sub(origin);
197       return v.normalize();
198    },

212 	/**
213 	 * Calculate an approximate 2D convex hull for the given array of points.
214 	 * <p/>
215 	 * Copyright 2001, softSurfer (www.softsurfer.com)
216 	 * This code may be freely used and modified for any purpose
217 	 * providing that this copyright notice is included with it.
218 	 * SoftSurfer makes no warranty for this code, and cannot be held
219 	 * liable for any real or imagined damage resulting from its use.
220 	 * Users of this code must verify correctness for their application.
221 	 * 
222 	 * @param points {Array} An array of {@link Point2D} instances
223 	 * @param k {Number} The approximation accuracy (larger = more accurate)
224 	 * @return {Array} An array of {@link Point2D} which contains the 
225 	 * 	approximate hull of the given points
226 	 */ 
227 	convexHull: function(points, k) {
228 		// Tests if a point is Left|On|Right of an infinite line.
229 		function isLeft(point0, point1, point2) {
230 			var p0 = point0.get(), p1 = point1.get(), p2 = point2.get(); 
231 			return (p1.x - p0.x)*(p2.y - p0.y) - (p2.x - p0.x)*(p1.y - p0.y);
232 		}
233 		
234 		var Bin = Base.extend({
235 			B: null,
236 			constructor: function(size) {
237 				this.B = [];
238 				for (var i = 0; i < size; i++) {
239 					this.B.push({
240 						min: 0,
241 						max: 0
242 					});
243 				}
244 			}
245 		});
246 		
247 		var NONE = -1;
248 		var minmin=0, minmax=0,
249 			 maxmin=0, maxmax=0,
250 			 xmin = points[0].get().x,  xmax = points[0].get().x,
251 			 cP, bot=0, top=(-1), n = points.length,  // indices for bottom and top of the stack
252 	 		 hull = [];
253 			 
254 		// Get the points with (1) min-max x-coord, and (2) min-max y-coord
255 		for ( i=1; i < n; i++) {
256 			cP = points[i].get();
257 			if (cP.x <= xmin) {
258 				if (cP.x < xmin) {        // new xmin
259 					xmin = cP.x;
260 					minmin = minmax = i;
261 				} else {                      // another xmin
262 					if (cP.y < points[minmin].get().y)
263 						minmin = i;
264 					else if (cP.y > points[minmax].get().y)
265 						minmax = i;
266 				}
267 			}
268 			
269 			if (cP.x >= xmax) {
270 				if (cP.x > xmax) {        // new xmax
271 					xmax = cP.x;
272 					maxmin = maxmax = i;
273 				} else {                      // another xmax
274 					if (cP.y < points[maxmin].get().y)
275 						maxmin = i;
276 					else if (cP.y > points[maxmax].get().y)
277 						maxmax = i;
278 				}
279 			}
280 		}
281 		
282 		if (xmin == xmax) {      // degenerate case: all x-coords == xmin
283 			hull[++top] = points[minmin];           // a point, or
284 			if (minmax != minmin)           // a nontrivial segment
285 				hull[++top] = points[minmax];
286 			return hull;                   // one or two points
287 		}
288 		
289 		// Next, get the max and min points in the k range bins
290 		var bin = new Bin(k+2);   // first allocate the bins
291 		bin.B[0].min = minmin;         bin.B[0].max = minmax;        // set bin 0
292 		bin.B[k+1].min = maxmin;       bin.B[k+1].max = maxmax;      // set bin k+1
293 		for (var b = 1; b <= k; b++) { // initially nothing is in the other bins
294 			bin.B[b].min = bin.B[b].max = NONE;
295 		}
296 		
297 		for (var b, i=0; i < n; i++) {
298 			var cPP = points[i]; 
299 			cP = cPP.get();
300 			if (cP.x == xmin || cP.x == xmax) // already have bins 0 and k+1 
301 				continue;
302 			
303 			// check if a lower or upper point
304 			if (isLeft(points[minmin], points[maxmin], cPP) < 0) {  // below lower line
305 				b = (k * (cP.x - xmin) / (xmax - xmin) ) + 1;  // bin #
306 				if (bin.B[b].min == NONE)       // no min point in this range
307 					bin.B[b].min = i;           // first min
308 				else if (cP.y < points[bin.B[b].min].get().y)
309 					bin.B[b].min = i;           // new min
310 				continue;
311 			}
312 			
313 			if (isLeft(points[minmax], points[maxmax], cPP) > 0) {  // above upper line
314 				b = (k * (cP.x - xmin) / (xmax - xmin) ) + 1;  // bin #
315 				if (bin.B[b].max == NONE)       // no max point in this range
316 					bin.B[b].max = i;           // first max
317 				else if (cP.y > points[bin.B[b].max].get().y)
318 					bin.B[b].max = i;           // new max
319 				continue;
320 			}
321 		}
322 		
323 		// Now, use the chain algorithm to get the lower and upper hulls
324 		// the output array hull[] will be used as the stack
325 		// First, compute the lower hull on the stack hull[]
326 		for (var i = 0; i <= k+1; ++i) {
327 			if (bin.B[i].min == NONE)  // no min point in this range
328 				continue;
329 			
330 			var cPP = points[bin.B[i].min];    // select the current min point
331 			cP = cPP.get();
332 
333 			while (top > 0) {        // there are at least 2 points on the stack
334 				// test if current point is left of the line at the stack top
335 				if (isLeft(hull[top-1], hull[top], cPP) > 0)
336 					break;         // cP is a new hull vertex
337 				else
338 					top--;         // pop top point off stack
339 			}
340 			hull[++top] = cPP;        // push current point onto stack
341 		}
342 		
343 		// Next, compute the upper hull on the stack H above the bottom hull
344 		if (maxmax != maxmin)      // if distinct xmax points
345 			hull[++top] = points[maxmax];  // push maxmax point onto stack
346 			
347 		bot = top;                 // the bottom point of the upper hull stack
348 		for (var i = k; i >= 0; --i) {
349 			if (bin.B[i].max == NONE)  // no max point in this range
350 				continue;
351 			
352 			var cPP = points[bin.B[i].max];   // select the current max point
353 			cP = cPP.get();
354 			
355 			while (top > bot) {      // at least 2 points on the upper stack
356 				// test if current point is left of the line at the stack top
357 				if (isLeft(hull[top-1], hull[top], cPP) > 0)
358 					break;         // current point is a new hull vertex
359 				else
360 					top--;         // pop top point off stack
361 			}
362 			hull[++top] = cPP;        // push current point onto stack
363 		}
364 		if (minmax != minmin)
365 			hull[++top] = points[minmin];  // push joining endpoint onto stack
366 		
367 		bin = null;                  // free bins before returning
368 		return hull;              // # of points on the stack
369 	},	
370 	
371 	/**
372 	 * Determine the Minkowski Difference of two convex hulls.  Useful for
373 	 * calculating collision response.
374 	 * 
375 	 * @param hullA {Array} An array of {@link Point2D}
376 	 * @param hullB {Array} An array of {@link Point2D}
377 	 * @return {Array} An array of {@link Point2D} which are the Minkowski Difference of
378 	 * 	the two hulls.
379 	 */
380 	minkDiff: function(hullA, hullB) {
381 		var cP = 0, minkDiff = new Array(hullA.length * hullB.length);
382 		for (var a in hullA) {
383 			for (var b in hullB) {
384 				var ha = hullA[a].get(), hb = hullB[b].get(),
385 					 pt = Point2D.create(hb.x - ha.x, hb.y - ha.y);
386 				minkDiff[cP++] = pt;	
387 			}
388 		}
389 		return minkDiff;	
390 	},

393 	DIMETRIC_SIDE_PROJECTION: 1,
394 	DIMETRIC_TOP_PROJECTION: 2
