 35 
 36 /**
 37  * @class A static class with methods and fields that are helpful
 38  * when dealing with two dimensional mathematics.
 39  *
 40  * @static
 41  */
 42 var Math2D = Base.extend(/** @scope Math2D.prototype */{
 43 
 44    constructor: null,
 45 
 46    /**
 47     * An approximation of PI for speedier calculations.
 48     */
 49    PI: 3.14159,
 50 
 51    /**
 52     * An approximation of the inverse of PI so we can
 53     * avoid divisions.
 54     */
 55    INV_PI: 0.31831,
 56 
 57    /**
 58     * Convert degrees to radians.
 59     * @param degrees {Number} An angle in degrees
 60     * @return {Number} The degrees value converted to radians
 61     */
 62    degToRad: function(degrees) {
 63       return (0.01745 * degrees);
 64    },
 65 
 66    /**
 67     * Convert radians to degrees.
 68     * @param radians {Number} An angle in radians
 69     * @return {Number} The radians value converted to degrees
 70     */
 71    radToDeg: function(radians) {
 72       return ((radians * 180) * Math2D.INV_PI);
 73    },
 74 
 75    /**
 76     * Perform AAB (axis-aligned box) to AAB collision testing, returning <tt>true</tt>
 77     * if the two boxes overlap.
 78     *
 79     * @param box1 {Rectangle2D} The collision box of object 1
 80     * @param box2 {Rectangle2D} The collision box of object 2
 81     * @return {Boolean} <tt>true</tt> if the rectangles overlap
 82     */
 83    boxBoxCollision: function(box1, box2) {
 84       return box1.isIntersecting(box2);
 85    },
 86 
 87    /**
 88     * Perform point to AAB collision, returning <code>true</code>
 89     * if a collision occurs.
 90     *
 91     * @param box {Rectangle2D} The collision box of the object
 92     * @param point {Point2D} The point to test, in world coordinates
 93     * @return {Boolean} <tt>true</tt> if the point is within the rectangle
 94     */
 95    boxPointCollision: function(box, point) {
 96       return box.containsPoint(point);
 97    },
 98 
 99    /**
100     * Check to see if a line intersects another
101     *
102     * @param p1 {Point2D} Start of line 1
103     * @param p2 {Point2D} End of line 1
104     * @param p3 {Point2D} Start of line 2
105     * @param p4 {Point2D} End of line 2
106     * @return {Boolean} <tt>true</tt> if the lines intersect
107     */
108    lineLineCollision: function(p1, p2, p3, p4) {
109       var d = ((p4.y - p3.y) * (p2.x - p1.x)) - ((p4.x - p3.x) * (p2.y - p1.y));
110       var n1 = ((p4.x - p3.x) * (p1.y - p3.y)) - ((p4.y - p3.y) * (p1.x - p3.x));
111       var n2 = ((p2.x - p1.x) * (p1.y - p3.y)) - ((p2.y - p1.y) * (p1.x - p3.x));
112 
113       if ( d == 0.0 )
114       {
115          if ( n1 == 0.0 && n2 == 0.0 )
116          {
117             return false;  //COINCIDENT;
118          }
119          return false;   // PARALLEL;
120       }
121       var ua = n1 / d;
122       var ub = n2 / d;
123 
124       return (ua >= 0.0 && ua <= 1.0 && ub >= 0.0 && ub <= 1.0);
125    },
126 
127    /**
128     * Test to see if a line intersects a Rectangle.
129     *
130     * @param p1 {Point2D} The start of the line
131     * @param p2 {Point2D} The end of the line
132     * @param rect {Rectangle} The box to test against
133     * @return {Boolean} <tt>true</tt> if the line intersects the box
134     */
135    lineBoxCollision: function(p1, p2, rect) {
136       // Convert the line to a box itself and do a quick box box test
137       var lRect = Rectangle2D.create(p1.x, p1.y, p2.x - p1.x, p2.y - p1.y);
138       return Math2D.boxBoxCollision(lRect, rect);
139    },
140 
141    /*
142     * Test to see if a line intersects a Rectangle.
143     *
144     * @param p1 {Point2D} The start of the line
145     * @param p2 {Point2D} The end of the line
146     * @param rect {Rectangle} The box to test against
147     * @return <tt>true</tt> if the line intersects the box
148     * @type Boolean
149    lineBoxCollision: function(p1, p2, rect)
150    {
151       if (Math2D.boxPointCollision(rect, p1) &&
152           Math2D.boxPointCollision(rect, p2))
153       {
154          // line inside
155          return true;
156       }
157 
158       // check each line for intersection
159       var topLeft = rect.getTopLeft();
160       var bottomRight = rect.getBottomRight();
161       var topRight = new Point2D(rect.x, rect.y).add(new Point2D(rect.width, 0));
162       var bottomLeft = new Point2D(rect.x, rect.y).add(new Point2D(0, rect.height));
163 
164       if (Math2D.lineLineCollision(p1, p2, topLeft, topRight)) return true;
165       if (Math2D.lineLineCollision(p1, p2, topRight, bottomRight)) return true;
166       if (Math2D.lineLineCollision(p1, p2, bottomRight, bottomLeft)) return true;
167       if (Math2D.lineLineCollision(p1, p2, bottomLeft, topLeft)) return true;
168 
169       return false;
170    },
171     */
172 
173    /**
174     * A static method used to calculate a direction vector
175     * from a heading angle.
176     *
177     * @param origin {Point2D} The origin of the shape
178     * @param baseVec {Vector2D} The base vector
179     * @param angle {Number} The rotation in degrees
180     * @return {Vector2D} The direction vector
181     */
182    getDirectionVector: function(origin, baseVec, angle) {
183       var r = Math2D.degToRad(angle);
184 
185       var x = Math.cos(r) * baseVec.x - Math.sin(r) * baseVec.y;
186       var y = Math.sin(r) * baseVec.x + Math.cos(r) * baseVec.y;
187 
188       var v = Vector2D.create(x, y).sub(origin);
189       return v.normalize();
190    },
191 
192    /**
193     * Given a {@link Rectangle2D}, generate a random point within it.
194     *
195     * @param rect {Rectangle2D} The rectangle
196     * @return {Point2D} A random point within the rectangle
197     */
198    randomPoint: function(rect) {
199       var r = rect.get();
200       return Point2D.create(Math.floor(r.x + Math.random() * r.w),
201                             Math.floor(r.y + Math.random() * r.h));
202    }
203 
204 });
205 
206 return Math2D;

276 
277 /**
278  * @class A 2D point class with helpful methods for manipulation
279  */
280 var Point2D = MathObject.extend(/** @scope Point2D.prototype */{
281 
282    _vec: null,
283 
284    x: 0,
285    y: 0,
286 
287    /**
288     * Create a new 2D point.
289     *
290     * @param x {Point2D/Number} If this arg is a Point2D, its values will be
291     *                           copied into the new point.
292     * @param y {Number} The Y coordinate of the point.  Only required if X
293     *                   was a number.
294     */
295    constructor: function(x, y) {
296       this.base("Point2D");
297       this.set(x, y);
298    },
299 
300    release: function() {
301       this.base();
302       this.x = 0;
303       this.y = 0;
304       this._vec.setElements([0,0]);
305    },
306 
307    /**
308     * Private method used to update underlying object.
309     * @private
310     */
311    upd: function() {
312       this.x = this._vec.e(1); this.y = this._vec.e(2);
313    },
314 
315    /**
316     * Returns <tt>true</tt> if this point is equal to the specified point.
317     *
318     * @param point {Point2D} The point to compare to
319     * @return {Boolean} <tt>true</tt> if the two points are equal
320     */
321    equals: function(point) {
322       return (this.x == point.x && this.y == point.y);
323    },
324 
325    /**
326     * Set the position of a 2D point.
327     *
328     * @param x {Point2D/Number} If this arg is a Point2D, its values will be
329     *                           copied into the new point.
330     * @param y {Number} The Y coordinate of the point.  Only required if X
331     *                   was a number.
332     */
333    set: function(x, y) {
334       if (x instanceof Point2D)
335       {
336          this._vec = x._vec.dup();
337       }
338       else
339       {
340          AssertWarn((y != null), "Undefined Y value for point initialized to zero.");
341          this._vec = $V([x, y || 0]);
342       }
343       this.upd();
344       return this;
345    },
346    
347    /**
348     * Get the elements of this point as an object with elements X and Y.
349     * @return {Object}
350     */
351    get: function() {
352       return { x: this._vec.e(1), y: this._vec.e(2) };
353    },
354 
355    /**
356     * Set the X coordinate.
357     *
358     * @param x {Number} The X coordinate
359     */
360    setX: function(x) {
361       this._vec.setElements([x, this._vec.e(2)]);
362       this.upd();
363    },
364 
365    /**
366     * Set the Y coordinate.
367     *
368     * @param y {Number} The Y coordinate
369     */
370    setY: function(y) {
371       this._vec.setElements([this._vec.e(1), y]);
372       this.upd();
373    },
374 
375    /**
376     * A method that mutates this point by adding the point to it.
377     *
378     * @param point {Point2D} A point
379     * @return {Point2D} This point
380     */
381    add: function(point) {
382       this._vec = this._vec.add(point._vec);
383       this.upd();
384       return this;
385    },
386 
387    /**
388     * A mutator method that adds the scalar value to each component of this point.
389     * @param scalar {Number} A number
390     * @return {Point2D} This point
391     */
392    addScalar: function(scalar) {
393       this._vec = this._vec.map(function(x) { return x + scalar; });
394       this.upd();
395       return this;
396    },
397 
398    /**
399     * A mutator method that subtracts the specified point from this point.
400     * @param point {Point2D} a point
401     * @return {Point2D} This point
402     */
403    sub: function(point) {
404       this._vec = this._vec.subtract(point._vec);
405       this.upd();
406       return this;
407    },
408 
409    /**
410     * A mutator method that multiplies the components of this point with another.
411     * @param point {Point2D} A point
412     * @return {Point2D} This point
413     */
414    convolve: function(point) {
415       this._vec = this._vec.map(function(x, i) { return x * (i == 1 ? point.x : point.y); });
416       this.upd();
417       return this;
418    },
419 
420    /**
421     * A mutator method that divides the components of this point by another.  The point 
422     * cannot contain zeros for its components.
423     * @param point {Point2D} A point
424     * @return {Point2D} This point
425     */
426    convolveInverse: function(point) {
427       Assert((point.x != 0 && point.y != 0), "Division by zero in Point.convolveInverse");
428       this._vec = this._vec.map(function(x, i) { return x / (i == 1 ? point.x : point.y); });
429       this.upd();
430       return this;
431    },
432 
433    /**
434     * A mutator methor that multiplies the components of this point by a scalar value.
435     * @param scalar {Number} A number
436     * @return {Point2D} This point
437     */
438    mul: function(scalar) {
439       this._vec = this._vec.map(function(x) { return x * scalar; });
440       this.upd();
441       return this;
442    },
443 
444    /**
445     * A mutator method that divides the components of this point by a scalar value.
446     * @param scalar {Number} A number - cannot be zero
447     * @return {Point2D} This point
448     */
449    div: function(scalar) {
450       Assert((scalar != 0), "Division by zero in Point.divScalar");
451 
452       this._vec = this._vec.map(function(x) { return x / scalar; });
453       this.upd();
454       return this;
455    },
456 
457    /**
458     * A mutator method that negates this point, inversing it's components.
459     * @return {Point2D} This point
460     */
461    neg: function() {
462       this._vec.setElements([ -this._vec.e(1), -this._vec.e(2) ]);
463       this.upd();
464       return this;
465    },
466 
467    /**
468     * Returns true if the point is the zero point.
469     * @return {Boolean} <tt>true</tt> if the point's elements are both zero.
470     */
471    isZero: function() {
472       return this._vec.eql(Vector.Zero);
473    },
474    
475    /**
476     * Returns the distance between this and another point.
477     * @param point {Point2D} The point to compare against
478     * @return {Number} The distance between the two points
479     */
480    dist: function(point) {
481       return this._vec.distanceFrom(point._vec);
482    },
483 
484    /**
485     * Returns a printable version of this object fixed to two decimal places.
486     * @return {String} Formatted as "x,y"
487     */
488    toString: function() {
489       return Number(this._vec.e(1)).toFixed(2) + "," + Number(this._vec.e(2)).toFixed(2);
490    }
491 
492 }, { /** @scope Point2D.prototype */
493 
494    /**
495     * Return the classname of the this object
496     * @return {String} "Point2D"
497     */
498    getClassName: function() {
499       return "Point2D";
500    }
501 });
502 
503 /**
504  * The "zero" point. This point should not be modified.
505  */
506 Point2D.ZERO = new Point2D(0,0);
507 
508 return Point2D;
509

587 Engine.initObject("Rectangle2D", "MathObject", function() {
588 
589 /**
590  * @class A 2D rectangle class with helpful manipulation methods.
591  */
592 var Rectangle2D = MathObject.extend(/** @scope Rectangle2D.prototype */{
593 
594    topLeft: null,
595    dims: null,
596 
597    /**
598     * Create a rectangle object specifying the X and Y position and
599     * the width and height.
600     *
601     * @param x {Number} The top-left X coordinate
602     * @param y {Number} The top-left Y coordinate
603     * @param width {Number} the width of the rectangle
604     * @param height {Number} The height of the rectangle
605     */
606    constructor: function(x, y, width, height) {
607       this.topLeft = Point2D.create(0,0);
608       this.dims = Point2D.create(0,0);
609       this.set(x,y,width,height);
610    },
611 
612    release: function() {
613       this.base();
614       this.topLeft = null;
615       this.dims = null;
616    },
617 
618    /**
619     * Set the values of this rectangle.
620     *
621     * @param x {Number} An optional value to initialize the X coordinate of the rectangle
622     * @param y {Number} An optional value to initialize the Y coordinate of the rectangle
623     * @param width {Number} An optional value to initialize the width of the rectangle
624     * @param height {Number} An optional value to initialize the height of the rectangle
625     */
626    set: function(x, y, width, height)
627    {
628       if (x instanceof Rectangle2D) {
629          this.topLeft.set(x.getTopLeft());
630          this.dims.set(x.getDims());
631       }
632       else
633       {
634          this.topLeft.set((x != null ? x : 0.0), (y != null ? y : 0.0));
635          this.dims.set((width != null ? width : 0.0), (height != null ? height : 0.0));
636       }
637    },
638 
639    /**
640     * Get an object with the elements containing left, top, width, height, right
641     * and bottom as the elements x, y, w, h, r, and b. 
642     *
643     * @return {Object} An object with the specified elements
644     */
645    get: function() {
646       var tl = this.getTopLeft();
647       var d = this.getDims();
648       return {x: tl.x, y: tl.y, w: d.x, h: d.y, r: tl.x + d.x, b: tl.y + d.y};
649    },
650 
651    /**
652     * Returns <tt>true</tt> if this rectangle is equal to the specified rectangle.
653     *
654     * @param rect {Rectangle2D} The rectangle to compare to
655     * @return {Boolean} <tt>true</tt> if the two rectangles are equal
656     */
657    equals: function(rect) {
658       return (this.topLeft.equals(rect.getTopLeft()) && this.dims.equals(rect.getDims()));
659    },
660 
661    /**
662     * A mutator method that offsets this rectangle by the given amount in the X and Y axis.  
663     * The first parameter can be either a point, or the value for the X axis.  If the X axis is 
664     * specified, the second parameter should be the amount to offset in the Y axis.
665     *
666     * @param offsetPtOrX {Point/int} Either a {@link Point} which contains the offset in X and Y, or an integer
667     *                                representing the offset in the X axis.
668     * @param offsetY {int} If <code>offsetPtOrX</code> is an integer value for the offset in the X axis, this should be
669     *                      the offset along the Y axis.
670     * @return {Rectangle2D} This rectangle
671     */
672    offset: function(offsetPtOrX, offsetY)
673    {
674       var offs = Point2D.create(0,0);
675       if (offsetPtOrX instanceof Point2D)
676       {
677          offs.set(offsetPtOrX);
678       }
679       else
680       {
681          offs.set(offsetPtOrX, offsetY);
682       }
683 
684       this.topLeft.add(offs);
685       return this;
686    },
687 
688    /**
689     * Set the top left of this rectangle to the point, or coordinates specified.
690     *
691     * @param ptOrX {Point2D/Number} The top left point, or the X coordinate
692     * @param y {Number} If the top left wasn't specified as the first argument, this is the Y coordinate
693     */
694    setTopLeft: function(ptOrX, y) {
695       if (ptOrX instanceof Point2D)
696       {
697          this.topLeft.set(ptOrX);
698       }
699       else
700       {
701          this.topLeft.set(ptOrX, y);
702       }
703    },
704 
705    /**
