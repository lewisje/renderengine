 45    /** @private */
 46    combiner: function() {
 47       var out = "";
 48       for (var a = 0; a < arguments.length; a++) {
 49          out += arguments[a].toString();
 50       }
 51       return out;
 52    },
 53 
 54    cleanup: function(o) {
 55       if (typeof o === "undefined") {
 56          return "";
 57       } else if (o === null) {
 58          return "null";
 59       } else if (typeof o == "function") {
 60          return "function";
 61       } else if (o.constructor == Array || (o.slice && o.join && o.splice)) { // An array
 62          var s = "[";
 63          for (var e in o) {
 64             s += (s.length > 1 ? "," : "") + this.cleanup(o[e]);
 65          }
 66          return s + "]";
 67       } else if (typeof o === "object") {
 68          var s = "{\n";
 69          for (var e in o) {
 70             s += e + ": " + this.cleanup(o[e]) + "\n";
 71          }
 72          return s + "}\n";
 73       } else {
 74          return o.toString();
 75       }
 76    },
 77 
 78    /** @private */
 79    fixArgs: function(a) {
 80       var x = [];
 81       for (var i=0; i < a.length; i++) {
 82          if (!a[i]) {
 83             x.push("null");
 84          } else {
 85             x.push(this.cleanup(a[i]));
 86          }
 87       }
 88       return x.join(" ");
 89    },

206             }
207          }
208          this.msgStore = null;
209       }

484 /**
485  * @class A class for logging messages to a console reference object.  There are
486  *        currently four supported console references:
487  *        <ul>
488  *        <li>Firebug - logs to the Firebug/Firebug Lite error console</li>
489  *        <li>OperaConsoleRef - logs to the Opera error console</li>
490  *        <li>HTMLConsoleRef - logs to an HTML div element in the body</li>
491  *        <li>SafariConsoleRef - logging for Apple's Safari browser</li>
492  *        </ul>
493  */
494 var Console = Base.extend(/** @scope Console.prototype */{
495    constructor: null,
496 
497    consoleRef: null,
498 
499    /**
500     * Output only errors to the console.
501     */
502    DEBUGLEVEL_ERRORS:      4,
503 
504    /**
505     * Output warnings and errors to the console.
506     */
507    DEBUGLEVEL_WARNINGS:    3,
508 
509    /**
510     * Output warnings, errors, and debug messages to the console.
511     */
512    DEBUGLEVEL_DEBUG:       2,
513 
514    /**
515     * Output warnings, errors, debug, and low-level info messages to the console.
516     */
517    DEBUGLEVEL_INFO:        1,
518 
519    /**
520     * Output all messages to the console.
521     */
522    DEBUGLEVEL_VERBOSE:     0,
523 
524    /**
525     * Output nothing to the console.
526     */
527    DEBUGLEVEL_NONE:       -1,
528 
529    /** @private */
530    verbosity: null,
531 
532    /**
533     * Starts up the console.
534     */
535    startup: function() {
536 		Console.verbosity = this.DEBUGLEVEL_ERRORS;
537 		
538       if (EngineSupport.checkBooleanParam("debug") && (EngineSupport.checkBooleanParam("simWii") || jQuery.browser.Wii)) {
539          this.consoleRef = new HTMLConsoleRef();
540       }
541       else if (typeof firebug != "undefined" || (typeof console != "undefined" && console.firebug)) {
542          // Firebug or firebug lite
543          this.consoleRef = new FirebugConsoleRef();
544       }
545       else if (typeof console != "undefined" && jQuery.browser.msie) {
546          this.consoleRef = new MSIEConsoleRef();
547       }
548       else if (jQuery.browser.chrome || jQuery.browser.safari) {
549          this.consoleRef = new SafariConsoleRef();
550       }
551       else if (jQuery.browser.opera) {

553       }
554       else {

556       }

667 /**
668  * Assert that a condition is <tt>true</tt>, stopping the engine if it is <tt>false</tt>.  
669  * If the condifion fails an exception will be thrown.
670  *
671  * @param test {Boolean} A simple test that should evaluate to <tt>true</tt>
672  * @param error {String} The error message to throw if the test fails
673  */
674 var Assert = function(test, error) {
675    var fail = false;
676    try {
677       if (!test)
678       {
679          fail = true;
680          Console.setDebugLevel(Console.DEBUGLEVEL_ERRORS);
681          if (arguments.length > 1) {
682             for (var a = 1; a < arguments.length; a++) {
683                Console.error("*ASSERT* ", arguments[a]);
684                Console.trace();
685             }
686          }
687 
688          Engine.shutdown();
689          
690       }
691    } catch (ex) {
692       var pr = Console.getDebugLevel();
693       Console.setDebugLevel(Console.DEBUGLEVEL_WARNINGS);
694       Console.warn("*ASSERT* 'test' would result in an exception: ", ex);
695       Console.setDebugLevel(pr);
696    }
697    
698    // This will provide a stacktrace for browsers that support it
699    if (fail) {
700       throw new Error(error);
701    }
702 };
703 
704 /**
705  * Assert that a condition is <tt>true</tt>, reporting a warning if the test fails.
706  *
707  * @param test {Boolean} A simple test that should evaluate to <tt>true</tt>
708  * @param error {String} The warning to display if the test fails
709  */
710 var AssertWarn = function(test, warning) {
711    try {
712       if (!test)
713       {
714          Console.setDebugLevel(Console.DEBUGLEVEL_WARNINGS);
715          if (arguments.length > 1) {
716             for (var a = 1; a < arguments.length; a++) {
717                Console.warn("*ASSERT-WARN* ", arguments[a]);
718             }
719          }
720          Console.warn(warning);
721       }
722    } catch (ex) {
723       var pr = Console.getDebugLevel();
724       Console.setDebugLevel(Console.DEBUGLEVEL_WARNINGS);
725       Console.warn("*ASSERT-WARN* 'test' would result in an exception: ", ex);
726       Console.setDebugLevel(pr);
727    }
728 };
