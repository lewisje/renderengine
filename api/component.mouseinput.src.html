 39 Engine.initObject("MouseInputComponent", "InputComponent", function() {
 40 
 41 	/**
 42 	 * @class A component which responds to mouse events and notifies
 43 	 * the host object when one of the events occurs.  The host should implement
 44 	 * any of the five methods listed below to be notified of the corresponding event:
 45 	 * <ul>
 46 	 * <li><tt>onMouseOver()</tt> - The mouse moved over the host object, or the object 
 47 	 *     moved under the mouse</li>
 48 	 * <li><tt>onMouseOut()</tt> - The mouse moved out of the host object (after being over it)</li>
 49 	 * <li><tt>onMouseDown()</tt> - A mouse button was depressed</li>
 50 	 * <li><tt>onMouseUp()</tt> - A mouse button was released</li>
 51 	 * <li><tt>onMouseMove()</tt> - The mouse was moved</li>
 52 	 * </ul>
 53 	 * Each event receives the "mouseInfo" object as its only argument which contains the following:
 54 	 * <ul>
 55 	 * <li><tt>position [Point2D]</tt> - The position of the mouse in screen coordinates</li>
 56 	 * <li><tt>lastPosition [Point2D]</tt> - The last updated position of the mouse</li>
 57 	 * <li><tt>button [Number]</tt> - One of the mouse button constants from {@link EventEngine}</li>
 58 	 * </ul>
 59 	 * Objects which wish to be notified via the <tt>onMouseOver()</tt> event handler will need
 60 	 * to define their bounding box.
 61 	 *
 62 	 * @param name {String} The unique name of the component.
 63 	 * @param priority {Number} The priority of the component among other input components.
 64 	 * @extends InputComponent
 65 	 * @constructor
 66 	 * @description Create a mouse input component.
 67 	 */
 68 	var MouseInputComponent = InputComponent.extend(/** @scope MouseInputComponent.prototype */{
 69 	
 70 	   /**
 71 	    * @private
 72 	    */
 73 	   constructor: function(name, priority) {
 74 	      this.base(name, priority);
 75 
 76 			// Assign the global handlers
 77 			MouseInputComponent.assignMouseHandlers();
 78 	   },
 79 	
 80 	   /**
 81 	    * Set the host object this component exists within.  Additionally, this component
 82 	    * sets some readable flags on the host object and establishes (if not already set)
 83 	    * a mouse listener on the render context.
 84 	    *
 85 	    * @param hostObject {HostObject} The object which hosts the component
 86 	    * @private
 87 	    */
 88 	   setHostObject: function(hostObject) {
 89 	      this.base(hostObject);
 90 	
 91 	      // Set some flags we can check
 92 	      hostObject.MouseInputComponent_mouseOver = false;
 93 	      hostObject.MouseInputComponent_mouseDown = false;
 94 	
 95 			// Remember if the host has any of the required handlers
 96 			if (this.getHostObject().onMouseOver ||
 97 	          this.getHostObject().onMouseOut ||
 98 	          this.getHostObject().onMouseDown ||
 99 	          this.getHostObject().onMouseUp) {
100 				hostObject.MouseInputComponent_hasHandlers = true;	 	
101 			}
102 	
103 	   },
104 	
105 	   /**
106 	    * Perform the checks on the mouse info object, and also perform
107 	    * intersection tests to be able to call mouse events.
108 	    * @private
109 	    */
110 	   execute: function(renderContext, time) {
111 	      // Objects may be in motion.  If so, we need to call the mouse
112 	      // methods for just such a case.
113 	      var mouseInfo = Engine.getDefaultContext().MouseInputComponent_mouseInfo;
114 	      var bBox = this.getHostObject().getWorldBox();
115 	      var mouseOver = false;
116 	      if (this.getHostObject().MouseInputComponent_hasHandlers && mouseInfo && bBox) {
117 	         mouseOver = Math2D.boxPointCollision(bBox, mouseInfo.position);
118 	      }
119 	
120 	      // Mouse position changed
121 	      if (this.getHostObject().onMouseMove && !mouseInfo.position.equals(mouseInfo.lastPosition)) {
122 	         this.getHostObject().onMouseMove(mouseInfo);
123 	      }
124 	
125 	      // Mouse is over object
126 	      if (this.getHostObject().onMouseOver && mouseOver && 
127 				 !this.getHostObject().MouseInputComponent_mouseOver) {
128 	         this.getHostObject().MouseInputComponent_mouseOver = true;
129 	         this.getHostObject().onMouseOver(mouseInfo);
130 	      }
131 	
132 	      // Mouse was over object
133 	      if (this.getHostObject().onMouseOut && !mouseOver &&
134 	          this.getHostObject().MouseInputComponent_mouseOver == true) {
135 	         this.getHostObject().MouseInputComponent_mouseOver = false;
136 	         this.getHostObject().onMouseOut(mouseInfo);
137 	      }
138 	
139 	      // Mouse button clicked 
140 	      if (this.getHostObject().onMouseDown && (mouseInfo.button != EventEngine.MOUSE_NO_BUTTON)) {
141 	         this.getHostObject().MouseInputComponent_mouseDown = true;
142 	         this.getHostObject().onMouseDown(mouseInfo);
143 	      }
144 	
145 	      // Mouse button released (and mouse was down)
146 	      if (this.getHostObject().onMouseUp && this.getHostObject().MouseInputComponent_mouseDown &&
147 			    (mouseInfo.button == EventEngine.MOUSE_NO_BUTTON)) {
148 	         this.getHostObject().MouseInputComponent_mouseDown = false;
149 	         this.getHostObject().onMouseUp(mouseInfo);
150 	      }
151 	   }
152 	}, /** @scope MouseInputComponent.prototype */{
153 	   /**
154 	    * Get the class name of this object
155 	    *
156 	    * @return {String} "MouseInputComponent"
157 	    */
158 	   getClassName: function() {
159 	      return "MouseInputComponent";
160 	   },
161 		
162 		/**
163 		 * Assigns the global mouse handlers.  The mouse information can be read from an
164 		 * object which is contained on the default context. Reading 
165 		 * <code>Engine.getDefaultContext().MouseInputComponent_mouseInfo</code> will return:
166 		 * <ul>
167 		 * <li><code>position</code> - {@link Point2D}</li>
168 		 * <li><code>lastPosition</code> - {@link Point2D}</li>
169 		 * <li><code>button</code> - {@link EventEngine#MOUSE_NO_BUTTON}, {@link EventEngine#MOUSE_LEFT_BUTTON},
170 		 * {@link EventEngine#MOUSE_RIGHT_BUTTON}, or {@link EventEngine#MOUSE_MIDDLE_BUTTON}</li>
171 		 * <li><code>lastOver</code> - {@link BaseObject} or <code>null</code></li>
172 		 * <li><code>moveVec</code> - {@link Vector2D} which represents the direction and distance
173 		 * of the mouse movement</li>
174 		 * </ul>
175 		 */
176 		assignMouseHandlers: function() {
177 	      // Assign handlers to the default context, making sure to only add
178 	      // the handler once.  This way we don't have hundreds of mouse move
179 	      // handlers taking up precious milliseconds.
180 	      var ctx = Engine.getDefaultContext();
181 	      if (!ctx.MouseInputComponent_mouseInfo)
182 	      {
183 	         ctx.MouseInputComponent_mouseInfo = {
184 	            position: Point2D.create(0,0),
185 	            lastPosition: Point2D.create(0,0),
186 					downPosition: Point2D.create(0,0),
187 	            button: EventEngine.MOUSE_NO_BUTTON,
188 					moveVec: Vector2D.create(0,0),
189 					dragVec: Vector2D.create(0,0),
190 	            lastOver: null,
191 					moveTimer: null
192 	         };
193 	
194 				ctx.addEvent(null, "mousemove", function(evt) {
195 	            var mouseInfo = Engine.getDefaultContext().MouseInputComponent_mouseInfo;
196 					if (mouseInfo.moveTimer != null) {
197 						mouseInfo.moveTimer.destroy();
198 						mouseInfo.moveTimer = null;						
199 					}
200    	         mouseInfo.lastPosition.set(mouseInfo.position);
201 					mouseInfo.position.set(evt.pageX, evt.pageY);
202 					mouseInfo.moveVec.set(mouseInfo.position);
203 					mouseInfo.moveVec.sub(mouseInfo.lastPosition);
204 					if (mouseInfo.button != EventEngine.MOUSE_NO_BUTTON) {
205 						mouseInfo.dragVec.set(mouseInfo.downPosition);
206 						mouseInfo.dragVec.sub(mouseInfo.position);
207 					}
208 					mouseInfo.moveTimer = Timeout.create("mouseMove", 33, function() {
209 						mouseInfo.moveVec.set(0,0);				
210 					});
211 	         });
212 	
213 	         ctx.addEvent(null, "mousedown", function(evt) {
214 	            var mouseInfo = Engine.getDefaultContext().MouseInputComponent_mouseInfo;
215 	            mouseInfo.button = evt.which;
216 					mouseInfo.downPosition.set(evt.pageX, evt.pageY);
217 					evt.preventDefault();
218 	         });
219 	
220 	         ctx.addEvent(null, "mouseup", function(evt) {
221 	            var mouseInfo = Engine.getDefaultContext().MouseInputComponent_mouseInfo;
222 	            mouseInfo.button = EventEngine.MOUSE_NO_BUTTON;
223 					mouseInfo.dragVec.set(0,0);
224 	         });
225 	
226 				// Clean up the events when the Engine is shutting down
227 				Engine.onShutdown(function() {
228 					ctx.removeEvent(null, "mousemove");
229 					ctx.removeEvent(null, "mousedown");
230 					ctx.removeEvent(null, "mouseup");
231 				});
232 	      }
233 		}
234 	});
235 	
236 	return MouseInputComponent;
237 
238 });
