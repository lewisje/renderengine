 40 /**
 41  * @class A base rendering context.  Game objects are rendered to a context
 42  * during engine runtime.  A render context is a container of all of the objects
 43  * added to it so that each object is given the chance to render.  A render context
 44  * is logically a scene graph.  While each context can have multiple contexts associated
 45  * with it, the root of the scene graph is always located at {@link Engine#getDefaultContext}.
 46  *
 47  * @param contextName {String} The name of this context.  Default: RenderContext
 48  * @param [surface] {HTMLElement} The surface node that all objects will be rendered to.
 49  * @see {CanvasContext}
 50  * @extends Container
 51  * @constructor
 52  * @description Creates a render context
 53  */
 54 var RenderContext = Container.extend(/** @scope RenderContext.prototype */{
 55 
 56    surface: null,
 57    transformStackDepth: 0,
 58    viewport: null,
 59    worldPosition: null,
 60    worldRotation: null,
 61    worldScale: null,
 62    staticCtx: null,
 63 	
 64 	safeRemoveList: null,

 66    /**
 67     * @private
 68     */
 69    constructor: function(contextName, surface) {
 70       this.base(contextName || "RenderContext");
 71       this.surface = surface;
 72       this.setElement(surface);
 73       this.worldScale = 1;
 74       this.worldPosition = Point2D.create(0, 0);
 75       this.worldRotation = 0;
 76       this.viewport = Rectangle2D.create(0, 0, 100, 100);
 77       this.staticCtx = false;
 78 		this.safeRemoveList = [];
 79    },
 80 
 81    /**
 82     * @private
 83     */
 84    release: function() {
 85       this.base();
 86       this.surface = null;
 87       this.transformStackDepth = 0;
 88       this.worldScale = null;
 89       this.worldPosition = null;
 90       this.worldRotation = null;
 91       this.staticCtx = null;
 92 		this.safeRemoveList = null;
 93    },

 95    /**
 96     * Destroy the rendering context, any objects within the context, and detach 
 97     * the surface from its parent container.
 98     */
 99    destroy: function() {
100       // Destroy all of the objects
101       this.cleanUp();
102       this.base();
103       this.surface = null;
104    },

125     * Set the context to be static.  Setting a context to be static effectively removes

129     * to the context.
130     * 
131     * @param state {Boolean} <tt>true</tt> to set the context to static
132     */
133    setStatic: function(state) {
134       this.staticCtx = state;
135    },

219    getViewport: function() {

223    /**
224     * Add an object to the context.  Only objects
225     * within the context will be rendered.  If an object declared
226     * an <tt>afterAdd()</tt> method, it will be called after the object
227     * has been added to the context.
228     *
229     * @param obj {BaseObject} The object to add to the render list
230     */
231    add: function(obj) {
232       this.base(obj);
233       if (obj instanceof HostObject)
234       {
235          obj.setRenderContext(this);
236          this.sort();
237 
238          // Create a structure to hold information that is related to
239          // the render context that keeps it separate from the rest of the object.
240          obj.RenderContext = {};
241       }
242 
243       if (obj.afterAdd) {
244          // If the object being added to the render context has
245          // an "afterAdd" method, call it (canvas uses this for IE emulation)
246          obj.afterAdd(this);
247       }
248    },
249 
250    /**
251     * Remove an object from the render context.  The object is
252     * not destroyed when it is removed from the container.  The removal
253     * occurs after each update to avoid disrupting the flow of object
254     * traversal.
255     *
256     * @param obj {Object} The object to remove from the container.
257     * @return {Object} The object that was removed
258     */
259    remove: function(obj) {
260 		this.safeRemoveList.push(obj);
261    },
262 
263    /**
264     * Remove an object from the render context at the specified index.
265     * The object is not destroyed when it is removed.  The removal
266     * occurs after each update to avoid disrupting the flow of object
267     * traversal.
268     *
269     * @param idx {Number} An index between zero and the size of the container minus 1.
270     * @return {Object} The object removed from the container.
271     */
272    removeAtIndex: function(idx) {
273 		this.safeRemoveList.push(this._find(idx));
274 	},
275 
276 	/**
277 	 * This method is called after the update to remove items from the
278 	 * context.
279 	 * @private
280 	 */
281 	_safeRemove: function() {
282 		var obj;
283 		while ((obj = this.safeRemoveList.shift()) != null) {
284 			Container.prototype.remove.call(this, obj);
285 		}
286 		this.safeRemoveList.length = 0;
287 	},

294     */

296       return obj.RenderContext;

306    /**
307     * [ABSTRACT] Clear the context and prepare it for rendering.  If you pass a
308     * rectangle, only that portion of the world will be cleared.  If
309     * you don't pass a rectangle, the entire context is cleared.
310     *
311     * @param rect {Rectangle2D} The area to clear in the context, or
312     *             <tt>null</tt> to clear the entire context.
313     */
314    reset: function(rect) {
315    },

317    /**
318     * Update the render context before rendering the objects to the surface.
319     * If the context isn't static, this will reset and then render the context.
320     * If the context is static, you'll need to perform the reset and render yourself.
321     * This allows you to updated objects in the world, skip the reset, and then
322     * render yourself.  This can be an effective way to handle redrawing the world
323     * only as needed.
324     *
325     * @param parentContext {RenderContext} A parent context, or <tt>null</tt>
326     * @param time {Number} The current render time in milliseconds from the engine.
327     */
328    update: function(parentContext, time)
329    {
330       if (!this.staticCtx) {
331          // Clear and render world
332          this.reset();
333          this.render(time);
334       }
335    },
336 
337    /**
338     * Called to render all of the objects to the context.
339     *
340     * @param time {Number} The current render time in milliseconds from the engine.
341     */
342    render: function(time) {
343    
344 		/* pragma:DEBUG_START */
345 		try {
346 			Profiler.enter("RenderContext.render(time)");
347 		/* pragma:DEBUG_END */
348 
349 			// Push the world transform
350 			this.pushTransform();
351 
352 			this.setupWorld(time);
353 
354 			// Run the objects if they are visible
355 			var objs = this.iterator();
356 			while (objs.hasNext()) {
357 				var o = objs.next();
358 				this.renderObject(o, time);
359 			}
360 
361 			objs.destroy();   
362 
363 			// Restore the world transform
364 			this.popTransform();
365 
366 			// Safely remove any objects that were removed from
367 			// the context while it was rendering
368 			if (this.safeRemoveList.length > 0) {
369 				this._safeRemove();
370 			}
371 		
372 		/* pragma:DEBUG_START */
373 		} finally {
374 			Profiler.exit();
375 		}
376 		/* pragma:DEBUG_END */
377    },
378 
379    /**
380     * Render a single object into the world for the given time.
381     * @param obj {BaseObject} An object to render
382     * @param time {Number} The world time, in milliseconds
383     */
384    renderObject: function(obj, time) {
385 		/* pragma:DEBUG_START */
386 		try {
387 			Profiler.enter("RenderContext.renderObject(obj, time)");
388 		/* pragma:DEBUG_END */
389 	
390 			obj.update(this, time);
391 			
392 		/* pragma:DEBUG_START */
393 		} finally {
394 			Profiler.exit();
395 		}
396 		/* pragma:DEBUG_END */
397    },

400     * [ABSTRACT] Gives the render context a chance to initialize the world.
401     * Use this method to change the world position, rotation, scale, etc.
402     *
403     * @param time {Number} The current world time
404     */
405    setupWorld: function(time) {
406    },

408    /**
409     * Increment the transform stack counter.
410     */
411    pushTransform: function() {
412       this.transformStackDepth++;
413    },
414 
415    /**
416     * Decrement the transform stack counter and ensure that the stack
417     * is not unbalanced.  An unbalanced stack can be indicative of
418     * objects that do not reset the state after rendering themselves.
419     */
420    popTransform: function() {
421       this.transformStackDepth--;
422       Assert((this.transformStackDepth >= 0), "Unbalanced transform stack!");
423    },
424 
425    /**
426     * This is a potentially expensive call, and can lead to rendering
427     * errors.  It is recommended against calling this method!
428     */
429    resetTransformStack: function() {
430       while (this.transformStackDepth > 0)
431       {
432          this.popTransform();
433       }
434    }

437    /**
438     * Sort the objects to draw from objects with the lowest
439     * z-index to the highest z-index.
440     * @static
441     */
442    sortFn: function(obj1, obj2) {
443       if (obj1.getZIndex && obj2.getZIndex) {
444          return obj1.getZIndex() - obj2.getZIndex();
445       }
446       return 0
447    },
