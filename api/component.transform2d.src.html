 33 // Includes
 34 Engine.include("/engine/engine.math2d.js");
 35 Engine.include("/components/component.base.js");

 37 Engine.initObject("Transform2DComponent", "BaseComponent", function() {
 38 
 39 /**
 40  * @class A simple component that maintains position, rotation, and scale.
 41  *
 42  * @param name {String} The name of the component
 43  * @param [priority=1.0] {Number} The priority of the component
 44  * @extends BaseComponent
 45  * @constructor
 46  * @description Create a 2d transformation component
 47  */
 48 var Transform2DComponent = BaseComponent.extend(/** @scope Transform2DComponent.prototype */{
 49 
 50    position: null,
 51    rotation: 0,
 52    scale: null,
 53    lastPosition: null,
 54    lastRenderPosition: null,
 55 	worldPos: null,
 56 
 57    /**
 58     * @private
 59     */
 60    constructor: function(name, priority) {
 61       this.base(name, BaseComponent.TYPE_TRANSFORM, priority || 1.0);
 62       this.position = Point2D.create(0,0);
 63 		this.worldPos = Point2D.create(0,0);
 64       this.lastPosition = Point2D.create(0,0);
 65       this.lastRenderPosition = Point2D.create(0,0);
 66       this.rotation = 0;
 67       this.scale = [1.0,1.0];
 68    },
 69 	
 70 	destroy: function() {
 71 		this.position.destroy();
 72 		this.worldPos.destroy();
 73 		this.lastPosition.destroy();
 74 		this.lastRenderPosition.destroy();
 75 		this.base();
 76 	},
 77 
 78    /**
 79     * Releases the component back into the object pool. See {@link PooledObject#release} for
 80     * more information.
 81     */
 82    release: function() {
 83       this.base();
 84       this.position = null;
 85       this.rotation = 0;
 86       this.scale = null;
 87       this.lastPosition = null;
 88       this.lastRenderPosition = null;
 89 		this.worldPos = null;
 90    },
 91 
 92    /**
 93     * Set the position of the transform.
 94     *
 95     * @param point {Point2D} The position
 96     */
 97    setPosition: function(point) {
 98       this.setLastPosition(this.getPosition());
 99       this.position.set(point);
100    },
101 
102    /**
103     * Returns the position of the transformation relative to the world.
104     * @return {Point2D}
105     */
106    getPosition: function() {
107       return this.position;
108    },
109 
110    /**
111     * Returns the position of the transformation relative to the viewport.  If the world is
112     * comprised of multiple viewports (wide and/or tall) the render position
113     * is relative to the current viewport's position.
114     * @return {Point2D}
115     */
116    getRenderPosition: function() {
117 		this.worldPos.set(this.getPosition());
118 		this.worldPos.sub(this.getHostObject().getRenderContext().getWorldPosition());
119       return this.worldPos;
120    },
121 
122    /**
123     * Set the last position that the transformation was at.
124     *
125     * @param point {Point2D} The last position
126     */
127    setLastPosition: function(point) {
128       this.lastPosition.set(point);
129    },
130 
131    /**
132     * Get the last position of the transformation relative to the world.
133     * @return {Point2D}
134     */
135    getLastPosition: function() {
136       return this.lastPosition;
137    },
138 
139    /**
140     * Get the last position of the transformation relative to the viewport.
141     * @return {Point2D}
142     */
143    getLastRenderPosition: function() {
144       return this.lastRenderPosition;
145    },
146 
147    /**
148     * Set the rotation of the transformation.
149     *
150     * @param rotation {Number} The rotation
151     */
152    setRotation: function(rotation) {
153       this.rotation = rotation;
154    },
155 
156    /**
157     * Get the rotation of the transformation.
158     * @return {Number}
159     */
160    getRotation: function() {
161       return this.rotation;
162    },
163 
164    /**
165     * Get the rotation of the transformation relative to the viewport.
166     * @return {Number}
167     */
168    getRenderRotation: function() {
169       var wR = this.getHostObject().getRenderContext().getWorldRotation();
170       return wR + this.getRotation();
171    },
172 
173    /**
174     * Set the scale of the transform.  You can apply a uniform scale by
175     * assigning only the first argument a value.  To use a non-uniform scale,
176     * use both the X and Y arguments.
177     *
178     * @param scaleX {Number} The scale of the transformation along the X-axis with 1.0 being 100%
179     * @param [scaleY] {Number} The scale of the transformation along the Y-axis. If provided, a 
180     *			non-uniform scale can be achieved by using a number which differs from the X-axis.
181     */
182    setScale: function(scaleX, scaleY) {
183    	scaleX = scaleX || 1.0;
184       this.scale = [scaleX, scaleY || scaleX];
185    },
186 
187    /**
188     * Get the uniform scale of the transformation.
189     * @return {Number}
190     */
191    getScale: function() {
192       return this.getScaleX();
193    },
194 
195 	/**
196 	 * Get the non-uniform scale along the X-axis of the transformation.
197 	 * @return {Number}
198 	 */
199 	getScaleX: function() {
200 		return this.scale[0];
201 	},
202 	
203 	/**
204 	 * Get the non-uniform scale along the Y-axis of the transformation.
205 	 * @return {Number}
206 	 */
207 	getScaleY: function() {
208 		return this.scale[1];
209 	},
210 
211    /**
212     * Get the uniform scale of the transformation relative to the viewport.
213     * @return {Number}
214     */
215    getRenderScale: function() {
216 //    var wS = this.getHostObject().getRenderContext().getWorldScale();
217 //      return wS * this.scale;
218       return this.scale[0];
219    },
220 
221    /**
222     * Get the uniform scale of the transformation relative to the viewport along the X-axis.
223     * @return {Number}
224     */
225    getRenderScaleX: function() {
226 //    var wS = this.getHostObject().getRenderContext().getWorldScale();
227 //      return wS * this.scale;
228       return this.scale[0];
229    },
230 
231    /**
232     * Get the uniform scale of the transformation relative to the viewport along the Y-axis.
233     * @return {Number}
234     */
235    getRenderScaleY: function() {
236 //    var wS = this.getHostObject().getRenderContext().getWorldScale();
237 //      return wS * this.scale;
238       return this.scale[1];
239    },
240 
241    /**
242     * Set the components of a transformation: position, rotation,
243     * and scale, within the rendering context.
244     *
245     * @param renderContext {RenderContext} The rendering context
246     * @param time {Number} The engine time in milliseconds
247     * @param rendering {Boolean} <tt>true</tt> during the rendering phase
248     */
249    execute: function(renderContext, time) {
250       renderContext.setPosition(this.getRenderPosition());
251       renderContext.setRotation(this.getRenderRotation());
252       renderContext.setScale(this.getRenderScaleX(), this.getRenderScaleY());
253    }
254    
255 }, /** @scope Transform2DComponent.prototype */{
256    /**
257     * Get the class name of this object
258     *
259     * @return {String} "Transform2DComponent"
260     */
261    getClassName: function() {
262       return "Transform2DComponent";
263    }
264 });
265 
266 return Transform2DComponent;
267 
268 });
