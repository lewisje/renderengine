engine.baseobject.js


Summary

The object from which most renderable engine objects will need to derive.
Class Summary
BaseObject The base object class which represents an object within the engine.

 38 Engine.initObject("BaseObject", "PooledObject", function() {
 39 
 40 /**
 41  * @class The base object class which represents an object within
 42  * the engine.  All objects should extend from this class mainly due to
 43  * the fact that the object will inherit a <tt>destroy</tt>
 44  * method.  This method gives the object an opportunity to clean up
 45  * object references, thus keeping the memory usage in check.
 46  * <p/>
 47  * This object also enhances the event handling provided by the {@link EventEngine}.
 48  * It will remember events assigned to the object so that they can be automatically
 49  * cleaned up when the object is destroyed.
 50  * <p/>
 51  * If you are working with an object that represents an HTML element (node),
 52  * this object is ideal to extend from.  It has methods for assigning and
 53  * accessing that element.
 54  * <p/>
 55  * The {@link #update} is called each time a frame is generated by the engine 
 56  * to update the object within the scene graph. In this method, you'll be able to 
 57  * update the components and perform general housekeeping.
 58  * 
 59  * @param name {String} The name of the object
 60  * @extends PooledObject
 61  * @constructor
 62  * @description Create a base object. 
 63  */
 64 var BaseObject = PooledObject.extend(/** @scope BaseObject.prototype */{
 65 
 66    element: null,
 67    
 68    jQObject: null,
 69 
 70    events: null,
 71 
 72    /**
 73     * @private
 74     */
 75    constructor: function(name) {
 76       this.base(name);
 77       this.events = {};
 78       this.jQObject = null;
 79    },
 80 
 81    /**
 82     * Destroy the object, cleaning up any events that have been
 83     * attached to this object.
 84     */
 85    destroy: function() {
 86       // We need to make sure to remove any event's attached to us
 87       // that weren't already cleaned up
 88       for (var ref in this.events) {
 89          var r = ref;
 90          var fn = this.events[r];
 91          var type = r.split(",")[1];
 92          if (fn) {
 93             EventEngine.clearHandler(this.getElement(), type, fn);
 94          }
 95       }
 96 
 97       this.base();
 98    },
 99 
100    /**
101     * Release the object back into the object pool.
102     */
103    release: function() {
104       this.base();
105       this.element = null;
106       this.events = null;
107       this.jQObject = null;
108    },
109 
110    /**
111     * Set the element which will represent this object within
112     * its rendering context.
113     *
114     * @param element {HTMLElement} The HTML element this object is associated with.
115     */
116    setElement: function(element) {
117       this.element = element;
118    },
119 
120    /**
121     * Get the element which represents this object within its rendering context.
122     *
123     * @return {HTMLElement} The HTML element
124     */
125    getElement: function() {
126       return this.element;
127    },
128    
129    /**
130     * A helper method to provide access to the jQuery object wrapping the
131     * element for this object.  This allows direct access to the DOM.
132     *
133     * @return {jQuery} A jQuery object
134     */
135    jQ: function() {
136       if (!this.jQObject && this.element) {
137          this.jQObject = $(this.element);
138       }
139       return this.jQObject;
140    },
141 
142    /**
143     * Abstract update method to set the state of the object.  This method
144     * will be called each frame that is generated by the engine.  The
145     * context where the object will be rendered is passed, along with the
146     * current engine time.  Use this method to update components and
147     * perform housekeeping on the object.
148     *
149     * @param renderContext {RenderContext} The context the object exists within
150     * @param time {Number} The current engine time, in milliseconds
151     */
152    update: function(renderContext, time) {
153    },
154 
155    /**
156     * Add an event handler to this object, as long as it has an associated HTML element.
157     *
158     * @param ref {Object} The object reference which is assigning the event
159     * @param type {String} The event type to respond to
160     * @param [data] {Array} Optional data to pass to the handler when it is invoked.
161     * @param fn {Function} The function to trigger when the event fires
162     */
163    addEvent: function(ref, type, data, fn) {
164 		if (ref == null) {
165 			// This is a global assignment to the document body.  Many listeners
166 			// may collect data from the event handler.
167 			Console.info("Global assignment of event '" + type + "'");
168 			EventEngine.setHandler(document.body, type, data || fn, fn);
169          this.events["document," + type] = func;
170 		} else if (this.getElement()) {
171          Console.info(ref.getName() + " attach event '" + type + "' to " + this.getName());
172          EventEngine.setHandler(this.getElement(), type, data || fn, fn);
173 
174          // Remember the handler by the reference object's name and event type
175          var func = $.isFunction(data) ? data : fn;
176          this.events[ref.getName() + "," + type] = func;
177       }
178    },
179 
180    /**
181     * Remove the event handler assigned to the object's associated HTML element
182     * for the given type.
183     *
184     * @param ref {Object} The object reference which assigned the event
185     * @param type {String} The event type to remove
186     */
187    removeEvent: function(ref, type) {
188    	if (ref == null) {
189    		// This was a global assignment to the document body.  Clean it up
190    		Console.info("Global event '" + type + "' removed");
191          var fn = this.events["document," + type];
192    		EventEngine.clearHandler(document.body, type, fn);
193    	} else if (ref != null && this.getElement()) {
194          // Find the handler to remove
195          var fn = this.events[ref.getName() + "," + type];
196          if (fn) {
197             Console.info(ref.getName() + " remove event '" + type + "' from " + this.getName());
198             EventEngine.clearHandler(this.getElement(), type, fn);
199          }
200          // Remove the reference
201          delete this.events[ref.getName() + "," + type];
202       }
203    }
204 
205  }, /** @scope BaseObject.prototype */{
206 
207    /**
208     * Get the class name of this object
209     *
210     * @return {String} "BaseObject"
211     */
212    getClassName: function() {
213       return "BaseObject";
214    }
215 
216 });
217 
218 return BaseObject;
219 
220 });
