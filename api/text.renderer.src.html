text.renderer.js


Summary

A text renderer object that uses a specific render object to produce text when a render context cannot.
Class Summary

 36 Engine.include("/engine/engine.object2d.js");

 40 Engine.initObject("TextRenderer", "Object2D", function() {
 41 
 42 /**
 43  * @class A 2d text rendering object.  The object hosts the given text
 44  *        renderer, and a way to position and size the text.  It is up
 45  *        to the renderer provided to present the text within the render
 46  *        context.
 47  *
 48  * @constructor
 49  * @description Create an instance of one of the text renderers.
 50  * @param renderer {AbstractTextRenderer} The text renderer to use
 51  * @param text {String} The text to render
 52  * @param size {Number} The size of the text to render
 53  * @see VectorText
 54  * @see BitmapText
 55  * @see ContextText
 56  */
 57 var TextRenderer = Object2D.extend(/** @scope TextRenderer.prototype */{
 58 
 59    drawMode: 0,
 60    
 61    renderer: null,
 62 
 63    /**
 64     * @private
 65     */
 66    constructor: function(renderer, text, size) {
 67 
 68       Assert((AbstractTextRenderer.isInstance(renderer)), "Text renderer must extend AbstractTextRenderer");
 69 
 70       this.base("TextRenderer");
 71 
 72       // Add components to move and draw the text
 73       this.renderer = renderer;
 74       if (!Engine.options.billboards || renderer.isNative()) {
 75          this.add(this.renderer);
 76       } else {
 77          this.add(Billboard2DComponent.create("billboard", this.renderer));
 78       }
 79 
 80       this.add(Transform2DComponent.create("transform"));
 81 
 82       renderer.setText(text || "");
 83       renderer.setSize(size || 1);
 84       this.drawMode = TextRenderer.DRAW_TEXT;
 85    },
 86 
 87    /**
 88     * @private
 89     */
 90    release: function() {
 91       this.base();
 92       this.drawMode = TextRenderer.DRAW_TEXT;
 93       this.renderer = null;
 94    },
 95 
 96    /**
 97     * Called to render the text to the context.
 98     *
 99     * @param renderContext {RenderContext} The context to render the text upon
100     * @param time {Number} The engine time in milliseconds
101     */
102    update: function(renderContext, time) {
103 
104       if (this.drawMode == TextRenderer.DRAW_TEXT)
105       {
106          renderContext.pushTransform();
107          this.base(renderContext, time);
108          renderContext.popTransform();
109       }
110 
111    },
112 
113    /**
114     * @private
115     */
116    regen: function() {
117       if (!Engine.options.billboards || this.renderer.isNative()) {
118          return;
119       } else {
120          this.getComponent("billboard").regenerate();
121       }
122    },
123 
124    /**
125     * Set the text for this object to render.  This method
126     * <i>must</i> be implemented by a text renderer.
127     *
128     * @param text {String} The text to render.
129     */
130    setText: function(text) {
131       this.renderer.setText(text);
132       this.regen();
133    },
134 
135    /**
136     * Get the text for this object to render.  This method
137     * <i>must</i> be implemented by a text renderer.
138     * 
139     * @return {String} The text to draw
140     */
141    getText: function() {
142       return this.renderer.getText();
143    },
144 
145    /**
146     * Set the size of the text to render.
147     *
148     * @param size {Number} Defaults to 1
149     */
150    setTextSize: function(size) {
151       this.renderer.setSize(size || 1);
152       this.regen();
153    },
154 
155    /**
156     * Get the size of the text to render.
157     * @return {Number} The size/scale of the text
158     */
159    getTextSize: function() {
160       return this.renderer.getSize();
161    },
162 
163    /**
164     * Set the weight (boldness) of the text.  This method
165     * is optional for a text renderer.
166     *
167     * @param weight {Object} The boldness of the given text renderer
168     */
169    setTextWeight: function(weight) {
170       this.renderer.setTextWeight(weight);
171       this.regen();
172    },
173 
174    /**
175     * Get the weight of the text.  This method is optional
176     * for a text renderer.
177     * @return {Object} The boldness of the given text renderer
178     */
179    getTextWeight: function() {
180       return this.renderer.getTextWeight();
181    },
182    
183    /**
184     * Set the font for the text.  This method is optional
185     * for a text renderer.
186     * @param font {String} The text font
187     */
188    setTextFont: function(font) {
189       this.renderer.setTextFont(font);
190       this.regen();
191    },
192    
193    /**
194     * Get the font for the text.  This method is optional
195     * for a text renderer.
196     * @return {String} The text font
197     */
198    getTextFont: function() {
199       return this.renderer.getTextFont();
200    },
201    
202    /**
203     * Set the style of the text.  This method is optional
204     * for a text renderer.
205     * @param style {String} The text style
206     */
207    setTextStyle: function(style) {
208       this.renderer.setTextStyle(style);
209       this.regen();
210    },
211 
212    /**
213     * Get the style for text.  This method is optional
214     * for a text renderer.
215     * @return {String} The text style
216     */   
217    getTextStyle: function() {
218       return this.renderer.getTextStyle();
219    },
220 
221    /**
222     * Get the position where the text will render.
223     * @return {Point2D} The position of the text
224     */
225    getPosition: function() {
226       return this.getComponent("transform").getPosition();
227    },
228 
229    /**
230     * Set the position where the text will render.
231     *
232     * @param point {Point2D} The position to render the text
233     */
234    setPosition: function(point) {
235       this.getComponent("transform").setPosition(point);
236       point.destroy();
237    },
238 
239    /**
240     * Set the horizontal alignment of the text.  This method is optional
241     * for a text renderer
242     *
243     * @param alignment {Object} A text alignment mode for the given text renderer.
244     */
245    setTextAlignment: function(alignment) {
246       this.renderer.setTextAlignment(alignment);
247       this.regen();
248    },
249 
250    /**
251     * Get the horizontal alignment of the text. This method is optional
252     * for a text renderer.
253     * @return {Number} The alignment mode for the given text renderer
254     */
255    getTextAlignment: function() {
256       return this.renderer.getTextAlignment();
257    },
258 
259    /**
260     * Set the color of the text to render.
261     *
262     * @param color {String} The color of the text
263     */
264    setColor: function(color) {
265       this.renderer.setColor(color);
266       this.regen();
267    },
268 
269    /**
270     * Get the color of the text to render
271     * @return {String} The text color
272     */
273    getColor: function() {
274       return this.renderer.getColor();
275    },
276 
277    /**
278     * Set the color of the text.
279     *
280     * @param textColor {String} Color of the text.
281     */
282    setTextColor: function(textColor) {
283       this.renderer.setColor(textColor);
284       this.regen();
285    },
286 
287    /**
288     * Get the text color
289     * @return {String} The color or style of the line
290     */
291    getTextColor: function() {
292       return this.renderer.getColor();
293    },
294 
295    /**
296     * Set the text drawing mode to either {@link #DRAW_TEXT} or {@link #NO_DRAW}.
297     *
298     * @param drawMode {Number} The drawing mode for the text.
299     */
300    setDrawMode: function(drawMode) {
301       this.drawMode = drawMode;
302    },
303 
304    /**
305     * Get the current drawing mode for the text.
306     * @return {Number} The text drawing mode
307     */
308    getDrawMode: function() {
309       return this.drawMode;
310    }
311 }, /** @scope TextRenderer.prototype */{
312 
313    /**
314     * Get the class name of this object
315     * @return {String} The string "TextRenderer"
316     */
317    getClassName: function() {
318       return "TextRenderer";
319    },
320 
321    /**
322     * Draw the text to the context.
323     * @type Number
324     */
325    DRAW_TEXT: 0,
326 
327    /**
328     * Don't draw the text to the context.
329     * @type Number
330     */
331    NO_DRAW: 1,
332    
333    /**
334     * The size of a text element, in pixels
335     * @type Number
336     */
337    BASE_TEXT_PIXELSIZE: 12
338 });
339 
340 return TextRenderer;
341 
342 });
