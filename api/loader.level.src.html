loader.level.js


Summary

An extension of the image resource loader for loading 2D levels with an associated collision map and object placement. Includes a class for working with loaded levels.
Class Summary
Level Creates an instance of a Level object.
LevelLoader Loads levels and makes them available to the system.

 40 Engine.initObject("LevelLoader", "ImageLoader", function() {
 41 
 42 /**
 43  * @class Loads levels and makes them available to the system.  Levels are defined
 44  *        by a specific type of resource file.  A level is comprised of its bitmap
 45  *        file, a collision map, and objects that make up the level with their
 46  *        constructor states.
 47  * <pre>
 48  * {
 49  *    // A level file
 50  *    bitmapImage: "level1.png",
 51  *    bitmapWidth: 6768,
 52  *    bitmapHeight: 448,
 53  *    collisionMap: [
 54  *       [0, 400, 6768, 48]
 55  *    ],
 56  *    objects: {}
 57  * }
 58  * </pre>
 59  *
 60  * @constructor
 61  * @param name {String=LevelLoader} The name of the resource loader
 62  * @extends ImageLoader
 63  */
 64 var LevelLoader = ImageLoader.extend(/** @scope LevelLoader.prototype */{
 65 
 66    levels: null,
 67 
 68    /**
 69     * @private
 70     */
 71    constructor: function(name) {
 72       this.base(name || "LevelLoader");
 73       this.levels = {};
 74    },
 75 
 76    /**
 77     * Load a level resource from a URL.
 78     *
 79     * @param name {String} The name of the resource
 80     * @param url {String} The URL where the resource is located
 81     */
 82    load: function(name, url, info, path) {
 83 
 84       if (url)
 85       {
 86          var loc = window.location;
 87          if (url.indexOf(loc.protocol) != -1 && url.indexOf(loc.host) == -1) {
 88             Assert(false, "Levels must be located on this server");
 89          }
 90 
 91          var thisObj = this;
 92 
 93          // Get the file from the server
 94          Engine.loadJSON(url, function(levelInfo) {
 95             // get the path to the resource file
 96             var path = url.substring(0, url.lastIndexOf("/"));
 97             thisObj.load(name, null, levelInfo, path + "/");
 98          });
 99       }
100       else
101       {
102          info.bitmapImage = path + info.bitmapImage;
103          Console.log("Loading level: " + name + " @ " + info.bitmapImage);
104 
105          // Load the level image file
106          this.base(name, info.bitmapImage, info.bitmapWidth, info.bitmapHeight);
107 
108          // Store the level info
109          this.levels[name] = info;
110       }
111    },
112 
113    /**
114     * Get the level resource with the specified name from the cache.  The
115     * object returned contains the bitmap as <tt>image</tt> and
116     * the level definition as <tt>info</tt>.
117     *
118     * @param name {String} The name of the object to retrieve
119     * @return {Object} The level resource specified by the name
120     */
121    get: function(name) {
122       var bitmap = this.base(name);
123       var level = {
124          image: bitmap,
125          info: this.levels[name]
126       };
127       return level;
128    },
129 
130    /**
131     * Creates a {@link Level} object representing the named level.
132     *
133     * @param level {String} A loaded level name
134     * @returns {Level} A {@link Level} object
135     */
136    getLevel: function(level) {
137       return Level.create(level, this.get(level));
138    },
139 
140    /**
141     * The name of the resource this loader will get.
142     * @returns {String} The string "level"
143     */
144    getResourceType: function() {
145       return "level";
146    }
147 
148 }, /** @scope LevelLoader.prototype */{
149    /**
150     * Get the class name of this object.
151     * @return {String} The string "LevelLoader"
152     */
153    getClassName: function() {
154       return "LevelLoader";
155    }
156 });
157 
158 return LevelLoader;
159 
160 });

162 Engine.initObject("Level", "PooledObject", function() {
163 
164 /**
165  * @class Creates an instance of a Level object.
166  *
167  * @constructor
168  * @param name {String} The name of the object
169  * @param levelResource {Object} The level resource loaded by the LevelLoader
170  * @extends PooledObject
171  */
172 var Level = PooledObject.extend(/** @scope Level.prototype */{
173 
174    // The level resource
175    levelResource: null,
176 
177    // The level frame
178    frame: null,
179 
180    // The map of all collision rects defined for the level
181    collisionMap: null,
182 
183    /**
184     * @private
185     */
186    constructor: function(name, levelResource) {
187 
188       this.levelResource = levelResource;
189       this.collisionMap = [];
190 
191       // Run through the collision map to recreate
192       // the collision rectangles
193       for (var r in levelResource.info.collisionMap) {
194          var rA = levelResource.info.collisionMap[r];
195          this.collisionMap.push(Rectangle2D.create(rA[0], rA[1], rA[2], rA[3]));
196       }
197 
198       return this.base(name);
199    },
200 
201    /**
202     * @private
203     */
204    release: function() {
205       this.base();
206       this.levelResource = null;
207       this.frame = null;
208       this.collisionMap = null;
209    },
210 
211    /**
212     * Gets a potential collision list (PCL) for the point and radius specified.
213     * This routine, and the entire collision mechanism for levels, could be optimized for speed
214     * using a BSP tree, or other structure.
215     *
216     * @param point {Point2D} The position to check for a collision
217     * @param radius {Number} The distance from the point to check for collisions
218     * @return {Array} An array of {@link Rectangle2D} instances which might be possible collisions
219     */
220    getPCL: function(point, radius) {
221       // Create a rectangle which represents the position and radius
222       var cRect = Rectangle2D.create(point.x - radius, point.y - radius, radius * 2, radius * 2);
223 
224       // Check the collision map for possible collisions
225       var pcl = [];
226       for (var r in this.collisionMap) {
227          if (this.collisionMap[r].isIntersecting(cRect)) {
228             pcl.push(this.collisionMap[r]);
229          }
230       }
231 
232       return pcl;
233    },
234 
235    /**
236     * Get the width of the level image.
237     * @return {Number} The width of the level in pixels
238     */
239    getWidth: function() {
240       return this.levelResource.info.bitmapWidth;
241    },
242 
243    /**
244     * Get the height of the level image.
245     * @return {Number} The height of the level in pixels
246     */
247    getHeight: function() {
248       return this.levelResource.info.bitmapHeight;
249    },
250 
251    /**
252     * Get a {@link Rectangle2D} which encloses this level.
253     * @return {Rectangle2D} A {@link Rectangle2D} which encloses the level
254     */
255    getFrame: function() {
256       if (!this.frame) {
257          this.frame = Rectangle2D.create(0, 0, this.getWidth(), this.getHeight());
258       }
259 
260       return this.frame;
261    },
262 
263    /**
264     * The source image loaded by the {@link LevelLoader} when the level was
265     * created.
266     * @return {HTMLImage} The source image of the level
267     */
268    getSourceImage: function() {
269       return this.levelResource.image;
270    }
271 
272 }, /** @scope Level.prototype */{
273    /**
274     * Gets the class name of this object.
275     * @return {String} The string "Level"
276     */
277    getClassName: function() {
278       return "Level";
279    }
280 });
281 
282 return Level;
283 
284 });
