 35 // Includes
 36 Engine.include("/components/component.collider.js");
 37 
 38 Engine.initObject("BoxColliderComponent", "ColliderComponent", function() {
 39 
 40 /**
 41  * @class An extension of the {@link ColliderComponent} which will check the
 42  *        object's bounding boxes for semi-precise collision.
 43  *
 44  * @param name {String} Name of the component
 45  * @param collisionModel {SpatialCollection} The collision model
 46  * @param priority {Number} Between 0.0 and 1.0, with 1.0 being highest
 47  *
 48  * @extends ColliderComponent
 49  * @constructor
 50  * @description Creates a collider component for box-box collision testing.  Each object
 51  *              must implement the {@link Object2D#getWorldBox} method and return a
 52  *              world-oriented bounding box.
 53  */
 54 var BoxColliderComponent = ColliderComponent.extend(/** @scope BoxColliderComponent.prototype */{
 55 
 56    /**
 57     * Call the host object's <tt>onCollide()</tt> method, passing the time of the collision
 58     * and the potential collision object.  The return value should either tell the collision
 59     * tests to continue, or to stop.
 60     * <p/>
 61     * A world bounding box collision must occur to trigger the <tt>onCollide()</tt> method.
 62     *
 63     * @param time {Number} The engine time (in milliseconds) when the potential collision occurred
 64     * @param collisionObj {HostObject} The host object with which the collision potentially occurs
 65     * @param hostMask {Number} The collision mask for the host object
 66     * @param targetMask {Number} The collision mask for <tt>collisionObj</tt>
 67     * @return {Number} A status indicating whether to continue checking, or to stop
 68     */
 69    testCollision: function(time, collisionObj, hostMask, targetMask) {
 70       if (this.getHostObject().getWorldBox && collisionObj.getWorldBox &&
 71           this.getHostObject().getWorldBox().isIntersecting(collisionObj.getWorldBox())) {
 72 
 73          return this.base(time, collisionObj, hostMask, targetMask);
 74       }
 75       
 76       return ColliderComponent.CONTINUE;
 77    }
 78 
 79 }, { /** @scope BoxColliderComponent.prototype */
 80 
 81    /**
 82     * Get the class name of this object
 83     * @return {String} "BoxColliderComponent"
 84     */
 85    getClassName: function() {
 86       return "BoxColliderComponent";
 87    }
 88    
 89 });
 90 
 91 return BoxColliderComponent;
 92 
 93 });
