 33 // Includes
 34 Engine.include("/components/component.base.js");
 35 
 36 Engine.initObject("InputComponent", "BaseComponent", function() {
 37 
 38 /**
 39  * @class A component which can read an input device and make those inputs
 40  *        available to a {@link HostObject}.
 41  *
 42  * @param name {String} The name of the component
 43  * @param [priority=1.0] {Number} The component's priority
 44  * @extends BaseComponent
 45  * @constructor
 46  * @description Create an input component.
 47  */
 48 var InputComponent = BaseComponent.extend(/** @scope InputComponent.prototype */{
 49 
 50    recording: false,
 51    playback: false,
 52    script: null,
 53    lastInputTime: 0,
 54 
 55    /**
 56     * @private
 57     */
 58    constructor: function(name, priority) {
 59       this.base(name, BaseComponent.TYPE_INPUT, priority || 1.0);
 60       this.recording = false;
 61       this.playback = false;
 62    },
 63    
 64    /**
 65     * @private
 66     */
 67    startRecording: function() {
 68       Console.debug("RECORDING INPUT");
 69       this.recording = true;
 70       this.lastInputTime = Engine.worldTime;
 71       this.script = [];
 72    },
 73    
 74    /**
 75     * @private
 76     */
 77    stopRecording: function() {
 78       Console.debug("RECORDING STOPPED");
 79       this.recording = false;
 80    },
 81    
 82    /**
 83     * @private
 84     */
 85    getScript: function() {
 86       return this.script;
 87    },
 88    
 89    /**
 90     * @private
 91     */
 92    setScript: function(script) {
 93       this.script = script;
 94    },
 95    
 96    /**
 97     * @private
 98     */
 99    playEvent: function() {
100       // ABSTRACT 
101    },
102    
103    /**
104     * @private
105     */
106    playScript: function(script) {
107       this.recording = false;
108       this.playback = true;
109       this.script = script;
110       
111       var popCall = function() {
112          if (arguments.callee.script.length == 0) {
113             return;
114          }
115          if (arguments.callee.e != null) {
116             Console.log("PLAYBACK:", arguments.callee.e.type);
117             arguments.callee.self.playEvent(arguments.callee.e);
118          }
119          arguments.callee.e = arguments.callee.script.shift();
120          setTimeout(arguments.callee, arguments.callee.e.delay);
121       };
122       popCall.script = this.script;
123       popCall.self = this;
124       popCall.e = null;
125       
126       popCall();
127    },
128    
129    /**
130     * @private
131     */
132    record: function(eventObj,parts) {
133       if (!this.recording) {
134          return;
135       }
136       var evtCall={};
137       for (var x in parts) {
138          evtCall[parts[x]] = eventObj[parts[x]];
139       }
140       evtCall.delay = Engine.worldTime - this.lastInputTime;
141       this.lastInputTime = Engine.worldTime;
142       evtCall.type = eventObj.type;
143       this.script.push(evtCall);
144    }
145    
146 }, /** @scope InputComponent.prototype */{
147    /**
148     * Get the class name of this object
149     *
150     * @return {String} "InputComponent"
151     */
152    getClassName: function() {
153       return "InputComponent";
154    }
155 });
156 
157 return InputComponent;
158
