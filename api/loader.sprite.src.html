loader.sprite.js


Summary

An extension of the image resource loader for handling sprites. Includes a class for working with loaded sprites.
Class Summary
Sprite Represents a sprite
SpriteLoader Loads sprites and makes them available to the system.

 39 Engine.initObject("SpriteLoader", "ImageLoader", function() {
 40 
 41 /**
 42  * @class Loads sprites and makes them available to the system.  Sprites are
 43  *        defined by an external JSON resource file.  A sprite definition file 
 44  *        is a JSON file which can support single-line comments.  The format 
 45  *        describes the image which contains the bitmap, the size of the bitmap,
 46  *        the version of the file, and the sprites.  Sprites can be either single
 47  *        frames or animations.  Animations are expected to be sequentially organized
 48  *        in the bitmap from left to right.  Each frame of an animation must be the exact
 49  *        same dimensions.
 50  *        <p/>
 51  *        A frame is simply defined by the upper left corner of the sprite and the
 52  *        width and height of the frame.  For an animation, the first four arguments are
 53  *        the same as a frame, followed by the frame count, the millisecond delay between
 54  *        frames, and the type of animation (either "loop" or "toggle").  A looped animation
 55  *        will play all frames, indicated by the frame count, and then start again from the
 56  *        beginning of the animation.  A toggled animation will play from the first to
 57  *        the last frame, then play from the last to the first, and then repeat.  The
 58  *        first and last frame will not be repeated in a toggled animation.  Thus, if
 59  *        the frames are A, B, C, then the toggle will play as A, B, C, B, A, B...
 60  * <pre>
 61  * {
 62  *    // Sprite definition file v2
 63  *    "bitmapImage": "bitmapFile.ext",
 64  *    "bitmapSize": [320, 320],
 65  *    "version": 2
 66  *    "sprites": {
 67  *        "stand": [0, 0, 32, 32],
 68  *        "walk": [32, 0, 32, 32, 3, 150, "loop"]
 69  *    }
 70  * }
 71  * </pre>
 72  *        <i>Note:</i> The new file structure is a bit more compact, and is indicated with
 73  *        the "version" key in the file, set to the value 2.  Version 1 will be deprecated
 74  *        and will not be supported in a future release of The Render Engine.
 75  *
 76  * @constructor
 77  * @param name {String=SpriteLoader} The name of the resource loader
 78  * @extends ImageLoader
 79  */
 80 var SpriteLoader = ImageLoader.extend(/** @scope SpriteLoader.prototype */{
 81 
 82    sprites: null,
 83    
 84    queuedSprites: 0,
 85 
 86    constructor: function(name) {
 87       this.base(name || "SpriteLoader");
 88       this.sprites = {};
 89       this.queuedSprites = 0;
 90    },
 91 
 92    /**
 93     * Load a sprite resource from a URL.
 94     *
 95     * @param name {String} The name of the resource
 96     * @param url {String} The URL where the resource is located
 97     */
 98    load: function(name, url, info, path) {
 99 
100       if (url)
101       {
102          var loc = window.location;
103          if (url.indexOf(loc.protocol) != -1 && url.indexOf(loc.host) == -1) {
104             Assert(false, "Sprites must be located on this server");
105          }
106 
107          this.queuedSprites++;
108          var thisObj = this;
109 
110          // Get the file from the server
111          Engine.loadJSON(url, function(spriteInfo, status) {
112             // get the path to the resource file
113             var path = url.substring(0, url.lastIndexOf("/"));
114             thisObj.load(name, null, spriteInfo, path + "/");
115          });
116       }
117       else
118       {
119          info.bitmapImage = path + info.bitmapImage;
120          Console.info("Loading sprite: " + name + " @ " + info.bitmapImage);
121 
122          // Load the sprite image file
123 			if (!info.version || info.version == 1) {
124 	         this.base(name, info.bitmapImage, info.bitmapWidth, info.bitmapHeight);
125 			} else if (info.version == 2) {
126 	         this.base(name, info.bitmapImage, info.bitmapSize[0], info.bitmapSize[1]);
127 			}
128 
129          // Store the sprite info
130          this.sprites[name] = info;
131          this.queuedSprites--;
132       }
133    },
134 
135    /**
136     * Get the sprite resource with the specified name from the cache.  The
137     * object returned contains the bitmap as <tt>image</tt> and
138     * the sprite definition as <tt>info</tt>.
139     *
140     * @param name {String} The name of the object to retrieve
141     * @return {Object} An object with two keys: "image" and "info"
142     */
143    get: function(name) {
144       var bitmap = this.base(name);
145       var sprite = {
146          image: bitmap,
147          info: this.sprites[name]
148       };
149       return sprite;
150    },
151 
152    /**
153     * Check to see if a named resource is "ready for use".
154     * @param name {String} The name of the resource to check ready status for,
155     *             or <tt>null</tt> for all resources in loader.
156     * @return {Boolean} <tt>true</tt> if the resource is loaded and ready to use
157     */
158    isReady: function(name) {
159       // If sprites are queued, we can't be totally ready
160       if (this.queuedSprites > 0) {
161          return false;
162       }
163       
164       return this.base(name);
165    },
166 
167    /**
168     * Creates a {@link Sprite} object representing the named sprite.
169     *
170     * @param resource {String} A loaded sprite resource
171     * @param sprite {String} The name of the sprite from the resource
172     * @return {Sprite} A {@link Sprite} instance
173     */
174    getSprite: function(resource, sprite) {
175 		var info = this.get(resource).info;
176       return Sprite.create(sprite, info.sprites[sprite], this.get(resource), info.version || 1);
177    },
178 
179    /**
180     * Get the names of all the sprites available in a resource.
181     *
182     * @param resource {String} The name of the resource
183     * @return {Array} All of the sprite names in the given loaded resource
184     */
185    getSpriteNames: function(resource) {
186       var s = [];
187       var spr = this.sprites[resource].sprites;
188       for (var i in spr) {
189          s.push(i);
190       }
191       return s;
192    },
193 
194 	exportAll: function(resource) {
195 		var o = this.base();
196 		var sprites = this.getSpriteNames(resource);
197 		for (var i in sprites) {
198 			o[sprites[i]] = this.getSprite(resource, sprites[i]);
199 		}
200 		return o;
201 	},
202 
203    /**
204     * The name of the resource this loader will get.
205     * @returns {String} The string "sprite"
206     */
207    getResourceType: function() {
208       return "sprite";
209    }
210 
211 }, /** @scope SpriteLoader.prototype */{
212    /**
213     * Get the class name of this object.
214     * @return {String} The string "SpriteLoader"
215     */
216    getClassName: function() {
217       return "SpriteLoader";
218    }
219 });
220 
221 return SpriteLoader;
222 
223 });

225 Engine.initObject("Sprite", "PooledObject", function() {
226 
227 /**
228  * @class Represents a sprite
229  *
230  * @constructor
231  * @param name {String} The name of the sprite within the resource
232  * @param spriteObj {Object} Passed in by a {@link SpriteLoader}.  An array which defines the
233  *                  sprite frame, and parameters.
234  * @param spriteResource {Object} The sprite resource loaded by the {@link SpriteLoader}
235  * @extends PooledObject
236  */
237 var Sprite = PooledObject.extend(/** @scope Sprite.prototype */{
238 
239    // The type of sprite: Single or Animation
240    type: -1,
241 
242    // Animation mode: loop or toggle
243    mode: -1,
244 
245    // Animation frame count
246    count: -1,
247 
248    // Animation speed
249    speed: -1,
250 
251    // The rect which defines the sprite frame
252    frame: null,
253 
254    // The image map that contains the sprite(s)
255    image: null,
256 
257    // The bounding box for the sprite
258    bbox: null,
259 
260    /**
261     * @private
262     */
263    constructor: function(name, spriteObj, spriteResource, fileVersion) {
264       this.base(name);
265 		
266 		if (fileVersion == 1) {
267 	  		this.type = (spriteObj["a"] ? Sprite.TYPE_ANIMATION : Sprite.TYPE_SINGLE);
268 		} else if (fileVersion == 2) {
269 			this.type = (spriteObj.length == 4 ? Sprite.TYPE_SINGLE : Sprite.TYPE_ANIMATION);
270 		}
271 
272       var s;
273 		if (fileVersion == 1) {
274 			s = (this.type == Sprite.TYPE_ANIMATION ? spriteObj["a"] : spriteObj["f"]);
275 		} else if (fileVersion == 2) {
276 			s = spriteObj;
277 		}
278 		
279       if (this.type == Sprite.TYPE_ANIMATION) {
280          this.mode = (s[Sprite.INDEX_TYPE] == "loop" ? Sprite.MODE_LOOP : Sprite.MODE_TOGGLE);
281          this.count = s[Sprite.INDEX_COUNT];
282          this.speed = s[Sprite.INDEX_SPEED];
283       }
284 
285       this.image = spriteResource.image;
286       this.frame = Rectangle2D.create(s[Sprite.INDEX_LEFT], s[Sprite.INDEX_TOP], s[Sprite.INDEX_WIDTH], s[Sprite.INDEX_HEIGHT]);
287       this.bbox = Rectangle2D.create(0, 0, s[Sprite.INDEX_WIDTH], s[Sprite.INDEX_HEIGHT]);
288    },
289 
290 	/**
291 	 * @private
292 	 */
293 	destroy: function() {
294 		this.bbox.destroy();
295 		this.frame.destroy();
296 		this.base();
297 	},
298 
299    /**
300     * @private
301     */
302    release: function() {
303       this.base();
304       this.mode = -1;
305       this.type = -1;
306       this.count = -1;
307       this.speed = -1;
308       this.frame = null;
309       this.image = null;
310       this.bbox = null;
311    },
312 
313    /**
314     * Returns <tt>true</tt> if the sprite is an animation.
315     * @return {Boolean} <tt>true</tt> if the sprite is an animation
316     */
317    isAnimation: function() {
318       return (this.type == Sprite.TYPE_ANIMATION);
319    },
320 
321    /**
322     * Returns <tt>true</tt> if the sprite is an animation and loops.
323     * @return {Boolean} <tt>true</tt> if the sprite is an animation and loops
324     */
325    isLoop: function() {
326       return (this.isAnimation() && this.mode == Sprite.MODE_LOOP);
327    },
328 
329    /**
330     * Returns <tt>true</tt> if the sprite is an animation and toggles.
331     * @return {Boolean} <tt>true</tt> if the sprite is an animation and toggles
332     */
333    isToggle: function() {
334       return (this.isAnimation() && this.mode == Sprite.MODE_TOGGLE);
335    },
336 
337    /**
338     * Get the bounding box for the sprite.
339     * @return {Rectangle2D} The bounding box which contains the entire sprite
340     */
341    getBoundingBox: function() {
342       return this.bbox;
343    },
344 
345    /**
346     * Gets the frame of the sprite. The frame is the rectangle that defining what
347     * portion of the image map the sprite frame occupies, given the specified time.
348     *
349     * @param time {Number} Current world time (can be obtained with {@link Engine#worldTime}
350     * @return {Rectangle2D} A rectangle which defines the frame of the sprite in
351     *         the source image map.
352     */
353    getFrame: function(time) {
354       if (!this.isAnimation()) {
355          return Rectangle2D.create(this.frame);
356       } else {
357          var f = Rectangle2D.create(this.frame);
358          var fn;
359          if (this.isLoop()) {
360             fn = Math.floor(time / this.speed) % this.count;
361          } else {
362             fn = Math.floor(time / this.speed) % (this.count * 2);
363             if (fn > this.count - 1) {
364                fn = this.count - (fn - (this.count - 1));
365             }
366          }
367          return f.offset(f.dims.x * fn, 0);
368       }
369    },
370 
371    /**
372     * Set the speed, in milliseconds, that an animation runs at.  If the sprite is
373     * not an animation, this has no effect.
374     *
375     * @param speed {Number} The number of milliseconds per frame of an animation
376     */
377    setSpeed: function(speed) {
378       if (speed >= 0) {
379          this.speed = speed;
380       }
381    },
382 
383    /**
384     * Get the number of milliseconds each frame is displayed for an animation
385     * @return {Number} The milliseconds per frame
386     */
387    getSpeed: function() {
388       return this.speed;
389    },
390 
391    /**
392     * The source image loaded by the {@link SpriteLoader} when the sprite was
393     * created.
394     * @return {HTMLImage} The source image the sprite is contained within
395     */
396    getSourceImage: function() {
397       return this.image;
398    }
399 
400 }, /** @scope Sprite.prototype */{
401    /**
402     * Gets the class name of this object.
403     * @return {String} The string "Sprite"
404     */
405    getClassName: function() {
406       return "Sprite";
407    },
408 
409    /** The sprite animation loops
410     * @type Number
411     */
412    MODE_LOOP: 0,
413 
414    /** The sprite animation toggles - Plays from the first to the last frame
415     *  then plays backwards to the first frame and repeats.
416     * @type Number
417     */
418    MODE_TOGGLE: 1,
419 
420    /** The sprite is a single frame
421     * @type Number
422     */
423    TYPE_SINGLE: 0,
424 
425    /** The sprite is an animation
426     * @type Number
427     */
428    TYPE_ANIMATION: 1,
429 
430    /** The field in the sprite definition file for the left pixel of the sprite frame
431     * @private
432     */
433    INDEX_LEFT: 0,
434 
435    /** The field in the sprite definition file for the top pixel of the sprite frame
436     * @private
437     */
438    INDEX_TOP: 1,
439 
440    /** The field in the sprite definition file for the width of the sprite frame
441     * @private
442     */
443    INDEX_WIDTH: 2,
444 
445    /** The field in the sprite definition file for the height of the sprite frame
446     * @private
447     */
448    INDEX_HEIGHT: 3,
449 
450    /** The field in the sprite definition file for the count of frames in the sprite
451     * @private
452     */
453    INDEX_COUNT: 4,
454 
455    /** The field in the sprite definition file for the speed in milliseconds that the sprite animates
456     * @private
457     */
458    INDEX_SPEED: 5,
459 
460    /** The field in the sprite definition file for the type of sprite animation
461     * @private
462     */
463    INDEX_TYPE: 6
464 });
465 
466 return Sprite;
467 
468 });
