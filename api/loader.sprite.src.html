loader.sprite.js


Summary

An extension of the image resource loader for handling sprites. Includes a class for working with loaded sprites.
Class Summary
Sprite Represents a sprite
SpriteLoader Loads sprites and makes them available to the system.

 39 Engine.initObject("SpriteLoader", "ImageLoader", function() {
 40 
 41 /**
 42  * @class Loads sprites and makes them available to the system.  Sprites are
 43  *        defined by an external resource that consist of the follwing definition.
 44  *        The definition file must include the source bitmap, the width and height
 45  *        of the bitmap, and a section that defines each sprite.
 46  *        <p/>
 47  *        Sprites can either be single frames, or animations:
 48  * <pre>
 49  * {
 50  *    // Frame (f): left, top, frameWidth, frameHeight
 51  *    // Animation (a): left, top, frameWidth, frameHeight, frameCount, speedMS, loop/toggle
 52  *    bitmapImage: "bitmapFile.ext",
 53  *    bitmapWidth: 320,
 54  *    bitmapHeight: 320,
 55  *    sprites: {
 56  *        "stand": {
 57  *           "f" : [0, 0, 32, 32]
 58  *        },
 59  *        "walk": {
 60  *           "a" : [32, 0, 32, 32, 3, 150, "loop"]
 61  *        }
 62  *    }
 63  * }
 64  * </pre>
 65  *
 66  * @constructor
 67  * @param name {String=SpriteLoader} The name of the resource loader
 68  * @extends ImageLoader
 69  */
 70 var SpriteLoader = ImageLoader.extend(/** @scope SpriteLoader.prototype */{
 71 
 72    sprites: null,
 73    
 74    queuedSprites: 0,
 75 
 76    constructor: function(name) {
 77       this.base(name || "SpriteLoader");
 78       this.sprites = {};
 79       this.queuedSprites = 0;
 80    },
 81 
 82    /**
 83     * Load a sprite resource from a URL.
 84     *
 85     * @param name {String} The name of the resource
 86     * @param url {String} The URL where the resource is located
 87     */
 88    load: function(name, url, info, path) {
 89 
 90       if (url)
 91       {
 92          var loc = window.location;
 93          if (url.indexOf(loc.protocol) != -1 && url.indexOf(loc.host) == -1) {
 94             Assert(false, "Sprites must be located on this server");
 95          }
 96 
 97          this.queuedSprites++;
 98 
 99          var thisObj = this;
100 
101          // Get the file from the server
102          $.get(url, function(data) {
103             var spriteInfo = EngineSupport.evalJSON(data);
104             // get the path to the resource file
105             var path = url.substring(0, url.lastIndexOf("/"));
106             thisObj.load(name, null, spriteInfo, path + "/");
107          });
108       }
109       else
110       {
111          info.bitmapImage = path + info.bitmapImage;
112          Console.info("Loading sprite: " + name + " @ " + info.bitmapImage);
113 
114          // Load the sprite image file
115          this.base(name, info.bitmapImage, info.bitmapWidth, info.bitmapHeight);
116 
117          // Store the sprite info
118          this.sprites[name] = info;
119          this.queuedSprites--;
120       }
121    },
122 
123    /**
124     * Get the sprite resource with the specified name from the cache.  The
125     * object returned contains the bitmap as <tt>image</tt> and
126     * the sprite definition as <tt>info</tt>.
127     *
128     * @param name {String} The name of the object to retrieve
129     * @return {Object} An object with two keys: "image" and "info"
130     */
131    get: function(name) {
132       var bitmap = this.base(name);
133       var sprite = {
134          image: bitmap,
135          info: this.sprites[name]
136       };
137       return sprite;
138    },
139 
140    /**
141     * Check to see if a named resource is "ready for use".
142     * @param name {String} The name of the resource to check ready status for,
143     *             or <tt>null</tt> for all resources in loader.
144     * @return {Boolean} <tt>true</tt> if the resource is loaded and ready to use
145     */
146    isReady: function(name) {
147       // If sprites are queued, we can't be totally ready
148       if (this.queuedSprites > 0) {
149          return false;
150       }
151       
152       return this.base(name);
153    },
154 
155    /**
156     * Creates a {@link Sprite} object representing the named sprite.
157     *
158     * @param resource {String} A loaded sprite resource
159     * @param sprite {String} The name of the sprite from the resource
160     * @return {Sprite} A {@link Sprite} instance
161     */
162    getSprite: function(resource, sprite) {
163       return Sprite.create(sprite, this.get(resource).info.sprites[sprite], this.get(resource));
164    },
165 
166    /**
167     * Get the names of all the sprites available in a resource.
168     *
169     * @param resource {String} The name of the resource
170     * @return {Array} All of the sprite names in the given loaded resource
171     */
172    getSpriteNames: function(resource) {
173       var s = [];
174       var spr = this.sprites[resource].sprites;
175       for (var i in spr) {
176          s.push(i);
177       }
178       return s;
179    },
180 
181    /**
182     * The name of the resource this loader will get.
183     * @returns {String} The string "sprite"
184     */
185    getResourceType: function() {
186       return "sprite";
187    }
188 
189 }, /** @scope SpriteLoader.prototype */{
190    /**
191     * Get the class name of this object.
192     * @return {String} The string "SpriteLoader"
193     */
194    getClassName: function() {
195       return "SpriteLoader";
196    }
197 });
198 
199 return SpriteLoader;
200 
201 });

203 Engine.initObject("Sprite", "PooledObject", function() {
204 
205 /**
206  * @class Represents a sprite
207  *
208  * @constructor
209  * @param name {String} The name of the sprite within the resource
210  * @param spriteObj {Object} Passed in by a {@link SpriteLoader}.  An array which defines the
211  *                  sprite frame, and parameters.
212  * @param spriteResource {Object} The sprite resource loaded by the {@link SpriteLoader}
213  * @extends PooledObject
214  */
215 var Sprite = PooledObject.extend(/** @scope Sprite.prototype */{
216 
217    // The type of sprite: Single or Animation
218    type: -1,
219 
220    // Animation mode: loop or toggle
221    mode: -1,
222 
223    // Animation frame count
224    count: -1,
225 
226    // Animation speed
227    speed: -1,
228 
229    // The rect which defines the sprite frame
230    frame: null,
231 
232    // The image map that contains the sprite(s)
233    image: null,
234 
235    // The bounding box for the sprite
236    bbox: null,
237 
238    /**
239     * @private
240     */
241    constructor: function(name, spriteObj, spriteResource) {
242       this.base(name);
243       this.type = (spriteObj["a"] ? Sprite.TYPE_ANIMATION : Sprite.TYPE_SINGLE);
244 
245       var s = (this.type == Sprite.TYPE_ANIMATION ? spriteObj["a"] : spriteObj["f"]);
246       if (this.type == Sprite.TYPE_ANIMATION) {
247          this.mode = (s[Sprite.INDEX_TYPE] == "loop" ? Sprite.MODE_LOOP : Sprite.MODE_TOGGLE);
248          this.count = s[Sprite.INDEX_COUNT];
249          this.speed = s[Sprite.INDEX_SPEED];
250       }
251 
252       this.image = spriteResource.image;
253       this.frame = Rectangle2D.create(s[Sprite.INDEX_LEFT], s[Sprite.INDEX_TOP], s[Sprite.INDEX_WIDTH], s[Sprite.INDEX_HEIGHT]);
254       this.bbox = Rectangle2D.create(0, 0, s[Sprite.INDEX_WIDTH], s[Sprite.INDEX_HEIGHT]);
255    },
256 
257    /**
258     * @private
259     */
260    release: function() {
261       this.base();
262       this.mode = -1;
263       this.type = -1;
264       this.count = -1;
265       this.speed = -1;
266       this.frame = null;
267       this.image = null;
268       this.bbox = null;
269    },
270 
271    /**
272     * Returns <tt>true</tt> if the sprite is an animation.
273     * @return {Boolean} <tt>true</tt> if the sprite is an animation
274     */
275    isAnimation: function() {
276       return (this.type == Sprite.TYPE_ANIMATION);
277    },
278 
279    /**
280     * Returns <tt>true</tt> if the sprite is an animation and loops.
281     * @return {Boolean} <tt>true</tt> if the sprite is an animation and loops
282     */
283    isLoop: function() {
284       return (this.isAnimation() && this.mode == Sprite.MODE_LOOP);
285    },
286 
287    /**
288     * Returns <tt>true</tt> if the sprite is an animation and toggles.
289     * @return {Boolean} <tt>true</tt> if the sprite is an animation and toggles
290     */
291    isToggle: function() {
292       return (this.isAnimation() && this.mode == Sprite.MODE_TOGGLE);
293    },
294 
295    /**
296     * Get the bounding box for the sprite.
297     * @return {Rectangle2D} The bounding box which contains the entire sprite
298     */
299    getBoundingBox: function() {
300       return this.bbox;
301    },
302 
303    /**
304     * Gets the frame of the sprite. The frame is the rectangle that defining what
305     * portion of the image map the sprite frame occupies, given the specified time.
306     *
307     * @param time {Number} Current world time (can be obtained with {@link Engine#worldTime}
308     * @return {Rectangle2D} A rectangle which defines the frame of the sprite in
309     *         the source image map.
310     */
311    getFrame: function(time) {
312       if (!this.isAnimation()) {
313          return this.frame;
314       } else {
315          var f = Rectangle2D.create(this.frame);
316          var fn;
317          if (this.isLoop()) {
318             fn = Math.floor(time / this.speed) % this.count;
319          } else {
320             fn = Math.floor(time / this.speed) % (this.count * 2);
321             if (fn > this.count - 1) {
322                fn = this.count - (fn - (this.count - 1));
323             }
324          }
325          return f.offset(f.dims.x * fn, 0);
326       }
327    },
328 
329    /**
330     * Set the speed, in milliseconds, that an animation runs at.  If the sprite is
331     * not an animation, this has no effect.
332     *
333     * @param speed {Number} The number of milliseconds per frame of an animation
334     */
335    setSpeed: function(speed) {
336       if (speed >= 0) {
337          this.speed = speed;
338       }
339    },
340 
341    /**
342     * Get the number of milliseconds each frame is displayed for an animation
343     * @return {Number} The milliseconds per frame
344     */
345    getSpeed: function() {
346       return this.speed;
347    },
348 
349    /**
350     * The source image loaded by the {@link SpriteLoader} when the sprite was
351     * created.
352     * @return {HTMLImage} The source image the sprite is contained within
353     */
354    getSourceImage: function() {
355       return this.image;
356    }
357 
358 }, /** @scope Sprite.prototype */{
359    /**
360     * Gets the class name of this object.
361     * @return {String} The string "Sprite"
362     */
363    getClassName: function() {
364       return "Sprite";
365    },
366 
367    /** The sprite animation loops
368     * @type Number
369     */
370    MODE_LOOP: 0,
371 
372    /** The sprite animation toggles - Plays from the first to the last frame
373     *  then plays backwards to the first frame and repeats.
374     * @type Number
375     */
376    MODE_TOGGLE: 1,
377 
378    /** The sprite is a single frame
379     * @type Number
380     */
381    TYPE_SINGLE: 0,
382 
383    /** The sprite is an animation
384     * @type Number
385     */
386    TYPE_ANIMATION: 1,
387 
388    /** The field in the sprite definition file for the left pixel of the sprite frame
389     * @private
390     */
391    INDEX_LEFT: 0,
392 
393    /** The field in the sprite definition file for the top pixel of the sprite frame
394     * @private
395     */
396    INDEX_TOP: 1,
397 
398    /** The field in the sprite definition file for the width of the sprite frame
399     * @private
400     */
401    INDEX_WIDTH: 2,
402 
403    /** The field in the sprite definition file for the height of the sprite frame
404     * @private
405     */
406    INDEX_HEIGHT: 3,
407 
408    /** The field in the sprite definition file for the count of frames in the sprite
409     * @private
410     */
411    INDEX_COUNT: 4,
412 
413    /** The field in the sprite definition file for the speed in milliseconds that the sprite animates
414     * @private
415     */
416    INDEX_SPEED: 5,
417 
418    /** The field in the sprite definition file for the type of sprite animation
419     * @private
420     */
421    INDEX_TYPE: 6
422 });
423 
424 return Sprite;
425 
426 });
