 33 // Includes
 34 Engine.include("/components/component.base.js");
 35 
 36 Engine.initObject("RenderComponent", "BaseComponent", function() {
 37 
 38 /**
 39  * @class The base component class for components which render
 40  *        to a context.  Rendering consists of anything which alters the
 41  *        visual state of the render context.
 42  *
 43  * @param name {String} The name of the component
 44  * @param priority {Number} The priority of the component between 0.0 and 1.0
 45  * @constructor
 46  * @extends BaseComponent
 47  * @description Creates a render component.
 48  */
 49 var RenderComponent = BaseComponent.extend(/** @scope RenderComponent.prototype */{
 50 
 51    drawMode: 0,
 52    
 53    oldDisplay: null,
 54 
 55    /**
 56     * @private
 57     */
 58    constructor: function(name, priority) {
 59       this.base(name, BaseComponent.TYPE_RENDERING, priority || 0.1);
 60       this.oldDisplay = null;
 61    },
 62 
 63    /**
 64     * Releases the component back into the object pool. See {@link PooledObject#release}
 65     * for more information.
 66     */
 67    release: function() {
 68       this.base();
 69       this.drawMode = 0;
 70    },
 71 
 72    /**
 73     * Set the draw mode of the component.  Currently this determines
 74     * if the component should render itself to the context or not.
 75     *
 76     * @param drawMode {Number} One of {@link RenderComponent#DRAW} or
 77     *                 {@link RenderComponent#NO_DRAW}.
 78     */
 79    setDrawMode: function(drawMode) {
 80       this.drawMode = drawMode;
 81    },
 82 
 83    /**
 84     * Get the drawing mode of the component.
 85     * @return {Number}
 86     */
 87    getDrawMode: function() {
 88       return this.drawMode;
 89    },
 90 
 91 	/**
 92 	 * Adjust the local transformation to accommodate the origin.
 93 	 * 
 94 	 * @param renderContext {RenderContext} The render context
 95 	 * @param before {Boolean} <code>true</code> if the transform is occurring before rendering
 96 	 */
 97 	transformOrigin: function(renderContext, before) {
 98 		if (this.getHostObject().getOrigin().isZero()) {
 99 			return;
100 		}
101 		
102 		if (before) {
103 			renderContext.pushTransform();
104 			var origin = Point2D.create(this.getHostObject().getOrigin());
105 			renderContext.setPosition(origin.neg());
106 			origin.destroy();
107 		} else {
108 			renderContext.popTransform();
109 		}
110 	},
111 
112    /**
113     * Handles whether or not the component should draw to the
114     * render context.
115     *
116     * @param renderContext {RenderContext} The rendering context
117     * @param time {Number} The engine time in milliseconds
118     */
119    execute: function(renderContext, time) {
120 
121       // Check visibility
122       if ((this.drawMode == RenderComponent.NO_DRAW) ||
123           this.getHostObject().getWorldBox &&
124           (!renderContext.getViewport().isIntersecting(this.getHostObject().getWorldBox())))
125       {
126          if (this.getHostObject().getElement() && !this.oldDisplay) {
127             this.oldDisplay = this.getHostObject().jQ().css("display");
128             this.getHostObject().jQ().css("display", "none");
129          }
130 
131          return false;
132       }
133 
134       if (this.getHostObject().getElement() && this.oldDisplay) {
135          this.getHostObject().jQ().css("display", this.oldDisplay);
136          this.oldDisplay = null;
137       }
138 
139       // The object is visible
140       Engine.vObj++;
141       return true;
142    }
143 
144 }, /** @scope RenderComponent.prototype */{ 
145 
146    /**
147     * Get the class name of this object
148     *
149     * @return {String} "RenderComponent"
150     */
151    getClassName: function() {
152       return "RenderComponent";
153    },
154 
155    /**
156     * The component should render itself to the rendering context.
157     * @type {Number}
158     */
159    DRAW: 0,
160 
161    /**
162     * The component <i>should not</i> render itself to the rendering context.
163     * @type {Number}
164     */
165    NO_DRAW: 1
166 
167 });
168 
169 return RenderComponent;
170 
171 });
