 33 Engine.include("/physics/common/b2Settings.js");
 34 Engine.include("/physics/collision/b2AABB.js");
 35 Engine.include("/physics/dynamics/b2World.js");
 36 Engine.include("/components/component.basebody.js");

 38 Engine.initObject("Simulation", "BaseObject", function() {
 39 
 40    /**
 41     * @class A representation of a physical world.  This object is used to
 42     *        introduce Box2dJS physics into your game by creating a world
 43     *        which supports the physical structures provided by Box2dJS.  You
 44     *        will need to create a <tt>Simulation</tt> before you can utilize
 45     *        physics in a game.
 46     *        <p/>
 47     *        See either "/demos/physics/" or "/demos/physics2" for examples
 48     *        of utilizing the <tt>Simulation</tt> object with rigid body components.
 49     *         
 50     * @param name {String} The name of the object
 51     * @param worldBoundary {Rectangle2D} The physical world boundary
 52     * @param [gravity] {Vector2D} The world's gravity vector. default: [0, 650]
 53     * @extends BaseObject
 54     * @constructor
 55     * @description Create a physical world for Box2dJS
 56     */
 57    var Simulation = BaseObject.extend(/** @scope Simulation.prototype */{
 58 
 59       worldAABB: null,
 60       world: null,
 61       gravity: null,
 62       doSleep: true,
 63       worldBoundary: null,
 64       integrations: 0,
 65 
 66       constructor: function(name, worldBoundary, gravity) {
 67          this.base(name);
 68          this.gravity = gravity || Vector2D.create(0, 650);
 69          this.worldAABB = new b2AABB();
 70          
 71          this.worldBoundary = worldBoundary;
 72          var wb = worldBoundary.get();
 73          this.worldAABB.minVertex.Set(-1000, -1000);
 74          this.worldAABB.maxVertex.Set(2000, 2000);
 75          this.doSleep = true;
 76          this.integrations = Simulation.DEFAULT_INTEGRATIONS;
 77 
 78          var g = this.gravity.get();
 79          var grav = new b2Vec2(g.x, g.y);
 80 
 81          // Create the world
 82          this.world = new b2World(this.worldAABB, grav, this.doSleep);     
 83       },
 84       
 85       destroy: function() {
 86          this.worldBoundary.destroy();
 87          this.gravity.destroy();
 88          this.base();
 89       },
 90       
 91       release: function() {
 92          this.worldAABB = null;
 93          this.gravity = null,
 94          this.worldBoundary = null;
 95          this.world = null;
 96          this.base();
 97       },
 98       
 99       update: function(renderContext, time) {
100          this.world.Step(1/Engine.getFPS(), this.integrations);   
101       },
102       
103       /**
104        * Support method to get the ground body for the world.
105        * @return {b2Body} The world's ground body
106        * @private
107        */
108       getGroundBody: function() {
109          return this.world.GetGroundBody();
110       },
111       
112       /**
113        * Support method to add a body to the simulation.  The body must be one of the
114        * box2d-js body types.  This method is intended to be used by {@link SimulationObject}.
115        * 
116        * @param b2jsBodyDef {b2BodyDef} A box2d-js Body definition object
117        * @private
118        */
119       addBody: function(b2jsBodyDef) {
120          return this.world.CreateBody(b2jsBodyDef);
121       },
122       
123       /**
124        * Support method to add a joint to the simulation.  The joint must be one of the
125        * box2d-js joint types.  This method is intended to be used by {@link SimulationObject}.
126        * 
127        * @param b2jsJointDef {b2JointDef} A box2d-js Joint definition object
128        * @private
129        */
130       addJoint: function(b2jsJointDef) {
131          return this.world.CreateJoint(b2jsJointDef);
132       },
133       
134       /**
135        * Support method to remove a body from the simulation.  The body must be one of the
136        * box2d-js body types.  This method is intended to be used by {@link SimulationObject}.
137        * 
138        * @param b2jsBody {b2Body} A box2d-js Body object
139        * @private
140        */
141       removeBody: function(b2jsBody) {
142          this.world.DestroyBody(b2jsBody);
143       },
144       
145       /**
146        * Support method to remove a joint from the simulation.  The joint must be one of the
147        * box2d-js joint types.  This method is intended to be used by {@link SimulationObject}.
148        * 
149        * @param b2jsJoint {b2Joint} A box2d-js Joint object
150        * @private
151        */
152       removeJoint: function(b2jsJoint) {
153          this.world.DestroyJoint(b2jsJoint);
154       },
155       
156       /**
157        * Set the number of integrations per frame.  A higher number will result
158        * in more accurate collisions, but will result in slower performance.
159        * 
160        * @param integrations {Number} The number of integrations per frame
161        */
162       setIntegrations: function(integrations) {
163          this.integrations = integrations || Simulation.DEFAULT_INTEGRATIONS;   
164       },
165       
166       /**
167        * Get the number of integrations per frame.
168        * @return {Number}
169        */
170       getIntegrations: function() {
171          return this.integrations;   
172       },
173       
174       /**
175        * Add a simple box body to the simulation.  The body doesn't have a visual
176        * representation, but exists in the simulation and can be interacted with.
177        *
178        * @param pos {Point2D} The position where the body's top/left is located
179        * @param extents {Point2D} The width and height of the body
180        * @param properties {Object} An object with up to three properties: <ul>
181        * 		<li>restitution - The bounciness of the body</li>
182        *			<li>friction - Friction against this body</li>
183        *			<li>density - The density of the object, defaults to zero (fixed in place)</li></ul>
184        *
185        * @return {b2BodyDef} A Box2D-JS body definition object representing the box
186        */
187       addSimpleBoxBody: function(pos, extents, properties) {
188          var e = extents.get(), p = pos.get();
189          properties = properties || {};
190          
191          var boxDef = new b2BoxDef();
192          boxDef.extents.Set(e.x, e.y);
193 
194 			// Set the properties
195          boxDef.restitution = properties.restitution || BaseBodyComponent.DEFAULT_RESTITUTION;
196 			boxDef.friction = properties.friction || BaseBodyComponent.DEFAULT_FRICTION;
197 			boxDef.density = properties.density || 0;
198 
199          var b2body = new b2BodyDef();
200          b2body.AddShape(boxDef);
201          b2body.position.Set(p.x, p.y);
202          return this.addBody(b2body);
203       },
204       
205       /**
206        * Add a simple circle body to the simulation.  The body doesn't have a visual
207        * representation, but exists in the simulation and can be interacted with.
208        *
209        * @param pos {Point2D} The position where the body's center is located
210        * @param radius {Point2D} The radius of the circle body
211        * @param properties {Object} An object with up to three properties: <ul>
212        * 		<li>restitution - The bounciness of the body</li>
213        *			<li>friction - Friction against this body</li>
214        *			<li>density - The density of the object, defaults to zero (fixed in place)</li></ul>
215        *
216        * @return {b2BodyDef} A Box2D-JS body definition object representing the circle
217        */
218       addSimpleCircleBody: function(pos, radius, properties) {
219          var p = pos.get();
220          properties = properties || {};
221          
222          var circleDef = new b2CircleDef();
223          circleDef.radius = radius;
224 
225 			// Set the properties
226          circleDef.restitution = properties.restitution || BaseBodyComponent.DEFAULT_RESTITUTION;
227 			circleDef.friction = properties.friction || BaseBodyComponent.DEFAULT_FRICTION;
228 			circleDef.density = properties.density || 0;
229 
230          var b2body = new b2BodyDef();
231          b2body.AddShape(circleDef);
232          b2body.position.Set(p.x, p.y);
233          return this.addBody(b2body);
234       }
235       
236    }, /** @scope Simulation.prototype */{
237       
238       /**
239        * Get the class name as a string.
240        * @return {String} "Simulation"
241        */
242       getClassName: function() {
243          return "Simulation";
244       },
245       
246       /**
247        * The default number of integrations per frame
248        */
249       DEFAULT_INTEGRATIONS: 10
250       
251    });
252 
253    return Simulation;
254    
255 });
