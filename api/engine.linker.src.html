 35  * @class A static class for processing class files, looking for dependencies, and
 36  *        ensuring that all dependencies exist before initializing a class.  The linker
 37  *        expects that files will follow a fairly strict format, so that patterns can be
 38  *        identified and all dependencies resolved.
 39  *        <p/>
 40  *        These methods handle object dependencies so that each object will be
 41  *        initialized as soon as its dependencies become available.  Using this method
 42  *        scripts can be loaded immediately, using the browsers threading model, and
 43  *        executed when dependencies are fulfilled.
 44  *
 45  * @static
 46  */

 80    /**
 81     * Initializes an object for use in the engine.  Calling this method is required to make sure
 82     * that all dependencies are resolved before actually instantiating an object of the specified
 83     * class.
 84     *
 85     * @param objectName {String} The name of the object class
 86     * @param primaryDependency {String} The name of the class for which the <tt>objectName</tt> class is
 87     *                                   dependent upon.  Specifying <tt>null</tt> will assume the <tt>Base</tt> class.
 88     * @param fn {Function} The function to run when the object can be initialized.
 89     */
 90    initObject: function(objectName, primaryDependency, fn) {
 91       var objDeps = Linker.findAllDependencies(objectName, fn);
 92 
 93       // Add the known dependency to the list of object
 94       // dependencies we discovered
 95       if (primaryDependency) {
 96          objDeps.push(primaryDependency);
 97       }
 98       
 99       var newDeps = [];
100 
101       // Check for nulls and circular references
102       for (var d in objDeps) {
103          if (objDeps[d] != null) {
104             if (objDeps[d].split(".")[0] != objectName) {
105                newDeps.push(objDeps[d]);
106             } else {
107                Console.warn("Object references itself: ", objectName);
108             }
109          }
110       }
111 
112       // Store the object for intialization when all dependencies are resolved
113       Linker.dependencyList[objectName] = {deps: newDeps, objFn: fn};
114       Linker.dependencyCount++;
115       Console.info(objectName, " depends on: ", newDeps);
116 
117       // Check for 1st level circular references
118       this.checkCircularRefs(objectName);
119 
120       // After a period of time has passed, we'll check our dependency list.
121       // If anything remains, we'll drop the bomb that certain files didn't resolve...
122       window.clearTimeout(Linker.dependencyTimer);
123       Linker.dependencyTimer = window.setTimeout(Linker.checkDependencyList, Linker.dependencyCheckTimeout);
124 
125       if (!Linker.dependencyProcessor) {
126          Linker.dependencyProcessor = window.setTimeout(Linker.processDependencies, Linker.dependencyProcessTimeout);
127       }
128    },

130    /**
131     * Check the object to make sure the namespace(s) are defined.
132     * @param objName {String} A dot-notation class name
133     * @private
134     */
135    checkNSDeps: function(objName) {
136       var objHr = objName.split(".");
137 
138       if (objHr.length == 0) {
139          return true;
140       }
141 
142       var o = window;
143       for (var i = 0; i < objHr.length - 1; i++) {
144          o = o[objHr[i]];
145          if (!o) {
146             return false;
147          }
148       }
149       return true;
150    },

169    /**
170     * The dependency processor and link checker.
171     * @private
172     */
173    processDependencies: function() {
174       var pDeps = [];
175       var dCount = 0;
176       for (var d in Linker.dependencyList) {
177          // Check to see if the dependencies of an object are loaded
178          // We also need to make sure that it's namespace(s) are initialized
179          dCount++;
180 
181          var deps = Linker.dependencyList[d].deps;
182          var resolved = [];
183          var miss = false;
184          for (var dep in deps) {
185             if (deps[dep] != null && window[deps[dep]] == null) {
186                miss = true;
187             } else {
188                resolved.push(deps[dep]);
189             }
190          }
191 
192          for (var x in resolved) {
193             EngineSupport.arrayRemove(Linker.dependencyList[d].deps, resolved[x]);
194          }
195 
196          if (!miss && Linker.checkNSDeps(d)) {
197             // We can initialize it now
198             Console.log("Initializing '", d, "'");
199             var parentObj = Linker.getParentClass(d);
200             parentObj[d] = Linker.dependencyList[d].objFn();
201             Console.info("-> Initialized '", d, "'");
202             Linker.loadedClasses.push(d);
203 
204             // Remember what we processed so we don't process them again
205             pDeps.push(d);
206             
207             // After it has been initialized, check to see if it has the
208             // resolved() class method
209             if (parentObj[d].resolved) {
210                parentObj[d].resolved();
211             }
212          }
213       }
214 
215       for (var i in pDeps) {
216          delete Linker.dependencyList[pDeps[i]];
217       }
218 
219       if (dCount != 0) {
220          Linker.dependencyProcessor = window.setTimeout(Linker.processDependencies, Linker.dependencyProcessTimeout);
221       } else {
222          window.clearTimeout(Linker.dependencyProcessor);
223          Linker.dependencyProcessor = null;
224       }
225    },

232    /**
233     * Find variables defined in a function:
234     *   var x = 10;
235     *   var y;
236     *   for (var z=3; z < 10; z++)
237     *   -- Still need to handle:  var x,y,z;
238     *
239     * @private
240     */
241    findVars: function(objectName, obj) {
242       // Find all variables explicitly defined
243       var def = obj.toString();
244       var vTable = [];
245       var nR = "([\\$_\\w\\.]*)";
246       var vR = new RegExp("(var\\s*" + nR + "\\s*)","g");
247       var m;
248       while ((m = vR.exec(def)) != null) {
249          vTable.push(m[2]); 
250       }
251       return vTable;
252    },

254    /**
255     * Find object dependencies in variables, arguments, and method usage.
256     * @private
257     */
258    findDependencies: function(objectName, obj) {
259       // Find all dependent objects
260       var def = obj.toString();
261       var dTable = [];
262       var nR = "([\\$_\\w\\.]*)";
263       var nwR = new RegExp("(new\\s+" + nR + ")","g");
264       var ctR = new RegExp("(" + nR + "\\.create\\()","g");
265       var fcR = new RegExp("(" + nR + "\\()", "g");
266       var inR = new RegExp("(intanceof\\s+"+ nR + ")", "g");
267       var m;
268 
269       // "new"-ing objects
270       while ((m = nwR.exec(def)) != null) {
271          if (EngineSupport.indexOf(dTable, m[2]) == -1) {
272             dTable.push(m[2]);
273          }
274       }
275 
276       // "create"-ing objects
277       while ((m = ctR.exec(def)) != null) {
278          if (EngineSupport.indexOf(dTable, m[2]) == -1) {
279             dTable.push(m[2]);
280          }
281       }
282 
283       // method dependencies
284       while ((m = fcR.exec(def)) != null) {
285          if (m[2].indexOf(".") != -1) {
286             var k = m[2].split(".")[0];
287             if (EngineSupport.indexOf(dTable, k) == -1) {
288                dTable.push(k);
289             }
290          }
291       }
292 
293       // "instanceof" checks
294       while ((m = inR.exec(def)) != null) {
295          if (EngineSupport.indexOf(dTable, m[2]) == -1) {
296             dTable.push(m[2]);
297          }
298       }
299       return dTable;
300    },

303     * Process anonymous functions, extracting the function arguments
304     * @private

306    findAnonArgs: function(objectName, str) {
307       var a = [];
308 
309       var aFnRE = new RegExp("function\\s*\\(([\\$\\w_, ]*?)\\)","g");
310       while ((m = aFnRE.exec(str)) != null) {
311          var args = m[1].split(",");
312 
313          for (var x in args) {
314 	         a.push(args[x].replace(" ",""));
315          }
316       }
317 
318       return a;

321    /**
322     * Finds all of the dependencies within an object class.
323     * @private
324     */
325    findAllDependencies: function(objectName, obj) {
326       var defs;
327       var fTable = {};
328       Console.warn("Process: " + objectName);
329 
330       try {
331          var k = obj();
332       } catch (ex) {
333          // The class probably extends a non-existent class. Replace the parent
334          // class with a known class and evaluate as a dummy class
335          var extRE = new RegExp("(var\\s*)?([\\$_\\w\\.]*?)\\s*=\\s*([\\$_\\w\\.]*?)\\.extend\\(");
336          var classDef = obj.toString();
337          var nm = null;
338          classDef = classDef.replace(extRE, function(str, varDef, classname, parent, offs, s) {
339             nm = classname;
340             return "return Base.extend(";
341          });
342          try {
343             k = eval("(" + classDef.replace("return " + nm + ";", "") + ")();");
344          } catch (inEx) {
345             Console.error("Cannot parse class '" + nm + "'");
346          }
347       }
348 
349 		var kInstance = null;
350       if ($.isFunction(k)) {
351          // If the class is an instance, get it's class object
352 			kInstance = k;
353          k = k.prototype;
354       }
355 
356       // Find the internal functions
357       for (var f in k) {
358          var def = k[f];
359 			if (kInstance && f == "constructor" && $.isFunction(kInstance) && k.hasOwnProperty(f)){
360 				// If it's an instance, we're looking at the constructor, and the
361 				// instance has its own constructor (not inherited)
362 				def = kInstance;
363 			}
364          if ($.isFunction(def) && k.hasOwnProperty(f)) {
365 				def = def.toString();
366             var fR = new RegExp("function\\s*\\(([\\$\\w_, ]*?)\\)\\s*\\{((.|\\s)*)","g");
367             var m = fR.exec(def);
368             if (m) {
369                // Remove strings, then comments
370                var fdef = m[2].replace(/(["|']).*?\1/g, "");
371                fdef = fdef.replace(/\/\/.*\r\n/g, "");
372                fdef = fdef.replace("\/\*(.|\s)*?\*\/", "");
373 
374                // Process anonymous function arguments
375                var anonArgs = Linker.findAnonArgs(objectName, fdef);
376 
377                // Process each function
378                fTable[f] = { vars: Linker.findVars(objectName, fdef),
379                              deps: Linker.findDependencies(objectName, fdef) };
380                fTable[f].deps = EngineSupport.filter(fTable[f].deps, function(e) {
381                   return (e != "" && e != "this" && e != "arguments");
382                });
383 
384 					// Consider arguments as local variables
385                var args = m[1].split(",");
386                var vs = fTable[f].vars;
387                for (var a in args) {
388                   if (EngineSupport.indexOf(vs, args[a]) == -1) {
389                      vs.push(args[a].replace(" ",""));
390                   }
391                }
392 
393                // Combine args with the anonymous function args
394                for (var a in anonArgs) {
395                   if (EngineSupport.indexOf(vs, anonArgs[a]) == -1) {
396                      vs.push(anonArgs[a]);
397                   }
398                }
399             }
400          }
401       }
402 
403       // This is useful for debugging dependency problems...
404       Console.log("DepTable: ", objectName, fTable);
405 
406       return Linker.procDeps(objectName, fTable);
407    },

409    /**
410     * Process dependencies and clear any that have been resolved.
411     * @private
412     */
413    procDeps: function(objectName, fTable) {
414       // Remove dependencies resolved by local variables and arguments
415       var r = [];
416       var allDeps = [];
417       for (var f in fTable) {
418          var deps = fTable[f].deps;
419          var vars = fTable[f].vars;
420 
421          for (var d in deps) {
422             var dp = deps[d].split(".")[0];
423             if (EngineSupport.indexOf(vars, dp) != -1 && EngineSupport.indexOf(r, deps[d]) == -1) {
424                r.push(deps[d]);
425             }
426          }
427 
428          fTable[f].deps = EngineSupport.filter(deps, function(e) {
429             return (EngineSupport.indexOf(r, e) == -1);
430          });
431 
432          for (var d in fTable[f].deps) {
433             if (EngineSupport.indexOf(allDeps, fTable[f].deps[d]) == -1) {
434                allDeps.push(fTable[f].deps[d]);
435             }
436          }
437       }
438 
439       return allDeps;
440    },

442    /**
443     * Perform a quick resolution on first-level circular references.
444     * @private
445     */
446    checkCircularRefs: function(objectName) {
447       var deps = Linker.dependencyList[objectName].deps;
448       var r = [];
449       for (var dep in deps) {
450          if (Linker.dependencyList[deps[dep]] && EngineSupport.indexOf(Linker.dependencyList[deps[dep]].deps, objectName) != -1) {
451             // Try removing the circular reference
452             EngineSupport.arrayRemove(Linker.dependencyList[objectName].deps, deps[dep]);
453          }
454       }

457    /**
458     * Check the dependency list for any unresolved dependencies.  Anything that hasn't
459     * been resolved will be dumped to the console as an error.
460     * @private
461     */
462    checkDependencyList: function() {
463       // Stop processing
464       window.clearTimeout(Linker.dependencyTimer);
465       Linker.dependencyTimer = null;
466       window.clearTimeout(Linker.dependencyProcessor);
467       Linker.dependencyProcessor = null;      
468 
469       // Build the list
470       var unresolved = [], unresDeps = "", dCount = 0;
471       for (var obj in Linker.dependencyList) {
472          dCount++;
473          unresDeps += "Object '" + obj + "' has the following unresolved dependencies: ";
474 			unresDeps += "(" + Linker.dependencyList[obj].deps.length + ") ";
475          for (var d in Linker.dependencyList[obj].deps) {
476             unresDeps += Linker.dependencyList[obj].deps[d] + " ";
477          }
478          unresolved.push(unresDeps);
479 			unresDeps = "";
480       }
481       
482       if (dCount != 0) {
483          // Dump the dependency list
484          Console.setDebugLevel(Console.DEBUGLEVEL_ERRORS);
485 			for (var ud in unresolved) {
486 	         Console.error(unresolved[ud]);
487 			}
488          Engine.shutdown();
489       }
490    },
