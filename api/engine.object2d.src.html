engine.object2d.js


Summary

An extension of the HostObject which is specifically geared towards 2d game development.
Class Summary
Object2D An object for use in a 2d environment.

 34 // Includes
 35 Engine.include("/engine/engine.hostobject.js");

 37 Engine.initObject("Object2D", "HostObject", function() {
 38 
 39 /**
 40  * @class An object for use in a 2d environment.  Methods for getting the position, rotation
 41  * and scale should be implemented within the extended class.
 42  * 
 43  * @param name {String} The name of the object
 44  * @extends HostObject
 45  * @constructor
 46  * @description Create a 2d host object
 47  */
 48 var Object2D = HostObject.extend(/** @scope Object2D.prototype */{
 49 
 50    /** @private */
 51    zIndex: 1,
 52 
 53    /** @private */
 54    bBox: null,
 55 
 56    /**
 57     * @private
 58     */
 59    constructor: function(name) {
 60       this.base(name);
 61       this.lastPosition = new Point2D(5,5);
 62       this.zIndex = 1;
 63    },
 64 
 65    /**
 66     * Release the object back into the pool.
 67     */
 68    release: function() {
 69       this.base();
 70       this.zIndex = 1;
 71       this.bBox = null;
 72    },
 73 
 74    /**
 75     * Set the bounding box of this object
 76     *
 77     * @param rect {Rectangle2D} The rectangle that completely encompasses
 78     *                           this object.
 79     */
 80    setBoundingBox: function(rect) {
 81       this.bBox = rect;
 82    },
 83 
 84    /**
 85     * Get the bounding box of this object
 86     *
 87     * @return {Rectangle2D} The object bounding rectangle
 88     */
 89    getBoundingBox: function() {
 90       return this.bBox;
 91    },
 92 
 93    /**
 94     * Get the world bounding box.
 95     * @return {Rectangle2D} The world bounding rectangle
 96     */
 97    getWorldBox: function() {
 98       return new Rectangle2D(this.getBoundingBox()).offset(this.getRenderPosition());
 99    },
100    
101    /**
102     * [ABSTRACT] Get the world bounding circle.
103     * @return {Circle2D}
104     */
105    getCircle: function() {
106       // ABSTRACT METHOD
107    },
108    
109    /**
110     * [ABSTRACT] Get the velocity of the object.
111     * @return {Vector2D}
112     */
113    getVelocity: function() {
114       // ABSTRACT METHOD
115    },
116 
117    /**
118     * [ABSTRACT] Set the position of the object
119     * @param point {Point2D} The position of the object
120     */
121    setPosition: function(point) {
122       // ABSTRACT
123    },
124 
125    /**
126     * Get the position of the object.
127     * @return {Point2D} The position
128     */
129    getPosition: function() {
130       return Point2D.ZERO;
131    },
132 
133    /**
134     * Get the render position of the object.
135     * @return {Point2D}
136     */
137    getRenderPosition: function() {
138       return Point2D.ZERO;
139    },
140 
141    /**
142     * Get the last position the object was rendered at.
143     * @return {Point2D}
144     */
145    getLastPosition: function() {
146       return Point2D.ZERO;
147    },
148 
149    /**
150     * [ABSTRACT] Set the rotation of the object
151     * @param angle {Number} The rotation angle
152     */
153    setRotation: function(angle) {
154    },
155 
156    /**
157     * Get the rotation of the object
158     * @return {Number}
159     */
160    getRotation: function() {
161       return 0;
162    },
163 
164    /**
165     * [ABSTRACT] Set the scale of the object along the X and Y axis.
166     * @param scaleX {Number} The scale along the X axis
167     * @param scaleY {Number} The scale along the Y axis
168     */
169    setScale: function(scaleX, scaleY) {
170    },
171 
172    /**
173     * Get the uniform scale of the object.
174     * @return {Number}
175     */
176    getScale: function() {
177       return 1;
178    },
179 
180    /**
181     * Get the scale of the object along the X axis
182     * @return {Number}
183     */
184    getScaleX: function() {
185       return 1;
186    },
187 
188    /**
189     * Get the scale of the object along the Y axis.
190     * @return {Number}
191     */
192    getScaleY: function() {
193       return 1;
194    },
195 
196    /**
197     * Set the depth at which this object will render to
198     * the context.  The lower the z-index, the further
199     * away from the front the object will draw.
200     *
201     * @param zIndex {Number} The z-index of this object
202     */
203    setZIndex: function(zIndex) {
204       this.zIndex = zIndex;
205       if (this.getRenderContext()) {
206          this.getRenderContext().sort();
207       }
208    },
209 
210    /**
211     * Get the depth at which this object will render to
212     * the context.
213     *
214     * @return {Number}
215     */
216    getZIndex: function() {
217       return this.zIndex;
218    },
219 
220    /**
221     * When editing objects, this method returns an object which
222     * contains the properties with their getter and setter methods.
223     * @return {Object} The properties object
224     */
225    getProperties: function() {
226       var self = this;
227       var prop = this.base(self);
228       return $.extend(prop, {
229          "ZIndex"       : [function() { return self.getZIndex(); },
230                            function(i){   self.setZIndex(i); }, true],
231          "BoundingBox"  : [function() { return self.getBoundingBox().toString(); },
232                            null, false],
233          "WorldBox"     : [function() { return self.getWorldBox().toString(); },
234                            null, false],
235          "Position"     : [function() { return self.getPosition(); },
236                            function(i) { var p = i.split(","); self.setPosition(Point2D.create(p[0],p[1])); }, true],
237          "RenderPos"    : [function() { return self.getRenderPosition() },
238                            null, false],
239          "Rotation"     : [function() { return self.getRotation(); },
240                            function(i) { self.setRotation(i); }, true],
241          "Scale"        : [function() { return self.getScale(); },
242                            function(i) {self.setScale(i,i); }, true]
243       });
244    }
245 
246 }, /** @scope Object2D.prototype */{
247    /**
248     * Get the class name of this object
249     *
250     * @return {String} "Object2D"
251     */
252    getClassName: function() {
253       return "Object2D";
254    }
255 });
256 
257 return Object2D;
258 
259 });
