engine.object2d.js


Summary

An extension of the HostObject which is specifically geared towards 2d game development.
Class Summary
Object2D An object for use in a 2d environment.

 34 // Includes
 35 Engine.include("/engine/engine.hostobject.js");

 37 Engine.initObject("Object2D", "HostObject", function() {
 38 
 39 /**
 40  * @class An object for use in a 2d environment.  Methods for getting the position, rotation
 41  * and scale should be implemented within the extended class.
 42  * 
 43  * @param name {String} The name of the object
 44  * @extends HostObject
 45  * @constructor
 46  * @description Create a 2d host object
 47  */
 48 var Object2D = HostObject.extend(/** @scope Object2D.prototype */{
 49 
 50    /** @private */
 51    zIndex: 1,
 52 
 53    /** @private */
 54    bBox: null,
 55 	wBox: null,
 56 	lastPosition: null,
 57 	
 58 	origin: null,
 59 
 60    /**
 61     * @private
 62     */
 63    constructor: function(name) {
 64       this.base(name);
 65       this.lastPosition = Point2D.create(5,5);
 66 		this.bBox = Rectangle2D.create(0,0,1,1);
 67 		this.wBox = Rectangle2D.create(0,0,1,1);
 68       this.zIndex = 1;
 69 		this.origin = Point2D.create(0,0);
 70    },
 71 	
 72 	/**
 73 	 * Destroy the object.
 74 	 */
 75 	destroy: function() {
 76 		this.bBox.destroy();
 77 		this.wBox.destroy();
 78 		this.lastPosition.destroy();
 79 		this.base();
 80 	},
 81 
 82    /**
 83     * Release the object back into the pool.
 84     */
 85    release: function() {
 86       this.base();
 87       this.zIndex = 1;
 88       this.bBox = null;
 89 		this.lastPosition = null;
 90    },
 91 
 92 	/**
 93 	 * Set the render origin of the object.  The render origin is where the object will be
 94 	 * centered around when drawing position and rotation.
 95 	 *  
 96 	 * @param point {Point2D} The render origin (default: 0,0 - top left corner)
 97 	 */
 98 	setOrigin: function(point) {
 99 		this.origin = point;
100 	},
101 	
102 	/**
103 	 * Get the render origin of the object.
104 	 * @return {Point2D}
105 	 */
106 	getOrigin: function() {
107 		return this.origin;
108 	},
109 
110    /**
111     * Set the bounding box of this object
112     *
113     * @param x {Number|Rectangle2D} The X coordinate, or the rectangle that completely encompasses
114     *                           		this object.
115     * @param y {Number} If X is a coordinate, this is the Y coordinate
116     * @param w {Number} If X is a coordinate, this is the width
117     * @param h {Number} If X is a coordinate, this is the height
118     */
119    setBoundingBox: function(x,y,w,h) {
120 		this.bBox.set(x,y,w,h);
121    },
122 
123    /**
124     * Get the bounding box of this object
125     *
126     * @return {Rectangle2D} The object bounding rectangle
127     */
128    getBoundingBox: function() {
129       return this.bBox;
130    },
131 
132    /**
133     * Get the world bounding box.
134     * @return {Rectangle2D} The world bounding rectangle
135     */
136    getWorldBox: function() {
137 		this.wBox.set(this.getBoundingBox());
138       return this.wBox.offset(this.getRenderPosition());
139    },
140    
141    /**
142     * [ABSTRACT] Get the world bounding circle.
143     * @return {Circle2D}
144     */
145    getCircle: function() {
146       // ABSTRACT METHOD
147    },
148    
149    /**
150     * [ABSTRACT] Get the velocity of the object.
151     * @return {Vector2D}
152     */
153    getVelocity: function() {
154       // ABSTRACT METHOD
155    },
156 
157    /**
158     * [ABSTRACT] Set the position of the object
159     * @param point {Point2D} The position of the object
160     */
161    setPosition: function(point) {
162       // ABSTRACT
163    },
164 
165    /**
166     * Get the position of the object.
167     * @return {Point2D} The position
168     */
169    getPosition: function() {
170       return Point2D.ZERO;
171    },
172 
173    /**
174     * Get the render position of the object.
175     * @return {Point2D}
176     */
177    getRenderPosition: function() {
178       return Point2D.ZERO;
179    },
180 
181    /**
182     * Get the last position the object was rendered at.
183     * @return {Point2D}
184     */
185    getLastPosition: function() {
186       return Point2D.ZERO;
187    },
188 
189    /**
190     * [ABSTRACT] Set the rotation of the object
191     * @param angle {Number} The rotation angle
192     */
193    setRotation: function(angle) {
194    },
195 
196    /**
197     * Get the rotation of the object
198     * @return {Number}
199     */
200    getRotation: function() {
201       return 0;
202    },
203 
204    /**
205     * [ABSTRACT] Set the scale of the object along the X and Y axis.
206     * @param scaleX {Number} The scale along the X axis
207     * @param scaleY {Number} The scale along the Y axis
208     */
209    setScale: function(scaleX, scaleY) {
210    },
211 
212    /**
213     * Get the uniform scale of the object.
214     * @return {Number}
215     */
216    getScale: function() {
217       return 1;
218    },
219 
220    /**
221     * Get the scale of the object along the X axis
222     * @return {Number}
223     */
224    getScaleX: function() {
225       return 1;
226    },
227 
228    /**
229     * Get the scale of the object along the Y axis.
230     * @return {Number}
231     */
232    getScaleY: function() {
233       return 1;
234    },
235 
236    /**
237     * Set the depth at which this object will render to
238     * the context.  The lower the z-index, the further
239     * away from the front the object will draw.
240     *
241     * @param zIndex {Number} The z-index of this object
242     */
243    setZIndex: function(zIndex) {
244       this.zIndex = zIndex;
245       if (this.getRenderContext()) {
246          this.getRenderContext().sort();
247       }
248    },
249 
250    /**
251     * Get the depth at which this object will render to
252     * the context.
253     *
254     * @return {Number}
255     */
256    getZIndex: function() {
257       return this.zIndex;
258    },
259 
260    /**
261     * When editing objects, this method returns an object which
262     * contains the properties with their getter and setter methods.
263     * @return {Object} The properties object
264     */
265    getProperties: function() {
266       var self = this;
267       var prop = this.base(self);
268       return $.extend(prop, {
269          "ZIndex"       : [function() { return self.getZIndex(); },
270                            function(i){   self.setZIndex(i); }, true],
271          "BoundingBox"  : [function() { return self.getBoundingBox().toString(); },
272                            null, false],
273          "WorldBox"     : [function() { return self.getWorldBox().toString(); },
274                            null, false],
275          "Position"     : [function() { return self.getPosition(); },
276                            function(i) { var p = i.split(","); self.setPosition(Point2D.create(p[0],p[1])); }, true],
277          "RenderPos"    : [function() { return self.getRenderPosition() },
278                            null, false],
279          "Rotation"     : [function() { return self.getRotation(); },
280                            function(i) { self.setRotation(i); }, true],
281          "Scale"        : [function() { return self.getScale(); },
282                            function(i) {self.setScale(i,i); }, true]
283       });
284    }
285 
286 }, /** @scope Object2D.prototype */{
287    /**
288     * Get the class name of this object
289     *
290     * @return {String} "Object2D"
291     */
292    getClassName: function() {
293       return "Object2D";
294    }
295 });
296 
297 return Object2D;
298 
299 });
