 34 // Includes
 35 Engine.include("/components/component.logic.js");
 36 
 37 Engine.initObject("NotifierComponent", "LogicComponent", function() {
 38 
 39 /**
 40  * @class A component which notifies objects when an action occurs.  The component
 41  *        uses a subscriber model to notify an object when certain actions occur.
 42  *        This component can be used so that multiple objects could subscribe to one
 43  *        to be notified when a particular event occurs.  The objects don't have to
 44  *        exist within the same scope.
 45  *        <p/>
 46  *        For example, a host object could publish an event of type "death" that other
 47  *        hosts are listening for.  Thus, when the host dies, it can pass relevant
 48  *        information that other objects (such as a life counter) could respond to.
 49  *        Rather than the host having to actively know about other parts of the
 50  *        game world, those other objects could "listen in" on the actions of the host
 51  *        and act accordingly.
 52  *
 53  * @param name {String} The name of the component
 54  * @param [priority=1.0] {Number} The priority of the component
 55  * @extends LogicComponent
 56  * @constructor
 57  * @description Create a notifier component
 58  */
 59 var NotifierComponent = LogicComponent.extend(/** @scope NotifierComponent.prototype */{
 60 
 61    notifyLists: null,
 62 
 63    /**
 64     * @private
 65     */
 66    constructor: function(name, priority) {
 67       this.base(name, priority || 1.0);
 68       this.notifyLists = {};
 69    },
 70 
 71    /**
 72     * Releases the component back into the object pool. See {@link PooledObject#release}
 73     * for more information.
 74     */
 75    release: function() {
 76       this.base();
 77       this.notifyLists = null;
 78    },
 79 
 80    /**
 81     * Subscribe to the event type specified.  The specified callback
 82     * will be passed the relative object and event object as its arguments.
 83     *
 84     * @param type {String} The type name of the event list.
 85     * @param thisObj {Object} The scope object for the callback, or <tt>null</tt>
 86     * @param fn {Function} The function to call when the event triggers.
 87     */
 88    subscribe: function(type, thisObj, fn) {
 89       if (this.notifyLists[type] == null) {
 90          this.notifyLists[type] = [];
 91       }
 92       this.notifyLists[type].push({parent: thisObj, func: fn});
 93    },
 94 
 95    /**
 96     * @deprecated
 97     * @see #subscribe
 98     */
 99    addRecipient: function(type, thisObj, fn) {
100       this.subscribe(type, thisObj, fn);
101    },
102 
103    /**
104     * Unsubscribe from the event type specified.  If you only
105     * pass the event type, all subscribers will be removed.  Passing
106     * the type and relative object will remove all listeners for the
107     * event type and scoped object.  Finally, passing the type, scoped
108     * object, and callback will remove the specific subscriber.
109     *
110     * @param type {String} The type name of the event list.
111     * @param [thisObj] {Object} The scope object for the callback
112     * @param [fn] {Function} The function to remove from the notification list.
113     */
114    unsubscribe: function(type, thisObj, fn) {
115       if (fn != null) {
116          // Remove a specific subscriber
117          var o = {parent: thisObj, func: fn};
118          EngineSupport.arrayRemove(this.notifyLists[type], o);
119       } else if (thisObj != null) {
120          // Remove all subscribers for the scope object
121          var newSubscribers = EngineSupport.filter(this.notifyLists[type], 
122             function(val, idx) {
123                return (val.parent !== thisObj);
124             });
125          this.notifyLists[type] = newSubscribers;
126       } else {
127          // Remove all subscribers for the event type
128          this.notifyLists[type] = [];
129       }
130    },
131    
132    /**
133     * @deprecated
134     * @see #unsubscribe
135     */
136    removeRecipient: function(type, thisObj, fn) {
137       
138    },
139 
140    /**
141     * Post a message of the given type, with the event object
142     * which subscribers will act upon.
143     *
144     * @param type {String} The type of the event
145     * @param eventObj {Object} An object which subscribers can use
146     */
147    post: function(type, eventObj) {
148       this.notifyRecipients(type, eventObj);      
149    },
150 
151    /**
152     * Run through the list of subscribers for the event type specified.  
153     * Optimized for speed if the list is large.
154     * @private
155     */
156    notifyRecipients: function(type, eventObj) {
157       if (this.notifyLists[type] == null)
158       {
159          // No handlers for this type
160          return;
161       }
162 
163       // Using Duff's device with loop inversion
164       var p = 0;
165       var s = null;
166       switch(this.notifyLists[type].length & 0x3)
167       {
168          case 3:
169             s = this.notifyLists[type][p++];
170             s.func.call(s.parent, eventObj);
171          case 2:
172             s = this.notifyLists[type][p++];
173             s.func.call(s.parent, eventObj);
174          case 1:
175             s = this.notifyLists[type][p++];
176             s.func.call(s.parent, eventObj);
177       }
178 
179       if (p < this.notifyLists[type].length)
180       {
181          do
182          {
183             s = this.notifyLists[type][p++];
184             s.func.call(s.parent, eventObj);
185 
186             s = this.notifyLists[type][p++];
187             s.func.call(s.parent, eventObj);
188 
189             s = this.notifyLists[type][p++];
190             s.func.call(s.parent, eventObj);
191 
192             s = this.notifyLists[type][p++];
193             s.func.call(s.parent, eventObj);
194 
195          } while (p < this.notifyLists[type].length);
196       }
197    }
198 }, /** @scope NotifierComponent.prototype */{
199    /**
200     * Get the class name of this object
201     *
202     * @return {String} "NotifierComponent"
203     */
204    getClassName: function() {
205       return "NotifierComponent";
206    }
207 });
208 
209 return NotifierComponent;
210 
211 });
