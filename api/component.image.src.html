 34 // Includes
 35 Engine.include("/engine/engine.math2d.js");
 36 Engine.include("/components/component.render.js");
 37 
 38 Engine.initObject("ImageComponent", "RenderComponent", function() {
 39 
 40 /**
 41  * @class A render component that renders an image.
 42  *
 43  * @param name {String} The name of the component
 44  * @param [priority=0.1] {Number} The render priority
 45  * @param imageLoader {ImageLoader} The image loader to get images from
 46  * @param [imageName] {String} The name of the image resource from the loader
 47  * @extends RenderComponent
 48  * @constructor
 49  * @description Creates a component which renders images from an {@link ImageLoader}.
 50  */
 51 var ImageComponent = RenderComponent.extend(/** @scope ImageComponent.prototype */{
 52 
 53    currentImage: null,
 54    bbox: null,
 55    imageLoader: null,
 56 
 57    /**
 58     * @private
 59     */
 60    constructor: function(name, priority, imageLoader, imageName) {
 61       if (priority instanceof ImageLoader) {
 62          imageName = imageLoader;
 63          imageLoader = priority;
 64          priority = 0.1;
 65       }
 66       this.base(name, priority);
 67       this.imageLoader = imageLoader;
 68       if (imageName != null) {
 69          this.currentImage = imageLoader.get(imageName);
 70          var dims = imageLoader.getDimensions(imageName);
 71          this.bbox = Rectangle2D.create(0,0,dims.x,dims.y);
 72       }
 73    },
 74 
 75    /**
 76     * Releases the component back into the object pool. See {@link PooledObject#release}
 77     * for more information.
 78     */
 79    release: function() {
 80       this.base();
 81       this.currentImage = null;
 82       this.bbox = null;
 83    },
 84 
 85    /**
 86     * Calculates the bounding box which encloses the image.
 87     * @private
 88     */
 89    calculateBoundingBox: function() {
 90       return this.bbox;
 91     },
 92 
 93    /**
 94     * Set the image the component will render from the {@link ImageLoader}
 95     * specified when creating the component.  This allows the user to change
 96     * the image on the fly.
 97     *
 98     * @param imageName {String} The image to render
 99     */
100    setImage: function(imageName) {
101       this.currentImage = this.imageLoader.get(imageName);
102       var dims = this.imageLoader.getDimensions(imageName);
103       this.bbox.setWidth(dims.x);
104       this.bbox.setHeight(dims.y);
105    },
106 
107    /**
108     * Get the image the component is rendering.
109     * @return {HTMLImage}
110     */
111    getImage: function() {
112       return this.currentImage;
113    },
114 
115    /**
116     * Draw the image to the render context.
117     *
118     * @param renderContext {RenderContext} The context to render to
119     * @param time {Number} The engine time in milliseconds
120     */
121    execute: function(renderContext, time) {
122 
123       if (!this.base(renderContext, time))
124       {
125          return;
126       }
127 
128       if (this.currentImage) {
129          renderContext.drawImage(this.bbox, this.currentImage, this.getHostObject());
130       }
131    }
132 }, /** @scope ImageComponent.prototype */{ 
133    /**
134     * Get the class name of this object
135     * @return {String} "ImageComponent"
136     */
137    getClassName: function() {
138       return "ImageComponent";
139    }
140 });
141 
142 return ImageComponent;
143 
144 });
