 33 // Includes
 34 Engine.include("/resourceloaders/loader.remote.js");
 35 
 36 Engine.initObject("XMLLoader", "RemoteLoader", function() {
 37 
 38 /**
 39  * @class Loads XML files from a specified URL.
 40  *
 41  * @constructor
 42  * @param name {String=XMLLoader} The name of the resource loader
 43  * @extends RemoteLoader
 44  */
 45 var XMLLoader = RemoteLoader.extend(/** @scope XMLLoader.prototype */{
 46 
 47    objects: null,
 48 
 49    /**
 50     * private
 51     */
 52    constructor: function(name) {
 53       this.base(name || "XMLLoader");
 54       this.objects = {};
 55    },
 56 
 57    /**
 58     * Load an XML file from a URL.
 59     *
 60     * @param name {String} The name of the resource
 61     * @param url {String} The URL where the resource is located
 62     * @param doc {Object} The document that was loaded
 63     */
 64    load: function(name, url, doc) {
 65 
 66       if (url) {
 67          Assert(url.indexOf("http") == -1, "XML must be located relative to this server");
 68          var thisObj = this;
 69 
 70          // Get the file from the server
 71          $.get(url, function(data) {
 72             // 2nd pass - store the XML
 73             thisObj.load(name, null, data);
 74          }, "xml");
 75       } else {
 76          // The object has been loaded and is ready for use
 77          this.setReady(true);
 78          this.base(name, doc);
 79       }
 80    },
 81 
 82    /**
 83     * The name of the resource this loader will get.
 84     * @returns {String} The string "object"
 85     */
 86    getResourceType: function() {
 87       return "xml";
 88    }
 89 
 90 }, /** @scope XMLLoader.prototype */{
 91    /**
 92     * Get the class name of this object.
 93     * @return {String} The string "SpriteLoader"
 94     */
 95    getClassName: function() {
 96       return "XMLLoader";
 97    }
 98 });
 99 
100 return XMLLoader;
101 
102 });
