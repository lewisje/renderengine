116    /**
117     * Updates the transformation of the component, setting the
118     * position and rotation based on the time that has elapsed since
119     * the last update.  This component handles frame rate independent
120     * updates so that lag between frames doesn't affect the relative
121     * position or rotation.
122     *
123     * @param renderContext {RenderContext} The render context for the component
124     * @param time {Number} The engine time in milliseconds
125     */
126    execute: function(renderContext, time) {
127       if (!this.isResting()) {
128          this.lPos.set(this.getPosition());
129          var rot = this.getRotation();
130    
131          // If we've just come into the world, we can short circuit with a
132          // quick addition of the velocity.
133          if (this.lastTime == -1)
134          {
135             this.setPosition(this.lPos.add(this.velocity).add(this.getAcceleration()).add(this.getGravity()));
136          }
137          else
138          {
139             if (this.getVelocityDecay() != 0 && this.velocity.len() > 0) {
140                // We need to decay the velocity by the amount
141                // specified until velocity is zero, or less than zero
142                var invVelocity = Vector2D.create(this.velocity).neg();
143                invVelocity.mul(this.getVelocityDecay());
144    
145                this.velocity.add(invVelocity);
146 					invVelocity.destroy();
147             }
148    
149             var oldVel = Vector2D.create(this.velocity);
150             this.velocity.add(this.getAcceleration());
151             this.velocity.add(this.getGravity());
152             if (this.maxVelocity != -1 && this.velocity.len() > this.maxVelocity) {
153                this.velocity.set(oldVel);
154             }
155 				oldVel.destroy();
156 
157             // Calculate lag?
158             var lag;
159             if (this.isCheckLag() && ((time - this.lastTime) > Engine.getFrameTime())) {
160                lag = (time - this.lastTime) * this.getLagAdjustment();
161             } else {
162                lag = 1;
163             }
164 
165             var vz = Vector2D.create(this.velocity).mul(lag);
166             this.setPosition(this.lPos.add(vz));
167             this.setRotation(rot + this.angularVelocity * (lag));
168 				vz.destroy();
169          }
170          
171          // Check rest state  
172          if (this.getCheckRestState()) {
173             if (this.getVelocity().len() < this.getRestingVelocity()) {
174                this.setVelocity(Point2D.ZERO);
175                this.atRest = true;
176             }
177          }   
178       }
179 
180       this.lastTime = time;
181       this.base(renderContext, time);
182    },
