 33 /**
 34  * @class The base object class which represents an object within
 35  * the engine.  All objects should extend from this class mainly due to
 36  * the fact that the object intelligently will inherit a <tt>destroy</tt>
 37  * method.  This method gives the object an opportunity to clean up
 38  * object references, thus keeping the memory requirements low.
 39  *
 40  */
 41 var BaseObject = PooledObject.extend(/** @scope BaseObject.prototype */{
 42 
 43    element: null,
 44 
 45    release: function() {
 46       this.base();
 47       this.element = null;
 48    },
 49 
 50    /**
 51     * Set the element which will represent this object within
 52     * its rendering context.
 53     *
 54     * @type element {BaseObject}
 55     * @memberOf BaseObject
 56     */
 57    setElement: function(element) {
 58       this.element = element;
 59    },
 60 
 61    /**
 62     * Get the element which represents this object within its rendering context.
 63     *
 64     * @return The element
 65     * @type BaseObject
 66     * @memberOf BaseObject
 67     */
 68    getElement: function() {
 69       return this.element;
 70    },
 71 
 72    /**
 73     * Update the state of the object.
 74     */
 75    update: function(renderContext, time) {
 76    }
 77 
 78 }, {
 79 
 80    /**
 81     * Get the class name of this object
 82     *
 83     * @type String
 84     * @memberOf BaseObject
 85     */
 86    getClassName: function() {
 87       return "BaseObject";
 88    }
 89 
 90 });
