engine.spatialcontainer.js


Summary

Spatial containers maintain a collection objects and can report on potential objects within a defined space of that container.
Class Summary
SpatialContainer An abstract class to represent spatial containers.
SpatialNode A single node within a spatial container.

 39 /**
 40  * An index assigned to a node.
 41  * @private
 42  */
 43 Engine.NodeIndex = 1;
 44 
 45 Engine.initObject("SpatialNode", null, function() {
 46 
 47 /**
 48  * @class A single node within a spatial container.  Has an index for fast node
 49  *        comparisons, and a list of objects contained within the node.
 50  */
 51 var SpatialNode = Base.extend(/** @scope SpatialNode.prototype */{
 52 
 53    idx: 0,
 54 
 55    objects: null,
 56 
 57    constructor: function() {
 58       this.idx = Engine.NodeIndex++;
 59       this.objects = [];
 60    },
 61 
 62    /**
 63     * Get the unique index of this node.
 64     * @type Number
 65     */
 66    getIndex: function() {
 67       return this.idx;
 68    },
 69 
 70    /**
 71     * Get an array of objects within this node.
 72     * @type Array
 73     */
 74    getObjects: function() {
 75       return this.objects;
 76    },
 77 
 78    /**
 79     * Add an object to this node.
 80     *
 81     * @param obj {BaseObject} The object to add to this node.
 82     */
 83    addObject: function(objId) {
 84       this.objects.push(objId);
 85    },
 86 
 87    /**
 88     * Remove an object from this node
 89     *
 90     * @param obj {BaseObject} The object to remove from this node
 91     */
 92    removeObject: function(obj) {
 93       EngineSupport.arrayRemove(this.objects, obj);
 94    },
 95 
 96    /**
 97     * Get the class name of this object
 98     *
 99     * @type String
100     * @memberOf Container
101     */
102    getClassName: function() {
103       return "SpatialNode";
104    }
105 
106 });
107 
108 return SpatialNode;
109 
110 });

112 Engine.initObject("SpatialContainer", "BaseObject", function() {
113 
114 /**
115  * @class An abstract class to represent spatial containers.  Spatial containers
116  *        contain objects and can report on potential objects within a defined
117  *        space of that container.
118  *
119  * @param name {String} The name of the container
120  * @param width {Number} The width of the container
121  * @param height {Number} The height of the container
122  * @extends BaseObject
123  */
124 var SpatialContainer = BaseObject.extend(/** @scope SpatialContainer.prototype */{
125 
126    root: null,
127 
128    width: 0,
129 
130    height: 0,
131 
132    constructor: function(name, width, height) {
133       this.base(name || "SpatialContainer");
134       this.width = width;
135       this.height = height;
136    },
137 
138    release: function() {
139       this.base();
140       this.root = null;
141       this.width = 0;
142       this.height = 0;
143    },
144 
145    /**
146     * Get the width of the container.
147     * @type Number
148     */
149    getWidth: function() {
150       return this.width;
151    },
152 
153    /**
154     * Get the height of the container.
155     * @type Number
156     */
157    getHeight: function() {
158       return this.height;
159    },
160 
161    /**
162     * Get the root of the container.
163     * @type Object
164     */
165    getRoot: function() {
166       return this.root;
167    },
168 
169    /**
170     * Set the root of the container.
171     *
172     * @param root {Object} The root object of this container
173     */
174    setRoot: function(root) {
175       this.root = root;
176    },
177 
178    /**
179     * Returns a potential collision list of objects that are contained
180     * within the defined sub-space of the container.
181     *
182     * @param point {Point2D} The point to build with
183     * @type HashContainer
184     */
185    getPCL: function(point) {
186       return new HashContainer();
187    }
188 
189 }, {
190    /**
191     * Get the class name of this object
192     *
193     * @type String
194     * @memberOf Container
195     */
196    getClassName: function() {
197       return "SpatialContainer";
198    }
199 
200 });
201 
202 return SpatialContainer;
203 
204 });
