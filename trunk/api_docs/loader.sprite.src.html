 31 /**
 32  * @class Loads sprites and makes them available to the system.  Sprites are
 33  *        defined by an external resource that consist of the follwing definition.
 34  *        The definition file must include the source bitmap, the width and height
 35  *        of the bitmap, and a section that defines each sprite.
 36  *        <p/>
 37  *        Sprites can either be single frames, or animations:
 38  * <pre>
 39  * {
 40  *    // Frame (f): left, top, frameWidth, frameHeight
 41  *    // Animation (a): left, top, frameWidth, frameHeight, frameCount, speedMS, loop/toggle
 42  *    bitmapImage: "bitmapFile.ext",
 43  *    bitmapWidth: 320,
 44  *    bitmapHeight: 320,
 45  *    sprites: {
 46  *        "stand": {
 47  *           "f" : [0, 0, 32, 32]
 48  *        },
 49  *        "walk": {
 50  *           "a" : [32, 0, 32, 32, 3, 150, "loop"]
 51  *        }
 52  *    }
 53  * }
 54  * </pre>
 55  *
 56  * @extends ImageResourceLoader
 57  */
 58 var SpriteLoader = ImageResourceLoader.extend(/** @scope SpriteLoader.prototype */{
 59 
 60    sprites: null,
 61 
 62    constructor: function(name) {
 63       this.base(name || "SpriteLoader");
 64       this.sprites = {};
 65    },
 66 
 67    /**
 68     * Load a sprite resource from a URL.
 69     *
 70     * @param name {String} The name of the resource
 71     * @param url {String} The URL where the resource is located
 72     */
 73    load: function(name, url, info, path) {
 74 
 75       if (url)
 76       {
 77          Assert(url.indexOf("http") == -1, "Sprites must be located on this server");
 78          var thisObj = this;
 79 
 80          // Get the file from the server
 81          $.get(url, function(data) {
 82             var spriteInfo = EngineSupport.parseJSON(data);
 83             // get the path to the resource file
 84             var path = url.substring(0, url.lastIndexOf("/"));
 85             thisObj.load(name, null, spriteInfo, path + "/");
 86          });
 87       }
 88       else
 89       {
 90          info.bitmapImage = path + info.bitmapImage;
 91          Console.log("Loading sprite: " + name + " @ " + info.bitmapImage);
 92 
 93          // Load the sprite image file
 94          this.base(name, info.bitmapImage, info.bitmapWidth, info.bitmapHeight);
 95 
 96          // Store the sprite info
 97          this.sprites[name] = info;
 98       }
 99    },
100 
101    /**
102     * Get the sprite resource with the specified name from the cache.  The
103     * object returned contains the bitmap as <tt>image</tt> and
104     * the sprite definition as <tt>info</tt>.
105     *
106     * @param name {String} The name of the object to retrieve
107     * @type Object
108     */
109    get: function(name) {
110       var bitmap = this.base(name);
111       var sprite = {
112          image: bitmap,
113          info: this.sprites[name]
114       };
115       return sprite;
116    },
117 
118    /**
119     * Creates a {@link Sprite} object representing the named sprite.
120     *
121     * @param resource {String} A loaded sprite resource
122     * @param sprite {String} The name of the sprite from the resource
123     * @returns A {@link Sprite} object
124     */
125    getSprite: function(resource, sprite) {
126       return Sprite.create(sprite, this.get(resource).info.sprites[sprite], this.get(resource));
127    },
128 
129    /**
130     * Get the names of all the sprites available in a resource.
131     *
132     * @param resource {String} The name of the resource
133     * @return An Array of sprite names in the given resource
134     */
135    getSpriteNames: function(resource) {
136       var s = [];
137       var spr = this.sprites[resource].sprites;
138       for (var i in spr) {
139          s.push(i);
140       }
141       return s;
142    },
143 
144    /**
145     * The name of the resource this loader will get.
146     * @returns A String that represents the resource type.
147     */
148    getResourceType: function() {
149       return "sprite";
150    }
151 
152 }, {
153    /**
154     * Get the class name of this object.
155     * @return The string <tt>SpriteLoader</tt>
156     * @type String
157     */
158    getClassName: function() {
159       return "SpriteLoader";
160    }
161 });

163 /**
164  * @class Represents a sprite
165  * @extends PooledObject
166  */
167 var Sprite = PooledObject.extend(/** @scope Sprite.prototype */{
168 
169    // The type of sprite: Single or Animation
170    type: -1,
171 
172    // Animation mode: loop or toggle
173    mode: -1,
174 
175    // Animation frame count
176    count: -1,
177 
178    // Animation speed
179    speed: -1,
180 
181    // The rect which defines the sprite frame
182    frame: null,
183 
184    // The image map that contains the sprite(s)
185    image: null,
186 
187    // The bounding box for the sprite
188    bbox: null,
189 
190    /**
191     * Creates an instance of a Sprite object.
192     *
193     * @param name {String} The name of the object
194     * @param spriteObj {Object} Passed in by a SpriteLoader
195     * @param spriteResource {Object} The sprite resource loaded by the SpriteLoader
196     */
197    constructor: function(name, spriteObj, spriteResource) {
198 
199       this.type = (spriteObj["a"] ? Sprite.TYPE_ANIMATION : Sprite.TYPE_SINGLE);
200 
201       var s = (this.type == Sprite.TYPE_ANIMATION ? spriteObj["a"] : spriteObj["f"]);
202       if (this.type == Sprite.TYPE_ANIMATION) {
203          this.mode = (s[Sprite.INDEX_TYPE] == "loop" ? Sprite.MODE_LOOP : Sprite.MODE_TOGGLE);
204          this.count = s[Sprite.INDEX_COUNT];
205          this.speed = s[Sprite.INDEX_SPEED];
206       }
207 
208       this.image = spriteResource.image;
209 
210       this.frame = Rectangle2D.create(s[Sprite.INDEX_LEFT], s[Sprite.INDEX_TOP], s[Sprite.INDEX_WIDTH], s[Sprite.INDEX_HEIGHT]);
211       this.bbox = Rectangle2D.create(0, 0, s[Sprite.INDEX_WIDTH], s[Sprite.INDEX_HEIGHT]);
212       return this.base(name);
213    },
214 
215    release: function() {
216       this.base();
217       this.mode = -1;
218       this.type = -1;
219       this.count = -1;
220       this.speed = -1;
221       this.frame = null;
222       this.image = null;
223       this.bbox = null;
224    },
225 
226    /**
227     * Returns <tt>true</tt> if the sprite is an animation.
228     * @return <tt>true</tt> if the sprite is an animation
229     */
230    isAnimation: function() {
231       return (this.type == Sprite.TYPE_ANIMATION);
232    },
233 
234    /**
235     * Returns <tt>true</tt> if the sprite is an animation and loops.
236     * @return <tt>true</tt> if the sprite is an animation and loops
237     */
238    isLoop: function() {
239       return (this.isAnimation() && this.mode == Sprite.MODE_LOOP);
240    },
241 
242    /**
243     * Returns <tt>true</tt> if the sprite is an animation and toggles.
244     * @return <tt>true</tt> if the sprite is an animation and toggles
245     */
246    isToggle: function() {
247       return (this.isAnimation() && this.mode == Sprite.MODE_TOGGLE);
248    },
249 
250    /**
251     * Get the bounding box for the sprite.
252     * @return A {@link Rectangle2D}
253     */
254    getBoundingBox: function() {
255       return this.bbox;
256    },
257 
258    /**
259     * Gets the frame (rectangle defining what portion of the image map
260     * the sprite frame occupies) of the sprite.
261     *
262     * @param time {Number} Current world time
263     * @return A {@link Rectangle2D} which defines the frame of the sprite in
264     *         the source image map.
265     */
266    getFrame: function(time) {
267       if (!this.isAnimation()) {
268          return this.frame;
269       } else {
270          var f = Rectangle2D.create(this.frame);
271          var fn;
272          if (this.isLoop()) {
273             fn = Math.floor(time / this.speed) % this.count;
274          } else {
275             fn = Math.floor(time / this.speed) % (this.count * 2);
276             if (fn > this.count - 1) {
277                fn = this.count - (fn - (this.count - 1));
278             }
279          }
280          return f.offset(f.dims.x * fn, 0);
281       }
282    },
283 
284    /**
285     * Set the speed, in milliseconds, that an animation runs at.
286     * @param speed {Number} The number of milliseconds per frame of an animation
287     */
288    setSpeed: function(speed) {
289       if (speed >= 0) {
290          this.speed = speed;
291       }
292    },
293 
294    /**
295     * Get the number of milliseconds each frame is displayed for an animation
296     * @type Number
297     */
298    getSpeed: function() {
299       return this.speed;
300    },
301 
302    /**
303     * The source image loaded by the {@link SpriteLoader} when the sprite was
304     * created.
305     * @return The source image the sprite is contained within
306     */
307    getSourceImage: function() {
308       return this.image;
309    }
310 
311 }, /** @scope Sprite */{
312    /**
313     * Gets the class name of this object.
314     * @return The string <tt>Sprite</tt>
315     */
316    getClassName: function() {
317       return "Sprite";
318    },
319 
320    /** The sprite animation loops */
321    MODE_LOOP: 0,
322 
323    /** The sprite animation toggles - Plays from the first to the last frame
324        then plays backwards to the first frame and repeats. */
325    MODE_TOGGLE: 1,
326 
327    /** The sprite is a single frame */
328    TYPE_SINGLE: 0,
329 
330    /** The sprite is an animation */
331    TYPE_ANIMATION: 1,
332 
333    /** The field in the sprite definition file for the left pixel of the sprite frame */
334    INDEX_LEFT: 0,
335 
336    /** The field in the sprite definition file for the top pixel of the sprite frame */
337    INDEX_TOP: 1,
338 
339    /** The field in the sprite definition file for the width of the sprite frame */
340    INDEX_WIDTH: 2,
341 
342    /** The field in the sprite definition file for the height of the sprite frame */
343    INDEX_HEIGHT: 3,
344 
345    /** The field in the sprite definition file for the count of frames in the sprite */
346    INDEX_COUNT: 4,
347 
348    /** The field in the sprite definition file for the speed in milliseconds that the sprite animates */
349    INDEX_SPEED: 5,
350 
351    /** The field in the sprite definition file for the type of sprite animation */
352    INDEX_TYPE: 6
353 });
