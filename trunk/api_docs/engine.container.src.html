 37 Engine.initObject("Iterator", "PooledObject", function() {
 38 
 39 /**
 40  * @class Create an iterator over a {@link Container} instance.
 41  *
 42  * @param container {Container} The container to iterate over.
 43  */
 44 var Iterator = PooledObject.extend(/** @scope Iterator.prototype */{
 45 
 46    idx: 0,
 47 
 48    c: null,
 49 
 50    objs: null,
 51 
 52    constructor: function(container) {
 53       this.base("Iterator");
 54       this.idx = 0;
 55       this.c = container;
 56 
 57       // Duplicate the elements in the container
 58       this.objs = new Array().concat(container.getObjects());
 59    },
 60 
 61    /**
 62     * Release the object back into the object pool.
 63     */
 64    release: function() {
 65       this.base();
 66       this.idx = 0;
 67       this.c = null;
 68       this.objs = null;
 69    },
 70 
 71    /**
 72     * Reverse the order of the elements in the container (non-destructive) before
 73     * iterating over them.  You cannot call this method after you have called {@link #next}.
 74     */
 75    reverse: function() {
 76       Assert(this.idx == 0, "Cannot reverse Iterator after calling next()");
 77       this.objs.reverse();
 78    },
 79 
 80    /**
 81     * Get the next element from the iterator.
 82     * @type Object
 83     */
 84    next: function() {
 85       if (this.idx < this.c.size())
 86       {
 87          return this.objs[this.idx++];
 88       }
 89       throw new Error("Index out of range");
 90    },
 91 
 92    /**
 93     * Returns <tt>true</tt> if the iterator has more elements.
 94     * @type Boolean
 95     */
 96    hasNext: function() {
 97       return (this.idx < this.c.size());
 98    }
 99 
100 }, /** @scope Iterator */{
101    /**
102     * Get the class name of this object
103     *
104     * @return {String} The string "Iterator"
105     */
106    getClassName: function() {
107       return "Iterator";
108    }
109 });
110 
111 return Iterator;

115 Engine.initObject("Container", "BaseObject", function() {
116 
117 /**
118  * @class A container is a logical collection of objects.  A container
119  *        is responsible for maintaining the list of objects within it.
120  *        When a container is destroyed, all objects within the container
121  *        are destroyed with it.
122  *
123  * @param {String} The name of the container. Default: Container
124  * @extends BaseObject
125  */
126 var Container = BaseObject.extend(/** @scope Container.prototype */{
127 
128    objects: null,
129 
130    constructor: function(containerName) {
131       this.base(containerName || "Container");
132       this.objects = [];
133    },
134 
135    /**
136     * Release the object back into the object pool.
137     */
138    release: function() {
139       this.base();
140       this.objects = null;
141    },
142 
143    /**
144     * Destroy all objects contained within this object.  Calls the
145     * <tt>destroy()</tt> method on each object, giving them a chance
146     * to perform clean up operations.
147     */
148    destroy: function() {
149       this.cleanUp();
150       this.base();
151    },
152 
153    /**
154     * Returns the count of the number of objects within the
155     * container.
156     *
157     * @type Number
158     */
159    size: function() {
160       return this.objects.length;
161    },
162 
163    /**
164     * Add an object to the container.
165     *
166     * @param obj {BaseObject} The object to add to the container.
167     */
168    add: function(obj) {
169       this.objects.push(obj);
170       if (obj.getId) {
171          Console.log("Added ", obj.getId(), "[", obj, "] to ", this.getId(), "[", this, "]");
172       }
173    },
174 
175    /**
176     * Remove an object from the container.  The object is
177     * not destroyed when it is removed from the container.
178     *
179     * @param obj {BaseObject} The object to remove from the container.
180     */
181    remove: function(obj) {
182       if (obj.getId) {
183          Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
184       }
185       EngineSupport.arrayRemove(this.objects, obj);
186    },
187 
188    /**
189     * Remove an object from the container at the specified index.
190     * The object is not destroyed when it is removed.
191     *
192     * @param idx {Number} An index between zero and the size of the container minus 1.
193     * @return The object removed from the container.
194     * @type BaseObject
195     */
196    removeAtIndex: function(idx) {
197       Assert((idx >= 0 && idx < this.size()), "Index of out range in Container");
198 
199       var obj = this.objects[idx];
200 
201       Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
202       this.objects.splice(idx, 1);
203 
204       return obj;
205    },
206 
207    /**
208     * Get the object at the index specified. If the container has been
209     * sorted, objects might not be in the position you'd expect.
210     *
211     * @param idx {Number} The index of the object to get
212     * @type BaseObject
213     */
214    get: function(idx) {
215       return this.objects[idx];
216    },
217 
218    /**
219     * Remove all objects from the container.  None of the objects are
220     * destroyed, only removed from this container.
221     */
222    clear: function() {
223       this.objects.length = 0;
224       this.objects = [];
225    },
226 
227    /**
228     * Remove and destroy all objects from the container.
229     */
230    cleanUp: function() {
231       while(this.objects.length > 0)
232       {
233          var o = this.objects.shift();
234          o.destroy();
235       }
236       this.clear();
237    },
238 
239    /**
240     * Get the array of all objects within this container.  If a filtering
241     * function is provided, only objects matching the filter will be
242     * returned from the object collection.
243     * <p/>
244     * The filter function needs to return <tt>true</tt> for each element
245     * that should be contained in the filtered set.  The function will be
246     * passed the following arguments:
247     * <ul>
248     * <li>element - The array element being operated upon</li>
249     * <li>index - The index of the element in the array</li>
250     * <li>array - The entire array of elements in the container</li>
251     * </ul>
252     * Say you wanted to filter a host objects components based on a
253     * particular type.  You might do something like the following:
254     * <pre>
255     * var logicComponents = host.getObjects(function(el, idx) {
256     *    if (el.getType() == BaseComponent.TYPE_LOGIC) {
257     *       return true;
258     *    }
259     * });
260     * </pre>
261     *
262     * @param filterFn {Function} A function to filter the set of
263     *                 elements with.  If you pass <tt>null</tt> the
264     *                 entire set of objects will be returned.
265     * @type Array
266     */
267    getObjects: function(filterFn) {
268       if (filterFn) {
269          return EngineSupport.filter(this.objects, filterFn);
270       } else {
271          return this.objects;
272       }
273    },
274 
275    /**
276     * Sort the objects within the container, using the provided function.
277     * The function will be provided object A and B.  If the result of the
278     * function is less than zero, A will be sorted before B.  If the result is zero,
279     * A and B retain their order.  If the result is greater than zero, A will
280     * be sorted after B.
281     *
282     * @param fn {Function} The function to sort with.
283     */
284    sort: function(fn) {
285       Assert((fn != null), "A function must be provided to sort the Container");
286       Console.log("Sorting ", this.getName(), "[" + this.getId() + "]");
287       this.objects.sort(fn);
288    },
289 
290    /**
291     * Returns a property object with accessor methods.
292     */
293    getProperties: function() {
294       var self = this;
295       var prop = this.base(self);
296       return $.extend(prop, {
297          "Contains"  : [function() { return self.objects.length; },
298                         null, false]
299       });
300    },
301 
302    /**
303     * Serializes a container to XML.
304     */
305    toString: function(indent) {
306       indent = indent ? indent : "";
307       var props = this.getProperties();
308       var xml = indent + "<" + this.constructor.getClassName();
309       for (var p in props) {
310          // If the value should be serialized, call it's getter
311          if (props[p][2]) {
312             xml += " " + p + "=\"" + props[p][0]().toString() + "\"";
313          }
314       }
315       xml += ">\n";
316 
317       // Dump children
318       for (var o in this.objects) {
319          xml += this.objects[o].toString(indent + "   ");
320       }
321 
322       // Closing tag
323       xml += indent + "</" + this.constructor.getClassName() + ">\n";
324       return xml;
325    }
326 
327 }, /** @scope Container */{
328    /**
329     * Get the class name of this object
330     *
331     * @return {String} The string "Container"
332     */
333    getClassName: function() {
334       return "Container";
335    }
336 });
337 
338 return Container;
339

342 Engine.initObject("HashContainer", "Container", function() {
343 
344 /**
345  * @class A hash container is a logical collection of objects.  A hash container
346  *        is a container with a backing object for faster lookups.  Objects within
347  *        the container must have unique names. When a container is destroyed, all
348  *        objects within the container are destroyed with it.
349  * @extends Container
350  */
351 var HashContainer = Container.extend(/** @scope HashContainer.prototype */{
352 
353    objHash: null,
354 
355    /**
356     * Create an instance of a container object.
357     *
358     * @param {String} The name of the container. Default: Container
359     * @constructor
360     */
361    constructor: function(containerName) {
362       this.base(containerName || "HashContainer");
363       this.objHash = {};
364    },
365 
366    /**
367     * Release the object back into the object pool.
368     */
369    release: function() {
370       this.base();
371       this.objHash = null;
372    },
373 
374    /**
375     * Returns <tt>true</tt> if the object name is already in
376     * the hash.
377     *
378     * @param name {String} The name of the hash to check
379     * @type Boolean
380     */
381    isInHash: function(key) {
382       return (this.objHash["_" + String(key)] != null);
383    },
384 
385    /**
386     * Add an object to the container.
387     *
388     * @param key {String} The name of the object to store.  Names must be unique
389     *                      or the object with that name will be overwritten.
390     * @param obj {BaseObject} The object to add to the container.
391     */
392    add: function(key, obj) {
393       AssertWarn(!this.isInHash(key), "Object already exists within hash!");
394 
395       if (this.isInHash(key)) {
396          // Remove the old one first
397          this.removeHash(key);
398       }
399 
400       // Some keys weren't being accepted (like "MOVE") so added
401       // an underscore to prevent keyword collisions
402       this.objHash["_" + String(key)] = obj;
403       this.base(obj);
404    },
405 
406    /**
407     * Remove an object from the container.  The object is
408     * not destroyed when it is removed from the container.
409     *
410     * @param obj {BaseObject} The object to remove from the container.
411     */
412    remove: function(obj) {
413       for (var o in this.objHash)
414       {
415          if (this.objHash[o] === obj)
416          {
417             this.removeHash(o);
418             break;
419          }
420       }
421 
422       this.base(obj);
423    },
424 
425    /**
426     * Remove the object with the given key name from the container.
427     *
428     * @param name {String} The object to remove
429     * @return the object removed from the hash
430     * @type BaseObject
431     */
432    removeHash: function(key) {
433       var obj = this.objHash["_" + String(key)];
434       EngineSupport.arrayRemove(this.objects, this.objHash["_" + String(key)]);
435       delete this.objHash["_" + String(key)];
436       return obj;
437    },
438 
439    /**
440     * Remove an object from the container at the specified index.
441     * The object is not destroyed when it is removed.
442     *
443     * @param idx {Number} An index between zero and the size of the container minus 1.
444     * @return The object removed from the container.
445     * @type BaseObject
446     */
447    removeAtIndex: function(idx) {
448       var obj = this.base(idx);
449       for (var o in this.objHash) {
450          if (this.objHash[o] === obj) {
451             this.removeHash(o);
452             break;
453          }
454       }
455 
456       return obj;
457    },
458 
459    /**
460     * If a number is provided, the request will be passed to the
461     * base object, otherwise a name is assumed and the hash will
462     * be retrieved.
463     *
464     * @param idx {Number/String} The index or hash of the object to get
465     * @type BaseObject
466     */
467    get: function(idx) {
468       if (typeof idx == 'string')
469       {
470          return this.objHash["_" + String(idx)];
471       }
472       else
473       {
474          return this.base(idx);
475       }
476    },
477 
478    /**
479     * Remove all objects from the container.  None of the objects are
480     * destroyed.
481     * @memberOf Container
482     */
483    clear: function() {
484       this.base();
485       this.objHash = {};
486    },
487 
488    /**
489     * Cleans up the references to the objects (destroys them) within
490     * the container.
491     */
492    cleanUp: function() {
493       this.base();
494       this.clear();
495    }
496 
497 }, /** @scope HashContainer.prototype */{
498    /**
499     * Get the class name of this object
500     *
501     * @return {String} The string "HashContainer"
502     */
503    getClassName: function() {
504       return "HashContainer";
505    }
506 });
507 
508 return HashContainer;
509
