 33 /**
 34  * @class Create an iterator over a container.
 35  *
 36  * @param container {Container} The container to iterate over.
 37  */
 38 var Iterator = Base.extend(/** @scope Iterator.prototype */{
 39 
 40    idx: 0,
 41 
 42    c: null,
 43 
 44    constructor: function(container) {
 45       this.idx = 0;
 46       this.c = container;
 47    },
 48 
 49    /**
 50     * Get the next element from the iterator.
 51     * @type Object
 52     */
 53    next: function() {
 54       if (this.idx < this.c.size())
 55       {
 56          return this.c.get(this.idx++);
 57       }
 58       throw new Error("Index out of range!");
 59    },
 60 
 61    /**
 62     * Returns <tt>true</tt> if the iterator has more elements.
 63     * @type Boolean
 64     */
 65    hasNext: function() {
 66       return (this.idx < this.c.size);
 67    },
 68 
 69    /**
 70     * Get the class name of this object
 71     *
 72     * @type String
 73     * @memberOf Container
 74     */
 75    getClassName: function() {
 76       return "Iterator";
 77    }
 78 
 79 });

 81 /**
 82  * @class A container is a logical collection of objects.  A container
 83  * is responsible for maintaining the list of objects within it.
 84  * When a container is destroyed, all objects within the container
 85  * are destroyed with it.
 86  *
 87  * @param {String} The name of the container. Default: Container
 88  * @extends BaseObject
 89  */
 90 var Container = BaseObject.extend(/** @scope Container.prototype */{
 91 
 92    objects: null,
 93 
 94    constructor: function(containerName) {
 95       this.base(containerName || "Container");
 96       this.objects = [];
 97    },
 98 
 99    /**
100     * Destroy all objects contained within this object.  Calls the
101     * <tt>destroy()</tt> method on each object, giving them a chance
102     * to perform clean up operations.
103     */
104    destroy: function() {
105 		this.cleanUp();
106       this.base();
107    },
108 
109    /**
110     * Returns the count of the number of objects within the
111     * container.
112     *
113     * @type Number
114     */
115    size: function() {
116       return this.objects.length;
117    },
118 
119    /**
120     * Add an object to the container.
121     *
122     * @param obj {BaseObject} The object to add to the container.
123     */
124    add: function(obj) {
125       this.objects.push(obj);
126       Console.log("Added ", obj.getId(), "[", obj, "] to ", this.getId(), "[", this, "]");
127    },
128 
129    /**
130     * Remove an object from the container.  The object is
131     * not destroyed when it is removed from the container.
132     *
133     * @param obj {BaseObject} The object to remove from the container.
134     */
135    remove: function(obj) {
136       Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
137       EngineSupport.arrayRemove(this.objects, obj);
138    },
139 
140    /**
141     * Remove an object from the container at the specified index.
142     * The object is not destroyed when it is removed.
143     *
144     * @param idx {Number} An index between zero and the size of the container minus 1.
145     * @return The object removed from the container.
146     * @type BaseObject
147     */
148    removeAtIndex: function(idx) {
149       Assert((idx >= 0 && idx < this.size()), "Index of out range in Container");
150 
151       var obj = this.objects[idx];
152 
153       Console.log("Removed ", obj.getId(), "[", obj, "] from ", this.getId(), "[", this, "]");
154       this.objects.splice(idx, 1);
155 
156       return obj;
157    },
158 
159    /**
160     * Get the object at the index specified. If the container has been
161     * sorted, objects might not be in the position you'd expect.
162     *
163     * @param idx {Number} The index of the object to get
164     * @type BaseObject
165     */
166    get: function(idx) {
167       return this.objects[idx];
168    },
169 
170    /**
171     * Remove all objects from the container.  None of the objects are
172     * destroyed, only removed from this container.
173     */
174    clear: function() {
175       this.objects.length = 0;
176    },
177 
178    /**
179     * Remove and destroy all objects from the container.
180     */
181    cleanUp: function() {
182       while(this.objects.length > 0)
183       {
184          var o = this.objects.shift();
185          o.destroy();
186       }
187 		this.clear();
188 	},
189 
190    /**
191     * Get the array of objects within this container.
192     *
193     * @type Array
194     */
195    getObjects: function() {
196       return this.objects;
197    },
198 
199    /**
200     * Sort the objects within the container, using the provided function.
201     * The function will be provided object A and B.  If the result of the
202     * function is less than zero, A will be sorted before B.  If the result is zero,
203     * A and B retain their order.  If the result is greater than zero, A will
204     * be sorted after B.
205     *
206     * @param fn {Function} The function to sort with.
207     */
208    sort: function(fn) {
209       Assert((fn != null), "A function must be provided to sort the Container");
210       Console.log("Sorting ", this.getClassName(), "[" + this.getId() + "]");
211       this.objects.sort(fn);
212    },
213 
214    /**
215     * Get the class name of this object
216     *
217     * @type String
218     */
219    getClassName: function() {
220       return "Container";
221    }
222 });
