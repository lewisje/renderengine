 35 Engine.include("/engine/engine.events.js");
 36 Engine.include("/components/component.input.js");

 38 Engine.initObject("MouseInputComponent", "InputComponent", function() {
 39 
 40 /**
 41  * @class A component which responds to mouse events and notifies
 42  * the host object when one of the events occurs.  The host object should implement
 43  * any of the five methods listed below to be notified of the corresponding event:
 44  * <ul>
 45  * <li>onMouseOver - The mouse moved over the host object, or the object moved under the mouse</li>
 46  * <li>onMouseOut - The mouse moved out of the host object (after being over it)</li>
 47  * <li>onMouseDown - A mouse button was depressed over the host object</li>
 48  * <li>onMouseUp - A mouse button was released over the host object</li>
 49  * <li>onMouseMove - The mouse was moved</li>
 50  * </ul>
 51  *
 52  * @extends InputComponent
 53  */
 54 var MouseInputComponent = InputComponent.extend(/** @scope MouseInputComponent.prototype */{
 55 
 56    /**
 57     * Create an instance of a mouse input component.
 58     *
 59     * @param name {String} The unique name of the component.
 60     * @param priority {Number} The priority of the component among other input components.
 61     * @constructor
 62     */
 63    constructor: function(name, priority) {
 64       this.base(name, priority);
 65 
 66    },
 67 
 68    /**
 69     * Set the host object this component exists within.  Additionally, this component
 70     * sets some readable flags on the host object and establishes (if not already set)
 71     * a mouse listener on the render context.
 72     *
 73     * @param hostObject {HostObject} The object which hosts the component
 74     */
 75    setHostObject: function(hostObject) {
 76       this.base(hostObject);
 77 
 78       // Set some flags we can check
 79       this.getHostObject().MouseInputComponent_mouseOver = false;
 80       this.getHostObject().MouseInputComponent_mouseDown = false;
 81 
 82       // Assign handlers to the context, making sure to only add
 83       // one handler.  This way we don't have hundreds of mouse move
 84       // handlers taking up precious milliseconds
 85       var context = this.getHostObject().getRenderContext();
 86       if (!context.MouseInputComponent_mouseInfo)
 87       {
 88          context.MouseInputComponent_mouseInfo = {
 89             position: new Point2D(0,0),
 90             lastPosition: new Point2D(0,0),
 91             button: EventEngine.MOUSE_NO_BUTTON,
 92             lastOver: null
 93          };
 94 
 95          EventEngine.setHandler(context, {owner: this}, "mousemove", function(eventObj) {
 96             var mouseInfo = eventObj.data.owner.MouseInputComponent_mouseInfo;
 97             mouseInfo.position.set(event.pageX, event.pageY);
 98             mouseInfo.button = event.which || EventEngine.MOUSE_NO_BUTTON;
 99          });
100 
101          EventEngine.setHandler(context, {owner: this}, "mousedown", function(eventObj) {
102             var mouseInfo = eventObj.data.owner.MouseInputComponent_mouseInfo;
103             mouseInfo.button = event.which;
104          });
105 
106          EventEngine.setHandler(context, {owner: this}, "mouseup", function(eventObj) {
107             var mouseInfo = eventObj.data.owner.MouseInputComponent_mouseInfo;
108             mouseInfo.button = EventEngine.MOUSE_NO_BUTTON;
109          });
110 
111       }
112    },
113 
114    /**
115     * Perform the checks on the mouse info object, and also perform
116     * intersection tests to be able to call mouse events.
117     *
118     * @private
119     */
120    execute: function(renderContext, time) {
121       // Objects may be in motion.  If so, we need to call the mouse
122       // methods for just such a case.
123       var mouseInfo = this.getHostObject().getRenderContext().MouseInputComponent_mouseInfo;
124       var bBox = this.getHostObject().getBoundingBox();
125       var pos = this.getHostObject().getPosition();
126       var mouseOver = false;
127       if (this.getHostObject().onMouseOver ||
128           this.getHostObject().onMouseOut ||
129           this.getHostObject().onMouseDown ||
130           this.getHostObject().onMouseUp)
131       {
132          mouseOver = Math2D.boxPointCollision(bBox, pos, mouseInfo.position);
133       }
134 
135       // Mouse position changed
136       if (this.getHostObject().onMouseMove && !mouseInfo.position.equals(mouseInfo.lastPosition))
137       {
138          this.getHostObject().onMouseMove(mouseInfo);
139          mouseInfo.lastPosition.set(mouseInfo.position);
140       }
141 
142       // Mouse is over object
143       if (this.getHostObject().onMouseOver && mouseOver)
144       {
145          this.getHostObject().MouseInputComponent_mouseOver = true;
146          this.getHostObject().onMouseOver(mouseInfo);
147       }
148 
149       // Mouse was over object
150       if (this.getHostObject().onMouseOut && !mouseOver &&
151           this.getHostObject().MouseInputComponent_mouseOver == true)
152       {
153          this.getHostObject().MouseInputComponent_mouseOver = false;
154          this.getHostObject().onMouseOut(mouseInfo);
155       }
156 
157       // Mouse button clicked and mouse over object
158       if (this.getHostObject().onMouseDown && mouseOver && (mouseInfo.button != EventEngine.MOUSE_NO_BUTTON))
159       {
160          this.getHostObject().MouseInputComponent_mouseDown = true;
161          this.getHostObject().onMouseDown(mouseInfo);
162       }
163 
164       // Mouse button released (and mouse was down over this object)
165       if (this.getHostObject().onMouseUp && this.getHostObject().MouseInputComponent_mouseDown  &&
166           (mouseInfo.button == EventEngine.MOUSE_NO_BUTTON))
167       {
168          this.getHostObject().MouseInputComponent_mouseDown = false;
169          this.getHostObject().onMouseUp(mouseInfo);
170       }
171    }
172 }, {
173    /**
174     * Get the class name of this object
175     *
176     * @type String
177     * @memberOf KeyboardInputComponent
178     */
179    getClassName: function() {
180       return "MouseInputComponent";
181    }
182 });
183 
184 return MouseInputComponent;
185
