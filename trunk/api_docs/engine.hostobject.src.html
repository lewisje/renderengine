 32 /**
 33  * @class A host object is a container of components which provide some sort
 34  * of action, be it rendering, collision detection, effects, or whatever.
 35  * This way, an object can be anything, depending on it's components.
 36  * @extends HashContainer
 37  */
 38 var HostObject = HashContainer.extend(/** @scope HostObject.prototype */{
 39 
 40    renderContext: null,
 41 
 42    release: function() {
 43       this.base();
 44       this.renderContext = null;
 45    },
 46 
 47    /**
 48     * Destroy all of the components within this object and
 49     * remove this object from it's render context.
 50     * @memberOf HostObject
 51     */
 52    destroy: function() {
 53       if (this.getRenderContext()) {
 54          this.getRenderContext().remove(this);
 55       }
 56       this.base();
 57    },
 58 
 59 
 60    /**
 61     * Set the rendering context this object will be drawn upon.
 62     *
 63     * @param renderContext {RenderContext} The context
 64     * @memberOf HostObject
 65     */
 66    setRenderContext: function(renderContext) {
 67       this.renderContext = renderContext;
 68    },
 69 
 70    /**
 71     * Get the rendering context this object will be drawn upon.
 72     *
 73     * @type RenderContext
 74     * @memberOf HostObject
 75     */
 76    getRenderContext: function() {
 77       return this.renderContext;
 78    },
 79 
 80    /**
 81     * Allows the object a chance to perform setup before all of
 82     * its components are executed.  Usually used to push the
 83     * render context's transform.
 84     *
 85     * @param renderContext {RenderContext} The context the object will be rendered within.
 86     * @param time {Number} The global time within the engine.
 87     * @memberOf HostObject
 88     */
 89    preUpdate: function(renderContext, time) {
 90    },
 91 
 92    /**
 93     * Update this object within the render context, and for the specified time.
 94     *
 95     * @param renderContext {RenderContext} The context the object will be rendered within.
 96     * @param time {Number} The global time within the engine.
 97     * @memberOf HostObject
 98     */
 99    update: function(renderContext, time) {
100       var components = this.getObjects();
101 
102       this.preUpdate(renderContext, time);
103 
104       for (var c in components) {
105          components[c].execute(renderContext, time);
106       }
107 
108       this.postUpdate(renderContext, time);
109    },
110 
111    /**
112     * Allows the object a chance to perform actions after all of
113     * its components have been executed.  Usually used to pop the
114     * transformations this object performed.
115     *
116     * @param renderContext {RenderContext} The context the object will be rendered within.
117     * @param time {Number} The global time within the engine.
118     * @memberOf HostObject
119     */
120    postUpdate: function(renderContext, time) {
121    },
122 
123    /**
124     * Add a component to the host object.  The components will be
125     * sorted based on their type, and priority within that type.
126     * Components with a higher priority will be sorted before components
127     * with a lower priority.  The sorting order for type is:
128     * <ul>
129     * <li>Input</li>
130     * <li>Logic</li>
131     * <li>Collision</li>
132     * <li>Rendering</li>
133     * </ul>
134     *
135     * @param component {BaseComponent} A component to add to the host
136     * @memberOf HostObject
137     */
138    add: function(component) {
139 
140       Assert((component instanceof BaseComponent), "Cannot add a non-component to a HostObject");
141       Assert(!this.isInHash(component.getName()), "Components must have a unique name within the host");
142 
143       this.base(component.getName(), component);
144 
145       component.setHostObject(this);
146       if (this.getObjects().length > 1)
147       {
148          this.sort(HostObject.componentSort);
149       }
150    },
151 
152    /**
153     * Get the component with the specified name from this object.
154     *
155     * @param name {String} The unique name of the component to get
156     * @type BaseComponent
157     * @memberOf HostObject
158     */
159    getComponent: function(name) {
160       return this.get(name.toUpperCase());
161    }
162 
163 }, /** @scope HostObject.prototype */{  // Interface
164 
165    /**
166     * Sort components within this object based upon their component
167     * type, and the priority within that type.  Components with a higher
168     * priority will be sorted before components with a lower priority.
169     * @memberOf HostObject
170     * @static
171     */
172    componentSort: function(component1, component2) {
173       return ((component1.getType() - component2.getType()) +
174               ((1 / component1.getPriority()) - (1 / component2.getPriority())));
175    },
176 
177    /**
178     * Get the class name of this object
179     *
180     * @type String
181     * @memberOf HostObject
182     */
183    getClassName: function() {
184       return "HostObject";
185    }
186 
187 });
