 35 // Includes
 36 Engine.include("/components/component.base.js");
 37 
 38 Engine.initObject("AbstractTextRenderer", "BaseComponent", function() {
 39 
 40 /**
 41  * @class Abstract class that provides the basic interface for all
 42  *        text render objects used by the {@link TextRenderer}.
 43  *
 44  * @constructor
 45  * @param componentName {String} The name of the renderer
 46  * @param priority {Number} The priority of the rendering order. Default: <tt>0.1</tt>
 47  */
 48 var AbstractTextRenderer = BaseComponent.extend(/** @scope AbstractTextRenderer.prototype */{
 49 
 50    text: null,
 51 
 52    color: "#000000",
 53 
 54    alignment: 0,
 55 
 56    weight: 1,
 57 
 58 	/**
 59 	 * @private
 60 	 */
 61    constructor: function(componentName, priority) {
 62       this.base(componentName || "TextRenderObject", BaseComponent.TYPE_RENDERING, priority || 0.1);
 63 
 64       this.text = "";
 65       this.weight = 1;
 66    },
 67 
 68 	/**
 69 	 * @private
 70 	 */
 71    release: function() {
 72       this.base();
 73       this.text = null;
 74       this.color= "#000000";
 75       this.alignment = 0;
 76       this.weight = 1;
 77    },
 78 
 79    /**
 80     * Get the text being rendered
 81     * @return {String} The text this renderer will draw
 82     */
 83    getText: function() {
 84       return this.text;
 85    },
 86 
 87    /**
 88     * Set the text to be rendered
 89     *
 90     * @param text {String} The text to render
 91     */
 92    setText: function(text) {
 93       this.text = text;
 94    },
 95 
 96    /**
 97     * Set the weight of the text to render.  Higher weights
 98     * are bolder text.
 99     *
100     * @param weight {Number} The weight of the text.
101     */
102    setTextWeight: function(weight) {
103       this.weight = weight;
104    },
105 
106    /**
107     * Get the weight of the text to render.
108     * @return {Number} The weight of the text
109     */
110    getTextWeight: function() {
111       return this.weight;
112    },
113 
114    /**
115     * Set the color of the text to render.
116     *
117     * @param color {String} The color of the text to render
118     */
119    setColor: function(color) {
120       this.color = color;
121    },
122 
123    /**
124     * Get the color of the text to render.
125     * @return {String} The text color
126     */
127    getColor: function() {
128       return this.color;
129    },
130 
131    /**
132     * Set the alignment of the text to one of the constants {@link #ALIGN_LEFT}, {@link #ALIGN_RIGHT}, or
133     *	{@link #ALIGN_CENTER}.
134     *
135     * @param alignment {Number} The alignment constant for the text
136     */
137    setAlignment: function(alignment) {
138       this.alignment = alignment;
139    },
140 
141    /**
142     * Get the alignment of the text.
143     * @return {Number} The alignment of the text
144     */
145    getAlignment: function() {
146       return this.alignment;
147    }
148 
149 }, /** @scope AbstractTextRenderer.prototype */{
150 
151    /**
152     * Get the class name of this object
153     * @return {String} The string "AbstractTextRenderer"
154     */
155    getClassName: function() {
156       return "AbstractTextRenderer";
157    },
158 
159    /**
160     * Align text with the left edge of the string at the point specified.
161     * @type Number
162     */
163    ALIGN_LEFT: 0,
164 
165    /**
166     * Align text with the right edge of the string at the point specified
167     * @type Number
168     */
169    ALIGN_RIGHT: 1,
170 
171    /**
172     * Align text with the center of the string at the point specified
173     * @type Number
174     */
175    ALIGN_CENTER: 2
176 
177 });
178 
179 return AbstractTextRenderer;
180 
181 });
