 33 /**
 34  * @class The base component class for components which render
 35  *        to a context.
 36  *
 37  * @param name {String} The name of the component
 38  * @param priority {Number} The priority of the component between 0.0 and 1.0
 39  */
 40 var RenderComponent = BaseComponent.extend(/** @scope RenderComponent.prototype */{
 41 
 42    drawMode: 0,
 43 
 44    constructor: function(name, priority) {
 45       this.base(name, BaseComponent.TYPE_RENDERING, priority || 0.1);
 46    },
 47 
 48    release: function() {
 49       this.base();
 50       this.drawMode = 0;
 51    },
 52 
 53    /**
 54     * Set the draw mode of the component.  Currently this determines
 55     * if the component should render itself to the context or not.
 56     *
 57     * @param drawMode {Number} One of <tt>RenderComponent.DRAW</tt> or
 58     *                 <tt>RenderComponent.NO_DRAW</tt>
 59     */
 60    setDrawMode: function(drawMode) {
 61       this.drawMode = drawMode;
 62    },
 63 
 64    /**
 65     * Get the drawing mode of the component.
 66     * @type Number
 67     */
 68    getDrawMode: function() {
 69       return this.drawMode;
 70    },
 71 
 72    /**
 73     * Handles whether or not the component should draw to the
 74     * render context, or not.
 75     *
 76     * @param renderContext {RenderContext} The rendering context
 77     * @param time {Number} The engine time in milliseconds
 78     */
 79    execute: function(renderContext, time) {
 80 
 81       if (this.drawMode == RenderContext.NO_DRAW)
 82       {
 83          return false;
 84       }
 85 
 86       // Set up drawing state
 87       // Xor, source-in, dest-in, etc...
 88 
 89       // Tear down drawing state
 90       return true;
 91    }
 92 
 93 }, { // Static
 94 
 95    /**
 96     * Get the class name of this object
 97     *
 98     * @type String
 99     */
100    getClassName: function() {
101       return "RenderComponent";
102    },
103 
104    /**
105     * The component should render itself to the rendering context.
106     */
107    DRAW: 0,
108 
109    /**
110     * The component <i>should not</i> render itself to the rendering context.
111     */
112    NO_DRAW: 1
113 
114 });
