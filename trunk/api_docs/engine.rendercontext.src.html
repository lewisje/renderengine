 32 // Includes
 33 Engine.include("/platform/engine.container.js");
 34 Engine.include("/platform/engine.math2d.js");
 35 Engine.include("/platform/engine.hostobject.js");

 37 Engine.initObject("RenderContext", "Container", function() {
 38 
 39 /**
 40  * @class A base rendering context.  Game objects are rendered to a context
 41  * during engine runtime.  A render context is a container of all of the objects
 42  * added to it so that each object is given the chance to render.
 43  *
 44  * @constructor
 45  * @param contextName {String} The name of this context.  Default: RenderContext
 46  * @param [surface] {HTMLElement} The surface node that all objects will be rendered to.
 47  * @see CanvasContext
 48  * @see DocumentContext
 49  * @extends Container
 50  */
 51 var RenderContext = Container.extend(/** @scope RenderContext.prototype */{
 52 
 53    surface: null,
 54 
 55    transformStackDepth: 0,
 56 
 57    viewport: null,
 58 
 59    worldPosition: null,
 60 
 61    worldRotation: null,
 62 
 63    worldScale: null,
 64 
 65    /**
 66     * @private
 67     */
 68    constructor: function(contextName, surface) {
 69       this.base(contextName || "RenderContext");
 70       this.surface = surface;
 71       this.setElement(surface);
 72       this.worldScale = 1;
 73       this.worldPosition = Point2D.create(0, 0);
 74       this.worldRotation = 0;
 75       this.viewport = Rectangle2D.create(0, 0, 100, 100);
 76    },
 77 
 78    /**
 79     * @private
 80     */
 81    release: function() {
 82       this.base();
 83       this.surface = null;
 84       this.transformStackDepth = 0;
 85       this.worldScale = null;
 86       this.worldPosition = null;
 87       this.worldRotation = null;
 88    },
 89 
 90    /**
 91     * Destroy the rendering context, and detach the surface from its
 92     * parent container.
 93     */
 94    destroy: function() {
 95       this.base();
 96       this.surface = null;
 97    },
 98 
 99    /**
100     * Set the surface element that objects will be rendered to.
101     *
102     * @param element {HTMLElement} The document node that all objects will be rendered to.
103     */
104    setSurface: function(element) {
105       this.surface = element;
106       this.setElement(element);
107    },
108 
109    /**
110     * Get the surface node that all objects will be rendered to.
111     * @return {HTMLElement} The document node that represents the rendering surface
112     */
113    getSurface: function() {
114       return this.surface;
115    },
116 
117    /**
118     * Set the scale of the rendering context.
119     *
120     * @param scaleX {Number} The scale along the X dimension
121     * @param scaleY {Number} The scale along the Y dimension
122     */
123    setScale: function(scaleX, scaleY) {
124    },
125 
126    /**
127     * Set the world scale of the rendering context.  All objects should
128     * be adjusted by this scale when the context renders.
129     *
130     * @param scale {Number} The uniform scale of the world
131     */
132    setWorldScale: function(scale) {
133       this.worldScale = scale;
134    },
135 
136    setWorldRotation: function(rotation) {
137       this.worldRotation = rotation;
138    },
139 
140    setWorldPosition: function(point) {
141       this.worldPosition.set(point);
142    },
143 
144    /**
145     * Gets an array representing the rendering scale of the world.
146     * @return {Array} The first element is the X axis, the second is the Y axis
147     */
148    getWorldScale: function() {
149       return this.worldScale;
150    },
151 
152    getWorldRotation: function() {
153       return this.worldRotation;
154    },
155 
156    getWorldPosition: function() {
157       return this.worldPosition;
158    },
159 
160    setViewport: function(rect) {
161       this.viewport = Rectangle2D.create(rect);
162    },
163 
164    getViewport: function() {
165       return this.viewport;
166    },
167 
168    /**
169     * Add an object to the render list.  Only objects
170     * within the render list will be rendered.  If an object declared
171     * an <tt>afterAdd</tt> method, it will be called after the object
172     * has been added to the context.
173     *
174     * @param obj {BaseObject} The object to add to the render list
175     */
176    add: function(obj) {
177       this.base(obj);
178       if (obj instanceof HostObject)
179       {
180          obj.setRenderContext(this);
181          this.sort();
182 
183          // Create a structure to hold information that is related to
184          // the render context that keeps it separate from the rest of the object.
185          obj.RenderContext = {};
186       }
187 
188       if (obj.afterAdd) {
189          // If the object being added to the render context has
190          // an "afterAdd" method, call it (canvas uses this for IE emulation)
191          obj.afterAdd(this);
192       }
193    },
194 
195    /**
196     * Returns the structure that contains information held about
197     * the rendering context.  This object allows a context to store
198     * extra information on an object that an object wouldn't know about.
199     * @return {Object} An object with data used by the context
200     */
201    getContextData: function(obj) {
202       return obj.RenderContext;
203    },
204 
205    /**
206     * Sort the render context's objects by their respective Z-index value.
207     */
208    sort: function() {
209       this.base(RenderContext.sortFn);
210    },
211 
212    /**
213     * Clear the context and prepare it for rendering.  If you pass a
214     * rectangle, only that portion of the world will be cleared.  If
215     * you don't pass a rectangle, the entire context is cleared.
216     *
217     * @param rect {Rectangle2D} The area to clear in the context, or
218     *             <tt>null</tt> to clear the entire context.
219     */
220    reset: function(rect) {
221    },
222 
223    /**
224     * Update the render context before rendering the objects to the surface.
225     *
226     * @param parentContext {RenderContext} A parent context, or <tt>null</tt>
227     * @param time {Number} The current render time in milliseconds from the engine.
228     */
229    update: function(parentContext, time)
230    {
231       this.render(time);
232    },
233 
234    /**
235     * Called after the update to render all of the objects to the context.
236     *
237     * @param time {Number} The current render time in milliseconds from the engine.
238     */
239    render: function(time) {
240       // Setup the world
241       this.reset();
242 
243       // Push the world transform
244       this.pushTransform();
245       this.setupWorld(time);
246 
247       // Run the objects if they are visible
248       var objs = this.getObjects();
249       for (var o in objs)
250       {
251          this.renderObject(objs[o], time);
252       }
253 
254       // Restore the world transform
255       this.popTransform();
256    },
257 
258    renderObject: function(obj, time) {
259       obj.update(this, time);
260    },
261 
262    /**
263     * Gives the render context a chance to initialize the world.
264     * Use this method to change the world position, rotation, scale, etc.
265     *
266     * @param time {Number} The current world time
267     */
268    setupWorld: function(time) {
269    },
270 
271    /**
272     * Increment the transform stack counter.
273     */
274    pushTransform: function() {
275       this.transformStackDepth++;
276    },
277 
278    /**
279     * Decrement the transform stack counter and ensure that the stack
280     * is not unbalanced.  An unbalanced stack can be indicative of
281     * objects that do not reset the state after rendering themselves.
282     */
283    popTransform: function() {
284       this.transformStackDepth--;
285       Assert((this.transformStackDepth >= 0), "Unbalanced transform stack!");
286    },
287 
288    /**
289     * This is a potentially expensive call, and can lead to rendering
290     * errors.  It is recommended against calling this method!
291     */
292    resetTransformStack: function() {
293       while (this.transformStackDepth > 0)
294       {
295          this.popTransform();
296       }
297    },
298 
299    /**
300     * Get a collection of object Id's that are near the object specified.
301     * The collection is a Javascript object that contains the Id's of the
302     * objects nearby.
303     *
304     * @param obj {Object2D} The object to test against
305     * @returns A collection that contains the names of objects nearby
306     * @deprecated
307     */
308    getNearObjects: function(obj) {
309       return {};
310    }
311 }, /** @scope RenderContext.prototype */{
312 
313    /**
314     * Sort the objects to draw from objects with the lowest
315     * z-index to the highest z-index.
316     * @static
317     */
318    sortFn: function(obj1, obj2) {
319       if (obj1 instanceof HostObject ||
320           obj2 instanceof HostObject)
321       {
322          return 0;
323       }
324 
325       return obj1.getZIndex() - obj2.getZIndex();
326    },
327 
328    /**
329     * Get the class name of this object
330     * @return {String} The string "RenderContext"
331     */
332    getClassName: function() {
333       return "RenderContext";
334    }
335 
336 });
337 
338 return RenderContext;
339 
340 });
