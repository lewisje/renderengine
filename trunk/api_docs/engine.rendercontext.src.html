 32 /**
 33  * @class A base rendering context.  Game objects are rendered to a context
 34  * during engine runtime.
 35  * @extends Container
 36  */
 37 var RenderContext = Container.extend(/** @scope RenderContext.prototype */{
 38 
 39    surface: null,
 40 
 41    transformStackDepth: 0,
 42 
 43    renderScaleX: 1,
 44 
 45    renderScaleY: 1,
 46 
 47    /**
 48     * Create an instance of this rendering context and optionally
 49     * assign the surface to which all rendering will occur.  This
 50     * is a base class and does not have any particular surface associated
 51     * with it.  Rendering contexts should extend this class.
 52     *
 53     * @param contextName {String} The name of this context.  Default: RenderContext
 54     * @param [surface] {HTMLElement} The surface node that all objects will be rendered to.
 55     * @see CanvasContext
 56     * @see DocumentContext
 57     * @constructor
 58     */
 59    constructor: function(contextName, surface) {
 60       this.base(contextName || "RenderContext");
 61       this.surface = surface;
 62       this.setElement(surface);
 63    },
 64 
 65    release: function() {
 66       this.base();
 67       this.surface = null;
 68       this.transformStackDepth = 0;
 69       this.renderScaleX = 1;
 70       this.renderScaleY = 1;
 71    },
 72 
 73    /**
 74     * Destroy the rendering context, and detach the surface from its
 75     * parent container.
 76     */
 77    destroy: function() {
 78       this.base();
 79       this.surface = null;
 80    },
 81 
 82    /**
 83     * Set the surface element that objects will be rendered to.
 84     *
 85     * @param element {HTMLElement} The surface node that all objects will be rendered to.
 86     */
 87    setSurface: function(element) {
 88       this.surface = element;
 89       this.setElement(element);
 90    },
 91 
 92    /**
 93     * Get the surface node that all objects will be rendered to.
 94     *
 95     * @type HTMLElement
 96     */
 97    getSurface: function() {
 98       return this.surface;
 99    },
100 
101    setScale: function(scaleX, scaleY) {
102    },
103 
104    setWorldScale: function(scaleX, scaleY) {
105       this.renderScaleX = scaleX;
106       this.renderScaleY = scaleY ? scaleY : scaleX;
107    },
108 
109    getWorldScale: function() {
110       return [this.renderScaleX, this.renderScaleY];
111    },
112 
113    /**
114     * Add an object to the render list.  Only objects
115     * within the render list will be rendered.
116     *
117     * @param obj {BaseObject} The object to add to the render list
118     */
119    add: function(obj) {
120       this.base(obj);
121       if (obj instanceof HostObject)
122       {
123          obj.setRenderContext(this);
124          this.sort(RenderContext.sortFn);
125 
126          // Create a structure to hold information that is related to
127          // the render context that keeps it separate from the rest of the object.
128          obj.RenderContext = {};
129       }
130    },
131 
132    /**
133     * Returns the structure that contains information held about
134     * the rendering context.  This object allows a context to store
135     * extra information on an object that an object wouldn't know about.
136     *
137     * @type Object
138     */
139    getContextData: function(obj) {
140       return obj.RenderContext;
141    },
142 
143    /**
144     * Update the render context before rendering the objects to the surface.
145     *
146     * @param parentContext {RenderContext} A parent context, or <tt>null</tt>
147     * @param time {Number} The current render time in milliseconds from the engine.
148     */
149    update: function(parentContext, time)
150    {
151       this.render(time);
152    },
153 
154    /**
155     * Called after the update to render all of the objects to the rendering context.
156     *
157     * @param time {Number} The current render time in milliseconds from the engine.
158     */
159    render: function(time) {
160 
161       // Push the world transform
162       this.pushTransform();
163 
164       // Set the scaling of the world
165       this.setScale(this.renderScaleX, this.renderScaleY);
166 
167       this.reset();
168 
169       // Render the objects into the world
170       var objs = this.getObjects();
171       for (var o in objs)
172       {
173          this.updateObject(objs[o], time);
174       }
175 
176       // Restore the world transform
177       this.popTransform();
178 
179    },
180 
181    /**
182     * Update an object in the render context
183     *
184     * @param obj {BaseObject} An object to update
185     * @param time {Number} The current time in milliseconds
186     */
187    updateObject: function(obj, time) {
188       obj.update(this, time);
189    },
190 
191    /**
192     * Increment the transform stack counter.
193     * @memberOf RenderContext
194     */
195    pushTransform: function() {
196       this.transformStackDepth++;
197    },
198 
199    /**
200     * Decrement the transform stack counter and ensure that the stack
201     * is not unbalanced.  An unbalanced stack can be indicative of
202     * objects that do not reset the state after rendering themselves.
203     */
204    popTransform: function() {
205       this.transformStackDepth--;
206       Assert((this.transformStackDepth >= 0), "Unbalanced transform stack!");
207    },
208 
209    /**
210     * This is a potentially expensive call, and can lead to rendering
211     * errors.  It is recommended against calling this method!
212     */
213    resetTransformStack: function() {
214       while (this.transformStackDepth > 0)
215       {
216          this.popTransform();
217       }
218    },
219 
220    /**
221     * Get a collection of object Id's that are near the object specified.
222     * The collection is a Javascript object that contains the Id's of the
223     * objects nearby.
224     *
225     * @param obj {Object2D} The object to test against
226     * @returns A collection that contains the names of objects nearby
227     * @type Object
228     */
229    getNearObjects: function(obj) {
230       return {};
231    }
232 }, /** @scope RenderContext.prototype */{ // Static
233 
234    /**
235     * Sort the objects to draw from objects with the lowest
236     * z-index to the highest z-index.
237     * @memberOf RenderContext
238     * @static
239     */
240    sortFn: function(obj1, obj2) {
241       if (obj1 instanceof HostObject ||
242           obj2 instanceof HostObject)
243       {
244          return 0;
245       }
246 
247       return obj1.getZIndex() - obj2.getZIndex();
248    },
249 
250    /**
251     * Get the class name of this object
252     *
253     * @type String
254     */
255    getClassName: function() {
256       return "RenderContext";
257    }
258 
259 });
