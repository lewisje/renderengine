 15       constructor: null,

 33          // Set the FPS of the game

 35 
 36          // Remove the "loading" message
 37          $("#loading").remove();
 38 
 39          // Create the render context
 40          this.fieldBox = Rectangle2D.create(0, 0, this.fieldWidth, this.fieldHeight);
 41          this.renderContext = CanvasContext.create("Playfield", this.fieldWidth, this.fieldHeight);
 42          this.renderContext.setBackgroundColor("black");
 43 
 44          // Add the new rendering context to the default engine context
 45          Engine.getDefaultContext().add(this.renderContext);
 46 			
 47 			// Add an instance of our game object to the render context
 48 			this.renderContext.add(MyObject.create());
 49       },
 50 
 51       /**
 52        * Called when a game is being shut down to allow it to clean up
 53        * any objects, remove event handlers, destroy the rendering context, etc.
 54        */
 55       teardown: function(){
 56          this.renderContext.destroy();
 57       },
 58 
 59       /**
 60        * Return a reference to the render context
 61        */
 62       getRenderContext: function(){
 63          return this.renderContext;
 64       },
 65 
 66       /**
 67        * Return a reference to the playfield box
 68        */
 69       getFieldBox: function() {

 71       }
 72 
 73    });
