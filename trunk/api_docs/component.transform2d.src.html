 33 // Includes
 34 Engine.include("/engine/engine.math2d.js");
 35 Engine.include("/components/component.base.js");
 36 
 37 Engine.initObject("Transform2DComponent", "BaseComponent", function() {
 38 
 39 /**
 40  * @class A simple component that maintains position, rotation, and scale.
 41  * @extends BaseComponent
 42  */
 43 var Transform2DComponent = BaseComponent.extend(/** @scope Transform2DComponent.prototype */{
 44 
 45    position: null,
 46 
 47    rotation: 0,
 48 
 49    scale: 1.0,
 50 
 51    lastPosition: null,
 52 
 53    lastRenderPosition: null,
 54 
 55    /**
 56     * @constructor
 57     * @memberOf Transform2DComponent
 58     */
 59    constructor: function(name, priority) {
 60       this.base(name, BaseComponent.TYPE_TRANSFORM, priority || 1.0);
 61       this.position = Point2D.create(0,0);
 62       this.lastPosition = Point2D.create(0,0);
 63       this.lastRenderPosition = Point2D.create(0,0);
 64       this.rotation = 0;
 65       this.scale = 1.0;
 66    },
 67 
 68    release: function() {
 69       this.base();
 70       this.position = null;
 71       this.rotation = 0;
 72       this.scale = 1.0;
 73       this.lastPosition = null;
 74       this.lastRenderPosition = null;
 75    },
 76 
 77    /**
 78     * Set the position of the transform.
 79     *
 80     * @param point {Point2D} The position
 81     */
 82    setPosition: function(point) {
 83       this.setLastPosition(this.getPosition());
 84       this.position.set(point);
 85    },
 86 
 87    /**
 88     * Returns the position of the transform.
 89     * @type Point2D
 90     */
 91    getPosition: function() {
 92       return this.position;
 93    },
 94 
 95    /**
 96     * Returns the render position of the transform.
 97     * @type Point2D
 98     */
 99    getRenderPosition: function() {
100       var wP = Point2D.create(this.getHostObject().getRenderContext().getWorldPosition());
101       var p = Point2D.create(this.position).sub(wP);
102       return p;
103    },
104 
105    /**
106     * Set the last position that the transform was at.
107     *
108     * @param point {Point2D} The last position
109     */
110    setLastPosition: function(point) {
111       this.lastPosition.set(point);
112       //var wP = Point2D.create(this.getHostObject().getRenderContext().getWorldPosition());
113       //this.lastRenderPosition.set(wP.sub(point));
114    },
115 
116    /**
117     * Get the last position of the transform.
118     * @type Point2D
119     */
120    getLastPosition: function() {
121       return this.lastPosition;
122    },
123 
124    /**
125     * Get the last position of the transform.
126     * @type Point2D
127     */
128    getLastRenderPosition: function() {
129       return this.lastRenderPosition;
130    },
131 
132    /**
133     * Set the rotation of the transform.
134     *
135     * @param rotation {Number} The rotation
136     */
137    setRotation: function(rotation) {
138       this.rotation = rotation;
139    },
140 
141    /**
142     * Get the rotation of the transform.
143     * @type Number
144     */
145    getRotation: function() {
146       return this.rotation;
147    },
148 
149    /**
150     * Get the render rotation of the transform.
151     * @type Number
152     */
153    getRenderRotation: function() {
154       var wR = this.getHostObject().getRenderContext().getWorldRotation();
155       return wR + this.rotation;
156    },
157 
158    /**
159     * Set the uniform scale of the transform.  The uniform
160     * scale applies to both the X and Y axis.
161     *
162     * @param scale {Number} The scale of the transform with 1.0 being 100%
163     */
164    setScale: function(scale) {
165       this.scale = scale;
166    },
167 
168    /**
169     * Get the uniform scale of the transform.
170     * @type Number
171     */
172    getScale: function() {
173       return this.scale;
174    },
175 
176    /**
177     * Get the uniform render scale of the transform.
178     * @type Number
179     */
180    getRenderScale: function() {
181 //    var wS = this.getHostObject().getRenderContext().getWorldScale();
182 //      return wS * this.scale;
183       return this.scale;
184    },
185 
186    /**
187     * Set the components of a transformation: position, rotation,
188     * and scale, within the rendering context.
189     *
190     * @param renderContext {RenderContext} The rendering context
191     * @param time {Number} The engine time in milliseconds
192     * @param rendering {Boolean} <tt>true</tt> during the rendering phase
193     */
194    execute: function(renderContext, time) {
195       renderContext.setPosition(this.getRenderPosition());
196       renderContext.setRotation(this.getRenderRotation());
197       var s = this.getRenderScale();
198       renderContext.setScale(s);
199    }
200 }, {
201    /**
202     * Get the class name of this object
203     *
204     * @type String
205     * @memberOf Transform2DComponent
206     */
207    getClassName: function() {
208       return "Transform2DComponent";
209    }
210 });
211 
212 return Transform2DComponent;
213 
214 });
