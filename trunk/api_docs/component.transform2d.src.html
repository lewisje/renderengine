 32 /**
 33  * @class A simple component that has position, rotation, and scale.
 34  * @extends BaseComponent
 35  */
 36 var Transform2DComponent = BaseComponent.extend(/** @scope Transform2DComponent.prototype */{
 37 
 38    position: null,
 39 
 40    rotation: 0,
 41 
 42    scale: 1.0,
 43 
 44    lastPosition: null,
 45 
 46    /**
 47     * @constructor
 48     * @memberOf Transform2DComponent
 49     */
 50    constructor: function(name, priority) {
 51       this.base(name, BaseComponent.TYPE_RENDERING, priority || 1.0);
 52       this.position = new Point2D(0,0);
 53       this.lastPosition = new Point2D(0,0);
 54    },
 55 
 56    release: function() {
 57       this.base();
 58       this.position = null;
 59       this.rotation = 0;
 60       this.scale = 1.0;
 61       this.lastPosition = null;
 62    },
 63 
 64    /**
 65     * Set the position of the transform.
 66     *
 67     * @param point {Point2D} The position
 68     */
 69    setPosition: function(point) {
 70       this.setLastPosition(this.getPosition());
 71       this.position.set(point);
 72    },
 73 
 74    /**
 75     * Returns the position of the transform.
 76     * @type Point2D
 77     */
 78    getPosition: function() {
 79       return this.position;
 80    },
 81 
 82    /**
 83     * Set the last position that the transform was at.
 84     *
 85     * @param point {Point2D} The last position
 86     */
 87    setLastPosition: function(point) {
 88       this.lastPosition.set(point);
 89    },
 90 
 91    /**
 92     * Get the last position of the transform.
 93     * @type Point2D
 94     */
 95    getLastPosition: function() {
 96       return this.lastPosition;
 97    },
 98 
 99    /**
100     * Set the rotation of the transform.
101     *
102     * @param rotation {Number} The rotation
103     */
104    setRotation: function(rotation) {
105       this.rotation = rotation;
106    },
107 
108    /**
109     * Get the rotation of the transform.
110     * @type Number
111     */
112    getRotation: function() {
113       return this.rotation;
114    },
115 
116    /**
117     * Set the uniform scale of the transform.  The uniform
118     * scale applies to both the X and Y axis.
119     *
120     * @param scale {Number} The scale of the transform with 1.0 being 100%
121     */
122    setScale: function(scale) {
123       this.scale = scale;
124    },
125 
126    /**
127     * Get the uniform scale of the transform.
128     * @type Number
129     */
130    getScale: function() {
131       return this.scale;
132    },
133 
134    /**
135     * Set the components of a transformation: position, rotation,
136     * and scale, within the rendering context.
137     *
138     * @param renderContext {RenderContext} The rendering context
139     * @param time {Number} The engine time in milliseconds
140     */
141    execute: function(renderContext, time) {
142       renderContext.setPosition(this.position);
143       renderContext.setRotation(this.rotation);
144       renderContext.setScale(this.scale, this.scale);
145    }
146 }, {
147    /**
148     * Get the class name of this object
149     *
150     * @type String
151     * @memberOf Transform2DComponent
152     */
153    getClassName: function() {
154       return "Transform2DComponent";
155    }
156 });
