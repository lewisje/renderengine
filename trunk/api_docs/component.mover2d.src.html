 32 /**
 33  * @class A simple mover component that adjusts the translation of a component
 34  * over time.  This component handles velocity, acceleration, and
 35  * deceleration.
 36  *
 37  * @param name {String} The name of the component
 38  * @param priority {Number} The priority of this component between 0.0 and 1.0
 39  * @extends Transform2DComponent
 40  */
 41 var Mover2DComponent = Transform2DComponent.extend(/** @scope Mover2DComponent.prototype */{
 42 
 43    lastTime: -1,
 44 
 45    velocity: null,
 46 
 47    vDecay: 0,
 48 
 49    angularVelocity: 0,
 50 
 51    lPos: null,
 52 
 53    constructor: function(name, priority) {
 54       this.velocity = new Vector2D(0,0);
 55       this.acceleration = new Vector2D(0,0);
 56       this.base(name, priority || 1.0);
 57       this.lPos = new Point2D(0,0);
 58       this.vDecay = 0;
 59    },
 60 
 61    release: function() {
 62       this.base();
 63       this.lastTime = -1;
 64       this.velocity = null;
 65       this.vDecay = 0;
 66       this.angularVelocity = 0;
 67       this.lPos = null;
 68    },
 69 
 70    /**
 71     * Updates the transformation of the component, setting the
 72     * position and rotation based on the time that has elapsed since
 73     * the last update.  This component handles frame rate independent
 74     * updates so that lag between frames doesn't affect the relative
 75     * position or rotation.
 76     *
 77     * @param renderContext {RenderContext} The render context for the component
 78     * @param time {Number} The engine time in milliseconds
 79     */
 80    execute: function(renderContext, time) {
 81       this.lPos.set(this.getPosition());
 82       var rot = this.getRotation();
 83 
 84       // If we've just come into the world, we can short circuit with a
 85       // quick addition of the velocity.
 86       if (this.lastTime == -1)
 87       {
 88          this.setPosition(this.lPos.add(this.velocity));
 89       }
 90       else
 91       {
 92          if (this.vDecay != 0 && this.velocity.len() > 0) {
 93             // We need to decay the velocity by the amount
 94             // specified until velocity is zero, or less than zero
 95             var invVelocity = new Vector2D(this.velocity).neg();
 96             invVelocity.mul(this.vDecay);
 97 
 98             this.velocity.add(invVelocity);
 99          }
100 
101          var diff = (time - this.lastTime) * 0.1;
102          var vz = new Vector2D(this.velocity).mul(diff);
103          this.setPosition(this.lPos.add(vz));
104 
105          this.setRotation(rot + this.angularVelocity * (diff));
106       }
107 
108       this.base(renderContext, time);
109       this.lastTime = time;
110    },
111 
112    /**
113     * Set the velocity vector of the component.
114     *
115     * @param vector {Vector2D} The velocity vector
116     */
117    setVelocity: function(vector) {
118       this.velocity.set(vector);
119    },
120 
121    /**
122     * Returns the velocity vector of the component
123     * @type Vector2D
124     */
125    getVelocity: function() {
126       return this.velocity;
127    },
128 
129    /**
130     * Set the decay rate at which the velocity will
131     * approach zero.  You can use this value to cause
132     * a moving object to eventually stop moving.
133     *
134     * @param decay {Number} The rate at which velocity decays
135     */
136    setVelocityDecay: function(decay) {
137       this.vDecay = decay;
138    },
139 
140    /**
141     * Get the rate at which velocity will decay to zero.
142     * @return The velocity decay rate
143     * @type Number
144     */
145    getVelocityDecay: function() {
146       return this.vDecay;
147    },
148 
149    /**
150     * Set the angular velocity as a number of degrees per second.
151     *
152     * @param angularVelocity {Number} The angle change per second
153     */
154    setAngularVelocity: function(angularVelocity) {
155       this.angularVelocity = angularVelocity;
156    },
157 
158    /**
159     * Returns the angle change per second of the component.
160     * @type Number
161     */
162    getAngularVelocity: function() {
163       return this.angularVelocity;
164    }
165 }, {
166    /**
167     * Get the class name of this object

174    }
