489  * Halts the engine if the test fails, throwing the error as a result.
490  *
491  * @param test {Boolean} A simple test that should evaluate to <tt>true</tt>
492  * @param error {String} The error message to throw if the test fails
493  */
494 var Assert = function(test, error) {
495    if (!test)
496    {
497       Engine.shutdown();
498       if (this.caller) {
499          var funcName = "";
500          if (this.caller.name)
501          {
502             funcName = this.caller.name;
503          }
504          else
505          {
506             // Try to determine the function name that called this Assert
507             var fRE = /((function\s+([\$_\.\w]+))|(var\s+([\$_\.\w]+)\s+=\s+function)|(function\(.*?\)))/;
508             var m = fRE.exec(this.caller.toString());
509             if (m)
510             {
511                funcName = (m[3] || m[5] || "anonymous");
512             }
513          }
514          error += "\nin function " + funcName;
515       }
516 
517       // This will provide a stacktrace for browsers that support it
518       throw new Error(error);
519    }
520 };

540 var EngineSupport = Base.extend(/** @scope EngineSupport.prototype */{
541    constructor: null,

543    /**
544     * Get the index of an element in the specified array.
545     *
546     * @param array {Array} The array to scan
547     * @param obj {Object} The object to find
548     * @param from {Number=0} The index to start at, defaults to zero.
549     * @memberOf EngineSupport
550     */
551    indexOf: function(array, obj, from) {
552       if (!array) {
553          return -1;
554       }
555 
556       if (Array.prototype.indexOf) {
557          return array.indexOf(obj, from);
558       }
559       else
560       {
561          var len = array.length;
562          var from = Number(from) || 0;
563          from = (from < 0)
564             ? Math.ceil(from)
565             : Math.floor(from);
566          if (from < 0)
567             from += len;
568 
569          for (; from < len; from++)
570          {
571             if (from in array && array[from] === obj)
572                return from;
573          }
574          return -1;
575       }
576    },
577 
578    /**
579     * Remove an element from an array.
580     *
581     * @param array {Array} The array to modify
582     * @param obj {Object} The object to remove
583     * @memberOf EngineSupport
584     */
585    arrayRemove: function(array, obj) {
586       if (!array) {
587          return;
588       }
589 
590       var idx = EngineSupport.indexOf(array, obj);
591       if (idx != -1)
592       {
593          array.splice(idx, 1);
594       }
595    },

597    /**
598     * Calls a provided callback function once for each element in
599     * an array, and constructs a new array of all the values for which
600     * callback returns a <tt>true</tt> value. callback is invoked only
601     * for indexes of the array which have assigned values; it is not invoked
602     * for indexes which have been deleted or which have never been assigned
603     * values. Array elements which do not pass the callback test are simply
604     * skipped, and are not included in the new array.
605     *
606     * @param array {Array} The array to filter.
607     * @param fn {Function} The callback to invoke.  It will be passed three
608     *                      arguments: The element value, the index of the element,
609     *                      and the array being traversed.
610     * @param thisp {Object} Used as <tt>this</tt> for each invocation of the
611     *                       callback.  Default: <tt>null</tt>
612     * @memberOf EngineSupport
613     */
614    filter: function(array, fn, thisp) {
615       if (!array) {
616          return null;
617       }
618 
619       if (Array.prototype.filter) {
620          return array.filter(fn, thisp)
621       }
622       else
623       {
624          var len = array.length;
625          if (typeof fn != "function")
626             throw new TypeError();
627 
628          var res = new Array();
629          for (var i = 0; i < len; i++)
630          {
631             if (i in array)
632             {
633                var val = array[i]; // in case fn mutates this
634                if (fn.call(thisp, val, i, array))
635                   res.push(val);
636             }
637          }
638 
639          return res;
640       }
641    },
642 
643    /**
644     * Executes a callback for each element within an array.
645     *
646     * @param array {Array} The array to operate on
647     * @param fn {Function} The function to apply to each element
648     * @param [thisp] {Object} An optional "this" pointer to use in the callback
649     * @memberOf EngineSupport
650     */
651    forEach: function(array, fn, thisp) {
652       if (!array) {
653          return;
654       }
655 
656       if (Array.prototype.forEach) {
657          array.forEach(fn, thisp);
658       }
659       else
660       {
661          var len = array.length;
662          if (typeof fn != "function")
663             throw new TypeError();
664 
665          for (var i = 0; i < len; i++)
666          {
667             if (i in array)
668                fn.call(thisp, array[i], i, array);
669          }
670       }
671    },

1979       var defs;

1981 
1982       try {
1983          var k = obj();
1984       } catch (ex) {
1985          // The class probably extends a non-existent class. Replace the parent

1987          var extRE = new RegExp("(var\\s*)?([\\$_\\w\\.]*?)\\s*=\\s*([\\$_\\w\\.]*?)\\.extend\\(");

2008          if ($.isFunction(k[f]) && k.hasOwnProperty(f)) {

2014             var strR = /(["|']).*?\1/g;
2015

2018                var fdef = m[2].replace(strR, "");

2021                var anonDeps = Engine.findAnonDeps(objectName, fdef);
2022                var aFnRE = new RegExp("function\\s*\\(([\\$\\w_, ]*?)\\)\\s*\\{((.|\\n)*?)\\};?","g");

2036                      vs.push(args[a].replace(" ",""));
2037                   }
2038                }
2039

2041                for (var a in anonDeps) {
