508  * @class Engine support class.  Provides extra functions the engine or games
509  *        can use.  Mainly contains functions that can manipulate arrays and
510  *        generate/read JSON.
511  */
512 var EngineSupport = Base.extend(/** @scope EngineSupport.prototype */{
513    constructor: null,
514 
515    /**
516     * Get the index of an element in the specified array.
517     *
518     * @param array {Array} The array to scan
519     * @param obj {Object} The object to find
520     * @param from {Number=0} The index to start at, defaults to zero.
521     * @memberOf EngineSupport
522     */
523    indexOf: function(array, obj, from) {
524       if (!array) {
525          return -1;
526       }
527 
528       if (Array.prototype.indexOf) {
529          return array.indexOf(obj, from);
530       }
531       else
532       {
533          var len = array.length;
534          var from = Number(from) || 0;
535          from = (from < 0)
536             ? Math.ceil(from)
537             : Math.floor(from);
538          if (from < 0)
539             from += len;
540 
541          for (; from < len; from++)
542          {
543             if (from in array && array[from] === obj)
544                return from;
545          }
546          return -1;
547       }
548    },
549 
550    /**
551     * Remove an element from an array.
552     *
553     * @param array {Array} The array to modify
554     * @param obj {Object} The object to remove
555     * @memberOf EngineSupport
556     */
557    arrayRemove: function(array, obj) {
558       if (!array) {
559          return;
560       }
561 
562       var idx = EngineSupport.indexOf(array, obj);
563       if (idx != -1)
564       {
565          array.splice(idx, 1);
566       }
567    },
568 
569    /**
570     * Calls a provided callback function once for each element in
571     * an array, and constructs a new array of all the values for which
572     * callback returns a <tt>true</tt> value. callback is invoked only
573     * for indexes of the array which have assigned values; it is not invoked
574     * for indexes which have been deleted or which have never been assigned
575     * values. Array elements which do not pass the callback test are simply
576     * skipped, and are not included in the new array.
577     *
578     * @param array {Array} The array to filter.
579     * @param fn {Function} The callback to invoke.  It will be passed three
580     *                      arguments: The element value, the index of the element,
581     *                      and the array being traversed.
582     * @param thisp {Object} Used as <tt>this</tt> for each invocation of the
583     *                       callback.  Default: <tt>null</tt>
584     * @memberOf EngineSupport
585     */
586    filter: function(array, fn, thisp) {
587       if (!array) {
588          return null;
589       }
590 
591       if (Array.prototype.filter) {
592          return array.filter(fn, thisp)
593       }
594       else
595       {
596          var len = array.length;
597          if (typeof fn != "function")
598             throw new TypeError();
599 
600          var res = new Array();
601          for (var i = 0; i < len; i++)
602          {
603             if (i in array)
604             {
605                var val = array[i]; // in case fn mutates this
606                if (fn.call(thisp, val, i, array))
607                   res.push(val);
608             }
609          }
610 
611          return res;
612       }
613    },
614 
615    /**
616     * Executes a callback for each element within an array.
617     *
618     * @param array {Array} The array to operate on
619     * @param fn {Function} The function to apply to each element.  It will be passed three
620     *                      arguments: The element value, the index of the element,
621     *                      and the array being traversed.
622     * @param [thisp] {Object} An optional "this" pointer to use in the callback
623     * @memberOf EngineSupport
624     */
625    forEach: function(array, fn, thisp) {
626       if (!array) {
627          return;
628       }
629 
630       if (Array.prototype.forEach) {
631          array.forEach(fn, thisp);
632       }
633       else
634       {
635          var len = array.length;
636          if (typeof fn != "function")
637             throw new TypeError();
638 
639          for (var i = 0; i < len; i++)
640          {
641             if (i in array)
642                fn.call(thisp, array[i], i, array);
643          }
644       }
645    },
646 
647 	/**
648 	 * Fill the specified array of <tt>size</tt> the
649 	 * <tt>value</tt> at each index.
650 	 * @param {Array} arr The array to fill
651 	 * @param {Number} size The size of the array to fill
652 	 * @param {Object} value The value to put at each index
653 	 */
654 	fillArray: function(arr, size, value) {
655 		for (var i = 0; i < size; i++) {
656 			arr[i] = value;
657 		}
658 	},
659 
660    /**
661     * Get the path from a fully qualified URL.
662     *
663     * @param url {String} The URL
664     * @return {String} The path
665     * @memberOf EngineSupport
666     */
667    getPath: function(url) {
668       var l = url.lastIndexOf("/");
669       return url.substr(0, l);
670    },
671 
672    /**
673     * Get the query parameters from the window location object.  The
674     * object returned will contain a key/value pair for each argument
675     * found.
676     *
677     * @return {Object} An <tt>Object</tt> with a key and value for each query argument.
678     * @memberOf EngineSupport
679     */
680    getQueryParams: function() {
681       if (!EngineSupport.parms) {
682          EngineSupport.parms = {};
683          var p = window.location.toString().split("?")[1];
684          if (p)
685          {
686             p = p.split("&");
687             for (var x = 0; x < p.length; x++)
688             {
689                var v = p[x].split("=");
690                EngineSupport.parms[v[0]] = (v.length > 1 ? v[1] : "");
691             }
692          }
693       }
694       return EngineSupport.parms;
695    },
696 
697    /**
698     * Check for a query parameter and to see if it evaluates to one of the following:
699     * <tt>true</tt>, <tt>1</tt>, <tt>yes</tt>, or <tt>y</tt>.  If so, returns <tt>true</tt>.
700     *
701     * @param paramName {String} The query parameter name
702     * @return {Boolean} <tt>true</tt> if the query parameter exists and is one of the specified values.
703     * @memberOf EngineSupport
704     */
705    checkBooleanParam: function(paramName) {
706       return (EngineSupport.getQueryParams()[paramName] &&
707               (EngineSupport.getQueryParams()[paramName] == "true" ||
708                EngineSupport.getQueryParams()[paramName] == "1" ||
709                EngineSupport.getQueryParams()[paramName].toLowerCase() == "yes" ||
710                EngineSupport.getQueryParams()[paramName].toLowerCase() == "y"));
711    },
