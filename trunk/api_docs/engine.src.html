363 /**
364  * Halts the engine if the test fails, throwing the error as a result.
365  *
366  * @param test {Boolean} A simple test that should evaluate to <tt>true</tt>
367  * @param error {String} The error message to throw if the test fails
368  */
369 var Assert = function(test, error) {
370    if (!test)
371    {
372       Engine.shutdown();
373       if (this.caller) {
374          var funcName = "";
375          if (this.caller.name)
376          {
377             funcName = this.caller.name;
378          }
379          else
380          {
381             // Try to determine the function name that called this Assert
382             var fRE = /((function\s+([\$_\.\w]+))|(var\s+([\$_\.\w]+)\s+=\s+function)|(function\(.*?\)))/;
383             var m = fRE.exec(this.caller.toString());
384             if (m)
385             {
386                funcName = (m[3] || m[5] || "anonymous");
387             }
388          }
389          error += "\nin function " + funcName;
390       }
391       throw new Error(error);
392    }
393 };

395 /**
396  * Reports a warning if the test fails.
397  *
398  * @param test {Boolean} A simple test that should evaluate to <tt>true</tt>
399  * @param error {String} The warning to display if the test fails
400  */
401 var AssertWarn = function(test, warning) {
402    if (!test)
403    {
404       Console.warn(warning);
405    }

409  * @class Engine support class.  Provides extra functions the engine or games
410  *        can use.  Mainly contains functions that can manipulate arrays and
411  *        generate/read JSON.
412  */
413 var EngineSupport = Base.extend(/** @scope EngineSupport.prototype */{
414    constructor: null,
415 
416    /**
417     * Get the index of an element in the specified array.
418     *
419     * @param array {Array} The array to scan
420     * @param obj {Object} The object to find
421     * @param from {Number=0} The index to start at, defaults to zero.
422     * @memberOf EngineSupport
423     */
424    indexOf: function(array, obj, from) {
425       if (!array) {
426          return -1;
427       }
428 
429       if (Array.prototype.indexOf) {
430          return array.indexOf(obj, from);
431       }
432       else
433       {
434          var len = array.length;
435          var from = Number(from) || 0;
436          from = (from < 0)
437             ? Math.ceil(from)
438             : Math.floor(from);
439          if (from < 0)
440             from += len;
441 
442          for (; from < len; from++)
443          {
444             if (from in array && array[from] === obj)
445                return from;
446          }
447          return -1;
448       }
449    },
450 
451    /**
452     * Remove an element from an array.
453     *
454     * @param array {Array} The array to modify
455     * @param obj {Object} The object to remove
456     * @memberOf EngineSupport
457     */
458    arrayRemove: function(array, obj) {
459       if (!array) {
460          return;
461       }
462 
463       var idx = EngineSupport.indexOf(array, obj);
464       if (idx != -1)
465       {
466          array.splice(idx, 1);
467       }
468    },
469 
470    /**
471     * Calls a provided callback function once for each element in
472     * an array, and constructs a new array of all the values for which
473     * callback returns a <tt>true</tt> value. callback is invoked only
474     * for indexes of the array which have assigned values; it is not invoked
475     * for indexes which have been deleted or which have never been assigned
476     * values. Array elements which do not pass the callback test are simply
477     * skipped, and are not included in the new array.
478     *
479     * @param array {Array} The array to filter.
480     * @param fn {Function} The callback to invoke.  It will be passed three
481     *                      arguments: The element value, the index of the element,
482     *                      and the array being traversed.
483     * @param thisp {Object} Used as <tt>this</tt> for each invocation of the
484     *                       callback.  Default: <tt>null</tt>
485     * @memberOf EngineSupport
486     */
487    filter: function(array, fn, thisp) {
488       if (!array) {
489          return null;
490       }
491 
492       if (Array.prototype.filter) {
493          return array.filter(fn, thisp)
494       }
495       else
496       {
497          var len = array.length;
498          if (typeof fn != "function")
499             throw new TypeError();
500 
501          var res = new Array();
502          for (var i = 0; i < len; i++)
503          {
504             if (i in array)
505             {
506                var val = array[i]; // in case fn mutates this
507                if (fn.call(thisp, val, i, array))
508                   res.push(val);
509             }
510          }
511 
512          return res;
513       }
514    },
515 
516    /**
517     * Executes a callback for each element within an array.
518     *
519     * @param array {Array} The array to operate on
520     * @param fn {Function} The function to apply to each element
521     * @param [thisp] {Object} An optional "this" pointer to use in the callback
522     * @memberOf EngineSupport
523     */
524    forEach: function(array, fn, thisp) {
525       if (!array) {
526          return;
527       }
528 
529       if (Array.prototype.forEach) {
530          array.forEach(fn, thisp);
531       }
532       else
533       {
534          var len = array.length;
535          if (typeof fn != "function")
536             throw new TypeError();
537 
538          for (var i = 0; i < len; i++)
539          {
540             if (i in array)
541                fn.call(thisp, array[i], i, array);
542          }
543       }
544    },
545 
546    /**
547     * Get the path from a fully qualified URL.
548     *
549     * @param url {String} The URL
550     * @type String
551     * @memberOf EngineSupport
552     */
553    getPath: function(url) {
554       var l = url.lastIndexOf("/");
555       return url.substr(0, l);
556    },
557 
558    /**
559     * Get the query parameters from the window location object.  The
560     * object returned will contain a key/value pair for each argument
561     * found.
562     *
563     * @type Object
564     * @return An <tt>Object</tt> with a key and value for each query argument.
565     * @memberOf EngineSupport
566     */
567    getQueryParams: function() {
568       if (!EngineSupport.parms) {
569          EngineSupport.parms = {};
570          var p = window.location.toString().split("?")[1];
571          if (p)
572          {
573             p = p.split("&");
574             for (var x = 0; x < p.length; x++)
575             {
576                var v = p[x].split("=");
577                EngineSupport.parms[v[0]] = (v.length > 1 ? v[1] : "");
578             }
579          }
580       }
581       return EngineSupport.parms;
582    },
583 
584    /**
585     * Check for a query parameter and to see if it evaluates to one of the following:
586     * <tt>true</tt>, <tt>1</tt>, <tt>yes</tt>, or <tt>y</tt>.  If so, returns <tt>true</tt>.
587     *
588     * @param paramName {String} The query parameter name
589     * @return <tt>true</tt> if the query parameter exists and is one of the specified values.
590     * @type Boolean
591     */
592    checkBooleanParam: function(paramName) {
593       return (EngineSupport.getQueryParams()[paramName] &&
594               (EngineSupport.getQueryParams()[paramName] == "true" ||
595                EngineSupport.getQueryParams()[paramName] == "1" ||
596                EngineSupport.getQueryParams()[paramName].toLowerCase() == "yes" ||
597                EngineSupport.getQueryParams()[paramName].toLowerCase() == "y"));
598    },
599 
600    /**
601     * Check for a query parameter and to see if it evaluates to the specified value.
602     * If so, returns <tt>true</tt>.
603     *

606     * @return <tt>true</tt> if the query parameter exists and is the value specified
607     * @type Boolean
608     */
609    checkStringParam: function(paramName, val) {
610       return (EngineSupport.getQueryParams()[paramName] && EngineSupport.getQueryParams()[paramName] == val);
611    },
612 
613    /**
614     * Check for a query parameter and to see if it evaluates to the specified number.
615     * If so, returns <tt>true</tt>.
616     *
617     * @param paramName {String} The query parameter name
618     * @param val {Number} The number to check for
619     * @return <tt>true</tt> if the query parameter exists and is the value specified
620     * @type Boolean
621     */
622    checkNumericParam: function(paramName, val) {
623       return (EngineSupport.getQueryParams()[paramName] && Number(EngineSupport.getQueryParams()[paramName]) == val);
624    },
625 
626    /**
627     * Returns specified object as a JavaScript Object Notation (JSON) string.
628     *
629     * Code to handle "undefined" type was delibrately not implemented, being that it is not part of JSON.
630     * "undefined" type is casted to "null".
631     *
632     * @param object {Object} Must not be undefined or contain undefined types and variables.
633     * @return String
634     * @memberOf EngineSupport
635     */
636    toJSONString: function(o)
637    {
638       if(o == null)
639       {
640          return "null";
641       }
642 
643       switch(o.constructor)
644       {
645          case Array:
646             var a = [], i;
647             for(i = 0; i < o.length; i++)
648             {
649                a[i] = EngineSupport.toJSONString(o[i]);
650             }
651             return "[" + a.join() + "]";
652          case String:
653             return EngineSupport.quoteString(o);
654          case Boolean:
655          case Number:
656             return o.toString();
657          default:
658             var a = [], i;
659             for(i in o)
660             {
661                if(o[i] == null)
662                {
663                   a.push(EngineSupport.quoteString(i) + ":null");
664                }
665                else if(o[i].constructor != Function)
666                {
667                   a.push(EngineSupport.quoteString(i) + ':' + EngineSupport.toJSONString(o[i]));
668                }
669             }
670             return '{' + a.join() + '}';
671       }
672    },
673 
674    /**
675     * Parses specified JavaScript Object Notation (JSON) string back into its corresponding object.
676     *
677     * @param jsonString
678     * @return Object
679     * @see http://www.json.org
680     * @memberOf EngineSupport
681     */
682    parseJSON: function(jsonString)
683    {
684       var obj = eval('(function(){return ' + jsonString + ';})();');

686    },

688    /**

690     *
691     * @param string
692     * @type String
693     * @memberOf EngineSupport
694     */
695    quoteString: function(o)
696    {
697       return '"'+o.replace(/[\\"\r\n]/g, function(s)
698          {
699             switch(s)
700             {
701                case "\\":return "\\\\";
702                case "\r":return "\\r";
703                case "\n":return "\\n";
704                case '"':return '\\"';
705             }
706          }
707       )+'"';
708    },

1008    /**
1009     * Load a script from the server and append it to
1010     * the head element of the browser.  Script names are
1011     * cached so they will not be loaded again.
1012     *
1013     * @param scriptPath {String} The URL of a script to load.
1014     * @memberOf Engine
1015     */
1016    loadScript: function(scriptPath) {
1017       var s = scriptPath.replace(/[\/\.]/g,"_");
1018       if (this.loadedScripts[s] == null)
1019       {
1020          // Store the request in the cache
1021          this.loadedScripts[s] = scriptPath;
1022 
1023          if (!Engine.scriptQueue) {
1024             // Create the queue
1025             Engine.scriptQueue = [];
1026          }
1027 
1028          // Put script into load queue
1029          Engine.scriptQueue.push(scriptPath);
1030 
1031          if (!Engine.scriptQueueTimer) {
1032             // Process any waiting scripts
1033             Engine.scriptQueueTimer = setInterval(function() {
1034                if (Engine.queuePaused) {
1035                   if (Engine.pauseReps++ > 500) {
1036                      // If after ~5 seconds the queue is still paused, unpause it and
1037                      // warn the user that the situation occurred
1038                      Console.error("Script queue was paused for 5 seconds and not resumed -- restarting...");
1039                      Engine.pauseReps = 0;
1040                      Engine.pauseQueue(false);
1041                   }
1042                   return;
1043                }
1044 
1045                Engine.pauseReps = 0;
1046 
1047                if (Engine.scriptQueue.length > 0) {
1048                   Engine.processScriptQueue();
1049                } else {
1050                   // Stop the queue timer if there are no scripts
1051                   clearInterval(Engine.scriptQueueTimer);
1052                   Engine.scriptQueueTimer = null;
1053                }
1054             }, 10);
1055 
1056             Engine.readyForNextScript = true;
1057          }
1058       }
1059    },

1093    /**
1094     * Process any scripts that are waiting to be loaded.
1095     * @private
1096     * @memberOf Engine
1097     */
1098    processScriptQueue: function() {
1099       if (Engine.scriptQueue.length > 0 && Engine.readyForNextScript) {
1100          // Hold the queue until the script is loaded
1101          Engine.readyForNextScript = false;
1102 
1103          // Get next script...
1104          var scriptPath = Engine.scriptQueue.shift();
1105 
1106          // If the queue element is a function, execute it and return
1107          if (typeof scriptPath == "function") {
1108             scriptPath();
1109             Engine.readyForNextScript = true;
1110             return;
1111          }
1112 
1113          // A hack to allow us to do filesystem testing
1114          if (!window.localDebugMode)
1115          {
1116             jQuery.getScript(scriptPath, function() {
1117                Console.debug("Loaded '" + scriptPath + "'");
1118                Engine.readyForNextScript = true;
1119             });
1120          }
1121          else
1122          {
1123             // If we're running locally, don't use jQuery to get the script.
1124             // This allows us to see and debug the loaded scripts.
1125             var n = document.createElement("script");
1126             n.src = scriptPath;
1127             n.type = "text/javascript";
1128             var fn = function() {
1129                if (!this.readyState || this.readyState == "loaded" || this.readyState == "complete") {
1130                   Console.debug("Loaded '" + scriptPath + "'");
1131                   Engine.readyForNextScript = true;
1132                }
1133             }
1134             if ($.browser.msie) {
1135                n.defer = true;
1136                n.onreadystatechange = fn;
1137             } else {
1138                n.onload = fn;
1139             }
1140 
1141             var h = document.getElementsByTagName("head")[0];
1142             h.appendChild(n);
1143          }
1144       }
1145    },

1349     * @private
1350     * @memberOf Engine
1351     */
1352    engineTimer: function() {
1353 
1354       if (!this.running) {
1355          return;
1356       }
1357 
1358       var b = new Date().getTime();
1359       Engine.worldTime = b;
1360 
1361       // Update the world
1362       if (Engine.getDefaultContext() != null)
1363       {
1364          Engine.getDefaultContext().update(null, new Date().getTime());
1365 
1366          if (this.showMetricsWindow && !this.metricDisplay) {
1367             this.metricDisplay = jQuery("<div/>").addClass("metrics");
1368             this.metricDisplay.appendTo($("body"));
1369          }

1371             this.metricDisplay.remove();
1372             this.metricDisplay = null;
1373          }
1374 
1375          if (this.showMetricsWindow && this.lastMetricSample-- == 0)
1376          {
1377             var d = new Date().getTime() - b;
1378             Engine.addMetric("FPS", Math.floor((1 / this.fpsClock) * 1000), false, "#");
1379             Engine.addMetric("aFPS", Math.floor((1 / d) * 1000), true, "#");
1380             Engine.addMetric("avail", this.fpsClock, false, "#ms");
1381             Engine.addMetric("frame", d, true, "#ms");
1382             Engine.addMetric("load", Math.floor((d / this.fpsClock) * 100), true, "#%");
1383 
1384             this.updateMetrics();
1385             this.lastMetricSample = this.metricSampleRate;
1386          }
1387       }
1388 
1389       // Another process interval
1390       Engine.globalTimer = setTimeout(function _engineTimer() { Engine.engineTimer(); }, this.fpsClock);
1391    },

1430    /**
1431     * Add a metric to the game engine that can be displayed
1432     * while it is running.  If smoothing is selected, a 3 point
1433     * running average will be used to smooth out jitters in the
1434     * value that is shown.  For the <tt>value</tt> argument,
1435     * you can provide a string which contains the pound sign "#"
1436     * that will be used to determine where the calculated value will
1437     * occur in the formatted string.
1438     *
1439     * @param metricName {String} The name of the metric to track
1440     * @param value {String/Number} The value of the metric.
1441     * @param smoothing {Boolean} <tt>true</tt> to use 3 point average smoothing
1442     * @memberOf Engine
1443     */
1444    addMetric: function(metricName, value, smoothing, fmt) {
1445       if (smoothing) {
1446          var vals = this.metrics[metricName] ? this.metrics[metricName].values : [];
1447          if (vals.length == 0) {
1448             // Init
1449             vals.push(value);
1450             vals.push(value);
1451             vals.push(value);
1452          }
1453          vals.shift();
1454          vals.push(value);
1455          var v = Math.floor((vals[0] + vals[1] + vals[2]) * 0.33);
1456          this.metrics[metricName] = { val: fmt.replace("#", v), values: vals };
1457       } else {
1458          this.metrics[metricName] = { val: fmt.replace("#", value) };
1459       }
1460    },
