 52    /**
 53     * Create an instance of a 2D rendering context using the canvas element.
 54     *
 55     * @param contextName {String} The name of this context.  Default: CanvasContext
 56     * @param width {Number} The width (in pixels) of the canvas context.
 57     * @param height {Number} The height (in pixels) of the canvas context.
 58     * @constructor
 59     */
 60    constructor: function(width, height) {
 61       Assert((width != null && height != null), "Width and height must be specified in CanvasContext");
 62 
 63       this.setWidth(width);
 64       this.setHeight(height);
 65 
 66       // Create the canvas element
 67       var canvas = document.createElement("div");
 68       var worldScale = this.getWorldScale();
 69       $(canvas)
 70       	.css({width: this.width * worldScale[0], height: this.height * worldScale[1]})
 71       	.attr("id", this.getId());
 72 
 73       this.jg = new jsGraphics(canvas);
 74       this.currentTransform = { translate: [0, 0],
 75                            	  rotate: 0,
 76                            	  scale: [1.0, 1.0] };
 77 
 78       this.transformStack = [];
 79 
 80       this.base("WZGraphicsContext", canvas);
 81    },
 82 
 83    setWorldScale: function(scaleX, scaleY) {
 84       this.base(scaleX, scaleY);
 85 
 86       scaleY = scaleY ? scaleY : scaleX;
 87       // Adjust the element accordingly
 88       $(this.getElement())
 89          .attr("width", this.getWidth() * scaleX)
 90          .attr("height", this.getHeight() * scaleY);
 91    },
 92 
 93    /**
 94     * Gets the surface context upon which all objects are drawn.
 95     */
 96    get2DContext: function() {
 97       return this.jg;
 98    },

100    /**
101     * Push a transform state onto the stack.
102 
103     */
104    pushTransform: function() {
105       this.base();
106       this.transformStack.push(this.currentTransform);
107    },

109    /**
110     * Pop a transform state off the stack.
111 
112     */
113    popTransform: function() {
114       this.base();
115       this.currentTransform = this.transformStack.pop();
116    },

227       this.base(rect);

230    /**
231     * Draw a filled arc on the context.  Arcs are drawn in clockwise
232     * order.
233     *
234     * @param point {Point2D} The point around which the arc will be drawn
235     * @param radius {Number} The radius of the arc in pixels
236     * @param startAngle {Number} The starting angle of the arc in degrees
237     * @param endAngle {Number} The end angle of the arc in degrees
238     */
239    drawFilledArc: function(point, radiusX, startAngle, endAngle) {
240 		var topLeft = new Point2D(point.x - (radiusX * 0.5), point.y - (radiusX * 0.5));
241       this.get2DContext().setColor(this.getFillStyle());
242 		this.get2DContext().fillArc(topLeft.x, topLeft.y, radiusX * 2, radiusX * 2, startAngle, endAngle);
243       this.get2DContext().setColor(this.getLineStyle());
244       this.base(point, radiusX, startAngle, endAngle);
245    },

288 	splitArray: function(pointArray) {
289 		// Split points into two arrays
290 		var xA = [];
291 		var yA = [];
292 		if (pointArray && pointArray.length) {
293 			for (var x=0; x<pointArray.length; x++) {
294 				if (pointArray[x]) {
295 					xA.push(pointArray[x].x);
296 					yA.push(pointArray[x].y);
297 				}
298 			}
299 		}
300 		return [xA, yA];
301 	},

328    drawFilledPolygon: function(pointArray) {
