  4 /**
  5  * XML Parser object.  Returns an {@link #XMLDOC.Parser.node} which is
  6  * the root element of the parsed document.
  7  * <p/>
  8  * By default, this parser will only handle well formed XML.  To
  9  * allow the parser to handle HTML, set the <tt>XMLDOC.Parser.strictMode</tt>
 10  * variable to <tt>false</tt> before calling <tt>XMLDOC.Parser.parse()</tt>.
 11  * <p/>
 12  * <i>Note: If you pass poorly formed XML, it will cause the parser to throw
 13  * an exception.</i>
 14  *
 15  * @author Brett Fattori (brettf@renderengine.com)
 16  * @author $Author$
 17  * @version $Revision$
 18  */
 19 XMLDOC.Parser = {};
 20 
 21 /**
 22  * Strict mode setting.  Setting this to false allows HTML-style source to
 23  * be parsed.  Normally, well formed XML has defined end tags, or empty tags
 24  * are properly formed.  Default: <tt>true</tt>
 25  * @type Boolean
 26  */
 27 XMLDOC.Parser.strictMode = true;

 29 /**
 30  * A node in an XML Document.  Node types are ROOT, ELEMENT, COMMENT, PI, and TEXT.
 31  * @param parent {XMLDOC.Parser.node} The parent node
 32  * @param name {String} The node name
 33  * @param type {String} One of the types
 34  */
 35 XMLDOC.Parser.node = function(parent, name, type)
 36 {
 37    this.name = name;
 38    this.type = type || "ELEMENT";
 39    this.parent = parent;
 40    this.charData = "";
 41    this.attrs = {};
 42    this.nodes = [];
 43    this.cPtr = 0;
 44 
 45    XMLDOC.Parser.node.prototype.getAttributeNames = function() {
 46       var a = [];
 47       for (var o in this.attrs)
 48       {
 49          a.push(o);
 50       }
 51 
 52       return a;
 53    };
 54 
 55    XMLDOC.Parser.node.prototype.getAttribute = function(attr) {
 56       return this.attrs[attr];
 57    };
 58 
 59    XMLDOC.Parser.node.prototype.setAttribute = function(attr, val) {
 60       this.attrs[attr] = val;
 61    };
 62 
 63    XMLDOC.Parser.node.prototype.parentNode = function() {
 64       return this.parent;
 65    };
 66 
 67    XMLDOC.Parser.node.prototype.node = function() {
 68       return this.nodes[this.cPtr];
 69    };
 70 
 71    XMLDOC.Parser.node.prototype.firstChild = function() {
 72       this.cPtr = 0;
 73       return this.node();
 74    };
 75 
 76    XMLDOC.Parser.node.prototype.lastChild = function() {
 77       this.cPtr = this.nodes.length - 1;
 78       return this.node();
 79    };
 80 
 81    XMLDOC.Parser.node.prototype.nextChild = function() {
 82       this.cPtr++;
 83       if (this.cPtr == this.nodes.length)
 84       {
 85          this.cPtr = this.nodes.length - 1;
 86          return null;
 87       }
 88       return this.node();
 89    };
 90 
 91    XMLDOC.Parser.node.prototype.prevChild = function() {
 92       this.cPtr--;
 93       if (this.cPtr == -1)
 94       {
 95          this.cPtr = 0;
 96          return null;
 97       }
 98       return this.node();
 99    };
100 };

102 /**
103  * Parse an XML Document from the specified source.  The XML should be
104  * well formed, unless strict mode is disabled, then the parser will
105  * handle HTML-style XML documents.
106  * @param src {String} The source to parse
107  */
108 XMLDOC.Parser.parse = function(src)
109 {
110    var A = [];
111 
112    // Normailize whitespace
113    A = src.split("\r\n");
114    src = A.join("\n");
115    A = src.split("\r");
116    src = A.join("\n");
117 
118    // Remove XML and DOCTYPE specifier
119    src.replace(/<\?XML .*\?>/i, "");
120    src.replace(/<!DOCTYPE .*\>/i, "");
121 
122    // The document is the root node and cannot be modified or removed
123    var doc = new XMLDOC.Parser.node(null, "ROOT", "DOCUMENT");
124 
125    // Let's break it down
126    XMLDOC.Parser.eat(doc, src);
127 
128    return doc;
129 };

131 /**
132  * The XML fragment processing routine.  This method is private and should not be called
133  * directly.
134  * @param parentNode {XMLDOC.Parser.node} The node which is the parent of this fragment
135  * @param src {String} The source within the fragment to process
136  * @private
137  */
138 XMLDOC.Parser.eat = function(parentNode, src)
139 {
140    // A simple tag def
141    var reTag = new RegExp("<(!|)(\\?|--|)((.|\\s)*?)\\2>","g");
142 
143    // Special tag types
144    var reCommentTag = /<!--((.|\s)*?)-->/;
145    var rePITag = /<\?((.|\s)*?)\?>/;
146 
147    // A start tag (with potential empty marker)
148    var reStartTag = /<(.*?)( +([\w_\-]*)=(\"|')(.*)\4)*(\/)?>/;
149 
150    // An empty HTML style tag (not proper XML, but we'll accept it so we can process HTML)
151    var reHTMLEmptyTag = /<(.*?)( +([\w_\-]*)=(\"|')(.*)\4)*>/;
152 
153    // Fully enclosing tag with nested tags
154    var reEnclosingTag = /<(.*?)( +([\w_\-]*)=(\"|')(.*?)\4)*>((.|\s)*?)<\/\1>/;
155 
156    // Breaks down attributes
157    var reAttributes = new RegExp(" +([\\w_\\-]*)=(\"|')(.*?)\\2","g");
158 
159    // Find us a tag
160    var tag;
161    while ((tag = reTag.exec(src)) != null)
162    {
163       if (tag.index > 0)
164       {
165          // The next tag has some text before it
166          var text = src.substring(0, tag.index).replace(/^[ \t\n]+((.|\n)*?)[ \t\n]+$/, "$1");
167 
168          if (text.length > 0 && (text != "\n"))
169          {
170             var txtnode = new XMLDOC.Parser.node(parentNode, "", "TEXT");
171             txtnode.charData = text;
172 
173             // Append the new text node
174             parentNode.nodes.push(txtnode);
175          }
176 
177          // Reset the lastIndex of reTag
178          reTag.lastIndex -= src.substring(0, tag.index).length;
179 
180          // Eat the text
181          src = src.substring(tag.index);
182       }
183 
184       if (reCommentTag.test(tag[0]))
185       {
186          // Is this a comment?
187          var comment = new XMLDOC.Parser.node(parentNode, "", "COMMENT");
188          comment.charData = reCommentTag.exec(tag[0])[1];
189 
190          // Append the comment
191          parentNode.nodes.push(comment);
192 
193          // Move the lastIndex of reTag
194          reTag.lastIndex -= tag[0].length;
195 
196          // Eat the tag
197          src = src.replace(reCommentTag, "");
198       }
199       else if (rePITag.test(tag[0]))
200       {
201          // Is this a processing instruction?
202          var pi = new XMLDOC.Parser.node(parentNode, "", "PI");
203          pi.charData = rePITag.exec(tag[0])[1];
204 
205          // Append the processing instruction
206          parentNode.nodes.push(pi);
207 
208          // Move the lastIndex of reTag
209          reTag.lastIndex -= tag[0].length;
210 
211          // Eat the tag
212          src = src.replace(rePITag, "");
213       }
214       else if (reStartTag.test(tag[0]))
215       {
216          // Break it down
217          var e = reStartTag.exec(tag[0]);
218          var elem = new XMLDOC.Parser.node(parentNode, e[1], "ELEMENT");
219 
220          // Get attributes from the tag
221          var a;
222          while ((a = reAttributes.exec(e[2])) != null )
223          {
224             elem.attrs[a[1]] = a[3];
225          }
226 
227          // Is this an empty XML-style tag?
228          if (e[6] == "/")
229          {
230             // Append the empty element
231             parentNode.nodes.push(elem);
232 
233             // Move the lastIndex of reTag (include the start tag length)
234             reTag.lastIndex -= e[0].length;
235 
236             // Eat the tag
237             src = src.replace(reStartTag, "");
238          }
239          else
240          {
241             // Check for malformed XML tags
242             var htmlParsed = false;
243             var htmlStartTag = reHTMLEmptyTag.exec(src);
244 
245             // See if there isn't an end tag within this block
246             var reHTMLEndTag = new RegExp("</" + htmlStartTag[1] + ">");
247             var htmlEndTag = reHTMLEndTag.exec(src);
248 
249             if (XMLDOC.Parser.strictMode && htmlEndTag == null)
250             {
251                // Poorly formed XML fails in strict mode
252                var err = new Error("Malformed XML passed to XMLDOC.Parser... Error contains malformed 'src'");
253                err.src = src;
254                throw err;
255             }
256             else if (htmlEndTag == null)
257             {
258                // This is an HTML-style empty tag, store the element for it in non-strict mode
259                parentNode.nodes.push(elem);
260 
261                // Eat the tag
262                src = src.replace(reHTMLEmptyTag, "");
263                htmlParsed = true;
264             }
265 
266             // If we didn't parse HTML-style, it must be an enclosing tag
267             if (!htmlParsed)
268             {
269                var enc = reEnclosingTag.exec(src);
270 
271                // Go deeper into the document
272                XMLDOC.Parser.eat(elem, enc[6]);
273 
274                // Append the new element node
275                parentNode.nodes.push(elem);
276 
277                // Eat the tag
278                src = src.replace(reEnclosingTag, "");
279             }
280          }
281 
282          // Reset the lastIndex of reTag
283          reTag.lastIndex = 0;
284       }
285    }
286 
287    // No tag was found... append the text if there is any
288    src = src.replace(/^[ \t\n]+((.|\n)*?)[ \t\n]+$/, "$1");
289    if (src.length > 0 && (src != "\n"))
290    {
291       var txtNode = new XMLDOC.Parser.node(parentNode, "", "TEXT");
292       txtNode.charData = src;
293 
294       // Append the new text node
295       parentNode.nodes.push(txtNode);
296    }
297 };
