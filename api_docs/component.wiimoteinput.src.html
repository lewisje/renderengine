 31 /**
 32  * @class A component which responds to the Wiimote (in the Opera browser)
 33  * is notifies its host object by calling a number of methods.  The host object
 34  * should implement any of the following methods to receive the corresponding events.
 35  * <br/>
 36  * The next few events are fired on the host object, if they exist, when
 37  * the corresponding button is pressed and released.  All methods take two
 38  * arguments: the controller number and a boolean indicating <tt>true</tt>
 39  * if the button has been pressed and <tt>false</tt> when released.
 40  * <br/>
 41  * <ul>
 42  * <li>onWiimoteLeft - Direction pad left</li>
 43  * <li>onWiimoteLeft - Direction pad right</li>
 44  * <li>onWiimoteUp - Direction pad up</li>
 45  * <li>onWiimoteDown - Direction pad down</li>
 46  * <li>onWiimotePlus - Plus button pressed/released</li>
 47  * <li>onWiimoteMinus - Minus button pressed/released</li>
 48  * <li>onWiimoteButton1 - Button 1 pressed/released</li>
 49  * <li>onWiimoteButton2 - Button 2 pressed/released</li>
 50  * <li>onWiimoteButtonA - Button A pressed/released</li>
 51  * <li>onWiimoteButtonB - Button B pressed/released</li>
 52  * <li>onWiimoteButtonC - Button C pressed/released</li>
 53  * <li>onWiimoteButtonZ - Button Z pressed/released</li>
 54  * </ul>
 55  * <br/><br/>
 56  * The following events are status events and take different
 57  * arguments:
 58  * <ul>
 59  * <li>onWiimoteEnabled - Enabled/disabled status (controller, state)</li>
 60  * <li>onWiimoteValidity - Validity of data transfer (controller, validity)</li>
 61  * <li>onWiimoteDistance - Distance from screen in meters (controller, dist)</li>
 62  * <li>onWiimotePosition - X/Y position (controller, x, y)</li>
 63  * <li>onWiimoteRoll - X-axis roll in radians (controller, roll)</li>
 64  * </ul>
 65  *
 66  * @extends InputComponent
 67  */

230    /**
231     * This will do the polling of the Wiimote and fire events when
232     * statuses change.
233     *
234     * @private
235     */
236    execute: function(renderContext, time) {
237       if (!(window.opera && opera.wiiremote)) {
238          // If this isn't Opera for Wii, don't do anything
239          return;
240       }
241 
242       // Run through the available Wiimotes
243       for (var w = 0; w < 4; w++) {
244          var remote = opera.wiiremote.update(w);
245          // Cannot perform this check on the primary remote,
246          // that's why this object extends the keyboard input component...
247          if (remote.isEnabled) {
248 
249             if (!this.enabledRemotes[w]) {
250                // Let the host know that a Wiimote became enabled
251                this._wmEnabled(w, true);
252             }
253 
254             if (!remote.isBrowsing) {
255                var chk;
256 
257                // Simple bitmask check to handle states and fire methods
258                this._wmLeft(w, remote.hold & 1);
259                this._wmRight(w, remote.hold & 2);
260                this._wmDown(w, remote.hold & 4);
261                this._wmUp(w, remote.hold & 8);
262                this._wmPlus(w, remote.hold & 16);
263                this._wmButton2(w, remote.hold & 256);
264                this._wmButton1(w, remote.hold & 512);
265                this._wmButtonB(w, remote.hold & 1024);
266                this._wmButtonA(w, remote.hold & 2048);
267                this._wmMinus(w, remote.hold & 4096);
268                this._wmButtonZ(w, remote.hold & 8192);
269                this._wmButtonC(w, remote.hold & 16384);
270             }
271 
272             // Set distance and validity
273             this._wmDistance(w, remote.dpdDistance);
274             this._wmValidity(w, remote.dpdValidity);
275 
276             // Get position and roll
277             this._wmPosition(w, remote.dpdScreenX, remote.dpdScreenY);
278             this._wmRoll(w, Math.atan2(remote.dpdRollY, remote.dpdRollX));
279          } else {
280             if (this.enabledRemotes[w]) {
281                // Let the host know that a Wiimote became disabled
282                this._wmEnabled(w, false);
283             }
284          }
285       }
286    },
