 32 /**
 33  * @class All components extend from this object class.  A component is one
 34  *        part of an operating whole object (a {@link HostObject}) which is
 35  *        responsible for only a portion of the overall operation of an in-
 36  *        game object.  Components are broken down into four major categories:
 37  *        <ul>
 38  *          <li><b>TYPE_INPUT</b> - Input from controllers (keyboard, mouse, etc.)</li>
 39  *          <li><b>TYPE_LOGIC</b> - Handles logical operations that are not related to
 40  *              input and collision</li>
 41  *          <li><b>TYPE_COLLIDER</b> - Determines what this object is possibly colliding
 42  *              with, and reports those collisions via callbacks to the host.</li>
 43  *          <li><b>TYPE_RENDER</b> - Performs some sort of rendering operation to the context</li>
 44  *        </ul>
 45  *        Components are executed in the order listed.  First, all inputs are
 46  *        checked, then logic is performed.  Logic may be internal to a host object
 47  *        itself, but some components perform an object-centric type of logic that
 48  *        can be reused.  Next, collisions are checked.  And finally, rendering can
 49  *        occur.
 50  *        <p/>
 51  *        Within each component type set, components can be prioritized so that
 52  *        one component will execute before others.  Such an ordering allows for
 53  *        multiple components of each type to perform their tasks in an order
 54  *        that the host defines.
 55  *
 56  *
 57  * @extends BaseObject
 58  */
 59 var BaseComponent = BaseObject.extend(/** @scope BaseComponent.prototype */{
 60 
 61    priority: 0,
 62 
 63    type: -1,
 64 
 65    host: null,
 66 
 67    /**
 68     * Create a new instance of a component, setting the name, type, and
 69     * update priority of this component compared to all other components
 70     * within the host.
 71     *
 72     * @param name {String} The name of the component
 73     * @param type {Number} The type of the component
 74     * @param priority {Number} A value between 0.0 and 1.0.  Default: 0.5
 75     */
 76    constructor: function(name, type, priority) {
 77       Assert((name != null), "You must assign a name to every Component.");
 78       name = name.toUpperCase();
 79 
 80       Assert((type != null && (type >= BaseComponent.TYPE_INPUT && type <= BaseComponent.TYPE_RENDERING)),
 81              "You must specify a type for BaseComponent");
 82 
 83       this.type = type;
 84 
 85       Assert((priority != null && (priority >= 0.0 && priority <= 1.0)),
 86              "Priority must be between 0.0 and 1.0 for BaseComponent");
 87 
 88       this.priority = priority || 0.5;
 89       this.base(name);
 90    },
 91 
 92    release: function() {
 93       this.base();
 94       this.priority = 0;
 95       this.type = -1;
 96       this.host = null;
 97    },
 98 
 99    /**
100     * Set the host object this component exists within.
101     *
102     * @param hostObject {HostObject} The object which hosts this component
103     */
104    setHostObject: function(hostObject) {
105       this.host = hostObject;
106    },
107 
108    /**
109     * Get the host object this component exists within.
110     *
111     * @type HostObject
112     */
113    getHostObject: function() {
114       return this.host;
115    },
116 
117    /**
118     * Get the type of this component.  Will be one of:
119     * {@link #TYPE_INPUT}, {@link #TYPE_LOGIC}, {@link #TYPE_COLLIDER}, {@link #TYPE_RENDERING}
120     *
121     * @type Number
122     */
123    getType: function() {
124       return this.type;
125    },
126 
127    /**
128     * Set the execution priority of this component with
129     * 1.0 being the highest priority and 0.0 being the lowest.  Components
130     * within a host object are sorted by type, and then priority.
131     *
132     * @param priority {Number} A value between 0.0 and 1.0
133     */
134    setPriority: function(priority) {
135       this.priority = priority;
136       this.getHost().sort();
137    },
138 
139    /**
140     * Returns the priority of this component.
141     *
142     * @type Number
143     */
144    getPriority: function() {
145       return this.priority;
146    },
147 
148    /**
149     * Run the component, updating its state.  Not all components will need an execute
150     * method.  However, it is important to include one if you need to run calculations
151     * or update the state of the component each engine cycle.
152     *
153     * @param renderContext {RenderContext} The context the component will render within.
154     * @param time {Number} The global engine time
155     */
156    execute: function(renderContext, time) {
157       // Does nothing...
158    },
159 
160    /**
161     * Converts the type name to a string.
162     *
163     * @type String
164     */
165    getTypeString: function() {
166       var ts = "";
167       switch (this.getType()) {
168          case BaseComponent.TYPE_INPUT: ts = "TYPE_INPUT"; break;
169          case BaseComponent.TYPE_LOGIC: ts = "TYPE_LOGIC"; break;
170          case BaseComponent.TYPE_COLLIDER: ts = "TYPE_COLLIDER"; break;
171          case BaseComponent.TYPE_RENDERING: ts = "TYPE_RENDERING"; break;
172          default: ts = "TYPE_UNKNOWN";
173       }
174 
175       return ts;
176    }
177 
178 
179 }, /** @scope BaseComponent.prototype */{
180 
181    /**
182     * Get the class name of this object
183     *
184     * @type String
185     */
186    getClassName: function() {
187       return "BaseComponent";
188    },
189 
190    /**
191     * An input component
192     * @type Number
193     */
194    TYPE_INPUT:          1,
195 
196    /**
197     * A logic component
198     * @type Number
199     */
200    TYPE_LOGIC:          2,
201 
202    /**
203     * A collider component
204     * @type Number
205     */
206    TYPE_COLLIDER:       3,
207 
208    /**
209     * A rendering component
210     * @type Number
211     */
212    TYPE_RENDERING:      4
213 });
