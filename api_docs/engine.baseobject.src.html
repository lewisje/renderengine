 33 /**
 34  * @class The base object class which represents an object within
 35  * the engine.  All objects should extend from this class mainly due to
 36  * the fact that the object intelligently will inherit a <tt>destroy</tt>
 37  * method.  This method gives the object an opportunity to clean up
 38  * object references, thus keeping the memory requirements low.
 39  *
 40  */
 41 var BaseObject = Base.extend(/** @scope BaseObject.prototype */{
 42    id: -1,
 43 
 44    name: "",
 45 
 46    element: null,
 47 
 48    /**
 49     * Returns <tt>true</tt> if the object is an object within the
 50     * Render Engine.  All objects should extend this class.
 51     * @memberOf BaseObject
 52     */
 53    isRenderEngineObject: function() {
 54       return true;
 55    },
 56 
 57    /**
 58     * Create an instance of this object, assigning a name to it.  An
 59     * object reference will be maintained by the {@link Engine} class,
 60     * which gives the object final responsibility for making sure the
 61     * object can be destroyed.
 62     *
 63     * @param name {String} The name of the object from which the Id will be generated.
 64     * @memberOf BaseObject
 65     * @constructor
 66     */
 67    constructor: function(name) {
 68       this.name = name;
 69       this.id = Engine.create(this);
 70    },
 71 
 72    /**
 73     * Destroy this object instance (remove it from the Engine).  The object
 74     * may still exist as a reference from another object, but will not
 75     * be managed by the Engine any longer.  Untracked objects pose the
 76     * potential for memory leaks.
 77     * @memberOf BaseObject
 78     */
 79    destroy: function() {
 80       // Remove any defined element from it's parent, unless it's the document body
 81       if (this.element && this.element.parentNode && this.element != document.body) {
 82          Console.log("DOM element ", this.element, " removed from ", this.element.parentNode);
 83          this.element.parentNode.removeChild(this.element);
 84       }
 85 
 86       // Clean up the reference to this object
 87       Engine.destroy(this);
 88       this.element = null;
 89    },
 90 
 91    /**
 92     * Get the managed Id of this object within the Engine.
 93     *
 94     * @return This object's engine Id
 95     * @type String
 96     * @memberOf BaseObject
 97     * @memberOf BaseObject
 98     */
 99    getId: function() {
100       return this.id;
101    },
102 
103    /**
104     * Get the original name this object was created with.
105     *
106     * @return The name used when creating this object
107     * @type String
108     * @memberOf BaseObject
109     */
110    getName: function() {
111       return this.name;
112    },
113 
114    /**
115     * Set the element which will represent this object within
116     * its rendering context.
117     *
118     * @type element {BaseObject}
119     * @memberOf BaseObject
120     */
121    setElement: function(element) {
122       this.element = element;
123    },
124 
125    /**
126     * Get the element which represents this object within its rendering context.
127     *
128     * @return The element
129     * @type BaseObject
130     * @memberOf BaseObject
131     */
132    getElement: function() {
133       return this.element;
134    },
135 
136    /**
137     * Update the state of the object.
138     */
139    update: function(renderContext, time) {
140    },
141 
142    /**
143     * Get the class name of this object
144     *
145     * @type String
146     * @memberOf BaseObject
147     */
148    getClassName: function() {
149       if (!this.hasOwnProperty("getClassName")) {
150          Console.warn("Object ", this.toString(), " is missing getClassName()");
151       }
152       return "BaseObject";
153    },
154 
155    /**
156     * Output the object in JSON format so it can be reconstructed from
157     * a text stream.
158     *
159     * @type String
160     */
161    toString: function() {
162       return EngineSupport.toJSONString(this);
163    }
164 });
