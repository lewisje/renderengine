 39 var ConsoleRef = Base.extend(/** @scope ConsoleRef.prototype */{

 44    /** @private */
 45    combiner: function() {
 46       var out = "";
 47       for (var a in arguments) {
 48          out += arguments[a].toString();
 49       }
 50       return out;
 51    },

 53    /** @private */
 54    out: function(msg) {
 55       if (this.dumpWindow == null)
 56       {
 57          this.dumpWindow = window.open("about:blank", "console", "width=640,height=480,resizeable=yes,toolbar=no,location=no,status=no");
 58          this.dumpWindow.document.body.innerHTML =
 59             "<style> " +
 60             "BODY { font-family: 'Lucida Console',Courier; font-size: 10pt; color: black; } " +
 61             ".debug { background: white; } " +
 62             ".warn { font-style: italics; background: #ffffdd; } " +
 63             ".error { color: white; background: red; font-weight: bold; } " +
 64             "</style>"
 65       }
 66 
 67       // Using jQuery to handle adding new messages
 68       $(this.dumpWindow.document.body).append(msg);
 69 
 70       // this.dumpWindow.document.body.innerHTML += this.dumpWindow.document.body.innerHTML + msg;
 71    },

 73    /** @private */
 74    dump: function(msg) {
 75       if (msg instanceof Array)
 76       {
 77          var a = "[";
 78          for (var o in msg) {
 79             a += (a.length > 1 ? ", " : "") + this.dump(msg[o]);
 80          }
 81          return a + "]";
 82       }
 83       else if (typeof msg === "function")
 84       {
 85          return "func[ " + this.fix(msg.toString().substring(0,30)) + "... ]";
 86       }
 87       else if (typeof msg === "object")
 88       {
 89          var a = "Object {\n";
 90          for (var o in msg)
 91          {
 92             a += o + ": " + this.dump(msg[o]) + "\n";
 93          }
 94          return a + "\n}";
 95       }
 96       else
 97       {
 98          return this.fix(msg);
 99       }
100    },
101 
102    /** @private */
103    fix: function(msg) {
104       return msg.replace(/\\n/g, "<br>").replace(/</g,"<").replace(/>/g,">");
105    },

287 /**
288  * Halts the engine if the test fails, throwing the error as a result.
289  *
290  * @param test {Boolean} A simple test that should evaluate to <tt>true</tt>
291  * @param error {String} The error message to throw if the test fails
292  */
293 var Assert = function(test, error) {
294    if (!test)
295    {
296       Engine.shutdown();
297       if (this.caller) {
298          var funcName = "";
299          if (this.caller.name)
300          {
301             funcName = this.caller.name;
302          }
303          else
304          {
305             // Try to determine the function name that called this Assert
306             var fRE = /((function\s+([\$_\.\w]+))|(var\s+([\$_\.\w]+)\s+=\s+function)|(function\(.*?\)))/;
307             var m = fRE.exec(this.caller.toString());
308             if (m)
309             {
310                funcName = (m[3] || m[5] || "anonymous");
311             }
312          }
313          error += "\nin function " + funcName;
314       }
315       throw new Error(error);
316    }
317 };

324  */
325 var AssertWarn = function(test, warning) {
326    if (!test)
327    {
328       Console.warn(warning);
329    }

332 /**
333  * @class Engine support class.  Provides extra functions the engine or games
334  *        can use.  Mainly contains functions that can manipulate arrays and
335  *        generate/read JSON.
336  */
337 var EngineSupport = Base.extend(/** @scope EngineSupport.prototype */{
338    constructor: null,
339 
340    /**
341     * Get the index of an element in the specified array.
342     *
343     * @param array {Array} The array to scan
344     * @param obj {Object} The object to find
345     * @param from {Number=0} The index to start at, defaults to zero.
346     * @memberOf EngineSupport
347     */
348    indexOf: function(array, obj, from) {
349       if (Array.prototype.indexOf) {
350          return array.indexOf(obj, from);
351       }
352       else
353       {
354          var len = array.length;
355          var from = Number(from) || 0;
356          from = (from < 0)
357             ? Math.ceil(from)
358             : Math.floor(from);
359          if (from < 0)
360             from += len;
361 
362          for (; from < len; from++)
363          {
364             if (from in array && array[from] === obj)
365                return from;
366          }
367          return -1;
368       }
369    },
370 
371    /**
372     * Remove an element from an array.
373     *
374     * @param array {Array} The array to modify
375     * @param obj {Object} The object to remove
376     * @memberOf EngineSupport
377     */
378    arrayRemove: function(array, obj) {
379       var idx = EngineSupport.indexOf(array, obj);
380       if (idx != -1)
381       {
382          array.splice(idx, 1);
383       }
384    },
385 
386    /**
387     * Calls a provided callback function once for each element in
388     * an array, and constructs a new array of all the values for which
389     * callback returns a <tt>true</tt> value. callback is invoked only
390     * for indexes of the array which have assigned values; it is not invoked
391     * for indexes which have been deleted or which have never been assigned
392     * values. Array elements which do not pass the callback test are simply
393     * skipped, and are not included in the new array.
394     *
395     * @param array {Array} The array to filter.
396     * @param fn {Function} The callback to invoke.  It will be passed three
397     *                      arguments: The element value, the index of the element,
398     *                      and the array being traversed.
399     * @param thisp {Object} Used as <tt>this</tt> for each invocation of the
400     *                       callback.  Default: <tt>null</tt>
401     * @memberOf EngineSupport
402     */
403    filter: function(array, fn, thisp) {
404       if (Array.prototype.filter) {
405          return array.filter(fn, thisp)
406       }
407       else
408       {
409          var len = array.length;
410          if (typeof fn != "function")
411             throw new TypeError();
412 
413          var res = new Array();
414          for (var i = 0; i < len; i++)
415          {
416             if (i in array)
417             {
418                var val = array[i]; // in case fn mutates this
419                if (fn.call(thisp, val, i, array))
420                   res.push(val);
421             }
422          }
423 
424          return res;
425       }
426    },
427 
428    /**
429     * Executes a callback for each element within an array.
430     *
431     * @param array {Array} The array to operate on
432     * @param fn {Function} The function to apply to each element
433     * @param [thisp] {Object} An optional "this" pointer to use in the callback
434     * @memberOf EngineSupport
435     */
436    forEach: function(array, fn, thisp) {
437       if (Array.prototype.forEach) {
438          array.forEach(fn, thisp);
439       }
440       else
441       {
442          var len = array.length;
443          if (typeof fn != "function")
444             throw new TypeError();
445 
446          for (var i = 0; i < len; i++)
447          {
448             if (i in array)
449                fn.call(thisp, array[i], i, array);
450          }
451       }
452    },
453 
454    /**
455     * Get the path from a fully qualified URL.
456     *
457     * @param url {String} The URL
458     * @type String
459     * @memberOf EngineSupport
460     */
461    getPath: function(url) {
462       var l = url.lastIndexOf("/");
463       return url.substr(0, l);
464    },
465 
466    /**
467     * Get the query parameters from the window location object.  The
468     * object returned will contain a key/value pair for each argument
469     * found.
470     *
471     * @type Object
472     * @return An <tt>Object</tt> with a key and value for each query argument.
473     * @memberOf EngineSupport
474     */
475    getQueryParams: function() {
476       var parms = {};
477       var p = window.location.toString().split("?")[1];
478       if (p)
479       {
480          p = p.split("&");
481          for (var x = 0; x < p.length; x++)
482          {
483             var v = p[x].split("=");
484             parms[v[0]] = (v.length > 1 ? v[1] : "");
485          }
486       }
487       return parms;
488    },
489 
490    /**
491     * Returns specified object as a JavaScript Object Notation (JSON) string.
492     *
493     * Code to handle "undefined" type was delibrately not implemented, being that it is not part of JSON.
494     * "undefined" type is casted to "null".
495     *
496     * @param object {Object} Must not be undefined or contain undefined types and variables.
497     * @return String
498     * @memberOf EngineSupport
499     */
500    toJSONString: function(o)
501    {
502       if(o == null)
503       {
504          return "null";
505       }
506 
507       switch(o.constructor)
508       {
509          case Array:
510             var a = [], i;
511             for(i = 0; i < o.length; i++)
512             {
513                a[i] = EngineSupport.toJSONString(o[i]);
514             }
515             return "[" + a.join() + "]";
516          case String:
517             return EngineSupport.quoteString(o);
518          case Boolean:
519          case Number:
520             return o.toString();
521          default:
522             var a = [], i;
523             for(i in o)
524             {
525                if(o[i] == null)
526                {
527                   a.push(i.quote() + ":null");
528                }
529                else if(o[i].constructor != Function)
530                {
531                   a.push(i.quote() + ':' + EngineSupport.toJSONString(o[i]));
532                }
533             }
534             return '{' + a.join() + '}';
535       }
536    },
537 
538    /**
539     * Parses specified JavaScript Object Notation (JSON) string back into its corresponding object.
540     *
541     * @param jsonString
542     * @return Object
543     * @see http://www.json.org
544     * @memberOf EngineSupport
545     */
546    parseJSONString: function(jsonString)
547    {
548       var obj = null;
549       eval.call(arguments.callee, 'obj = (function(){return ' + jsonString + ';})();');
550       return obj;
551    },
552 
553    /**
554     * Return a string, enclosed in quotes.
555     *
556     * @param string
557     * @type String
558     * @memberOf EngineSupport
559     */
560    quoteString: function(o)
561    {
562       return '"'+this.replace(/[\\"\r\n]/g, function(s)
563          {
564             switch(s)
565             {
566                case "\\":return "\\\\";
567                case "\r":return "\\r";
568                case "\n":return "\\n";
569                case '"':return '\\"';
570             }
571          }
572       )+'"';
573    },
574 });

848     * Load a script from the server and append it to
849     * the head element of the browser.  Script names are
850     * cached so they will not be loaded again.
851     *
852     * @param scriptPath {String} The URL of a script to load.
853     * @memberOf Engine
854     */
855    loadScript: function(scriptPath) {
856       var s = scriptPath.replace(/[\/\.]/g,"_");
857       if (this.loadedScripts[s] == null)
858       {
859          // Store the request in the cache
860          this.loadedScripts[s] = scriptPath;
861 
862          if (!Engine.scriptQueue) {
863             // Create the queue
864             Engine.scriptQueue = [];
865          }
866 
867          // Put script into load queue
868          Engine.scriptQueue.push(scriptPath);
869 
870          if (!Engine.scriptQueueTimer) {
871             // Process any waiting scripts
872             Engine.scriptQueueTimer = setInterval(function() {
873                if (Engine.queuePaused) {
874                   if (Engine.pauseReps++ > 500) {
875                      // If after ~5 seconds the queue is still paused, unpause it and
876                      // warn the user that the situation occurred
877                      Console.error("Script queue was paused for 5 seconds and not resumed -- restarting...");
878                      Engine.pauseReps = 0;
879                      Engine.pauseQueue(false);
880                   }
881                   return;
882                }
883 
884                Engine.pauseReps = 0;
885 
886                if (Engine.scriptQueue.length > 0) {
887                   Engine.processScriptQueue();
888                } else {
889                   // Stop the queue timer if there are no scripts
890                   clearInterval(Engine.scriptQueueTimer);
891                   Engine.scriptQueueTimer = null;
892                }
893             }, 10);
894 
895             Engine.readyForNextScript = true;
896          }
897       }

939

942

945

1219    /**
1220     * Add a metric to the game engine that can be displayed
1221     * while it is running.  If smoothing is selected, a 3 point
1222     * running average will be used to smooth out jitters in the
1223     * value that is shown.  For the <tt>value</tt> argument,
1224     * you can provide a string which contains the pound sign "#"
1225     * that will be used to determine where the calculated value will
1226     * occur in the formatted string.
1227     *
1228     * @param metricName {String} The name of the metric to track
1229     * @param value {String/Number} The value of the metric.
1230     * @param smoothing {Boolean} <tt>true</tt> to use 3 point average smoothing
1231     * @memberOf Engine
1232     */
1233    addMetric: function(metricName, value, smoothing, fmt) {
1234       if (smoothing) {
1235          var vals = this.metrics[metricName] ? this.metrics[metricName].values : [];
1236          if (vals.length == 0) {
1237             // Init
1238             vals.push(value);
1239             vals.push(value);
1240             vals.push(value);
1241          }
1242          vals.shift();
1243          vals.push(value);
1244          var v = Math.floor((vals[0] + vals[1] + vals[2]) * 0.33);
1245          this.metrics[metricName] = { val: fmt.replace("#", v), values: vals };
1246       } else {
1247          this.metrics[metricName] = { val: fmt.replace("#", value) };
1248       }
1249    },
