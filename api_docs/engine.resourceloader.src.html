 31 /**
 32  * @class  A resource loader is a generalized interface used by all resource
 33  *         loaders.  It is designed to provide a common set of routines for
 34  *         loading resources (fonts, images, game data, etc...) from some
 35  *         location.
 36  *
 37  * @param name {String} The name of the resource loader. Default: ResourceLoader
 38  */
 39 var ResourceLoader = BaseObject.extend(/** @scope ResourceLoader.prototype */{
 40 
 41    cache: {},
 42 
 43    length: 0,
 44 
 45    constructor: function(name) {
 46       this.base(name || "ResourceLoader");
 47    },
 48 
 49    release: function() {
 50       this.base();
 51       this.cache = {};
 52       this.length = 0;
 53    },
 54 
 55    destroy: function() {
 56       this.clear();
 57       this.base();
 58    },
 59 
 60    /**
 61     * Load an object via this resource loader, and add it to the cache.
 62     *
 63     * @param name {String} The name to refer to the loaded object
 64     * @param data {Object} The data to store in the cache
 65     * @param isReady {Boolean} A flag that states whether or not a resource
 66     *                          is ready to use.
 67     */
 68    load: function(name, data, isReady) {
 69       var obj = { data: data, ready: isReady || false};
 70       this.cache[name] = obj;
 71       this.length++;
 72       Console.log("Loading " + this.getResourceType() + ": " + name);
 73    },
 74 
 75    setReady: function(name, isReady) {
 76       this.cache[name].isReady = isReady;
 77    },
 78 
 79    isReady: function(name) {
 80       return this.cache[name] ? this.cache[name].isReady : false;
 81    },
 82 
 83    /**
 84     * Unload an object from this resource loader.  Removes the object
 85     * from the cache.
 86     *
 87     * @param name {String} The name of the object to remove
 88     */
 89    unload: function(name) {
 90       this.cache[name] = null;
 91       delete this.cache[name];
 92       this.length--;
 93    },
 94 
 95    /**
 96     * Get the object with the specified name from the cache.
 97     *
 98     * @param name {String} The name of the object to retrieve
 99     * @type Object
100     */
101    get: function(name) {
102       if (this.cache[name]) {
103          return this.cache[name].data;
104       } else {
105          return null;
106       }
107    },
108 
109    /**
110     * Clear the objects contained in the cache.
111     */
112    clear: function() {
113       for (var o in this.cache) {
114          this.cache[o] = null;
115       }
116 
117       this.cache = {};
118       this.length = 0;
119    },
120 
121    /**
122     * The name of the resource this loader will get.
123     * @returns A String that represents the resource type.
124     */
125    getResourceType: function() {
126       return "default";
127    }
128 }, {
129    /**
130     * Get the class name of this object
131     *
132     * @type String
133     */
134    getClassName: function() {
135       return "ResourceLoader";
136    }
137 
138 });
