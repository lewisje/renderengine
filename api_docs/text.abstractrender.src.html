 35 // Includes
 36 Engine.include("/components/component.base.js");

 38 Engine.initObject("AbstractTextRenderer", "BaseComponent", function() {
 39 
 40 /**
 41  * @class Abstract class that provides the basic interface for all
 42  *        text render objects used by the {@link TextRenderer}.
 43  *
 44  * @constructor
 45  * @param componentName {String} The name of the renderer
 46  * @param priority {Number} The priority of the rendering order. Default: <tt>0.1</tt>
 47  */
 48 var AbstractTextRenderer = BaseComponent.extend(/** @scope AbstractTextRenderer.prototype */{
 49 
 50    text: null,
 51    color: "#000000",
 52    alignment: null,
 53    weight: null,
 54 	size: 1,
 55 	font: null,
 56 	style: null,
 57 
 58 	/**
 59 	 * @private
 60 	 */
 61    constructor: function(componentName, priority) {
 62       this.base(componentName || "TextRenderObject", BaseComponent.TYPE_RENDERING, priority || 0.1);
 63 
 64       this.text = "";
 65 		this.size = 1;
 66       this.weight = null;
 67 		this.font = null;
 68 		this.style = null;
 69 		this.alignment = null;
 70    },
 71 
 72 	/**
 73 	 * @private
 74 	 */
 75    release: function() {
 76       this.base();
 77       this.text = null;
 78       this.color= "#000000";
 79 		this.size = 1;
 80       this.weight = null;
 81 		this.font = null;
 82 		this.style = null;
 83 		this.alignment = null;
 84    },
 85 
 86    /**
 87     * Get the text being rendered
 88     * @return {String} The text this renderer will draw
 89     */
 90    getText: function() {
 91       return this.text;
 92    },
 93 
 94    /**
 95     * Set the text to be rendered
 96     *
 97     * @param text {String} The text to render
 98     */
 99    setText: function(text) {
100       this.text = text;
101    },
102 
103 	/**
104 	 * Set the font of the text to be renderer
105 	 * @param font {String} The font name
106 	 */
107 	setTextFont: function(font) {
108 		this.font = font;		
109 	},
110 	
111 	/**
112 	 * Get the font of the text to be rendered
113 	 * @return {String} The font name
114 	 */
115 	getTextFont: function() {
116 		return this.font;
117 	},
118 
119    /**
120     * Set the weight of the text to render.  Higher weights
121     * are bolder text.
122     *
123     * @param weight {Number} The weight of the text.
124     */
125    setTextWeight: function(weight) {
126       this.weight = weight;
127    },
128 
129    /**
130     * Get the weight of the text to render.
131     * @return {Number} The weight of the text
132     */
133    getTextWeight: function() {
134       return this.weight;
135    },
136 	
137 	/**
138 	 * Set the style of the text, usually italics or normal, for the text renderer.
139 	 * @param style {Object} The style of the text
140 	 */
141 	setTextStyle: function(style) {
142 		this.style = style;	
143 	},
144 	
145 	/**
146 	 * Get the style of the text for the renderer.
147 	 * @return {Object} The style of the text
148 	 */
149 	getTextStyle: function() {
150 		return this.style;
151 	},
152 
153    /**
154     * Set the alignment of the text.
155     *
156     * @param alignment {Object} The alignment for the text renderer
157     */
158    setTextAlignment: function(alignment) {
159       this.alignment = alignment;
160    },
161 
162    /**
163     * Get the alignment of the text.
164     * @return {Object} The alignment of the text renderer
165     */
166    getTextAlignment: function() {
167       return this.alignment;
168    },
169 	
170 	/**
171 	 * Set the scaling of the text
172 	 * @param size {Number}
173 	 */
174 	setSize: function(size) {
175 		this.size = size;
176 	},
177 	
178 	/**
179 	 * Get the scaling of the text
180 	 * @return {Number}
181 	 */
182 	getSize: function() {
183 		return this.size;
184 	},
185 
186    /**
187     * Set the color of the text to render.
188     *
189     * @param color {String} The color of the text to render
190     */
191    setColor: function(color) {
192       this.color = color;
193    },
194 
195    /**
196     * Get the color of the text to render.
197     * @return {String} The text color
198     */
199    getColor: function() {
200       return this.color;
201    }
202 
203 }, /** @scope AbstractTextRenderer.prototype */{
204 
205    /**
206     * Get the class name of this object
207     * @return {String} The string "AbstractTextRenderer"
208     */
209    getClassName: function() {
210       return "AbstractTextRenderer";
211    },
212 
213    /**
214     * Align text with the left edge of the string at the point specified.
215     * @type Number
216     */
217    ALIGN_LEFT: 0,
218 
219    /**
220     * Align text with the right edge of the string at the point specified
221     * @type Number
222     */
223    ALIGN_RIGHT: 1,
224 
225    /**
226     * Align text with the center of the string at the point specified
227     * @type Number
228     */
229    ALIGN_CENTER: 2
230 
231 });
232 
233 return AbstractTextRenderer;
234 
235 });
