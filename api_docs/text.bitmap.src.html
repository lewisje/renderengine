 32 /**
 33  * @class A text renderer which draws text from a bitmap font file.
 34  * @extends AbstractTextRenderer
 35  */
 36 var BitmapText = AbstractTextRenderer.extend(/** @scope BitmapText.prototype */{
 37 
 38    font: null,
 39 
 40    spacing: 0,
 41 
 42    constructor: function(font) {
 43       this.base();
 44       this.font = font;
 45    },
 46 
 47    release: function() {
 48       this.base();
 49       this.font = null;
 50       this.spacing = 0;
 51    },
 52 
 53    /**
 54     * Calculate the bounding box for the text and set it on the host object.
 55     * @private
 56     */
 57    calculateBoundingBox: function() {
 58       return;
 59 
 60       var x1 = 0;
 61       var x2 = 0;
 62       var y1 = 0;
 63       var y2 = 0;
 64       for (var p = 0; p < this.rText.length; p++)
 65       {
 66          var pt = this.rText[p];
 67 
 68          if (pt.x < x1)
 69          {
 70             x1 = pt.x;
 71          }
 72          if (pt.x > x2)
 73          {
 74             x2 = pt.x;
 75          }
 76          if (pt.y < y1)
 77          {
 78             y1 = pt.y;
 79          }
 80          if (pt.y > y2)
 81          {
 82             y2 = pt.y;
 83          }
 84       }
 85 
 86       this.getHostObject().setBoundingBox(new Rectangle2D(x1, y1, Math.abs(x1) + x2, Math.abs(y1) + y2));
 87    },
 88 
 89    /**
 90     * Set the text to render.
 91     *
 92     * @param text {String} The text to render
 93     */
 94    setText: function(text) {
 95       // If the font only supports uppercase letters
 96       text = (this.font.upperCaseOnly ? String(text).toUpperCase() : text);
 97 
 98       // Replace special chars
 99       text = text.replace(/©/gi,"(C)").replace(/®/gi,"(R)");
100 
101       this.base(text);
102 
103       this.calculateBoundingBox();
104    },
105 
106    /**
107     * @private
108     */
109    execute: function(renderContext, time) {
110 
111       if (this.getText().length == 0)
112       {
113          return;
114       }
115 
116       var text = this.getText();
117       var lCount = text.length;
118       var align = this.getAlignment();
119       var letter = (align == AbstractTextRenderer.ALIGN_RIGHT ? text.length - 1 : 0);
120       var kern = (align == AbstractTextRenderer.ALIGN_RIGHT ? -this.font.info.kerning : this.font.info.kerning);
121       var space = new Point2D((align == AbstractTextRenderer.ALIGN_RIGHT ? -this.font.info.space : this.font.info.space), 0);
122       var cW, cH = this.font.info.height;
123       var cS = 0;
124 
125       // Render the text
126       var pc = new Point2D(0,0);
127 
128       // 1st pass: The text
129       pc = new Point2D(0,0);
130       letter = (align == AbstractTextRenderer.ALIGN_RIGHT ? text.length - 1 : 0);
131       lCount = text.length;
132 
133       renderContext.get2DContext().globalCompositeOperation = "source-over";
134       while (lCount-- > 0)
135       {
136          var glyph = text.charCodeAt(letter) - 32;
137          if (glyph == 0)
138          {
139             // A space
140             pc.add(space);
141          }
142          else
143          {
144             // Draw the text
145             cS = this.font.info.letters[glyph - 1];
146             cW = this.font.info.letters[glyph] - cS;
147             //debugger;
148             renderContext.get2DContext().drawImage(this.font.image, cS, 0, cW, cH, pc.x, pc.y, cW, cH);
149             pc.add(new Point2D(cW, 0).mul(kern));
150          }
151 
152          letter += (align == AbstractTextRenderer.ALIGN_RIGHT ? -1 : 1);
153       }
154 
155       // 2nd pass: The color
156       renderContext.get2DContext().globalCompositeOperation = "source-atop";
157       lCount = text.length;
158       pc = new Point2D(0,0);
159       letter = (align == AbstractTextRenderer.ALIGN_RIGHT ? text.length - 1 : 0);
160       while (lCount-- > 0)
161       {
162          var glyph = text.charCodeAt(letter) - 32;
163          if (glyph == 0)
164          {
165             // A space
166             pc.add(space);
167          }
168          else
169          {
170             // Draw a box the color we want and the size of the character
171             cS = this.font.info.letters[glyph - 1];
172             cW = this.font.info.letters[glyph] - cS;
173             var r = new Rectangle2D(pc.x, pc.y, cW, cH);
174             renderContext.setFillStyle(this.getColor());
175             renderContext.drawFilledRectangle(r);
176             pc.add(new Point2D(cW, 0).mul(kern));
177          }
178 
179          letter += (align == AbstractTextRenderer.ALIGN_RIGHT ? -1 : 1);
180       }
181 
182 
183       // Reset the composition operation
184       renderContext.get2DContext().globalCompositeOperation = "source-over";
185    }
186 }, {
187    /**
188     * Get the class name of this object
189     * @type String
190     */
191    getClassName: function() {
192       return "BitmapText";
193    }
194 });
