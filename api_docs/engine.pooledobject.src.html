engine.pooledobject.js


Summary

An object that has functionality to assist in keeping memory usage down and to minimize the effect of the Javascript garbage collector.
Class Summary
PooledObject Pooled objects are created as needed, and reused from a static pool of all object types.

 35 Engine.initObject("PooledObject", null, function() {
 36 
 37 /**
 38  * @class Pooled objects are created as needed, and reused from a static pool
 39  *        of all object types.  When an object is created, if one is not available
 40  *        in the pool, a new object is created.  When the object is destroyed,
 41  *        it is returned to the pool so it can be used again.  This has the
 42  *        effect of minimizing the requirement of garbage collection, reducing
 43  *        cycles needed to clean up dead objects.
 44  *
 45  */
 46 var PooledObject = Base.extend(/** @scope PooledObject.prototype */{
 47 
 48    // The Id assigned by the engine
 49    id: -1,
 50 
 51    // The name of the object
 52    name: "",
 53 
 54 
 55    /**
 56     * Create an instance of this object, assigning a name to it.  An
 57     * object reference will be maintained by the {@link Engine} class,
 58     * which gives the object final responsibility for making sure the
 59     * object can be destroyed.
 60     *
 61     * @param name {String} The name of the object from which the Id will be generated.
 62     * @memberOf PooledObject
 63     * @constructor
 64     */
 65    constructor: function(name) {
 66       this.name = name;
 67       this.id = Engine.create(this);
 68    },
 69 
 70    /**
 71     * When a pooled object is destroyed, it will be released so it
 72     * can clean up variables before being put back into the pool.
 73     * the variables should be returned to an "uninitialized" state.
 74     * @memberOf PooledObject
 75     */
 76    release: function() {
 77       this.name = "";
 78       this.id = -1;
 79    },
 80 
 81    /**
 82     * Destroy this object instance (remove it from the Engine).  The object
 83     * is returned to the pool of objects so it can be used again.
 84     * @memberOf PooledObject
 85     */
 86    destroy: function() {
 87 
 88       PooledObject.returnToPool(this);
 89 
 90       // Clean up the engine reference to this object
 91       Engine.destroy(this);
 92 
 93       // Reset any variables on the object before putting
 94       // it back in the pool.
 95       this.release();
 96    },
 97 
 98    /**
 99     * Get the managed Id of this object within the Engine.
100     *
101     * @return This object's engine Id
102     * @type String
103     * @memberOf BaseObject
104     * @memberOf BaseObject
105     */
106    getId: function() {
107       return this.id;
108    },
109 
110    /**
111     * Get the original name this object was created with.
112     *
113     * @return The name used when creating this object
114     * @type String
115     * @memberOf BaseObject
116     */
117    getName: function() {
118       return this.name;
119    },
120 
121 	/**
122 	 * Returns an object that assigns getter and setter methods
123 	 * for exposed properties of an object.
124 	 * @param {BaseObject} self The reference to the object being
125 	 * 					 inspected.
126 	 * @return {Object} An object which contains getter and setter methods.
127 	 */
128 	getProperties: function() {
129 		var self = this;
130 		var prop = {};
131 		return $.extend(prop, {
132 	  		"Id" 				: [function() { return self.getId(); },
133 							 		null, false],
134 	  		"ObjectName" 	: [function() { return self.getName(); },
135 							 		function(i) { self.setName(i); }, true]
136 		});
137 	},
138 
139    /**
140     * Serialize the object to XML.
141     * @type String
142     */
143    toString: function(indent) {
144 		indent = indent ? indent : "";
145 		var props = this.getProperties();
146 		var xml = indent + "<" + this.constructor.getClassName();
147 		for (var p in props) {
148 			// If the value should be serialized, call it's getter
149 			if (props[p][2]) {
150 				xml += " " + p + "=\"" + props[p][0]().toString() + "\"";
151 			}
152 		}
153 
154 		xml += "/>\n";
155 		return xml;
156    }
157 
158 }, /** @scope PooledObject.prototype **/{
159 
160    /**
161     * <tt>true</tt> for all objects within the engine.
162     */
163    isRenderEngineObject: true,
164 
165    /**
166     * <tt>true</tt> for all objects that are pooled.
167     */
168    isPooledObject: true,
169 
170    poolNew: 0,
171 
172    poolSize: 0,
173 
174    /**
175     * Similar to a constructor, all pooled objects should implement
176     * a create method which is used to either create a new instance of
177     * the object, or retrieve one from the pool.
178     * <p/>
179     * Usage: <tt>var obj = [ObjectClass].create(arg1, arg2, arg3...);</tt>
180     *
181     * @memberOf PooledObject
182     */
183    create: function() {
184       // Check the pool for the object type
185       if (PooledObject.objectPool[this.getClassName()] && PooledObject.objectPool[this.getClassName()].length != 0) {
186          PooledObject.poolSize--;
187          Engine.addMetric("poolUse", Math.floor((PooledObject.poolSize / PooledObject.poolNew) * 100), false, "#%");
188          var obj = PooledObject.objectPool[this.getClassName()].shift();
189          obj.constructor.apply(obj, arguments);
190          return obj;
191       } else {
192          PooledObject.poolNew++;
193          Engine.addMetric("newObjs", PooledObject.poolNew, false, "#");
194          // Any more than 10 arguments and we're pooched!
195          return new this(arguments[0],arguments[1],arguments[2],arguments[3],arguments[4],arguments[5],arguments[6],arguments[7],arguments[8],arguments[9]);
196       }
197    },
198 
199    /**
200     * During object destruction, it needs to return itself to the
201     * pool so it can be used again.  Instead of creating new instances
202     * of each object, we utilize a pooled object so we don't need the
203     * garbage collector to be invoked for removed objects.
204     * @private
205     */
206    returnToPool: function(obj) {
207       // If there is no pool for objects of this type, create one
208       if (!PooledObject.objectPool[obj.constructor.getClassName()]) {
209          PooledObject.objectPool[obj.constructor.getClassName()] = new Array;
210       }
211 
212       // Push this object into the pool
213       PooledObject.poolSize++;
214       PooledObject.objectPool[obj.constructor.getClassName()].push(obj);
215    },
216 
217    /**
218     * The pool of all objects.  Each object is stored
219     * by it's classname, retrieved from <tt>getClassName()</tt>
220     * that should be defined by all objects.
221     */
222    objectPool: {},
223 
224    /**
225     * Get the class name of this object
226     *
227     * @return {String}
228     * @memberOf PooledObject
229     */
230    getClassName: function() {
231       if (!this.hasOwnProperty("getClassName")) {
232          Console.warn("Object is missing getClassName()");
233       }
234       return "PooledObject";
235    }
236 });
237 
238 return PooledObject;
239 
240 });
