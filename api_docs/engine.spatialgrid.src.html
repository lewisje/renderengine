 70 /**
 71  * @class A structure which divides a finite space up into a more
 72  *        coarse grid to assist in quickly finding objects within that
 73  *        space.
 74  *
 75  * @param width {Number} The width of the area
 76  * @param height {Number} The height of the area
 77  * @param divisions {Number} The number of divisions along both axis
 78  * @extends SpatialContainer
 79  */
 80 var SpatialGrid = SpatialContainer.extend(/** @scope SpatialGrid.prototype */{
 81 
 82    divisions: 1,
 83 
 84    xLocator: 1,
 85    yLocator: 1,
 86 
 87    constructor: function(width, height, divisions) {
 88       this.base("SpatialGrid", width, height);
 89 
 90       // Divide the space up into a grid
 91       var gX = Math.floor(width / divisions);
 92       var gY = Math.floor(height / divisions);
 93       this.divisions = divisions;
 94       this.xLocator = 1 / gX;
 95       this.yLocator = 1 / gY;
 96 
 97       var grid = [];
 98       this.setRoot(grid);
 99 
100       for (var y = 0; y < this.divisions; y++)
101       {
102          for (var x = 0; x < this.divisions; x++)
103          {
104             var rect = new Rectangle2D(x * gX, y * gY, gX, gY);
105             grid[x + (y * this.divisions)] = new GridNode(rect);
106          }
107       }
108    },
109 
110    /**
111     * Find the node that contains the specified point.
112     *
113     * @param point {Point2D} The point to locate the node for
114     * @type SpatialNode
115     */
116    findNodePoint: function(point) {
117       return this.getRoot()[Math.floor(point.x * this.xLocator) + (Math.floor(point.y * this.yLocator) * this.divisions)];
118    },
119 
120    /**
121     * Get the node within the grid.
122     * @param x {Number} The virtual X coordinate in our grid
123     * @param y {Number} The virtual Y coordinate in our grid
124     * @type Number
125     * @private
126     */
127    getNode: function(x, y) {
128       return this.getRoot()[x + (y * this.divisions)];
129    },
130 
131    /**
132     * Get the list of objects with respect to the point given.  Objects will
133     * be returned from the cross that makes up the grid node containing
134     * the point, and the four adjacent points.  For example, if you had a
135     * 3x3 grid with the object in the center node, the nodes marked with
136     * asterisks below would be included in the result set:
137     * <pre>
138     *  +---+---+---+
139     *  |   | * |   |
140     *  +---+---+---+
141     *  | * | * | * |
142     *  +---+---+---+
143     *  |   | * |   |
144     *  +---+---+---+
145     * </pre>
146     *
147     * @param point {Point2D} The point to begin the search at.
148     * @returns An array of objects found within the cross
149     * @type Array
150     */
151    getPCL: function(point) {
152 
153       // We'll build our list from the 5 node cross section
154       var x = Math.floor(point.x * this.xLocator);
155       var y = Math.floor(point.y * this.yLocator);
156 
157       // if our borders cross the margin, we can drop up to two nodes
158       var nodes = [];
159       nodes.push(this.getNode(x, y));
160       if (x > 0) { nodes.push(this.getNode(x - 1, y)); }
161       if (x < this.divisions) { nodes.push(this.getNode(x + 1, y)); }
162       if (y > 0) { nodes.push(this.getNode(x, y - 1)); }
163       if (y < this.divisions) { nodes.push(this.getNode(x, y + 1)); }
164 
165       var o = [];
166       if (nodes.length > 0)
167       {
168          for (var n = 0; n < nodes.length; n++)
169          {
170             if (nodes[n])
171             {
172                o = o.concat(nodes[n].getObjects());
173             }
174          }
175       }
176       return o;
177    },
178 
179    /**
180     * Get the class name of this object
181     *
182     * @type String
183     * @memberOf Container
184     */
185    getClassName: function() {
186       return "SpatialGrid";
187    }
188 
189 });
