 33 Engine.include("/engine/engine.container.js");
 34 Engine.include("/components/component.base.js");

 36 Engine.initObject("HostComponent", "LogicComponent", function() {
 37 
 38 /**
 39  * @class A component that can execute host objects.  Allows embedding
 40  *        of multiple objects into one object.
 41  * @extends LogicComponent
 42  */
 43 var HostComponent = LogicComponent.extend(/** @scope HostComponent.prototype */{
 44 
 45    objects: null,
 46 
 47    /**
 48     * @constructor
 49     * @memberOf Transform2DComponent
 50     */
 51    constructor: function(name, priority) {
 52       this.base(name, priority || 1.0);
 53       this.objects = new HashContainer();
 54    },
 55 
 56    release: function() {
 57       this.base();
 58       this.objects = null;
 59    },
 60 
 61    destroy: function() {
 62       this.objects.destroy();
 63       this.base();
 64    },
 65 
 66    /**
 67     * Add a {@link HostObject} to the component to be processed when
 68     * this component is executed.
 69     *
 70     * @param name {String} A unique name to refer to the object by
 71     * @param obj {HostObject} The host object reference
 72     */
 73    add: function(name, obj) {
 74       this.objects.add(name.toUpperCase(), obj);
 75    },
 76 
 77    /**
 78     * Retrieves the {@link HostObject} that is associated with the
 79     * given name from the component.
 80     *
 81     * @param name {String} The unique name of the object
 82     * @type HostObject
 83     */
 84    get: function(name) {
 85       return this.objects.get(name.toUpperCase());
 86    },
 87 
 88    /**
 89     * Remove the host object from the component.
 90     *
 91     * @param obj {HostObject} The host object reference
 92     */
 93    remove: function(obj) {
 94       return this.objects.remove(obj);
 95    },
 96 
 97    /**
 98     * Update each of the host objects hosted in this component.
 99     *
100     * @param renderContext {RenderContext} The rendering context
101     * @param time {Number} The engine time in milliseconds
102     */
103    execute: function(renderContext, time) {
104       var objs = this.objects.getObjects();
105       for (var c in objs) {
106          objs[c].update(renderContext, time);
107       }
108    }
109 }, {
110    /**
111     * Get the class name of this object
112     *
113     * @type String
114     * @memberOf Transform2DComponent
115     */
116    getClassName: function() {
117       return "HostComponent";
118    }
119 });
120 
121 return HostComponent;
122 
123 });
