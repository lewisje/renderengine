 31 /**
 32  * @class Loads sounds and stores the reference to them.  Provides a simple
 33  *        way to play and stop the sounds that have been loaded.
 34  *        <p/>
 35  *        Sounds must be 44.1KHz and have a bitrate of 192k to play correctly.
 36  *
 37  * @extends ResourceLoader
 38  */
 39 var SoundResourceLoader = ResourceLoader.extend(/** @scope SoundResourceLoader.prototype */{
 40 
 41    init: false,
 42 
 43    playingSounds: null,
 44 
 45    constructor: function() {
 46       this.base("SoundLoader");
 47       this.init = false;
 48       this.playingSounds = {};
 49    },
 50 
 51    /**
 52     * Load a sound resource from a URL.
 53     *
 54     * @param name {String} The name of the resource
 55     * @param url {String} The URL where the resource is located
 56     */
 57    load: function(name, url) {
 58 
 59       if (!Engine.isSoundEnabled()) {
 60          return;
 61       }
 62 
 63       // Only MP3 files are supported
 64       Assert(url.indexOf(".mp3") > 0, "Only MP3 sound format is supported!");
 65 
 66       // Create the sound object
 67       var sound = Engine.soundManager.createSound({
 68          "id": name,
 69          "url": url,
 70          "autoPlay": false,
 71          "volume": 50
 72       });
 73 
 74 
 75       if (!this.init) {
 76          Engine.getDefaultContext().add(this);
 77          this.init = true;
 78       }
 79 
 80       this.base(name, sound);
 81    },
 82 
 83    /**
 84     * Get the sound object.  If the sound object isn't found and the
 85     * sound engine didn't initialize, this will return a safe object
 86     * whose methods can be called without exceptions.
 87     *
 88     * @param name {String} The name of the sound to retrieve.
 89     * @type SMSound
 90     */
 91    get: function(name) {
 92       var sound = this.base(name);
 93       if (!sound && !Engine.isSoundEnabled()) {
 94          console.debug("in here");
 95          // If the sound engine didn't load, this will return an object
 96          // that won't throw exceptions when methods are called.
 97          sound = {
 98             play: function() {},
 99             stop: function() {}
100          };
101       }
102 
103       return sound;
104    },
105 
106    /**
107     * The name of the resource this loader will get.
108     * @returns A String that represents the resource type.
109     */
110    getResourceType: function() {
111       return "sound";
112    }
113 }, {
114    /**
115     * Get the class name of this object
116     *
117     * @type String
118     */
119    getClassName: function() {
120       return "SoundResourceLoader";
121    }
122 });
