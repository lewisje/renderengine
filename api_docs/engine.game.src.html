 34 /**
 35  * @class The game object represents an instance of a game.  It is
 36  * the controlling entity for all of a game and is responsible
 37  * for setup and teardown of the game.
 38  */
 39 var Game = Base.extend(/** @scope Game.prototype */{
 40 
 41    constructor: null,
 42 
 43    /**
 44     * Initialize the game.
 45     * @memberOf Game
 46     * @static
 47     */
 48    setup: function() {
 49    },
 50 
 51    /**
 52     * Shut down the game.
 53     * @memberOf Game
 54     * @static
 55     */
 56    tearDown: function() {
 57    },
 58 
 59    /**
 60     * Get the display name of the game.
 61     * @memberOf Game
 62     * @static
 63     */
 64    getName: function() {
 65       return "Game";
 66    },
 67 
 68    /**
 69     * Load a script relative to this game.  Scripts cannot be specified
 70     * with an absolute URL.
 71     *
 72     * @param scriptSource {String} The relative path to the script to load.
 73     * @memberOf Game
 74     * @static
 75     */
 76    load: function(scriptSource) {
 77       Assert((scriptSource.indexOf("http") == -1), "Game scripts can only be loaded relative to the game's path");
 78       Engine.loadScript( (scriptSource.charAt(0) != "/" ? "./" : ".") + scriptSource );
 79    },
 80 
 81    /**
 82     * Get the number of players this game supports.
 83     *
 84     * @type Number
 85     * @memberOf Game
 86     * @static
 87     */
 88    getPlayers: function() {
 89       return 1;
 90    }
 91 });
