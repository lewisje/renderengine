 32 // Includes
 33 Engine.include("/rendercontexts/context.render2d.js");

 35 Engine.initObject("HTMLElementContext", "RenderContext2D", function() {
 36 
 37 /**
 38  * @class A reference to the <tt>document.body</tt> element as a rendering context.
 39  * Aside from being The Render Engine's default rendering context, the context
 40  * is essentially a wrapper for the HTML document.  Wrapping, in this way, allows
 41  * us to update not only this context, but all other contexts during an engine frame.
 42  *
 43  * @extends RenderContext2D
 44  */
 45 var HTMLElementContext = RenderContext2D.extend(/** @scope DocumentContext.prototype */{
 46 
 47    /**
 48     * Create an instance of an HTML element rendering context.  This context
 49     * represents any HTML element.
 50     * @memberOf HTMLElementContext
 51     * @constructor
 52     */
 53    constructor: function(name, element) {
 54       this.base(name || "HTMLElementContext", element);
 55    },
 56 
 57    /**
 58     * Eliminate the elements from container element.
 59     * @memberOf HTMLElementContext
 60     */
 61    destroy: function() {
 62       var objs = this.getObjects();
 63       for (var o in objs)
 64       {
 65          var e = objs[o].getElement();
 66          if (e && e != document.body) {
 67             Console.log("DOM element ", e, " removed from ", this.getSurface());
 68             this.getSurface().removeChild(e);
 69          }
 70       }
 71 
 72       this.base();
 73    },
 74 
 75    /**
 76     * @memberOf HTMLElementContext
 77     */
 78    add: function(obj) {
 79       if (obj.getElement())
 80       {
 81          this.getSurface().appendChild(obj.getElement());
 82       }
 83       this.base(obj);
 84    },
 85 
 86    /**
 87     * @memberOf HTMLElementContext
 88     */
 89    remove: function(obj) {
 90       if (obj.getElement())
 91       {
 92          this.getSurface().removeChild(obj.getElement());
 93       }
 94       this.base(obj);
 95    }
 96 }, /** @scope HTMLElementContext */{
 97 
 98    /**
 99     * Get the class name of this object
100     *
101     * @type String
102     */
103    getClassName: function() {
104       return "HTMLElementContext";
105    }
106 });
107 
108 return HTMLElementContext;
109 
110 });
