 36 Engine.initObject("HTMLElementContext", "RenderContext2D", function() {
 37 
 38 /**
 39  * @class A reference to the <tt>document.body</tt> element as a rendering context.
 40  * Aside from being The Render Engine's default rendering context, the context
 41  * is essentially a wrapper for the HTML document.  Wrapping, in this way, allows
 42  * us to update not only this context, but all other contexts during an engine frame.
 43  *
 44  * @extends RenderContext2D
 45  */
 46 var HTMLElementContext = RenderContext2D.extend(/** @scope HTMLElementContext.prototype */{
 47 
 48    transformStack: null,
 49    
 50    cursorPos: null,
 51    
 52    jQObj: null,
 53 
 54    /**
 55     * Create an instance of an HTML element rendering context.  This context
 56     * represents any HTML element.
 57     * @memberOf HTMLElementContext
 58     * @constructor
 59     */
 60    constructor: function(name, element) {
 61       this.base(name || "HTMLElementContext", element);
 62       element.id = this.getId();
 63       this.cursorPos = Point2D.create(0,0);
 64       this.transformStack = [];
 65       this.pushTransform();
 66       this.jQObj = null;
 67       this.setViewport(Rectangle2D.create(0, 0, this.jQ().width(), this.jQ().height()));
 68    },
 69 
 70    /**
 71     * Eliminate the elements from container element.
 72     * @memberOf HTMLElementContext
 73     */
 74    destroy: function() {
 75       var objs = this.getObjects();
 76       for (var o in objs)
 77       {
 78          var e = objs[o].getElement();
 79          if (e && e != document.body) {
 80             Console.log("DOM element ", e, " removed from ", this.getSurface());
 81             this.getSurface().removeChild(e);
 82          }
 83       }
 84 
 85       this.base();
 86    },
 87 
 88    /**
 89     * Retrieve the jQuery object which represents the element.
 90     * @return jQuery Object
 91     * @type {jQuery}
 92     */
 93    jQ: function() {
 94       if (this.jQObj == null) {
 95          this.jQObj = $(this.getSurface());
 96       }
 97       return this.jQObj;
 98    },
 99 
100    /**
101     * @memberOf HTMLElementContext
102     */
103    add: function(obj) {
104       if (!obj.getElement())
105       {
106 			// Create an element for the object
107 			obj.setElement($("<div>").css("position", "absolute"));
108       }
109       this.jQ().append(obj.getElement());
110       this.base(obj);
111    },
112 
113    /**
114     * @memberOf HTMLElementContext
115     */
116    remove: function(obj) {
117       if (obj.getElement())
118       {
119          this.jQ().remove(obj.getElement());
120       }
121       this.base(obj);
122    },
123    
124    /**
125     * Push a transform state onto the stack.
126     */
127    pushTransform: function() {
128       this.base();
129       this.transformStack.push(this.cursorPos); 
130    },
131    
132    /**
133     * Pop a transform state off the stack.
134     */
135    popTransform: function() {
136       this.base();
137       this.cursorPos = this.transformStack.pop();
138    },
139 
140    //================================================================
141    // Drawing functions
142 
143    /**
144     * Set the background color of the context.
145     *
146     * @param color {String} An HTML color
147     */
148    setBackgroundColor: function(color) {
149       this.base(color);
150       this.jQ().css("background-color", color);
151    },
152 
153    /**
154     * Set the current transform position (translation).
155     *
156     * @param point {Point2D} The translation
157     */
158    setPosition: function(point) {
159       this.cursorPos = Point2D.create(point);
160       this.base(point);
161    },
162 
163   /**
164     * Set the width of the context drawing area.
165     *
166     * @param width {Number} The width in pixels
167     */
168    setWidth: function(width) {
169       this.base(width);
170       this.jQ().width(width);
171    },
172 
173    /**
174     * Set the height of the context drawing area
175     *
176     * @param height {Number} The height in pixels
177     */
178    setHeight: function(height) {
179       this.base(height);
180       this.jQ().height(height);
181    },
182 
183    /**
184     * Draw an un-filled rectangle on the context.
185     *
186     * @param rect {Rectangle2D} The rectangle to draw
187     */
188    drawRectangle: function(rect) {
189       var rD = rect.getDims();
190       var d = $("<div>").css({
191          borderWidth: this.getLineWidth(),
192          borderColor: this.getLineStyle(),
193          left: rD.l,
194          top: rD.t,
195          width: rD.w,
196          height: rD.h,
197          position: "absolute"
198       });
199       this.jQ().append(d);
200       return d;
201    },
202 
203    /**
204     * Draw a filled rectangle on the context.
205     *
206     * @param rect {Rectangle2D} The rectangle to draw
207     */
208    drawFilledRectangle: function(rect) {
209       var rD = rect.getDims();
210       var d = $("<div>").css({
211          borderWidth: this.getLineWidth(),
212          borderColor: this.getLineStyle(),
213          backgroundColor: this.getFillStyle(),
214          left: rD.l,
215          top: rD.t,
216          width: rD.w,
217          height: rD.h,
218          position: "absolute"
219       });
220       this.jQ().append(d);
221       return d;
222    },
223 
224    /**
225     * Draw a point on the context.
226     *
227     * @param point {Point2D} The position to draw the point
228     */
229    drawPoint: function(point) {
230       return this.drawFilledRectangle(Rectangle2D.creat(point.x, point.y, 1, 1));
231    },
232 
233    /**
234     * Draw a sprite on the context.
235     *
236     * @param obj {Object} A reference object, or <tt>null</tt>
237     * @param sprite {Sprite} The sprite to draw
238     * @param time {Number} The current world time
239     */
240    drawSprite: function(obj, sprite, time) {
241       var f = sprite.getFrame(time);
242       var tl = f.getTopLeft();
243       var rD = f.getDims();
244 
245       // If the reference object is a host object it
246       // will give us a reference to the HTML element which we can then
247       // just modify the displayed image for.
248       if (obj && (obj instanceof HostObject) && obj.jQ()) {
249          obj.jQ().css({
250             top: this.cursorPos.y,
251             left: this.cursorPos.x,
252             width: rD.w,
253             height: rD.h,
254             backgroundPosition: -tl.x + "px " + tl.y + "px"
255          });
256       } else {
257          // Otherwise, just draw a new div with the sprite on it
258          var d = $("<div>").css({
259             position: "absolute",
260             top: this.cursorPos.y,
261             left: this.cursorPos.x,
262             width: rD.w,
263             height: rD.h,
264             background: "url(" + f.getSourceImage().src + ")",
265             backgroundPosition: -tl.x + "px " + tl.y + "px"
266          });
267          this.jQ().append(d);
268          return d;
269       }
270    },
271 
272    /**
273     * Draw an image on the context.
274     *
275     * @param obj {Object} A reference object, or <tt>null</tt>
276     * @param rect {Rectangle2D} The rectangle that specifies the position and
277     *             dimensions of the image rectangle.
278     * @param image {Object} The image to draw onto the context
279     * @param [srcRect] {Rectangle2D} <i>[optional]</i> The source rectangle within the image, if
280     *                <tt>null</tt> the entire image is used
281     */
282    drawImage: function(obj, rect, image, srcRect) {
283       var rD = rect.getDims();
284 		var sD = srcRect ? srcRect.getDims() : rect.getDims();
285 
286       // If the reference object is a host object it
287       // will give us a reference to the HTML element which we can then
288       // just modify the displayed image for.
289       if (obj && (obj instanceof HostObject) && obj.jQ()) {
290          obj.jQ().css({
291             top: rD.t,
292             left: rD.l,
293             width: rD.w,
294             height: rD.h});
295          // Only modify the source, width, and height if they change it
296          if (obj.jQ().attr("src") != image.src) {
297             obj.jQ().attr("src", image.src);
298          }
299          if (obj.jQ().attr("width") != rD.w) {
300             obj.jQ().attr("width", rD.w)
301          }
302          if (obj.jQ().attr("height") != rD.h) {
303             obj.jQ().attr("height", rD.h)
304          }
305       } else {
306          // Otherwise, just draw a new image
307          var i = $(image).clone().css({
308             position: "absolute",
309             left: sD.l,
310             top: sD.t,
311             width: sD.w,
312             height: sD.h
313          });
314 			i.attr("width", sD.w);
315 			i.attr("height", sD.h);
316          this.jQ().append(i);
317          return i;
318       }
319    },
320 
321    /**
322     * Draw text on the context.
323     *
324     * @param point {Point2D} The top-left position to draw the image.
325     * @param text {String} The text to draw
326     */
327    drawText: function(point, text) {
328 		this.base(point, text);
329       var d = $("<span>").css({
330          font: this.getNormalizedFont(),
331 			left: point.x,
332          top: point.y,
333          position: "absolute"
334       }).text(text);
335       this.jQ().append(d);
336       return d;
337    }

350 	/**
351 	 * When the HTMLElementContext's dependencies are resolved, check the browser
352 	 * type and load a specific object which handles specific processing.
353 	 */
354 	resolved: function() {
355 		var version = parseFloat(EngineSupport.sysInfo().version);
356 		switch (EngineSupport.sysInfo().browser) {
357 			case "safari" :
358 					if (version >= 3) {
359 						// Load support for webkit transforms
360 						Engine.include("/rendercontexts/support/context.htmlelement.safari.js");
361 					}
362 				break;
363 			case "mozilla" :
364 					if (version >= 1.9) {
365 						// Load support for gecko transforms
366 						Engine.include("/rendercontexts/support/context.htmlelement.gecko.js");
367 					}
368 				break;
369 			default:
370 				break;
371 		};
372 		
373 	}
