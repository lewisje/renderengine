 33 /**
 34  * @class The base abstract class for all timers.
 35  *
 36  * @param name {String} The name of the timer
 37  * @param interval {Number} The interval for the timer, in milliseconds
 38  * @param callback {Function} The function to call when the interval is reached
 39  *
 40  * @extends BaseObject
 41  */
 42 var Timer = BaseObject.extend(/** @scope Timer.prototype */{
 43 
 44    timer: null,
 45 
 46    running: false,
 47 
 48    constructor: function(name, interval, callback) {
 49       callback = name instanceof Number ? interval : callback;
 50       interval = name instanceof Number ? name : interval;
 51       name = name instanceof Number ? "Timer" : name;
 52 
 53       Assert((typeof callback == "function"), "Callback must be a function in Timer");
 54 
 55       this.base(name);
 56       this.interval = interval;
 57       this.callback = callback;
 58       this.restart();
 59    },
 60 
 61    release: function() {
 62       this.base();
 63       this.timer = null;
 64       this.running = false;
 65    },
 66 
 67    /**
 68     * Stop the timer and remove it from the system
 69     */
 70    destroy: function() {
 71       this.timer = null;
 72       this.base();
 73    },
 74 
 75    /**
 76     * Get the underlying system timer object.
 77     */
 78    getTimer: function() {
 79       return this.timer;
 80    },
 81 
 82    /**
 83     * Set the underlying system timer object.
 84     *
 85     * @param timer {Object} The timer object
 86     */
 87    setTimer: function(timer) {
 88       this.timer = timer;
 89    },
 90 
 91    /**
 92     * Returns <tt>true</tt> if the timer is currently running.
 93     * @type Boolean
 94     */
 95    isRunning: function() {
 96       return this.running;
 97    },
 98 
 99    /**
100     * Cancel the timer.
101     */
102    cancel: function() {
103       this.timer = null;
104       this.running = false;
105    },
106 
107    /**
108     * Cancel the running timer and restart it.
109     */
110    restart: function() {
111       this.cancel();
112       this.running = true;
113    },
114 
115    /**
116     * Set the callback function for this timer.  If the timer is
117     * currently running, it will be restarted.
118     *
119     * @param callback {Function} A function object to call
120     */
121    setCallback: function(callback) {
122       Assert((typeof callback == "function"), "Callback must be a function in Timer.setCallback");
123       this.callback = callback;
124       if (this.isRunning)
125       {
126          this.restart();
127       }
128    },
129 
130    /**
131     * Get the callback function for this timer.
132     * @type Function
133     */
134    getCallback: function() {
135       return this.callback;
136    },
137 
138    /**
139     * Set the interval of this timer.  If the timer is running, it
140     * will be cancelled.
141     *
142     * @param interval {Number} The interval of this timer, in milliseconds
143     */
144    setInterval: function(interval) {
145       this.cancel();
146       this.interval = interval;
147    },
148 
149    /**
150     * Get the interval of this timer, in milliseconds.
151     * @type Number
152     */
153    getInterval: function() {
154       return this.interval;
155    }
156 }, {
157    /**
158     * Get the class name of this object
159     * @type String
160     */
161    getClassName: function() {
162       return "Timer";
163    }
164 });

166 /**
167  * @class A timer that wraps the <tt>window.setTimeout</tt> method.
168  * @extends Timer
169  */
170 var Timeout = Timer.extend(/** @scope Timeout.prototype */{
171 
172    /**
173     * Cancel this timeout timer.
174     */
175    cancel: function() {
176       window.clearTimeout(this.getTimer());
177       this.base();
178    },
179 
180    destroy: function() {
181       this.cancel();
182       this.base();
183    },
184 
185    /**
186     * Restart this timeout timer
187     */
188    restart: function() {
189       this.setTimer(window.setTimeout(this.getCallback(), this.getInterval()));
190    }
191 }, {
192    /**
193     * Get the class name of this object
194     * @type String
195     */
196    getClassName: function() {
197       return "Timeout";
198    }

201 /**
202  * @class A one-shot timer that cannot be restarted and will
203  *        self-destroy after it completes its interval..
204  *
205  * @param name {String} The name of the timer
206  * @param interval {Number} The interval for the timer, in milliseconds
207  * @param callback {Function} The function to call when the interval is reached
208  */
209 var OneShotTimeout = Timeout.extend(/** @scope OneShotTimeout.prototype */{
210 
211    constructor: function(name, interval, fn) {
212 
213       var cb = function() {
214          arguments.callee.timer.destroy();
215          arguments.callee.cbFn();
216       };
217       cb.cbFn = fn;
218       cb.timer = this;
219 
220       this.base(name, interval, cb);
221    },
222 
223    /**
224     * This timer cannot be restarted.
225     * @private
226     */
227    restart: function() {
228       if (this.running)
229       {
230          return;
231       }
232 
233       this.base();
234    }
235 }, {
236 
237    /**
238     * Get the class name of this object
239     * @type String
240     */
241    getClassName: function() {
242       return "OneShotTimeout";
243    }
244 });
