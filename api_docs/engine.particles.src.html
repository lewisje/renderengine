 38 Engine.initObject("Particle", "PooledObject", function() {
 39 
 40 /**
 41  * @class Base particle class.  A particle need only implement the
 42  *        draw method.  The remainder of the functionality is
 43  *        handled by this abstract class.
 44  */
 45 var Particle = PooledObject.extend(/** @scope Particle.prototype */{
 46 
 47    life: 0,
 48 
 49    engine: null,
 50 
 51    birth: 0,
 52 
 53    constructor: function(lifetime) {
 54       this.base("Particle");
 55       this.life = lifetime;
 56       this.birth = 0;
 57    },
 58 
 59    release: function() {
 60       this.base();
 61       this.life = 0;
 62       this.engine = null;
 63       this.birth = 0;
 64    },
 65 
 66    init: function(pEngine, time) {
 67       this.engine = pEngine;
 68       this.life += time;
 69       this.birth = time;
 70    },
 71 
 72    update: function(renderContext, time) {
 73       if (time < this.life)
 74       {
 75          // Draw it ( I think this is an optimization point )
 76          renderContext.pushTransform();
 77          this.draw(renderContext, time);
 78          renderContext.popTransform();
 79       }
 80       else
 81       {
 82          // Remove it
 83          this.engine.removeParticle(this);
 84       }
 85    },
 86 
 87    getTTL: function() {
 88 		return this.life;
 89 	},
 90 
 91 	getBirth: function() {
 92 		return this.birth;
 93 	},
 94 
 95    draw: function(renderContext, time) {
 96    }
 97 }, {
 98    /**
 99     * Get the class name of this object
100     *
101     * @type String
102     */
103    getClassName: function() {
104       return "Particle";
105    }
106 });
107 
108 return Particle;
109 
110 });

112 Engine.initObject("ParticleEngine", "BaseObject", function() {
113 
114 /**
115  * @class The particle engine is simply a system for updating, and expiring
116  *        particles within a game environment.  This is registered with the
117  *        render context so it will be updated at regular intervals.
118  *
119  */
120 var ParticleEngine = BaseObject.extend(/** @scope ParticleEngine.prototype */{
121 
122    particles: null,
123 
124    lastTime: 0,
125 
126    dead: null,
127 
128    constructor: function() {
129       this.base("ParticleEngine");
130       this.particles = [];
131       this.dead = [];
132    },
133 
134    release: function() {
135       this.base();
136       this.particles = null,
137       this.lastTime = 0;
138       this.dead = null;
139    },
140 
141    /**
142     * Add a particle to the system.
143     *
144     * @param particle {Particle} A particle to animate
145     */
146    addParticle: function(particle) {
147       this.particles.push(particle);
148       particle.init(this, this.lastTime);
149    },
150 
151    /**
152     * Mark a particle as dead so we can remove it after
153     * the update.
154     *
155     * @param particle {Particle} A particle to animate
156     */
157    removeParticle: function(particle) {
158       this.dead.push(particle);
159    },
160 
161    /**
162     * Kill off all of the dead particles.
163     * @private
164     */
165    kill: function() {
166       for (var d = 0; d < this.dead.length; d++)
167       {
168          this.dead[d].destroy();
169          EngineSupport.arrayRemove(this.particles, this.dead[d]);
170       }
171       this.dead = [];
172    },
173 
174    /**
175     * Update the particles within the render context, and for the specified time.
176     *
177     * @param renderContext {RenderContext} The context the particles will be rendered within.
178     * @param time {Number} The global time within the engine.
179     */
180    update: function(renderContext, time) {
181       var p = 1;
182 
183       this.lastTime = time;
184       if (this.particles.length == 0)
185       {
186          return;
187       }
188 
189 
190       // Using Duff's device with loop inversion
191       switch((this.particles.length - 1) & 0x3)
192       {
193          case 3:
194             this.particles[p++].update(renderContext, time);
195          case 2:
196             this.particles[p++].update(renderContext, time);
197          case 1:
198             this.particles[p++].update(renderContext, time);
199       }
200 
201       if (p < this.particles.length)
202       {
203          do
204          {
205             this.particles[p++].update(renderContext, time);
206             this.particles[p++].update(renderContext, time);
207             this.particles[p++].update(renderContext, time);
208             this.particles[p++].update(renderContext, time);
209          } while (p < this.particles.length);
210       }
211 
212       if (this.dead.length)
213       {
214          // Kill off dead particles
215          this.kill();
216       }
217    },
218 
219 	getProperties: function() {
220 		var self = this;
221 		var prop = this.base(self);
222 		return $.extend(prop, {
223 			"Count" : [function() { return self.particles.length; },
224 						  null, false]
225 		});
226 	}
227 
228 
229 }, {
230    /**
231     * Get the class name of this object
232     *
233     * @type String
234     */
235    getClassName: function() {
236       return "ParticleEngine";
237    }
238 });
239 
240 return ParticleEngine;
241
